## 1001. Grid Illumination
https://leetcode.com/problems/grid-illumination/

哈希表计数行、列、两条对角线上的灯。

```python
from collections import defaultdict
def gridIllumination(self, n, lamps, queries):
    rows, cols, diag1, diag2 = defaultdict(int), defaultdict(int), defaultdict(int), defaultdict(int)
    lit = set()
    for r, c in lamps:
        if (r, c) not in lit:
            lit.add((r, c))
            rows[r] += 1; cols[c] += 1; diag1[r-c] += 1; diag2[r+c] += 1
    res = []
    for r, c in queries:
        if rows[r] or cols[c] or diag1[r-c] or diag2[r+c]:
            res.append(1)
        else:
            res.append(0)
        for dr in range(-1, 2):
            for dc in range(-1, 2):
                nr, nc = r+dr, c+dc
                if (nr, nc) in lit:
                    lit.remove((nr, nc))
                    rows[nr] -= 1; cols[nc] -= 1; diag1[nr-nc] -= 1; diag2[nr+nc] -= 1
    return res
```

```python
def commonChars(self, words):
    common = Counter(words[0])
    for w in words[1:]:
        common &= Counter(w)
```

```python
    left = 0
    for right in range(len(nums)):
```

```python
def largestSumAfterKNegations(self, nums, k):

    for i in range(len(nums)):
        if nums[i] < 0 and k > 0:
            nums[i] = -nums[i]; k -= 1
```

## 1006. Clumsy Factorial

```python
def clumsy(self, n):
    stack = [n]
    ops = ['*', '/', '+', '-']
    idx = 0
    for i in range(n - 1, 0, -1):
        op = ops[idx % 4]
        if op == '*': stack[-1] *= i
        elif op == '/': stack[-1] = int(stack[-1] / i)
        elif op == '+': stack.append(i)
        else: stack.append(-i)
        idx += 1
    return sum(stack)
```

```python
def minDominoRotations(self, tops, bottoms):
    def check(val):
        rot_top = rot_bot = 0
        for t, b in zip(tops, bottoms):
            if t != val and b != val: return float('inf')
            elif t != val: rot_top += 1
            elif b != val: rot_bot += 1
        return min(rot_top, rot_bot)
    res = min(check(tops[0]), check(bottoms[0]))
    return res if res != float('inf') else -1
```

```python
        d, cur = 1, 0
        for w in weights:
            if cur + w > cap:
                d += 1; cur = 0
            cur += w
```

```python
def numEnclaves(self, grid):
    m, n = len(grid), len(grid[0])
    def dfs(i, j):
        if i < 0 or i >= m or j < 0 or j >= n or grid[i][j] == 0: return
        grid[i][j] = 0
        for di, dj in [(-1,0),(1,0),(0,-1),(0,1)]: dfs(i+di, j+dj)
    for i in range(m):
        for j in range(n):
            if (i in (0, m-1) or j in (0, n-1)) and grid[i][j] == 1:
                dfs(i, j)
    return sum(sum(row) for row in grid)
```

```python
    for s, e in clips:
        if s <= time:
            max_reach[s] = max(max_reach[s], e)
```

```python
def longestArithSeqLength(self, nums):
    n = len(nums)
    dp = [{} for _ in range(n)]
    res = 2
    for i in range(n):
        for j in range(i):
            d = nums[i] - nums[j]

            res = max(res, dp[i][d])
```

```python
class StreamChecker:
    def __init__(self, words):
        self.trie = {}
        for w in words:
            node = self.trie
            for c in reversed(w):
                node = node.setdefault(c, {})
            node['#'] = True
        self.stream = []
    def query(self, letter):
        self.stream.append(letter)
        node = self.trie
        for c in reversed(self.stream):

            node = node[c]
            if '#' in node: return True
        return False
```

```python
def colorBorder(self, grid, row, col, color):

    borders = []
    def dfs(i, j):
```

```python
def minScoreTriangulation(self, values):
    @lru_cache(None)
    def dp(i, j):
        if j - i < 2: return 0
```

```python
def lastStoneWeight(self, stones):

    heapq.heapify(h)
    while len(h) > 1:
```

```python
        for j in range(target, s - 1, -1):
            dp[j] = dp[j] or dp[j - s]
    for j in range(target, -1, -1):
        if dp[j]: return total - 2 * j
```

```python
def addNegabinary(self, arr1, arr2):
    res = []; carry = 0
    i, j = len(arr1) - 1, len(arr2) - 1
    while i >= 0 or j >= 0 or carry:
        s = carry
        if i >= 0: s += arr1[i]; i -= 1
        if j >= 0: s += arr2[j]; j -= 1
        res.append(s & 1)
        carry = -(s >> 1)
    while len(res) > 1 and res[-1] == 0: res.pop()
    return res[::-1]
```

```python
def sufficientSubset(self, root, limit):
    def dfs(node, cur):
        if not node: return None
        cur += node.val
        if not node.left and not node.right:
            return node if cur >= limit else None
        node.left = dfs(node.left, cur)
        node.right = dfs(node.right, cur)
        return node if node.left or node.right else None
    return dfs(root, 0)
```

```python
def smallestSubsequence(self, s):
    last = {c: i for i, c in enumerate(s)}
    stack = []; seen = set()
    for i, c in enumerate(s):
        if c in seen: continue
        while stack and c < stack[-1] and last[stack[-1]] > i:
            seen.remove(stack.pop())
        stack.append(c); seen.add(c)
    return ''.join(stack)
```

## 1091. Shortest Path in Binary Matrix
https://leetcode.com/problems/shortest-path-in-binary-matrix/

BFS 八方向搜索。

```python
from collections import deque
def shortestPathBinaryMatrix(self, grid):
    n = len(grid)
    if grid[0][0] or grid[n-1][n-1]: return -1
    q = deque([(0, 0, 1)])
    grid[0][0] = 1
    while q:
        i, j, d = q.popleft()
        if i == n-1 and j == n-1: return d
        for di in range(-1, 2):
            for dj in range(-1, 2):
                ni, nj = i+di, j+dj
                if 0 <= ni < n and 0 <= nj < n and grid[ni][nj] == 0:
                    grid[ni][nj] = 1
                    q.append((ni, nj, d+1))
    return -1
```

## 1092. Shortest Common Supersequence
https://leetcode.com/problems/shortest-common-supersequence/

先求 LCS，再根据 LCS 构建最短公共超序列。

```python
def shortestCommonSupersequence(self, str1, str2):
    m, n = len(str1), len(str2)
    dp = [[''] * (n + 1) for _ in range(m + 1)]
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if str1[i-1] == str2[j-1]:
                dp[i][j] = dp[i-1][j-1] + str1[i-1]
            else:
                dp[i][j] = dp[i-1][j] if len(dp[i-1][j]) > len(dp[i][j-1]) else dp[i][j-1]
    lcs = dp[m][n]
    res = []; i = j = 0
    for c in lcs:
        while i < m and str1[i] != c: res.append(str1[i]); i += 1
        while j < n and str2[j] != c: res.append(str2[j]); j += 1
        res.append(c); i += 1; j += 1
    return ''.join(res) + str1[i:] + str2[j:]
```

## 1095. Find in Mountain Array
https://leetcode.com/problems/find-in-mountain-array/

三次二分：找峰值、左半搜、右半搜。

```python
def findInMountainArray(self, target, mountain_arr):
    n = mountain_arr.length()
    # 找峰值
    lo, hi = 0, n - 1
    while lo < hi:
        mid = (lo + hi) // 2
        if mountain_arr.get(mid) < mountain_arr.get(mid + 1): lo = mid + 1
        else: hi = mid
    peak = lo
    # 左半升序搜索
    lo, hi = 0, peak
    while lo <= hi:
        mid = (lo + hi) // 2
        val = mountain_arr.get(mid)
        if val == target: return mid
        elif val < target: lo = mid + 1
        else: hi = mid - 1
    # 右半降序搜索
    lo, hi = peak, n - 1
    while lo <= hi:
        mid = (lo + hi) // 2
        val = mountain_arr.get(mid)
        if val == target: return mid
        elif val > target: lo = mid + 1
        else: hi = mid - 1
    return -1
```

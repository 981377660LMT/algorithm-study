## 3604. Minimum Time to Reach Destination in Directed Graph

https://leetcode.cn/problems/minimum-time-to-reach-destination-in-directed-graph/

```python
import heapq
def minTime(self, n, edges, patience):
    adj = [[] for _ in range(n)]
    for u, v, w in edges:
        adj[u].append((v, w)); adj[v].append((u, w))
    dist = [float('inf')] * n; dist[0] = 0
    heap = [(0, 0)]
    while heap:
        d, u = heapq.heappop(heap)
        if d > dist[u]: continue
        for v, w in adj[u]:
            if d + w < dist[v]:
                dist[v] = d + w; heapq.heappush(heap, (dist[v], v))
    # 每个节点 i 每隔 patience[i] 重发消息, 直到收到回复(时间 2*dist[i])
    # 最后一次重发时间 = ((2*dist[i]-1) // patience[i]) * patience[i]
    # 网络安静时间 = 最后重发时间 + 2*dist[i] + 1
    res = 0
    for i in range(1, n):
        last_resend = ((2 * dist[i] - 1) // patience[i]) * patience[i]
        quiet = last_resend + 2 * dist[i] + 1
        res = max(res, quiet)
    return res
```

```python
def minMoves(self, rooks):
    n = len(rooks)
    rows = sorted(r[0] for r in rooks)
    cols = sorted(r[1] for r in rooks)
    return sum(abs(rows[i] - i) for i in range(n)) + sum(abs(cols[i] - i) for i in range(n))
```

## 3612. Peaks in Array

https://leetcode.cn/problems/peaks-in-array/

```python
class BIT:
    def __init__(self, n):
        self.n = n; self.tree = [0]*(n+1)
    def update(self, i, delta):
        i += 1
        while i <= self.n: self.tree[i] += delta; i += i & (-i)
    def query(self, i):
        i += 1; s = 0
        while i > 0: s += self.tree[i]; i -= i & (-i)
        return s
    def range_query(self, l, r):
        return self.query(r) - (self.query(l-1) if l > 0 else 0)

def countOfPeaks(self, nums, queries):
    n = len(nums)
    bit = BIT(n)
    def is_peak(i):
        return 0 < i < n-1 and nums[i] > nums[i-1] and nums[i] > nums[i+1]
    for i in range(1, n-1):
        if is_peak(i): bit.update(i, 1)
    res = []
    for q in queries:
        if q[0] == 1:
            l, r = q[1], q[2]
            res.append(bit.range_query(l+1, r-1) if r - l >= 2 else 0)
        else:
            idx, val = q[1], q[2]
            for i in range(max(1, idx-1), min(n-1, idx+2)):
                if is_peak(i): bit.update(i, -1)
            nums[idx] = val
            for i in range(max(1, idx-1), min(n-1, idx+2)):
                if is_peak(i): bit.update(i, 1)
    return res
```

```python
def minOperations(self, nums, k):
    from collections import Counter
    cnt = Counter(nums)
    ops = 0
    for v, c in cnt.items():
        if v < k: return -1
        if v > k: ops += c
    return ops
```

## 3633. Maximum XOR Score of a Subarray

https://leetcode.cn/problems/maximum-xor-score-of-a-subarray/

思路兜底：区间 DP。xor_score[l][r] = xor_score[l][r-1] ^ xor_score[l+1][r]。自底向上 DP，然后取最大值。

```python
def maximumXorScore(self, nums, k):
    n = len(nums)
    # dp[l][r] = xor score of subarray [l, r]
    dp = [[0]*n for _ in range(n)]
    for i in range(n): dp[i][i] = nums[i]
    for length in range(2, n+1):
        for l in range(n-length+1):
            r = l + length - 1
            dp[l][r] = dp[l][r-1] ^ dp[l+1][r]
    # mx[l][r] = max xor score in subarray starting in [l..r]
    mx = [[0]*n for _ in range(n)]
    for i in range(n): mx[i][i] = dp[i][i]
    for length in range(2, n+1):
        for l in range(n-length+1):
            r = l + length - 1
            mx[l][r] = max(dp[l][r], mx[l][r-1], mx[l+1][r])
    return mx[0][n-1]
```

## 3639. Zero Array Transformation III

https://leetcode.cn/problems/zero-array-transformation-iii/

```python
import heapq
def maxRemoval(self, nums, queries):
    n = len(nums)
    queries.sort()  # 按左端点排序
    avail = []  # max-heap by right endpoint
    active = [0] * (n + 1)  # difference array for active queries
    j = 0; cur = 0; used = 0
    for i in range(n):
        cur += active[i]
        while j < len(queries) and queries[j][0] <= i:
            heapq.heappush(avail, -queries[j][1])
            j += 1
        while cur < nums[i] and avail and -avail[0] >= i:
            r = -heapq.heappop(avail)
            cur += 1; active[r + 1] -= 1; used += 1
        if cur < nums[i]: return -1
    return len(queries) - used
```

## 3650. Maximize Win From Two Segments

https://leetcode.cn/problems/maximize-win-from-two-segments/

```python
from bisect import bisect_right
def maximizeWin(self, prizePositions, k):
    n = len(prizePositions)
    # dp[i] = 一个窗口以 i 或之前结尾的最多奖品
    dp = [0] * (n + 1)
    res = 0
    for i in range(n):
        j = bisect_right(prizePositions, prizePositions[i] - k - 1)
        # 窗口 [prizePositions[j], prizePositions[i]] 内有 i - j + 1 个
        cnt = i - j + 1  # 不对，用 bisect_left
    # 重写
    dp = [0] * (n + 1); res = 0
    j = 0
    for i in range(n):
        while prizePositions[i] - prizePositions[j] > k: j += 1
        cnt = i - j + 1
        dp[i + 1] = max(dp[i], cnt)
        res = max(res, cnt + dp[j])
    return res
```

```python
def minOperationsToMakeMedianK(self, nums, k):
    nums.sort(); n = len(nums); m = n // 2; ops = 0
    if nums[m] < k:
        for i in range(m, n):
            if nums[i] < k: ops += k - nums[i]

            if nums[i] > k: ops += nums[i] - k
    return ops
```

```python
def getMaximumXor(self, nums, maximumBit):
    mask = (1 << maximumBit) - 1
    xor_all = 0
    for x in nums: xor_all ^= x
    res = []
    for i in range(len(nums)-1, -1, -1):
        res.append(xor_all ^ mask)
        xor_all ^= nums[i]
    return res
```

## 3685. Minimum Obstacle Removal to Reach Corner

https://leetcode.cn/problems/minimum-obstacle-removal-to-reach-corner/

```python
from collections import deque
def minimumObstacles(self, grid):
    m, n = len(grid), len(grid[0])
    dist = [[float('inf')]*n for _ in range(m)]
    dist[0][0] = 0; dq = deque([(0, 0, 0)])
    while dq:
        d, r, c = dq.popleft()
        if d > dist[r][c]: continue
        for dr, dc in [(0,1),(0,-1),(1,0),(-1,0)]:
            nr, nc = r+dr, c+dc
            if 0<=nr<m and 0<=nc<n:
                nd = d + grid[nr][nc]
                if nd < dist[nr][nc]:
                    dist[nr][nc] = nd
                    if grid[nr][nc] == 0: dq.appendleft((nd, nr, nc))
                    else: dq.append((nd, nr, nc))
    return dist[m-1][n-1]
```

## 3688. Count Unguarded Cells in the Grid

https://leetcode.cn/problems/count-unguarded-cells-in-the-grid/

```python
def countUnguarded(self, m, n, guards, walls):
    grid = [[0]*n for _ in range(m)]
    for r, c in walls: grid[r][c] = 2
    for r, c in guards: grid[r][c] = 3
    for r, c in guards:
        for dr, dc in [(0,1),(0,-1),(1,0),(-1,0)]:
            nr, nc = r+dr, c+dc
            while 0<=nr<m and 0<=nc<n and grid[nr][nc] < 2:
                grid[nr][nc] = 1; nr += dr; nc += dc
    return sum(1 for i in range(m) for j in range(n) if grid[i][j] == 0)
```

```python
def decrypt(self, code, k):
    n = len(code); res = [0]*n
    for i in range(n):
        if k > 0: res[i] = sum(code[(i+j) % n] for j in range(1, k+1))
```

## 3697. Maximize Total Cost of Alternating Subarrays

https://leetcode.cn/problems/maximize-total-cost-of-alternating-subarrays/

```python
def maximumTotalCost(self, nums):
    n = len(nums)
    # dp: 交替 +/- 符号的子数组划分
    # pos[i] = 以 i 结尾, nums[i] 取 + 的最大和
    # neg[i] = 以 i 结尾, nums[i] 取 - 的最大和
    pos = nums[0]; neg = float('-inf')
    for i in range(1, n):
        new_pos = max(pos, neg) + nums[i]  # 新子数组开始或取+
        new_neg = pos - nums[i]  # 延续上一个, 取-
        pos, neg = new_pos, new_neg
    return max(pos, neg)
```

## 3699. Count Almost Equal Pairs I / II

https://leetcode.cn/problems/count-almost-equal-pairs-i/

```python
from collections import Counter
def countPairs(self, nums):
    cnt = Counter(nums); res = 0
    for x, c in cnt.items():
        if c >= 2: res += c * (c - 1) // 2
    # 还需要考虑交换一位后相等的对
    # 对每个数生成所有可能的 swap 结果
    seen = Counter()
    for x in nums:
        s = str(x)
        variants = {x}
        ds = list(s)
        for i in range(len(ds)):
            for j in range(i+1, len(ds)):
                ds[i], ds[j] = ds[j], ds[i]
                variants.add(int(''.join(ds)))
                ds[i], ds[j] = ds[j], ds[i]
        for v in variants:
            if v in seen and v != x:
                res += seen[v]
        seen[x] += 1
    return res
```

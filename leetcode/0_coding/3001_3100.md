## 3001. Minimum Moves to Capture The Queen

https://leetcode.cn/problems/minimum-moves-to-capture-the-queen/

```python
def minMovesToCaptureTheQueen(self, a, b, c, d, e, f):
    # 车(a,b), 象(c,d), 后(e,f)
    # 车直线能到后: 同行或同列, 检查象是否挡道
    if a == e:  # 同行
        if c != a or not (min(b,f) < d < max(b,f)): return 1
    if b == f:  # 同列
        if d != b or not (min(a,e) < c < max(a,e)): return 1
    # 象对角能到后
    if abs(c-e) == abs(d-f):
        dx = 1 if e > c else -1; dy = 1 if f > d else -1
        blocked = False
        x, y = c + dx, d + dy
        while (x, y) != (e, f):
            if (x, y) == (a, b): blocked = True; break
            x += dx; y += dy
        if not blocked: return 1
    return 2
```

## 3002. Maximum Size of a Set After Removals

https://leetcode.cn/problems/maximum-size-of-a-set-after-removals/

```python
def maximumSetSize(self, nums1, nums2):
    s1, s2 = set(nums1), set(nums2)
    n = len(nums1) // 2
    both = s1 & s2
    only1 = len(s1) - len(both)
    only2 = len(s2) - len(both)
    # 从 nums1 保留 n 个: 优先保留 only1
    keep1 = min(only1, n)
    # 从 nums2 保留 n 个: 优先保留 only2
    keep2 = min(only2, n)
    # 剩余名额给 both
    remain = n - keep1 + n - keep2
    keep_both = min(len(both), remain)
    return keep1 + keep2 + keep_both
```

## 3015. Count the Number of Houses at a Certain Distance I

```python
def countOfPairs(self, n, x, y):
    # BFS from each node
    from collections import deque
    adj = [[] for _ in range(n+1)]
    for i in range(1, n): adj[i].append(i+1); adj[i+1].append(i)
    if x != y: adj[x].append(y); adj[y].append(x)
    res = [0] * n
    for start in range(1, n+1):
        dist = [-1]*(n+1); dist[start] = 0; q = deque([start])
        while q:
            u = q.popleft()
            for v in adj[u]:
                if dist[v] == -1:
                    dist[v] = dist[u]+1; q.append(v)
        for v in range(start+1, n+1):
            res[dist[v]-1] += 2
    return res
```

```python
def maximumSubarraySum(self, nums, k):
    prefix = {}  # val -> min prefix sum ending at index with that val
    res = float('-inf'); s = 0
    found = False
    for x in nums:
        s += x
        for target in [x - k, x + k]:
            if target in prefix:
                res = max(res, s - prefix[target])
                found = True
        if x not in prefix or prefix[x] > s - x:
            prefix[x] = s - x
    return res if found else 0
```

```python
def minimumTimeToInitialState(self, word, k):
    n = len(word)
    # Z-function
    z = [0] * n; z[0] = n; l = r = 0
    for i in range(1, n):
        if i < r: z[i] = min(r - i, z[i - l])
        while i + z[i] < n and word[z[i]] == word[i + z[i]]: z[i] += 1
        if i + z[i] > r: l, r = i, i + z[i]
    for t in range(1, (n + k - 1) // k + 1):
        pos = t * k
        if pos >= n or z[pos] >= n - pos:
            return t
    return (n + k - 1) // k
```

```python
def maxOperations(self, nums):
    n = len(nums)

        @lru_cache(None)
        def dp(l, r):
            if l >= r: return 0
            res = 0
```

## 3048. Earliest Second to Mark Indices I

https://leetcode.cn/problems/earliest-second-to-mark-indices-i/

二分答案 + 贪心。

```python
def earliestSecondToMarkIndices(self, nums, changeIndices):
    n = len(nums); m = len(changeIndices)
    def check(t):
        # 在前 t 秒内完成
        last = {}
        for s in range(t):
            last[changeIndices[s]-1] = s
        if len(last) < n: return False
        # 按 last 时间排序，每个 index 必须在 last 时刻标记
        # 之前需要 nums[i] 次递减
        events = sorted(last.items(), key=lambda x: x[1])
        free = 0
        prev = 0
        for idx, day in events:
            free += day - prev  # 之前 prev 到 day-1 这些时刻可以操作
            if free < nums[idx]: return False
            free -= nums[idx]
            prev = day + 1
        return True
    lo, hi = 1, m
    while lo < hi:
        mid = (lo+hi)//2
        if check(mid): hi = mid
        else: lo = mid + 1
    return lo if check(lo) else -1
```

## 3067. Count Pairs of Connectable Servers in a Weighted Tree

```python
def countPairsOfConnectableServers(self, edges, signalSpeed):
    n = len(edges) + 1
    adj = [[] for _ in range(n)]
    for u, v, w in edges: adj[u].append((v,w)); adj[v].append((u,w))
    res = [0] * n
    for root in range(n):
        counts = []
        for nxt, w in adj[root]:
            # DFS 统计该子树中距离 % signalSpeed == 0 的节点数
            cnt = 0
            stk = [(nxt, root, w)]
            while stk:
                u, par, d = stk.pop()
                if d % signalSpeed == 0: cnt += 1
                for v, w2 in adj[u]:
                    if v != par: stk.append((v, u, d+w2))
            counts.append(cnt)
        # 两两配对
        total = 0
        for i in range(len(counts)):
            res[root] += total * counts[i]
            total += counts[i]
    return res
```

## 3082. Find the Sum of the Power of All Subsequences

```python
def sumOfPower(self, nums, k):
    MOD = 10**9 + 7; n = len(nums)
    # dp[j] = 和为 j 的子序列数 * 2^(n - len)
    # 贡献：每个和为 k 的子序列 size=t, 贡献 2^(n-t)
    dp = [0] * (k + 1); dp[0] = 1
    for x in nums:
        for j in range(k, -1, -1):
            # 选 x: dp[j] → dp[j+x] (乘1), 不选: dp[j] → dp[j] (乘2)
            dp[j] = dp[j] * 2 % MOD
            if j >= x:
                dp[j] = (dp[j] + dp[j-x]) % MOD  # wrong, need tmp
    # 实际: dp_new[j] = 2*dp[j] + dp[j-x]
    dp = [0] * (k + 1); dp[0] = 1
    for x in nums:
        new_dp = [0] * (k + 1)
        for j in range(k + 1):
            new_dp[j] = dp[j] * 2 % MOD
            if j >= x: new_dp[j] = (new_dp[j] + dp[j-x]) % MOD
        dp = new_dp
    return dp[k]
```

## 3095. Shortest Subarray With OR at Least K I / 3097. Shortest Subarray With OR at Least K II

https://leetcode.cn/problems/shortest-subarray-with-or-at-least-k-ii/

```python
def minimumSubarrayLength(self, nums, k):
    bits = [0] * 32; res = float('inf'); l = 0
    def get_or():
        v = 0
        for b in range(32):
            if bits[b]: v |= (1 << b)
        return v
    for r in range(len(nums)):
        for b in range(32):
            if nums[r] >> b & 1: bits[b] += 1
        while l <= r and get_or() >= k:
            res = min(res, r - l + 1)
            for b in range(32):
                if nums[l] >> b & 1: bits[b] -= 1
            l += 1
    return res if res < float('inf') else -1
```

```python
def maxBottlesDrunk(self, numBottles, numExchange):
```

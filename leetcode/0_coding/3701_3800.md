## 3702. Sum of Prefix Scores of Strings

https://leetcode.cn/problems/sum-of-prefix-scores-of-strings/

```python
def sumPrefixScores(self, words):
    # Trie, 每个节点记录经过次数
    trie = {}
    for w in words:
        node = trie
        for c in w:
            if c not in node: node[c] = {'#': 0}
            node = node[c]; node['#'] += 1
    res = []
    for w in words:
        node = trie; s = 0
        for c in w:
            node = node[c]; s += node['#']
        res.append(s)
    return res
```

## 3707. Maximum Number of Points From Grid Queries

https://leetcode.cn/problems/maximum-number-of-points-from-grid-queries/

```python
import heapq
def maxPoints(self, grid, queries):
    m, n = len(grid), len(grid[0])
    sorted_q = sorted(enumerate(queries), key=lambda x: x[1])
    res = [0] * len(queries)
    heap = [(grid[0][0], 0, 0)]
    visited = [[False]*n for _ in range(m)]
    visited[0][0] = True; cnt = 0
    for idx, q in sorted_q:
        while heap and heap[0][0] < q:
            val, r, c = heapq.heappop(heap)
            cnt += 1
            for dr, dc in [(0,1),(0,-1),(1,0),(-1,0)]:
                nr, nc = r+dr, c+dc
                if 0<=nr<m and 0<=nc<n and not visited[nr][nc]:
                    visited[nr][nc] = True
                    heapq.heappush(heap, (grid[nr][nc], nr, nc))
        res[idx] = cnt
    return res
```

## 3709. Design a Number Container System

https://leetcode.cn/problems/design-a-number-container-system/

```python
from collections import defaultdict
from sortedcontainers import SortedList
class NumberContainers:
    def __init__(self):
        self.idx_to_num = {}
        self.num_to_idx = defaultdict(SortedList)
    def change(self, index, number):
        if index in self.idx_to_num:
            old = self.idx_to_num[index]
            self.num_to_idx[old].remove(index)
        self.idx_to_num[index] = number
        self.num_to_idx[number].add(index)
    def find(self, number):
        if self.num_to_idx[number]:
            return self.num_to_idx[number][0]
        return -1
```

from collections import Counter
def maxSum(self, nums, m, k):
    cnt = Counter(); cur = 0; res = 0
    for i in range(len(nums)):
        cnt[nums[i]] += 1; cur += nums[i]
        if i >= k:
            cnt[nums[i-k]] -= 1
            if cnt[nums[i-k]] == 0: del cnt[nums[i-k]]

import heapq
def getFinalState(self, nums, k, multiplier):
    heap = [(x, i) for i, x in enumerate(nums)]
    heapq.heapify(heap)
    for _ in range(k):
        val, idx = heapq.heappop(heap)
        nums[idx] = val * multiplier
        heapq.heappush(heap, (nums[idx], idx))

def delNodes(self, root, to_delete):
    to_del = set(to_delete); res = []
    def dfs(node, is_root):
        if not node: return None
        deleted = node.val in to_del
        if is_root and not deleted: res.append(node)
        node.left = dfs(node.left, deleted)
        node.right = dfs(node.right, deleted)
        return None if deleted else node
    dfs(root, True)
    return res

## 3730. Minimum Cost to Make At Least One Valid Path in a Grid

https://leetcode.cn/problems/minimum-cost-to-make-at-least-one-valid-path-in-a-grid/

```python
from collections import deque
def minCost(self, grid):
    m, n = len(grid), len(grid[0])
    dirs = [(0,1),(0,-1),(1,0),(-1,0)]  # 1,2,3,4 -> right,left,down,up
    dist = [[float('inf')]*n for _ in range(m)]
    dist[0][0] = 0; dq = deque([(0, 0, 0)])
    while dq:
        d, r, c = dq.popleft()
        if d > dist[r][c]: continue
        for i, (dr, dc) in enumerate(dirs):
            nr, nc = r+dr, c+dc
            if 0<=nr<m and 0<=nc<n:
                cost = 0 if grid[r][c] == i+1 else 1
                if d + cost < dist[nr][nc]:
                    dist[nr][nc] = d + cost
                    if cost == 0: dq.appendleft((d, nr, nc))
                    else: dq.append((d+1, nr, nc))
    return dist[m-1][n-1]
```

```python
import heapq
def unmarkedSumArray(self, nums, queries):
    n = len(nums); total = sum(nums)
    marked = [False] * n
    heap = [(nums[i], i) for i in range(n)]
    heapq.heapify(heap)
    res = []
    for idx, k in queries:
        if not marked[idx]:
            marked[idx] = True; total -= nums[idx]
        while k > 0 and heap:
            val, i = heapq.heappop(heap)
            if marked[i]: continue
            marked[i] = True; total -= val; k -= 1
        res.append(total)
    return res
```

import heapq
def minOperations(self, nums, k):

    while nums[0] < k:

def distributeCandies(self, n, limit):

## 3765. Lexicographically Minimum String After Removing Stars

https://leetcode.cn/problems/lexicographically-minimum-string-after-removing-stars/

```python
import heapq
def clearStars(self, s):
    # 每颗星删除左边最小字符(相同则最右)
    from collections import defaultdict
    stacks = defaultdict(list)  # char -> list of indices
    removed = set()
    for i, c in enumerate(s):
        if c == '*':
            removed.add(i)
            for ch in 'abcdefghijklmnopqrstuvwxyz':
                if stacks[ch]:
                    removed.add(stacks[ch].pop())
                    break
        else:
            stacks[c].append(i)
    return ''.join(c for i, c in enumerate(s) if i not in removed)
```

def minimumArea(self, grid):
    m, n = len(grid), len(grid[0])
    r1, r2, c1, c2 = m, 0, n, 0

            if grid[i][j] == 1:

import heapq
def resultsArray(self, queries, k):
    heap = []; res = []
    for x, y in queries:
        d = abs(x) + abs(y)
        if len(heap) < k:
            heapq.heappush(heap, -d)
        elif d < -heap[0]:
            heapq.heapreplace(heap, -d)

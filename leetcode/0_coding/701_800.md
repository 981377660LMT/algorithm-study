class KthLargest:
    def __init__(self, k, nums):
        self.k = k
        self.heap = nums
        heapq.heapify(self.heap)
        while len(self.heap) > k:
            heapq.heappop(self.heap)

        self.size = 1000
        self.table = [[] for _ in range(self.size)]

        return key % self.size
    def put(self, key, value):
        h = self._hash(key)
        for i, (k, v) in enumerate(self.table[h]):
            if k == key:
                self.table[h][i] = (key, value); return
        self.table[h].append((key, value))

        h = self._hash(key)
        for k, v in self.table[h]:
            if k == key: return v
        return -1

        self.table[h] = [(k, v) for k, v in self.table[h] if k != key]

    def __init__(self): self.head = None; self.size = 0
    def _get(self, index):
        cur = self.head
        for _ in range(index): cur = cur.next
        return cur

        if index < 0 or index >= self.size: return -1
        return self._get(index).val
    def addAtHead(self, val): self.addAtIndex(0, val)
    def addAtTail(self, val): self.addAtIndex(self.size, val)

        index = max(0, index)
        node = ListNode(val)
        if index == 0: node.next = self.head; self.head = node
        else: prev = self._get(index - 1); node.next = prev.next; prev.next = node
        self.size += 1
    def deleteAtIndex(self, index):
        if index < 0 or index >= self.size: return
        if index == 0: self.head = self.head.next

        self.size -= 1

        self.m = n - len(blacklist)
        black = set(blacklist)

def minimumDeleteSum(self, s1, s2):
    m, n = len(s1), len(s2)
    dp = [0] * (n + 1)
    for j in range(1, n + 1): dp[j] = dp[j-1] + ord(s2[j-1])

        prev = dp[0]; dp[0] += ord(s1[i-1])

from sortedcontainers import SortedDict
class RangeModule:
    def __init__(self): self.d = SortedDict()
    def addRange(self, left, right):
        i = self.d.bisect_left(left)
        if i > 0 and self.d.values()[i-1] >= left:
            i -= 1; left = self.d.keys()[i]
        while i < len(self.d) and self.d.keys()[i] <= right:
            right = max(right, self.d.values()[i])
            self.d.popitem(i)
        self.d[left] = right
    def queryRange(self, left, right):
        i = self.d.bisect_right(left) - 1
        return i >= 0 and self.d.values()[i] >= right
    def removeRange(self, left, right):
        i = self.d.bisect_left(left)
        if i > 0 and self.d.values()[i-1] > left: i -= 1
        new_intervals = []
        while i < len(self.d) and self.d.keys()[i] < right:
            k, v = self.d.keys()[i], self.d.values()[i]
            self.d.popitem(i)
            if k < left: new_intervals.append((k, left))
            if v > right: new_intervals.append((right, v))
        for k, v in new_intervals: self.d[k] = v

from collections import defaultdict
def accountsMerge(self, accounts):
    parent = {}
    def find(x):
        while parent[x] != x:
            parent[x] = parent[parent[x]]
            x = parent[x]
        return x
    def union(x, y):
        parent[find(x)] = find(y)
    email_to_name = {}
    for acc in accounts:
        name = acc[0]
        for email in acc[1:]:
            if email not in parent: parent[email] = email
            email_to_name[email] = name
            union(email, acc[1])
    groups = defaultdict(list)
    for email in parent:
        groups[find(email)].append(email)
    return [[email_to_name[root]] + sorted(emails) for root, emails in groups.items()]

```python
def evaluate(self, expression):
    def parse(expr, scope):
        if expr[0] != '(':
            return scope.get(expr, int(expr)) if not expr.lstrip('-').isdigit() else int(expr)
        expr = expr[1:-1]
        tokens = tokenize(expr)
        if tokens[0] == 'let':
            new_scope = dict(scope)
            i = 1
            while i < len(tokens) - 1:
                new_scope[tokens[i]] = parse(tokens[i+1], new_scope)
                i += 2
            return parse(tokens[-1], new_scope)
        elif tokens[0] == 'add':
            return parse(tokens[1], scope) + parse(tokens[2], scope)
        else:  # mult
            return parse(tokens[1], scope) * parse(tokens[2], scope)
    def tokenize(expr):
        tokens = []; bal = 0; start = 0
        for i, c in enumerate(expr):
            if c == '(': bal += 1
            elif c == ')': bal -= 1
            elif c == ' ' and bal == 0:
                tokens.append(expr[start:i]); start = i + 1
        tokens.append(expr[start:])
        return tokens
    return parse(expression, {})
```

def cherryPickup(self, grid):
    n = len(grid)
    from functools import lru_cache
    @lru_cache(maxsize=None)
    def dp(r1, c1, r2):
        c2 = r1 + c1 - r2
        if r1 >= n or c1 >= n or r2 >= n or c2 >= n: return -float('inf')
        if grid[r1][c1] == -1 or grid[r2][c2] == -1: return -float('inf')
        if r1 == c1 == n - 1: return grid[r1][c1]
        val = grid[r1][c1] + (grid[r2][c2] if (r1, c1) != (r2, c2) else 0)
        return val + max(dp(r1+1,c1,r2+1), dp(r1+1,c1,r2),
                         dp(r1,c1+1,r2+1), dp(r1,c1+1,r2))

## 743. Network Delay Time

import heapq
from collections import defaultdict
def networkDelayTime(self, times, n, k):
    graph = defaultdict(list)
    for u, v, w in times:
        graph[u].append((v, w))
    dist = {}
    h = [(0, k)]
    while h:
        d, u = heapq.heappop(h)
        if u in dist: continue
        dist[u] = d
        for v, w in graph[u]:
            if v not in dist:
                heapq.heappush(h, (d + w, v))
    return max(dist.values()) if len(dist) == n else -1
```

import heapq
from collections import Counter
def reorganizeString(self, s):
    cnt = Counter(s)
    if max(cnt.values()) > (len(s) + 1) // 2: return ''
    h = [(-v, k) for k, v in cnt.items()]
    heapq.heapify(h)
    res = []
    while len(h) >= 2:
        c1, ch1 = heapq.heappop(h)
        c2, ch2 = heapq.heappop(h)
        res.extend([ch1, ch2])
        if c1 + 1 < 0: heapq.heappush(h, (c1 + 1, ch1))
        if c2 + 1 < 0: heapq.heappush(h, (c2 + 1, ch2))
    if h: res.append(h[0][1])
    return ''.join(res)
```

import heapq
def swimInWater(self, grid):
    n = len(grid)
    visited = [[False] * n for _ in range(n)]
    heap = [(grid[0][0], 0, 0)]
    visited[0][0] = True
    while heap:
        t, i, j = heapq.heappop(heap)
        if i == n - 1 and j == n - 1: return t

def reachingPoints(self, sx, sy, tx, ty):
    while tx >= sx and ty >= sy:
        if tx == ty: break
        if tx > ty:
            if ty == sy: return (tx - sx) % ty == 0
            tx %= ty
        else:
            if tx == sx: return (ty - sy) % tx == 0

    return tx == sx and ty == sy

def isBipartite(self, graph):
    color = [0] * len(graph)
    for i in range(len(graph)):
        if color[i]: continue
        color[i] = 1
        q = [i]
        while q:

def rotatedDigits(self, n):
    cnt = 0
    for i in range(1, n + 1):
        s = str(i)
        if any(d in s for d in '347'): continue
        if any(d in s for d in '2569'): cnt += 1
    return cnt

## 790. Domino and Tromino Tiling

https://leetcode.cn/problems/domino-and-tromino-tiling/

```python
def numTilings(self, n):
    MOD = 10**9 + 7
    dp = [0] * (n + 3)
    dp[0] = 1; dp[1] = 1; dp[2] = 2
    for i in range(3, n + 1):
        dp[i] = (2 * dp[i-1] + dp[i-3]) % MOD
    return dp[n]
```

def customSortString(self, order, s):

## 792. Number of Matching Subsequences

https://leetcode.cn/problems/number-of-matching-subsequences/

```python
from collections import defaultdict
def numMatchingSubseq(self, s, words):
    waiting = defaultdict(list)
    for w in words: waiting[w[0]].append(iter(w[1:]))
    for c in s:
        old = waiting.pop(c, [])
        for it in old:
            nxt = next(it, None)
            if nxt: waiting[nxt].append(it)
            else: waiting['#'].append(it)
    return len(waiting.get('#', []))
```

## 793. Preimage Size of Factorial Zeroes Function

https://leetcode.cn/problems/preimage-size-of-factorial-zeroes-function/

二分搜索。

```python
def preimageSizeFZF(self, k):
    def zeros(n):
        cnt = 0
        while n: n //= 5; cnt += n
        return cnt
    lo, hi = 0, 5 * (k + 1)
    while lo < hi:
        mid = (lo + hi) // 2
        z = zeros(mid)
        if z < k: lo = mid + 1
        elif z > k: hi = mid
        else: return 5
    return 0
```

## 797. All Paths From Source to Target

https://leetcode.cn/problems/all-paths-from-source-to-target/

DFS/回溯遍历 DAG。

```python
def allPathsSourceTarget(self, graph):
    res = []
    def dfs(node, path):
        if node == len(graph) - 1:
            res.append(path[:]); return
        for nb in graph[node]:
            path.append(nb)
            dfs(nb, path)
            path.pop()
    dfs(0, [0])
    return res
```

```python
from sortedcontainers import SortedList
def minimumDistance(self, points):
    sx = SortedList(); sy = SortedList()
    for x, y in points:
        sx.add(x + y); sy.add(x - y)
    res = float('inf')
    for x, y in points:
        sx.remove(x + y); sy.remove(x - y)
        d = max(sx[-1] - sx[0], sy[-1] - sy[0])
        res = min(res, d)
        sx.add(x + y); sy.add(x - y)
```

## 3108. Minimum Cost Walk in Weighted Graph

https://leetcode.cn/problems/minimum-cost-walk-in-weighted-graph/

AND 性质：走越多边 AND 越小。同一连通分量内答案 = 所有边权 AND。

```python
def minimumCost(self, n, edges, query):
    parent = list(range(n)); rank = [0]*n; cost = [-1]*n  # -1 = all bits 1
    def find(x):
        while parent[x] != x: parent[x] = parent[parent[x]]; x = parent[x]
        return x
    def union(a, b, w):
        pa, pb = find(a), find(b)
        cost[pa] &= w; cost[pb] &= w
        if pa == pb: return
        if rank[pa] < rank[pb]: pa, pb = pb, pa
        parent[pb] = pa; cost[pa] &= cost[pb]
        if rank[pa] == rank[pb]: rank[pa] += 1
    for u, v, w in edges: union(u, v, w)
    res = []
    for s, t in query:
        if find(s) != find(t): res.append(-1)
        else: res.append(cost[find(s)])
    return res
```

## 3112. Minimum Time to Visit Disappearing Nodes

https://leetcode.cn/problems/minimum-time-to-visit-disappearing-nodes/

Dijkstra 加判断是否消失。

```python
import heapq
def minimumTime(self, n, edges, disappear):
    adj = [[] for _ in range(n)]
    for u, v, w in edges: adj[u].append((v,w)); adj[v].append((u,w))
    dist = [float('inf')] * n; dist[0] = 0
    heap = [(0, 0)]
    while heap:
        d, u = heapq.heappop(heap)
        if d > dist[u]: continue
        for v, w in adj[u]:
            nd = d + w
            if nd < dist[v] and nd < disappear[v]:
                dist[v] = nd; heapq.heappush(heap, (nd, v))
    return [d if d < float('inf') else -1 for d in dist]
```

## 3116. Kth Smallest Amount With Single Denomination Combination

https://leetcode.cn/problems/kth-smallest-amount-with-single-denomination-combination/

二分 + 容斥。

```python
from math import gcd
from itertools import combinations
def findKthSmallest(self, coins, k):
    def count(x):
        # 1~x 中能整除 coins 中至少一个的数的个数 (容斥)
        total = 0
        for r in range(1, len(coins)+1):
            for combo in combinations(coins, r):
                lcm = combo[0]
                for c in combo[1:]: lcm = lcm * c // gcd(lcm, c)
                total += (-1)**(r+1) * (x // lcm)
        return total
    lo, hi = 1, min(coins) * k
    while lo < hi:
        mid = (lo + hi) // 2
        if count(mid) >= k: hi = mid
        else: lo = mid + 1
    return lo
```

```python
from collections import Counter
def minimumOperations(self, grid):
    m, n = len(grid), len(grid[0])
    # cost[j][d] = 将第 j 列全变为 d 的代价
    col_cost = []
    for j in range(n):
        cnt = Counter(grid[i][j] for i in range(m))
        costs = [m - cnt[d] for d in range(10)]
        col_cost.append(costs)
    # DP: dp[j][d] = 前 j 列, 第 j 列值为 d 的最小总代价
    dp = col_cost[0][:]
    for j in range(1, n):
        new_dp = [0] * 10
        for d in range(10):
            new_dp[d] = col_cost[j][d] + min(dp[d2] for d2 in range(10) if d2 != d)
        dp = new_dp
    return min(dp)
```

## 3123. Find Edges in Shortest Paths

https://leetcode.cn/problems/find-edges-in-shortest-paths/

```python
import heapq
def findAnswer(self, n, edges):
    adj = [[] for _ in range(n)]
    for i, (u, v, w) in enumerate(edges):
        adj[u].append((v, w, i)); adj[v].append((u, w, i))
    def dijkstra(src):
        dist = [float('inf')] * n; dist[src] = 0
        heap = [(0, src)]
        while heap:
            d, u = heapq.heappop(heap)
            if d > dist[u]: continue
            for v, w, _ in adj[u]:
                if d + w < dist[v]:
                    dist[v] = d + w; heapq.heappush(heap, (d+w, v))
        return dist
    d0 = dijkstra(0); dn = dijkstra(n-1)
    shortest = d0[n-1]
    res = [False] * len(edges)
    if shortest == float('inf'): return res
    for i, (u, v, w) in enumerate(edges):
        if d0[u] + w + dn[v] == shortest or d0[v] + w + dn[u] == shortest:
            res[i] = True
    return res
```

```python
from functools import lru_cache
def numberOfStableArrays(self, zero, one, limit):
    MOD = 10**9 + 7
    @lru_cache(None)
    def dp(z, o, last, consec):
        if z == 0 and o == 0: return 1
        res = 0
        if last == 0:
            if z > 0 and consec < limit:
                res += dp(z-1, o, 0, consec+1)
            if o > 0:
                res += dp(z, o-1, 1, 1)
        else:
            if o > 0 and consec < limit:
                res += dp(z, o-1, 1, consec+1)
            if z > 0:
                res += dp(z-1, o, 0, 1)
        return res % MOD
    return (dp(zero-1, one, 0, 1) + dp(zero, one-1, 1, 1)) % MOD
```

## 3133. Minimum Array End

https://leetcode.cn/problems/minimum-array-end/

```python
def minEnd(self, n, x):
    # 结果必须所有位都包含 x 的1位, 在0位上填 n-1 的二进制
    n -= 1; res = x
    bit = 0
    while n > 0:
        while (res >> bit) & 1: bit += 1  # 跳过 x 的1位
        if n & 1: res |= (1 << bit)
        n >>= 1; bit += 1
    return res
```

```python
    total = n * (n + 1) // 2; need = (total + 1) // 2
```

## 3154. Find Number of Ways to Reach the K-th Stair

```python
from math import comb
def waysToReachStair(self, k):
    # 从 stair 1 开始, op1: 下到 i-1(不能连续), op2: 上到 i+2^jump, jump++
    # 经过 j 次 op2, 到达 1+2^0+2^1+...+2^(j-1) = 2^j
    # 然后做 d 次 op1 (d = 2^j - k), 在 j+1 个空隙中选 d 个 (每个空隙最多1次)
    res = 0
    for j in range(32):
        pos = (1 << j)
        d = pos - k
        if d < 0: continue
        if d > j + 1: break
        res += comb(j + 1, d)
    return res
```

```python
from collections import Counter
def numberOfPairs(self, nums1, nums2, k):
    cnt2 = Counter(b * k for b in nums2)
    res = 0
    for a in nums1:
        d = 1
        while d * d <= a:
            if a % d == 0:
                if d in cnt2: res += cnt2[d]
                if d != a // d and a // d in cnt2: res += cnt2[a // d]
            d += 1
    return res
```

## 3186. Maximum Total Damage With Spell Casting

https://leetcode.cn/problems/maximum-total-damage-with-spell-casting/

```python
from collections import Counter
def maximumTotalDamage(self, power):
    cnt = Counter(power)
    vals = sorted(cnt)
    n = len(vals)
    dp = [0] * (n + 1)
    j = 0
    for i in range(n):
        dp[i+1] = dp[i]  # skip
        # 选 vals[i], 找最大 j 使得 vals[j] < vals[i]-2
        while j < i and vals[j] < vals[i] - 2: j += 1
        # j 是第一个 >= vals[i]-2 的
        # 前一个安全位置的 dp 索引
        safe = j  # dp[safe] = 不选 vals[j]...vals[i-1] 的最大值
        dp[i+1] = max(dp[i+1], dp[safe] + vals[i] * cnt[vals[i]])
    # 需要更精确的 dp 定义
    # dp[i] = 考虑前 i 个不同值的最大伤害
    dp = [0] * (n + 1)
    for i in range(n):
        dp[i+1] = dp[i]
        # 二分找最大 k 使得 vals[k] <= vals[i] - 3
        lo2, hi2 = 0, i - 1; best = -1
        while lo2 <= hi2:
            mid = (lo2 + hi2) // 2
            if vals[mid] <= vals[i] - 3: best = mid; lo2 = mid + 1
            else: hi2 = mid - 1
        dp[i+1] = max(dp[i+1], dp[best+1] + vals[i] * cnt[vals[i]])
    return dp[n]
```

```python
def maximumTotalCost(self, nums):
    # dp: 选择 + 或 - 当前元素
    pos = nums[0]; neg = float('-inf')  # pos: 当前为+, neg: 当前为-
    for i in range(1, len(nums)):
        new_pos = max(pos, neg) + nums[i]
        new_neg = pos - nums[i]
        pos, neg = new_pos, new_neg
    return max(pos, neg)
```

## 3197. Find the Minimum Area to Cover All Ones in a Binary Grid

```python
def minimumArea(self, grid):
    m, n = len(grid), len(grid[0])
    r1=m; r2=0; c1=n; c2=0
    for i in range(m):
        for j in range(n):
            if grid[i][j]:
                r1=min(r1,i); r2=max(r2,i)
                c1=min(c1,j); c2=max(c2,j)
    return (r2-r1+1)*(c2-c1+1)
```

## 3200. Maximum Height of a Triangle

https://leetcode.cn/problems/maximum-height-of-a-triangle/

```python
def maxHeightOfTriangle(self, red, blue):
    def check(a, b):
        row = 1
        while True:
            if row % 2 == 1:
                if a >= row: a -= row
                else: return row - 1
            else:
                if b >= row: b -= row
                else: return row - 1
            row += 1
    return max(check(red, blue), check(blue, red))
```

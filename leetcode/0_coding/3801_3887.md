```python
from collections import Counter
def minDistinctFreqPair(self, nums):
    cnt = Counter(nums)
    vals = sorted(cnt.keys())
    for i in range(len(vals)):
        for j in range(i + 1, len(vals)):
            if cnt[vals[i]] != cnt[vals[j]]:
                return [vals[i], vals[j]]
    return [-1, -1]
```

### 3871. Count Commas in Range II (Premium)
https://leetcode.com/problems/count-commas-in-range-ii/

Weekly 493 Q2。大范围版本，用数学公式按位数分段。

```python
def countCommas(self, num1, num2):
    def f(n):
        # commas(x) = (len(str(x)) - 1) // 3
        # 统计 [1..n] 所有数的逗号总和
        if n <= 0: return 0
        s = str(n); L = len(s); res = 0
        # 对每个位数 d，d 位数的范围是 [10^(d-1), 10^d - 1]
        for d in range(1, L):
            cnt = 10**d - 10**(d-1)  # d 位数的个数
            res += cnt * ((d - 1) // 3)
        # L 位数: [10^(L-1), n]
        cnt = n - 10**(L-1) + 1
        res += cnt * ((L - 1) // 3)
        return res
    return f(num2) - f(num1 - 1)
```

## 3875. Construct Uniform Parity Array I
https://leetcode.com/problems/construct-uniform-parity-array-i/

Weekly 494 Q1。

```python
def uniformArray(self, nums1):
    # 对每个元素可以保持或减去任意其它元素
    # 奇-奇=偶, 奇-偶=奇, 偶-奇=奇, 偶-偶=偶
    n = len(nums1)
    odds = sum(x % 2 for x in nums1)
    evens = n - odds
    # 尝试全偶: 奇数元素需要 -（奇数元素）变偶，需要另一个奇数
    # 尝试全奇: 偶数元素需要 -（奇数元素）变奇，需要至少1个奇数
    if odds == 0: return True  # 全偶
    if odds == n: return True  # 全奇
    # 全奇可行: 偶数减奇数变奇，需要1个奇数; 奇数保持
    if odds >= 1: return True
    return False
```

## 3876. Construct Uniform Parity Array II
https://leetcode.com/problems/construct-uniform-parity-array-ii/

Weekly 494 Q2。大范围版，同样逻辑 + 需要验证 a-b >= 1。

```python
def uniformArray(self, nums1):
    nums1.sort()
    n = len(nums1)
    odds = [x for x in nums1 if x % 2 == 1]
    evens = [x for x in nums1 if x % 2 == 0]
    # 尝试全偶: 每个奇数 x 减去某个奇数 y(y!=x) 得偶数，需要 x-y>=1
    # 尝试全奇: 每个偶数 x 减去某个奇数 y 得奇数，需要 x-y>=1
    def canAllEven():
        if not odds: return True
        if len(odds) < 2: return False  # 单个奇数无法减去另一个奇数
        mn = odds[0]  # 最小奇数
        for x in odds:
            # 减去最小奇数(不等于自己)
            y = mn if x != mn else (odds[1] if len(odds) > 1 else -1)
            if y == -1 or x - y < 1: return False
        return True
    def canAllOdd():
        if not evens: return True
        if not odds: return False
        mn_odd = odds[0]
        for x in evens:
            if x - mn_odd < 1: return False
        return True
    return canAllEven() or canAllOdd()
```

## 3877. Minimum Removals to Achieve Target XOR
https://leetcode.com/problems/minimum-removals-to-achieve-target-xor/

Weekly 494 Q3。n≤40，折半搜索 (meet in the middle)。

```python
from collections import Counter
def minRemovals(self, nums, target):
    total_xor = 0
    for x in nums: total_xor ^= x
    # 移除集合 S，剩余 XOR = total_xor ^ xor(S) = target
    # 即 xor(S) = total_xor ^ target
    need = total_xor ^ target
    n = len(nums)
    # 折半: 枚举前半/后半所有子集
    def half_xor(arr):
        res = Counter()
        for mask in range(1 << len(arr)):
            xr = 0; cnt = 0
            for i in range(len(arr)):
                if mask >> i & 1:
                    xr ^= arr[i]; cnt += 1
            res[xr] = min(res.get(xr, float('inf')), cnt) if xr in res else cnt
            # 保存最小移除数
        return res
    if need == 0: return 0
    a, b = nums[:n//2], nums[n//2:]
    ha, hb = half_xor(a), half_xor(b)
    ans = float('inf')
    for xr, ca in ha.items():
        rem = need ^ xr
        if rem in hb:
            ans = min(ans, ca + hb[rem])
    # 也检查只用一半
    if need in ha: ans = min(ans, ha[need])
    if need in hb: ans = min(ans, hb[need])
    return ans if ans != float('inf') else -1
```

## 3881. Direction Assignments with Exactly K Visible People
https://leetcode.com/problems/direction-assignments-with-exactly-k-visible-people/

Biweekly 179 Q2。组合数学。

```python
def countVisiblePeople(self, n, pos, k):
    MOD = 10**9 + 7
    from math import comb
    a, b = pos, n - pos - 1  # 左边/右边人数
    res = 0
    for i in range(min(a, k) + 1):
        j = k - i
        if j > b: continue
        # i 人左边选 L，j 人右边选 R，其余自由选
        ways = comb(a, i) * comb(b, j) % MOD
        free = a - i + b - j + 1  # pos 自己也自由
        ways = ways * pow(2, free, MOD) % MOD
        res = (res + ways) % MOD
    return res
```

## 3882. Minimum XOR Path in a Grid
https://leetcode.com/problems/minimum-xor-path-in-a-grid/

Biweekly 179 Q3。DP + bitset。值域 ≤1023，用 set 存可达 XOR 值。

```python
def minCost(self, grid):
    m, n = len(grid), len(grid[0])
    # dp[i][j] = set of reachable XOR values
    dp = [[set() for _ in range(n)] for _ in range(m)]
    dp[0][0].add(grid[0][0])
    for i in range(m):
        for j in range(n):
            for xr in dp[i][j]:
                if i + 1 < m:
                    dp[i+1][j].add(xr ^ grid[i+1][j])
                if j + 1 < n:
                    dp[i][j+1].add(xr ^ grid[i][j+1])
    return min(dp[m-1][n-1])
```

## 3885. Design Event Manager
https://leetcode.com/problems/design-event-manager/

Weekly 495 Q2。堆 + 懒删除。

```python
import heapq
class EventManager:
    def __init__(self, events):
        self.pri = {}  # eventId -> priority
        self.heap = []  # (-priority, eventId)
        for eid, p in events:
            self.pri[eid] = p
            heapq.heappush(self.heap, (-p, eid))
    
    def updatePriority(self, eventId, newPriority):
        self.pri[eventId] = newPriority
        heapq.heappush(self.heap, (-newPriority, eventId))
    
    def pollHighest(self):
        while self.heap:
            neg_p, eid = heapq.heappop(self.heap)
            if eid in self.pri and self.pri[eid] == -neg_p:
                del self.pri[eid]
                return eid
        return -1
```

## 3886. Sum of Sortable Integers
https://leetcode.com/problems/sum-of-sortable-integers/

Weekly 495 Q3。枚举 n 的每个因子 k，检查每个 block 是否可通过循环旋转匹配排序后的结果。

```python
def sortableIntegers(self, nums):
    n = len(nums)
    s = sorted(nums)
    res = 0
    for k in range(1, n + 1):
        if n % k != 0: continue
        ok = True
        for i in range(0, n, k):
            block = nums[i:i+k]
            target = s[i:i+k]
            # 检查 target 是否是 block 的循环旋转
            doubled = block + block
            found = False
            for start in range(k):
                if doubled[start:start+k] == target:
                    found = True; break
            if not found:
                ok = False; break
        if ok: res += k
    return res
```

## 3887. Incremental Even-Weighted Cycle Queries
https://leetcode.com/problems/incremental-even-weighted-cycle-queries/

Weekly 495 Q4。加权并查集 (weighted DSU)。维护节点到根的奇偶距离。加边时如果两端在同一连通分量，检查路径奇偶是否与新边权一致；不同分量则合并。

```python
class Solution:
    def numberOfEdgesAdded(self, n, edges):
        parent = list(range(n))
        rank = [0] * n
        parity = [0] * n  # parity[x] = x 到 parent[x] 路径的奇偶
        
        def find(x):
            if parent[x] != x:
                root = find(parent[x])
                parity[x] ^= parity[parent[x]]
                parent[x] = root
            return parent[x]
        
        cnt = 0
        for u, v, w in edges:
            ru, rv = find(u), find(v)
            if ru == rv:
                # 检查 parity[u] ^ parity[v] == w
                if parity[u] ^ parity[v] == w:
                    cnt += 1
            else:
                # 合并
                if rank[ru] < rank[rv]:
                    ru, rv = rv, ru
                    u, v = v, u
                parent[rv] = ru
                parity[rv] = parity[u] ^ parity[v] ^ w
                if rank[ru] == rank[rv]:
                    rank[ru] += 1
                cnt += 1
        return cnt
```

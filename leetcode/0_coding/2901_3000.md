## 2901. Longest Unequal Adjacent Groups Subsequence II

https://leetcode.cn/problems/longest-unequal-adjacent-groups-subsequence-ii/

```python
def getWordsInLongestSubsequence(self, n, words, groups):
    # 相邻组不同且 hamming distance = 1
    dp = [1] * n; prev = [-1] * n
    for i in range(n):
        for j in range(i):
            if groups[i] != groups[j] and len(words[i]) == len(words[j]):
                if sum(a != b for a, b in zip(words[i], words[j])) == 1:
                    if dp[j] + 1 > dp[i]:
                        dp[i] = dp[j] + 1; prev[i] = j
    idx = max(range(n), key=lambda i: dp[i])
    res = []
    while idx != -1: res.append(words[idx]); idx = prev[idx]
    return res[::-1]
```

## 2902. Count of Sub-Multisets With Bounded Sum

https://leetcode.cn/problems/count-of-sub-multisets-with-bounded-sum/

多重背包 DP。

```python
from collections import Counter
def countSubMultisets(self, nums, l, r):
    MOD = 10**9 + 7
    cnt = Counter(nums)
    zeros = cnt.pop(0, 0)
    dp = [0] * (r + 1); dp[0] = 1
    for val, c in cnt.items():
        # 多重背包：前缀和优化
        new_dp = [0] * (r + 1)
        for rem in range(val):
            s = 0
            j = rem
            while j <= r:
                s = (s + dp[j]) % MOD
                if j - val * (c + 1) >= 0:
                    s = (s - dp[j - val * (c + 1)]) % MOD
                new_dp[j] = s
                j += val
        dp = new_dp
    return (zeros + 1) * sum(dp[l:r+1]) % MOD
```

```python
        j = i - indexDifference
```

## 2906. Construct Product Matrix

https://leetcode.cn/problems/construct-product-matrix/

```python
def constructProductMatrix(self, grid):
    MOD = 12345
    m, n = len(grid), len(grid[0])
    flat = [grid[i][j] for i in range(m) for j in range(n)]
    N = len(flat)
    prefix = [1] * (N + 1); suffix = [1] * (N + 1)
    for i in range(N): prefix[i+1] = prefix[i] * flat[i] % MOD
    for i in range(N-1, -1, -1): suffix[i] = suffix[i+1] * flat[i] % MOD
    res = [[0]*n for _ in range(m)]
    for i in range(m):
        for j in range(n):
            k = i * n + j
            res[i][j] = prefix[k] * suffix[k+1] % MOD
    return res
```

## 2910. Minimum Number of Groups to Create a Valid Assignment

https://leetcode.cn/problems/minimum-number-of-groups-to-create-a-valid-assignment/

```python
from collections import Counter
from math import ceil
def minGroupsForValidAssignment(self, nums):
    cnt = Counter(nums)
    mn = min(cnt.values())
    for sz in range(mn, 0, -1):
        # 每组大小 sz 或 sz+1
        total = 0; ok = True
        for v in cnt.values():
            # x*(sz+1) + y*sz = v, x+y 最小, x,y>=0
            # y = (v - x*(sz+1)) / sz, 需 v % (sz+1) 对
            q, r = divmod(v, sz + 1)
            if r == 0:
                total += q
            elif q >= (sz - r):
                # 可以拆: 把 sz-r 个 sz+1 组变成 sz 组 + 1
                total += q + ceil(r / sz)
            else:
                # 不行
                # 另一种算法
                groups = ceil(v / (sz + 1))
                if v - groups * sz < 0 or groups * (sz+1) < v:
                    ok = False; break
                total += groups
        if ok: return total
    return len(nums)
```

## 2939. Maximum Xor Product

https://leetcode.cn/problems/maximum-xor-product/

```python
def maximumXorProduct(self, a, b, n):
    MOD = 10**9 + 7
    # 从高到低贪心设 x 的每一位
    x_a, x_b = a, b  # x XOR a 和 x XOR b
    for i in range(n-1, -1, -1):
        ba = (x_a >> i) & 1; bb = (x_b >> i) & 1
        if ba == bb:
            # 都设为1
            if ba == 0:
                x_a |= (1 << i); x_b |= (1 << i)
        else:
            # 给小的那个设1
            if x_a < x_b:
                x_a |= (1 << i); x_b &= ~(1 << i)
            else:
                x_b |= (1 << i); x_a &= ~(1 << i)
    return x_a % MOD * (x_b % MOD) % MOD
```

```python
from collections import deque
def minimumCoins(self, prices):
    n = len(prices)
    dp = [float('inf')] * (n + 1)
    dp[0] = 0
    for i in range(1, n + 1):
        # 买第 i 个水果, 可以免费拿 i+1 到 2i
        for j in range(max(0, (i-1)//2), i):
            # 从 j 买了,可以免费到 2j
            pass
    # 另一种DP: dp[i] = 买到第i个的最小代价, 可以选择买i然后跳到 i+i+1 或继续
    dp = [0] * (n + 2)
    dq = deque()  # 单调递增队列
    for i in range(n, 0, -1):
        while dq and dq[0] > 2 * i: dq.popleft()
        dp[i] = prices[i-1] + (dp[dq[0]] if dq else 0)
        while dq and dp[dq[-1]] >= dp[i]: dq.pop()
        dq.append(i)
    return dp[1]
```

```python
def lexicographicallySmallestArray(self, nums, limit):
    n = len(nums)
    idx = sorted(range(n), key=lambda i: nums[i])
    res = [0] * n
    i = 0
    while i < n:
        j = i
        while j + 1 < n and nums[idx[j+1]] - nums[idx[j]] <= limit: j += 1
        positions = sorted(idx[i:j+1])
        values = sorted(nums[p] for p in positions)
        for k, p in enumerate(positions):
            res[p] = values[k]
        i = j + 1
    return res
```

## 2959. Number of Possible Sets of Closing Branches

https://leetcode.cn/problems/number-of-possible-sets-of-closing-branches/

枚举关闭的分支子集 + Floyd 检查最短路。

```python
def numberOfSets(self, n, maxDistance, roads):
    res = 0
    for mask in range(1 << n):
        nodes = [i for i in range(n) if mask >> i & 1]
        dist = [[float('inf')]*n for _ in range(n)]
        for i in nodes: dist[i][i] = 0
        for u, v, w in roads:
            if mask >> u & 1 and mask >> v & 1:
                dist[u][v] = min(dist[u][v], w)
                dist[v][u] = min(dist[v][u], w)
        for k in nodes:
            for i in nodes:
                for j in nodes:
                    dist[i][j] = min(dist[i][j], dist[i][k]+dist[k][j])
        if all(dist[i][j] <= maxDistance for i in nodes for j in nodes):
            res += 1
    return res
```

```python
    while not is_pal(lo): lo -= 1
    while not is_pal(hi): hi += 1
    return min(cost(lo), cost(hi))
```

## 2968. Apply Operations to Maximize Frequency Score

https://leetcode.cn/problems/apply-operations-to-maximize-frequency-score/

排序 + 二分/滑动窗口 + 前缀和。

```python
def maxFrequencyScore(self, nums, k):
    nums.sort(); n = len(nums)
    prefix = [0] * (n + 1)
    for i in range(n): prefix[i+1] = prefix[i] + nums[i]
    def cost(l, r):
        mid = (l + r) // 2
        left = nums[mid] * (mid - l) - (prefix[mid] - prefix[l])
        right = (prefix[r+1] - prefix[mid+1]) - nums[mid] * (r - mid)
        return left + right
    l = res = 0
    for r in range(n):
        while cost(l, r) > k: l += 1
        res = max(res, r - l + 1)
    return res
```

## 2970. Count the Number of Incremovable Subarrays I / 2972. Count the Number of Incremovable Subarrays II

https://leetcode.cn/problems/count-the-number-of-incremovable-subarrays-ii/

```python
from bisect import bisect_left
def incremovableSubarrayCount(self, nums):
    n = len(nums)
    # 找最长严格递增前缀和后缀
    l = 0
    while l + 1 < n and nums[l] < nums[l+1]: l += 1
    if l == n - 1: return n * (n + 1) // 2  # 全递增
    r = n - 1
    while r > 0 and nums[r-1] < nums[r]: r -= 1
    # 删除 [i+1, j-1] 使得 nums[:i+1] + nums[j:] 严格递增
    res = (n - r + 1) + (l + 2)  # 删到末尾 + 删从头
    for i in range(l + 1):
        # 找最小 j >= r 使得 nums[j] > nums[i]
        j = bisect_left(nums, nums[i] + 1, r, n)
        res += n - j
    return res
```

## 2975. Maximum Square Area by Removing Fences From a Field

https://leetcode.cn/problems/maximum-square-area-by-removing-fences-from-a-field/

```python
def maximizeSquareArea(self, m, n, hFences, vFences):
    MOD = 10**9 + 7
    def get_diffs(fences, limit):
        pts = sorted([1] + fences + [limit])
        diffs = set()
        for i in range(len(pts)):
            for j in range(i+1, len(pts)):
                diffs.add(pts[j] - pts[i])
        return diffs
    h = get_diffs(hFences, m)
    v = get_diffs(vFences, n)
    common = h & v
    return max(common) ** 2 % MOD if common else -1
```

## 2976. Minimum Cost to Convert String I

```python
def minimumCost(self, source, target, original, changed, cost):
    dist = [[float('inf')]*26 for _ in range(26)]
    for i in range(26): dist[i][i] = 0
    for o, c, w in zip(original, changed, cost):
        i, j = ord(o)-97, ord(c)-97
        dist[i][j] = min(dist[i][j], w)
    for k in range(26):
        for i in range(26):
            for j in range(26):
                dist[i][j] = min(dist[i][j], dist[i][k]+dist[k][j])
    res = 0
    for s, t in zip(source, target):
        d = dist[ord(s)-97][ord(t)-97]
        if d == float('inf'): return -1
        res += d
    return res
```

## 2981. Find Longest Special Substring That Occurs Thrice I / 2982. Find Longest Special Substring That Occurs Thrice II

https://leetcode.cn/problems/find-longest-special-substring-that-occurs-thrice-ii/

```python
from collections import defaultdict
def maximumLength(self, s):
    groups = defaultdict(list)  # char -> list of run lengths
    i = 0
    while i < len(s):
        j = i
        while j < len(s) and s[j] == s[i]: j += 1
        groups[s[i]].append(j - i)
        i = j
    res = -1
    for c, runs in groups.items():
        runs.sort(reverse=True)
        # 3种方式获得长度 L 出现 >= 3 次
        # 1. 最长 run 贡献 L, L, L-1: runs[0]-2
        # 2. 两个最长 run: min(runs[0]-1, runs[1])
        # 3. 三个 run: runs[2] if exists
        cands = [runs[0] - 2]
        if len(runs) >= 2: cands.append(min(runs[0]-1, runs[1]))
        if len(runs) >= 3: cands.append(runs[2])
        best = max(cands)
        if best > 0: res = max(res, best)
    return res
```

## 2998. Minimum Number of Operations to Make X and Y Equal

```python
from functools import lru_cache
def minimumOperationsToMakeEqual(self, x, y):
    if x <= y: return y - x
    @lru_cache(None)
    def dp(n):
        if n <= y: return y - n
        res = n - y
        for d in [5, 11]:
            res = min(res, n % d + 1 + dp(n // d))
            res = min(res, d - n % d + 1 + dp(n // d + 1))
        return res
    return dp(x)
```

## 2999. Count the Number of Powerful Integers

数位 DP。

```python
def numberOfPowerfulInt(self, start, finish, limit, s):
    def count(num):
        # count of powerful integers <= num
        t = str(num)
        n = len(t); m = len(s)
        if n < m: return 0
        pre_len = n - m
        suffix = t[pre_len:]
        # 前 pre_len 位自由选(受 limit 限制), 后缀 = s
        # 数位DP
        from functools import lru_cache
        @lru_cache(None)
        def dp(pos, tight):
            if pos == n:
                return 1
            if pos >= pre_len:
                # 必须匹配 s
                d = int(s[pos - pre_len])
                if tight:
                    hi = int(t[pos])
                    if d > hi: return 0
                    return dp(pos + 1, d == hi)
                else:
                    return dp(pos + 1, False)
            hi = int(t[pos]) if tight else 9
            hi = min(hi, limit)
            res = 0
            for d in range(0, hi + 1):
                res += dp(pos + 1, tight and d == int(t[pos]))
            return res
        return dp(0, True)
    return count(finish) - count(start - 1)
```

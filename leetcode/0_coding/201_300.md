## 201. Bitwise AND of Numbers Range

https://leetcode.cn/problems/bitwise-and-of-numbers-range/

找公共前缀：不断右移直到相等，再左移回来。

```python
def rangeBitwiseAnd(self, left, right):
    shift = 0
    while left != right:
        left >>= 1; right >>= 1
        shift += 1
    return left << shift
```

```python
def countPrimes(self, n):
    if n < 2: return 0
    is_prime = [True] * n
    is_prime[0] = is_prime[1] = False
    for i in range(2, int(n**0.5) + 1):
        if is_prime[i]:
            is_prime[i*i:n:i] = [False] * len(range(i*i, n, i))
    return sum(is_prime)
```

## 207. Course Schedule

拓扑排序 BFS 检测环。

```python
from collections import deque, defaultdict
def canFinish(self, numCourses, prerequisites):
    graph = defaultdict(list)
    indeg = [0] * numCourses
    for a, b in prerequisites:
        graph[b].append(a)
        indeg[a] += 1
    q = deque(i for i in range(numCourses) if indeg[i] == 0)
    count = 0
    while q:
        node = q.popleft(); count += 1
        for nb in graph[node]:
            indeg[nb] -= 1
            if indeg[nb] == 0: q.append(nb)
    return count == numCourses
```

```python
class WordDictionary:

        self.root = {}
    def addWord(self, word):

            node = node.setdefault(c, {})
```

## 212. Word Search II

https://leetcode.cn/problems/word-search-ii/

Trie + DFS 回溯在矩阵中搜索。

```python
def findWords(self, board, words):
    root = {}
    for w in words:
        node = root
        for c in w:
            node = node.setdefault(c, {})
        node['#'] = w
    m, n = len(board), len(board[0])
    res = []
    def dfs(i, j, node):
        c = board[i][j]
        if c not in node: return
        node = node[c]
        if '#' in node:
            res.append(node['#'])
            del node['#']
        board[i][j] = '.'
        for di, dj in [(-1,0),(1,0),(0,-1),(0,1)]:
            ni, nj = i+di, j+dj
            if 0 <= ni < m and 0 <= nj < n and board[ni][nj] != '.':
                dfs(ni, nj, node)
        board[i][j] = c
    for i in range(m):
        for j in range(n):
            dfs(i, j, root)
    return res
```

```python
def rob(self, nums):
    if len(nums) == 1: return nums[0]
    def helper(arr):
        prev2 = prev1 = 0
        for x in arr:
            prev2, prev1 = prev1, max(prev1, prev2 + x)
        return prev1
    return max(helper(nums[:-1]), helper(nums[1:]))
```

## 214. Shortest Palindrome

https://leetcode.cn/problems/shortest-palindrome/

KMP：找 `s + '#' + rev_s` 的最长前缀后缀。

```python
def shortestPalindrome(self, s):
    rev = s[::-1]
    t = s + '#' + rev
    lps = [0] * len(t)
    for i in range(1, len(t)):
        j = lps[i - 1]
        while j > 0 and t[i] != t[j]: j = lps[j - 1]
        if t[i] == t[j]: j += 1
        lps[i] = j
    return rev[:len(s) - lps[-1]] + s
```

## 218. The Skyline Problem

https://leetcode.cn/problems/the-skyline-problem/

扫描线 + 最大堆。

```python
import heapq
def getSkyline(self, buildings):
    events = []
    for l, r, h in buildings:
        events.append((l, -h, r))
        events.append((r, 0, 0))
    events.sort()
    res = [[0, 0]]
    heap = [(0, float('inf'))]  # (-height, end)
    for x, neg_h, r in events:
        if neg_h:  # 建筑开始
            heapq.heappush(heap, (neg_h, r))
        while heap[0][1] <= x:  # 移除已结束
            heapq.heappop(heap)
        max_h = -heap[0][0]
        if max_h != res[-1][1]:
            res.append([x, max_h])
    return res[1:]
```

## 220. Contains Duplicate III

桶排序：桶大小 `t+1`，检查相邻桶。

```python
def containsNearbyAlmostDuplicate(self, nums, indexDiff, valueDiff):
    if valueDiff < 0: return False
    buckets = {}
    w = valueDiff + 1
    for i, x in enumerate(nums):
        b = x // w
        if b in buckets: return True
        if b - 1 in buckets and abs(x - buckets[b - 1]) < w: return True
        if b + 1 in buckets and abs(x - buckets[b + 1]) < w: return True
        buckets[b] = x
        if i >= indexDiff: del buckets[nums[i - indexDiff] // w]
    return False
```

## 221. Maximal Square

https://leetcode.cn/problems/maximal-square/

DP：`dp[i][j] = min(上, 左, 左上) + 1`。

```python
def maximalSquare(self, matrix):
    m, n = len(matrix), len(matrix[0])
    dp = [0] * (n + 1)
    res = 0
    for i in range(m):
        new_dp = [0] * (n + 1)
        for j in range(n):
            if matrix[i][j] == '1':
                new_dp[j + 1] = min(dp[j], dp[j + 1], new_dp[j]) + 1
                res = max(res, new_dp[j + 1])
        dp = new_dp
    return res * res
```

```python
def countNodes(self, root):
    if not root: return 0
    l = r = root
    hl = hr = 0
    while l: l = l.left; hl += 1
    while r: r = r.right; hr += 1
    if hl == hr: return (1 << hl) - 1
```

## 223. Rectangle Area

```python
def computeArea(self, ax1, ay1, ax2, ay2, bx1, by1, bx2, by2):
    area = (ax2-ax1)*(ay2-ay1) + (bx2-bx1)*(by2-by1)
    overlap_w = max(0, min(ax2, bx2) - max(ax1, bx1))
    overlap_h = max(0, min(ay2, by2) - max(ay1, by1))
    return area - overlap_w * overlap_h
```

```python
def majorityElement(self, nums):
    c1 = c2 = 0; n1 = n2 = None
    for x in nums:
        if x == n1: c1 += 1
        elif x == n2: c2 += 1
        elif c1 == 0: n1, c1 = x, 1
        elif c2 == 0: n2, c2 = x, 1
```

```python
def singleNumber(self, nums):
```

```python
    i2 = i3 = i5 = 0
    for i in range(1, n):
        dp[i] = min(dp[i2]*2, dp[i3]*3, dp[i5]*5)
        if dp[i] == dp[i2]*2: i2 += 1
        if dp[i] == dp[i3]*3: i3 += 1
        if dp[i] == dp[i5]*5: i5 += 1
    return dp[-1]
```

```python
def numSquares(self, n):
    dp = [0] + [float('inf')] * n
    for i in range(1, n + 1):
        j = 1
        while j * j <= i:
            dp[i] = min(dp[i], dp[i - j*j] + 1)
```

```python
def addOperators(self, num, target):
    res = []
    def bt(idx, path, val, prev):
        if idx == len(num):
            if val == target: res.append(path)
            return
        for i in range(idx + 1, len(num) + 1):
            s = num[idx:i]
            n = int(s)
            if len(s) > 1 and s[0] == '0': break
            if idx == 0:
                bt(i, s, n, n)
            else:
                bt(i, path+'+'+s, val+n, n)
                bt(i, path+'-'+s, val-n, -n)
                bt(i, path+'*'+s, val-prev+prev*n, prev*n)
    bt(0, '', 0, 0)
    return res
```

```python
def gameOfLife(self, board):
    m, n = len(board), len(board[0])
    for i in range(m):
        for j in range(n):
            live = 0
            for di in range(-1, 2):
                for dj in range(-1, 2):
                    if di == 0 and dj == 0: continue
                    ni, nj = i+di, j+dj
                    if 0 <= ni < m and 0 <= nj < n and board[ni][nj] in (1, 2):
                        live += 1
            if board[i][j] == 1 and (live < 2 or live > 3): board[i][j] = 2
            if board[i][j] == 0 and live == 3: board[i][j] = 3
    for i in range(m):
        for j in range(n):
            board[i][j] %= 2
```

```python
import heapq
class MedianFinder:
    def __init__(self):
        self.lo = []  # max heap (negated)
        self.hi = []  # min heap
    def addNum(self, num):
        heapq.heappush(self.lo, -num)
        heapq.heappush(self.hi, -heapq.heappop(self.lo))
        if len(self.hi) > len(self.lo):
            heapq.heappush(self.lo, -heapq.heappop(self.hi))
    def findMedian(self):
        if len(self.lo) > len(self.hi):
            return -self.lo[0]
        return (-self.lo[0] + self.hi[0]) / 2
```

```python
import bisect
def lengthOfLIS(self, nums):
    tails = []
    for x in nums:
        pos = bisect.bisect_left(tails, x)
        if pos == len(tails):
            tails.append(x)
        else:
            tails[pos] = x
    return len(tails)
```

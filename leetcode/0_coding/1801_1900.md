## 1802. Maximum Value at a Given Index in a Bounded Array
https://leetcode.com/problems/maximum-value-at-a-given-index-in-a-bounded-array/

二分答案，计算最小总和。

```python
def maxValue(self, n, index, maxSum):
    def calc(v, cnt):
        if v >= cnt: return v * cnt - cnt * (cnt - 1) // 2
        return v * (v + 1) // 2 + (cnt - v)
    lo, hi = 1, maxSum
    while lo < hi:
        mid = (lo + hi + 1) // 2
        left = calc(mid - 1, index)
        right = calc(mid - 1, n - index - 1)
        if left + right + mid <= maxSum: lo = mid
        else: hi = mid - 1
    return lo
```

## 1818. Minimum Absolute Sum Difference
https://leetcode.com/problems/minimum-absolute-sum-difference/

排序 + 二分找最优替换。

```python
import bisect
def minAbsoluteSumDiff(self, nums1, nums2):
    MOD = 10**9 + 7
    total = sum(abs(a - b) for a, b in zip(nums1, nums2))
    sorted1 = sorted(nums1)
    max_improve = 0
    for a, b in zip(nums1, nums2):
        cur = abs(a - b)
        idx = bisect.bisect_left(sorted1, b)
        for j in [idx - 1, idx]:
            if 0 <= j < len(sorted1):
                new = abs(sorted1[j] - b)
                max_improve = max(max_improve, cur - new)
    return (total - max_improve) % MOD
```

```python
def countDifferentSubsequenceGCDs(self, nums):
    s = set(nums); mx = max(nums); res = 0
    for g in range(1, mx + 1):
        cur = 0
```

```python
import heapq
def getOrder(self, tasks):
    indexed = sorted(enumerate(tasks), key=lambda x: x[1][0])
    res = []; heap = []; i = 0; time = 0
    while i < len(indexed) or heap:
        if not heap and i < len(indexed) and indexed[i][1][0] > time:
            time = indexed[i][1][0]
        while i < len(indexed) and indexed[i][1][0] <= time:
            idx, (_, dur) = indexed[i]

        dur, idx = heapq.heappop(heap)
        time += dur
        res.append(idx)
    return res
```

## 1840. Maximum Building Height
https://leetcode.com/problems/maximum-building-height/

排序限制后双向传播。

```python
def maxBuilding(self, n, restrictions):
    restrictions.append([1, 0])
    restrictions.sort()
    if not restrictions or restrictions[-1][0] != n:
        restrictions.append([n, n - 1])
    # 正向传播
    for i in range(1, len(restrictions)):
        restrictions[i][1] = min(restrictions[i][1],
            restrictions[i-1][1] + restrictions[i][0] - restrictions[i-1][0])
    # 反向传播
    for i in range(len(restrictions)-2, -1, -1):
        restrictions[i][1] = min(restrictions[i][1],
            restrictions[i+1][1] + restrictions[i+1][0] - restrictions[i][0])
    res = 0
    for i in range(1, len(restrictions)):
        p1, h1 = restrictions[i-1]
        p2, h2 = restrictions[i]
        # 两点之间最大高度
        res = max(res, (h1 + h2 + p2 - p1) // 2)
    return res
```

```python
        if idx == len(s): return cnt >= 2
        for i in range(idx, len(s)):
            num = int(s[idx:i+1])
```

## 1851. Minimum Interval to Include Each Query

```python
import heapq
def minInterval(self, intervals, queries):
    intervals.sort()
    sorted_q = sorted(enumerate(queries), key=lambda x: x[1])
    res = [-1] * len(queries)
    heap = []; i = 0
    for qi, q in sorted_q:
        while i < len(intervals) and intervals[i][0] <= q:
            l, r = intervals[i]
            heapq.heappush(heap, (r - l + 1, r))
            i += 1
        while heap and heap[0][1] < q:
            heapq.heappop(heap)
        if heap:
            res[qi] = heap[0][0]
    return res
```

## 1856. Maximum Subarray Min-Product
https://leetcode.com/problems/maximum-subarray-min-product/

单调栈 + 前缀和。

```python
def maxSumMinProduct(self, nums):
    MOD = 10**9 + 7
    n = len(nums)
    prefix = [0] * (n + 1)
    for i in range(n): prefix[i+1] = prefix[i] + nums[i]
    left = [-1] * n; right = [n] * n
    stack = []
    for i in range(n):
        while stack and nums[stack[-1]] >= nums[i]:
            right[stack.pop()] = i
        if stack: left[i] = stack[-1]
        stack.append(i)
    res = 0
    for i in range(n):
        s = prefix[right[i]] - prefix[left[i] + 1]
        res = max(res, nums[i] * s)
    return res % MOD
```

## 1857. Largest Color Value in a Directed Graph
https://leetcode.com/problems/largest-color-value-in-a-directed-graph/

拓扑排序 + DP。

```python
from collections import deque
def largestPathValue(self, colors, edges):
    n = len(colors)
    graph = [[] for _ in range(n)]
    indeg = [0] * n
    for u, v in edges:
        graph[u].append(v); indeg[v] += 1
    dp = [[0]*26 for _ in range(n)]
    q = deque()
    for i in range(n):
        if indeg[i] == 0: q.append(i)
    processed = 0
    for i in range(n):
        dp[i][ord(colors[i]) - ord('a')] = 1
    while q:
        u = q.popleft(); processed += 1
        for v in graph[u]:
            for c in range(26):
                dp[v][c] = max(dp[v][c], dp[u][c] + (1 if c == ord(colors[v]) - ord('a') else 0))
            indeg[v] -= 1
            if indeg[v] == 0: q.append(v)
    if processed != n: return -1
    return max(max(row) for row in dp)
```

```python
def rearrangeSticks(self, n, k):
    MOD = 10**9 + 7
    dp = [0] * (k + 1); dp[0] = 1
    for i in range(1, n + 1):
        new_dp = [0] * (k + 1)
```

## 1879. Minimum XOR Sum of Two Arrays
https://leetcode.com/problems/minimum-xor-sum-of-two-arrays/

状压 DP。

```python
from functools import lru_cache
def minimumXORSum(self, nums1, nums2):
    n = len(nums1)
    @lru_cache(maxsize=None)
    def dp(i, mask):
        if i == n: return 0
        res = float('inf')
        for j in range(n):
            if not (mask & (1 << j)):
                res = min(res, (nums1[i] ^ nums2[j]) + dp(i+1, mask | (1 << j)))
        return res
    return dp(0, 0)
```

## 1882. Process Tasks Using Servers
https://leetcode.com/problems/process-tasks-using-servers/

两个堆：空闲服务器和忙碌服务器。

```python
import heapq
def assignTasks(self, servers, tasks):
    free = [(w, i) for i, w in enumerate(servers)]
    heapq.heapify(free)
    busy = []; res = []; time = 0
    for t, dur in enumerate(tasks):
        time = max(time, t)
        if not free:
            time = max(time, busy[0][0])
        while busy and busy[0][0] <= time:
            _, w, i = heapq.heappop(busy)
            heapq.heappush(free, (w, i))
        w, i = heapq.heappop(free)
        res.append(i)
        heapq.heappush(busy, (time + dur, w, i))
    return res
```

```python
    n = len(s); s2 = s + s
    t1 = ''.join('01'[i%2] for i in range(2*n))
```

```python
def maximumRemovals(self, s, p, removable):
    def check(k):
        removed = set(removable[:k])
        j = 0
        for i in range(len(s)):
            if i in removed: continue
            if j < len(p) and s[i] == p[j]: j += 1
        return j == len(p)
    lo, hi = 0, len(removable)
    while lo < hi:
        mid = (lo + hi + 1) // 2
        if check(mid): lo = mid
        else: hi = mid - 1
    return lo
```

```python
def mergeTriplets(self, triplets, target):
```

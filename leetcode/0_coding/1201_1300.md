## 1201. Ugly Number III

https://leetcode.cn/problems/ugly-number-iii/

二分 + 容斥原理。

```python
from math import gcd
def nthUglyNumber(self, n, a, b, c):
    def lcm(x, y): return x * y // gcd(x, y)
    ab, ac, bc = lcm(a,b), lcm(a,c), lcm(b,c)
    abc = lcm(ab, c)
    def count(x):
        return x//a + x//b + x//c - x//ab - x//ac - x//bc + x//abc
    lo, hi = 1, 2 * 10**9
    while lo < hi:
        mid = (lo + hi) // 2
        if count(mid) < n: lo = mid + 1
        else: hi = mid
    return lo
```

## 1202. Smallest String With Swaps

https://leetcode.cn/problems/smallest-string-with-swaps/

并查集分组，每组内排序。

```python
from collections import defaultdict
def smallestStringWithSwaps(self, s, pairs):
    n = len(s)
    parent = list(range(n))
    def find(x):
        while parent[x] != x:
            parent[x] = parent[parent[x]]
            x = parent[x]
        return x
    def union(a, b):
        parent[find(a)] = find(b)
    for a, b in pairs: union(a, b)
    groups = defaultdict(list)
    for i in range(n): groups[find(i)].append(i)
    res = list(s)
    for indices in groups.values():
        chars = sorted(res[i] for i in indices)
        for i, idx in enumerate(sorted(indices)):
            res[idx] = chars[i]
    return ''.join(res)
```

```python
def equalSubstring(self, s, t, maxCost):
    left = cost = res = 0
    for right in range(len(s)):
        cost += abs(ord(s[right]) - ord(t[right]))
        while cost > maxCost:
            cost -= abs(ord(s[left]) - ord(t[left]))
            left += 1
        res = max(res, right - left + 1)
```

```python
def removeDuplicates(self, s, k):
    stack = []
    for c in s:
        if stack and stack[-1][0] == c:
            stack[-1][1] += 1
            if stack[-1][1] == k: stack.pop()
```

```python
def queensAttacktheKing(self, queens, king):
    queens = set(map(tuple, queens))
    res = []
    for dx in range(-1, 2):
        for dy in range(-1, 2):
            if dx == 0 and dy == 0: continue
            x, y = king[0] + dx, king[1] + dy
```

## 1223. Dice Roll Simulation

https://leetcode.cn/problems/dice-roll-simulation/

DP：`dp[i][j][k]` = 第 i 次掷骰子，结果 j，连续 k 次 j。

```python
from functools import lru_cache
def dieSimulator(self, n, rollMax):
    MOD = 10**9 + 7
    @lru_cache(None)
    def dp(pos, last, count):
        if pos == n: return 1
        res = 0
        for d in range(6):
            if d == last:
                if count < rollMax[d]:
                    res += dp(pos+1, d, count+1)
            else:
                res += dp(pos+1, d, 1)
        return res % MOD
    return dp(0, -1, 0)
```

```python
    target = n // 4
```

```python
import bisect
def jobScheduling(self, startTime, endTime, profit):
    jobs = sorted(zip(endTime, startTime, profit))
    dp = [(0, 0)]  # (end_time, max_profit)
    for e, s, p in jobs:
        i = bisect.bisect(dp, (s, float('inf'))) - 1
        new_profit = dp[i][1] + p
        if new_profit > dp[-1][1]:
            dp.append((e, new_profit))
    return dp[-1][1]
```

## 1239. Maximum Length of a Concatenated String with Unique Characters

https://leetcode.cn/problems/maximum-length-of-a-concatenated-string-with-unique-characters/

回溯 + bitmask。

```python
def maxLength(self, arr):
    masks = []
    for s in arr:
        mask = 0; dup = False
        for c in s:
            bit = 1 << (ord(c) - ord('a'))
            if mask & bit: dup = True; break
            mask |= bit
        if not dup: masks.append(mask)
    self.res = 0
    def bt(idx, cur, length):
        self.res = max(self.res, length)
        for i in range(idx, len(masks)):
            if cur & masks[i] == 0:
                bt(i+1, cur | masks[i], length + bin(masks[i]).count('1'))
    bt(0, 0, 0)
    return self.res
```

```python
        if a == 'x' and b == 'y': xy += 1
```

```python
def reconstructMatrix(self, upper, lower, colsum):
    n = len(colsum)
    row1 = [0] * n; row2 = [0] * n
    for i in range(n):
        if colsum[i] == 2:
            row1[i] = row2[i] = 1; upper -= 1; lower -= 1
        elif colsum[i] == 1:
            if upper >= lower:
                row1[i] = 1; upper -= 1

                row2[i] = 1; lower -= 1
    return [row1, row2] if upper == 0 and lower == 0 else []
```

```python
    def dfs(i, j):
        if i < 0 or i >= m or j < 0 or j >= n or grid[i][j] == 1: return
        grid[i][j] = 1
        for di, dj in [(-1,0),(1,0),(0,-1),(0,1)]: dfs(i+di, j+dj)
```

```python
from collections import Counter
def maxScoreWords(self, words, letters, score):
    avail = Counter(letters)
    def bt(idx, cnt):
        if idx == len(words): return 0
        # 不选
        res = bt(idx + 1, cnt)
        # 选
        wc = Counter(words[idx])
        new_cnt = cnt.copy()
        for c, v in wc.items():
            new_cnt[c] = new_cnt.get(c, 0) - v
        if all(v >= 0 for v in new_cnt.values()):
            s = sum(score[ord(c)-ord('a')] * v for c, v in wc.items())
            res = max(res, s + bt(idx + 1, new_cnt))
        return res
    return bt(0, avail)
```

```python
def numWays(self, steps, arrLen):
    MOD = 10**9 + 7
    max_pos = min(arrLen, steps // 2 + 1)
    dp = [0] * max_pos
    dp[0] = 1
    for _ in range(steps):
        ndp = [0] * max_pos
```

## 1278. Palindrome Partitioning III

https://leetcode.cn/problems/palindrome-partitioning-iii/

DP：`dp[i][k]` = 前 i 个字符分 k 段的最小改变次数。

```python
def palindromePartition(self, s, k):
    n = len(s)
    # cost[i][j] = 将 s[i..j] 变为回文需要的改变次数
    cost = [[0]*n for _ in range(n)]
    for length in range(2, n+1):
        for i in range(n-length+1):
            j = i + length - 1
            cost[i][j] = cost[i+1][j-1] + (0 if s[i]==s[j] else 1)
    dp = [[float('inf')]*(k+1) for _ in range(n+1)]
    dp[0][0] = 0
    for i in range(1, n+1):
        for kk in range(1, min(i, k)+1):
            for j in range(kk, i+1):
                dp[i][kk] = min(dp[i][kk], dp[j-1][kk-1] + cost[j-1][i-1])
    return dp[n][k]
```

```python
def minFallingPathSum(self, grid):
    n = len(grid)
    prev = grid[0]
    for i in range(1, n):
        # 找最小和次小
        idx = sorted(range(n), key=lambda x: prev[x])
        cur = [0] * n
        for j in range(n):
            cur[j] = grid[i][j] + (prev[idx[0]] if j != idx[0] else prev[idx[1]])
        prev = cur
    return min(prev)
```

## 1292. Maximum Side Length of a Square with Sum Less than or Equal to Threshold

https://leetcode.cn/problems/maximum-side-length-of-a-square-with-sum-less-than-or-equal-to-threshold/

前缀和 + 二分。

```python
def maxSideLength(self, mat, threshold):
    m, n = len(mat), len(mat[0])
    prefix = [[0]*(n+1) for _ in range(m+1)]
    for i in range(m):
        for j in range(n):
            prefix[i+1][j+1] = mat[i][j]+prefix[i][j+1]+prefix[i+1][j]-prefix[i][j]
    def get_sum(r1,c1,r2,c2):
        return prefix[r2+1][c2+1]-prefix[r1][c2+1]-prefix[r2+1][c1]+prefix[r1][c1]
    res = 0
    for i in range(m):
        for j in range(n):
            while res+1 <= min(m-i, n-j) and get_sum(i,j,i+res,j+res) <= threshold:
                res += 1
    return res
```

## 1293. Shortest Path in a Grid with Obstacles Elimination

https://leetcode.cn/problems/shortest-path-in-a-grid-with-obstacles-elimination/

BFS 状态 = (r, c, 剩余消除次数)。

```python
from collections import deque
def shortestPath(self, grid, k):
    m, n = len(grid), len(grid[0])
    if m == 1 and n == 1: return 0
    q = deque([(0, 0, k, 0)])
    visited = {(0, 0, k)}
    while q:
        r, c, rem, steps = q.popleft()
        for dr, dc in [(-1,0),(1,0),(0,-1),(0,1)]:
            nr, nc = r+dr, c+dc
            if 0 <= nr < m and 0 <= nc < n:
                nk = rem - grid[nr][nc]
                if nk >= 0 and (nr, nc, nk) not in visited:
                    if nr == m-1 and nc == n-1: return steps+1
                    visited.add((nr, nc, nk))
                    q.append((nr, nc, nk, steps+1))
    return -1
```

```python
from collections import Counter
def isPossibleDivide(self, nums, k):
    cnt = Counter(nums)
    for x in sorted(cnt):
        if cnt[x] > 0:
            for i in range(k-1, -1, -1):
                cnt[x+i] -= cnt[x]
                if cnt[x+i] < 0: return False
```

## 1297. Maximum Number of Occurrences of a Substring

```python
from collections import Counter
def maxFreq(self, s, maxLetters, minSize, maxSize):
    cnt = Counter()
    for i in range(len(s) - minSize + 1):
        sub = s[i:i+minSize]
        if len(set(sub)) <= maxLetters:
            cnt[sub] += 1
    return max(cnt.values()) if cnt else 0
```

## 1300. Sum of Mutated Array Closest to Target

https://leetcode.cn/problems/sum-of-mutated-array-closest-to-target/

二分查找 value。

```python
def findBestValue(self, arr, target):
    arr.sort()
    prefix = [0]
    for x in arr: prefix.append(prefix[-1] + x)
    n = len(arr)
    lo, hi = 0, max(arr)
    res = lo; best_diff = float('inf')
    while lo <= hi:
        mid = (lo + hi) // 2
        import bisect
        i = bisect.bisect_right(arr, mid)
        s = prefix[i] + mid * (n - i)
        diff = abs(s - target)
        if diff < best_diff or (diff == best_diff and mid < res):
            best_diff = diff; res = mid
        if s < target: lo = mid + 1
        else: hi = mid - 1
    return res
```

## 2202. Maximize the Topmost Element After K Moves
https://leetcode.com/problems/maximize-the-topmost-element-after-k-moves/

```python
def maximumTop(self, nums, k):
    n = len(nums)
    if n == 1: return -1 if k % 2 else nums[0]
    candidates = []
    # 可以取走前 min(k-1, n) 个，留其中最大的
    if k - 1 >= 0:
        candidates.extend(nums[:min(k-1, n)])
    # 或者取走前 k 个后露出 nums[k]
    if k < n: candidates.append(nums[k])
    return max(candidates) if candidates else -1
```

## 2203. Minimum Weighted Subgraph With the Required Paths
https://leetcode.com/problems/minimum-weighted-subgraph-with-the-required-paths/

三次 Dijkstra：从 src1 正向，从 src2 正向，从 dest 反向。

```python
import heapq
from collections import defaultdict
def minimumWeight(self, n, edges, src1, src2, dest):
    g = defaultdict(list); rg = defaultdict(list)
    for u, v, w in edges:
        g[u].append((v, w)); rg[v].append((u, w))
    def dijkstra(graph, src):
        dist = [float('inf')] * n; dist[src] = 0
        h = [(0, src)]
        while h:
            d, u = heapq.heappop(h)
            if d > dist[u]: continue
            for v, w in graph[u]:
                if d + w < dist[v]:
                    dist[v] = d + w; heapq.heappush(h, (dist[v], v))
        return dist
    d1 = dijkstra(g, src1); d2 = dijkstra(g, src2); dd = dijkstra(rg, dest)
    res = min(d1[i] + d2[i] + dd[i] for i in range(n))
    return res if res < float('inf') else -1
```

## 2209. Minimum White Tiles After Covering With Carpets
https://leetcode.com/problems/minimum-white-tiles-after-covering-with-carpets/

DP。

```python
def minimumWhiteTiles(self, floor, numCarpets, carpetLen):
    n = len(floor)
    pre = [0] * (n + 1)
    for i in range(n): pre[i+1] = pre[i] + int(floor[i])
    # dp[i][j] = 前 i 格用了 j 块毯子，最少白色
    dp = [[0] * (numCarpets + 1) for _ in range(n + 1)]
    for i in range(1, n + 1):
        for j in range(numCarpets + 1):
            # 不铺毯子
            dp[i][j] = dp[i-1][j] + int(floor[i-1])
            # 铺毯子覆盖 [i-carpetLen, i-1]
            if j > 0:
                prev = max(0, i - carpetLen)
                dp[i][j] = min(dp[i][j], dp[prev][j-1])
    return dp[n][numCarpets]
```

def maximumBobPoints(self, numArrows, aliceArrows):
    best_score = 0; best_mask = 0
    for mask in range(1, 1 << 12):
        score = 0; arrows = 0
        for i in range(12):
            if mask & (1 << i):
                score += i; arrows += aliceArrows[i] + 1
        if arrows <= numArrows and score > best_score:
            best_score = score; best_mask = mask
    res = [0] * 12
    remaining = numArrows
    for i in range(12):

            remaining -= res[i]
    res[0] += remaining
    return res

def maxValueOfCoins(self, piles, k):
    dp = [0] * (k + 1)
    for pile in piles:
        prefix = [0]
        for x in pile:
            prefix.append(prefix[-1] + x)
            if len(prefix) - 1 > k: break
        new_dp = dp[:]
        for j in range(1, min(len(prefix), k + 1)):
            for m in range(j, k + 1):
                new_dp[m] = max(new_dp[m], dp[m - j] + prefix[j])
        dp = new_dp
    return dp[k]
```

from collections import Counter
class Encrypter:
    def __init__(self, keys, values, dictionary):
        self.enc_map = dict(zip(keys, values))
        # 预计算所有 dictionary 单词加密后的结果
        self.cnt = Counter()
        for w in dictionary:
            e = self.encrypt(w)
            if e: self.cnt[e] += 1
    def encrypt(self, word1):

class ATM:
    def __init__(self):
        self.bills = [0] * 5
        self.denoms = [20, 50, 100, 200, 500]
    def deposit(self, banknotesCount):
        for i in range(5): self.bills[i] += banknotesCount[i]

        res = [0] * 5

            res[i] = min(self.bills[i], amount // self.denoms[i])

def maximumScore(self, scores, edges):
    n = len(scores)
    # 每个节点保留 score 最大的3个邻居
    top = [[] for _ in range(n)]
    for u, v in edges:
        top[u].append((scores[v], v)); top[v].append((scores[u], u))
    for i in range(n):
        top[i] = sorted(top[i], reverse=True)[:3]
    res = -1
    for u, v in edges:
        for _, a in top[u]:
            if a == v: continue
            for _, b in top[v]:
                if b == u or b == a: continue
                res = max(res, scores[a] + scores[u] + scores[v] + scores[b])
    return res
```

from collections import defaultdict
def longestPath(self, parent, s):
    graph = defaultdict(list)
    for i in range(1, len(parent)):
        graph[parent[i]].append(i)
    self.res = 1
    def dfs(u):
        top2 = [0, 0]
        for v in graph[u]:
            child_len = dfs(v)
            if s[u] != s[v]:
                if child_len > top2[0]:
                    top2 = [child_len, top2[0]]
                elif child_len > top2[1]:
                    top2[1] = child_len
        self.res = max(self.res, top2[0] + top2[1] + 1)
        return top2[0] + 1
    dfs(0)
    return self.res
```

def countUnguarded(self, m, n, guards, walls):
    grid = [[0]*n for _ in range(m)]
    for r, c in guards: grid[r][c] = 1  # guard
    for r, c in walls: grid[r][c] = 2   # wall
    for r, c in guards:
        for dr, dc in [(0,1),(0,-1),(1,0),(-1,0)]:
            nr, nc = r+dr, c+dc
            while 0 <= nr < m and 0 <= nc < n and grid[nr][nc] != 1 and grid[nr][nc] != 2:
                if grid[nr][nc] == 0: grid[nr][nc] = 3
                nr += dr; nc += dc
    return sum(grid[r][c] == 0 for r in range(m) for c in range(n))
```

    if (m + n) % 2 == 0: return False  # 总长度必须是偶数
    from functools import lru_cache

    def dp(i, j, bal):

        if i == m or j == n: return False

## 2276. Count Integers in Intervals

```python
from sortedcontainers import SortedList
class CountIntervals:
    def __init__(self):
        self.intervals = SortedList()
        self.cnt = 0
    def add(self, left, right):
        # 合并重叠区间
        i = self.intervals.bisect_left((left,))
        if i > 0 and self.intervals[i-1][1] >= left:
            i -= 1
        while i < len(self.intervals) and self.intervals[i][0] <= right:
            l, r = self.intervals[i]
            left = min(left, l); right = max(right, r)
            self.cnt -= r - l + 1
            self.intervals.pop(i)
        self.intervals.add((left, right))
        self.cnt += right - left + 1
    def count(self):
        return self.cnt
```

## 2290. Minimum Obstacle Removal to Reach Corner
https://leetcode.com/problems/minimum-obstacle-removal-to-reach-corner/

0-1 BFS。

```python
from collections import deque
def minimumObstacles(self, grid):
    m, n = len(grid), len(grid[0])
    dist = [[float('inf')] * n for _ in range(m)]
    dist[0][0] = 0; dq = deque([(0, 0, 0)])
    while dq:
        d, r, c = dq.popleft()
        if d > dist[r][c]: continue
        for dr, dc in [(0,1),(0,-1),(1,0),(-1,0)]:
            nr, nc = r+dr, c+dc
            if 0 <= nr < m and 0 <= nc < n:
                nd = d + grid[nr][nc]
                if nd < dist[nr][nc]:
                    dist[nr][nc] = nd
                    if grid[nr][nc] == 0: dq.appendleft((nd, nr, nc))
                    else: dq.append((nd, nr, nc))
    return dist[m-1][n-1]
```

def minMaxGame(self, nums):

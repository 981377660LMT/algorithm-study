## 2709. Greatest Common Divisor Traversal

https://leetcode.cn/problems/greatest-common-divisor-traversal/

质因数分解 + 并查集。

```python
def canTraverseAllPairs(self, nums):
    if len(nums) == 1: return True
    if 1 in nums: return False
    parent = {}
    def find(x):
        while parent.setdefault(x, x) != x:
            parent[x] = parent[parent[x]]; x = parent[x]
        return x
    def union(a, b):
        pa, pb = find(a), find(b)
        if pa != pb: parent[pa] = pb
    for num in nums:
        tmp = num; d = 2
        while d * d <= tmp:
            if tmp % d == 0:
                union(num, d)
                while tmp % d == 0: tmp //= d
            d += 1
        if tmp > 1: union(num, tmp)
    return len(set(find(x) for x in nums)) == 1
```

            tl = set(); r, c = i-1, j-1

            br = set(); r, c = i+1, j+1

    m, n = len(mat), len(mat[0])
    vals = defaultdict(list)

        for j in range(n):
            vals[mat[i][j]].append((i, j))

        cells = vals[v]

## 2719. Count of Integers

https://leetcode.cn/problems/count-of-integers/

数位 DP。

```python
def count(self, num1, num2, min_sum, max_sum):
    MOD = 10**9 + 7
    def solve(s):
        from functools import lru_cache
        @lru_cache(None)
        def dp(pos, total, tight):
            if total > max_sum: return 0
            if pos == len(s):
                return 1 if min_sum <= total <= max_sum else 0
            limit = int(s[pos]) if tight else 9
            ans = 0
            for d in range(0, limit + 1):
                ans += dp(pos + 1, total + d, tight and d == limit)
            return ans % MOD
        return dp(0, 0, True)
    # subtract solve(num1-1)
    def sub1(s):
        s = list(s)
        i = len(s) - 1
        while i >= 0 and s[i] == '0': s[i] = '9'; i -= 1
        s[i] = str(int(s[i]) - 1)
        return ''.join(s).lstrip('0') or '0'
    return (solve(num2) - solve(sub1(num1))) % MOD
```

## 2732. Find a Good Subset of the Matrix

```python
def goodSubsetofBinaryMatrix(self, grid):
    seen = {}
    for i, row in enumerate(grid):
        mask = 0
        for j, v in enumerate(row):
            if v: mask |= (1 << j)
        if mask == 0: return [i]
        for prev_mask, idx in seen.items():
            if prev_mask & mask == 0: return sorted([idx, i])
        seen[mask] = i
    return []
```

```python
from functools import lru_cache
def minimizeConcatenatedLength(self, words):
    @lru_cache(None)
    def dp(i, first, last):
        if i == len(words): return 0
        w = words[i]; wf, wl = w[0], w[-1]; wn = len(w)
        # words[i] 接在后面
        opt1 = wn - (1 if last == wf else 0) + dp(i+1, first, wl)
        # words[i] 接在前面
        opt2 = wn - (1 if wl == first else 0) + dp(i+1, wf, last)
        return min(opt1, opt2)
    return len(words[0]) + dp(1, words[0][0], words[0][-1])
```

def countServers(self, n, logs, x, queries):
    logs.sort(key=lambda l: l[1])
    idx = sorted(range(len(queries)), key=lambda i: queries[i])

## 2800. Shortest String That Contains Three Strings

from itertools import permutations
def minimumString(self, a, b, c):
    def merge(s, t):
        if t in s: return s
        for i in range(min(len(s), len(t)), 0, -1):
            if s.endswith(t[:i]):
                return s + t[i:]
        return s + t
    res = None
    for p in permutations([a, b, c]):
        merged = merge(merge(p[0], p[1]), p[2])
        if res is None or len(merged) < len(res) or (len(merged) == len(res) and merged < res):
            res = merged
    return res

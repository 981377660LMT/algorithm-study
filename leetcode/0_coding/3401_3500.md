```python
def minimumOperations(self, grid):
    m, n = len(grid), len(grid[0]); ops = 0
    for j in range(n):
        for i in range(1, m):
            if grid[i][j] <= grid[i-1][j]:
                diff = grid[i-1][j] - grid[i][j] + 1
                grid[i][j] += diff; ops += diff
    return ops
```

## 3404. Count Special Subsequences

https://leetcode.cn/problems/count-special-subsequences/

```python
from collections import defaultdict
def numberOfSubsequences(self, nums):
    n = len(nums); res = 0
    # p * s == q * r where p=nums[a], q=nums[b], r=nums[c], s=nums[d]
    # p/q == r/s => 枚举 (b,c), 中间跳一个
    # 对每个 r(index c), 预计算右边 s 的频率，维护 p/q 的频率
    # 正确做法: 枚举中间两个 q=nums[b], r=nums[c], b+2<=c
    freq = defaultdict(int)
    for c in range(4, n):
        # 加入 b = c-2 的所有 a 的 p/q
        b = c - 2
        for a in range(b):
            from math import gcd
            g = gcd(nums[a], nums[b])
            freq[(nums[a]//g, nums[b]//g)] += 1
        # 枚举 d = c+1..n-1 不对，我们固定 c 然后枚举 d
        pass
    # 重新思考：固定 b, enumrate a on left, enumerate (c,d) on right
    # p/q = r/s => 用分数做 key
    from math import gcd
    res = 0
    for b in range(1, n-2):
        # 左边: a < b, 分数 nums[a]/nums[b]
        # 右边: c > b+1, d > c, 分数 nums[c]/nums[d]
        # 预计算右边
        right = defaultdict(int)
        for c in range(b+2, n):
            for d in range(c+1, n):
                g = gcd(nums[c], nums[d])
                right[(nums[c]//g, nums[d]//g)] += 1
        for a in range(b):
            g = gcd(nums[a], nums[b])
            res += right[(nums[a]//g, nums[b]//g)]
    return res
```

```python
    # 枚举 c(即 r 的位置), 滚动维护左侧 (a,b) 的分数频率
```

## 3408. Design Task Manager

https://leetcode.cn/problems/design-task-manager/

```python
from sortedcontainers import SortedList
class TaskManager:
    def __init__(self, tasks):
        self.task_map = {}  # taskId -> (userId, priority)
        self.sl = SortedList()
        for uid, tid, pri in tasks:
            self.task_map[tid] = (uid, pri)
            self.sl.add((pri, tid))
    def add(self, userId, taskId, priority):
        self.task_map[taskId] = (userId, priority)
        self.sl.add((priority, taskId))
    def edit(self, taskId, newPriority):
        uid, old_pri = self.task_map[taskId]
        self.sl.remove((old_pri, taskId))
        self.task_map[taskId] = (uid, newPriority)
        self.sl.add((newPriority, taskId))
    def rmv(self, taskId):
        uid, pri = self.task_map.pop(taskId)
        self.sl.remove((pri, taskId))
    def execTop(self):
        if not self.sl: return -1
        pri, tid = self.sl.pop()
        uid, _ = self.task_map.pop(tid)
        return uid
```

## 3409. Longest Subsequence With Decreasing Adjacent Difference

https://leetcode.cn/problems/longest-subsequence-with-decreasing-adjacent-difference/

```python
def longestSubsequence(self, nums):
    # dp[val][diff] = 以值 val、相邻差 diff 结尾的最长子序列
    MAX_VAL = 301
    dp = [[0]*(MAX_VAL) for _ in range(MAX_VAL)]
    res = 1
    for x in nums:
        for d in range(MAX_VAL):
            for prev in [x - d, x + d]:
                if 0 <= prev < MAX_VAL and dp[prev][d] > 0:
                    dp[x][d] = max(dp[x][d], dp[prev][d] + 1)
            dp[x][d] = max(dp[x][d], 1)
            res = max(res, dp[x][d])
        # suffix max: diff 可以递减, 所以从大 diff 传播到小 diff
        for d in range(MAX_VAL - 2, -1, -1):
            dp[x][d] = max(dp[x][d], dp[x][d+1])
    return res
```

## 3413. Maximum Coins From K Consecutive Bags

https://leetcode.cn/problems/maximum-coins-from-k-consecutive-bags/

```python
def maximumCoins(self, coins, k):
    coins.sort()
    n = len(coins); res = 0; cur = 0; j = 0
    # 滑动窗口: 窗口 [start, start+k-1]
    # 枚举窗口左端对齐某个区间的左端
    events = []
    for l, r, c in coins: events.append((l, r, c))

    # 法1: 枚举左端为每个区间起点
    prefix_coins = [0] * (n + 1)
    for i in range(n):
        prefix_coins[i+1] = prefix_coins[i] + coins[i][2] * (coins[i][1] - coins[i][0] + 1)

    import bisect
    starts = [c[0] for c in coins]
    ends = [c[1] for c in coins]

    for i in range(n):
        L = coins[i][0]; R = L + k - 1
        # 完全包含的区间 [i, j), 部分包含 j
        j = bisect.bisect_right(ends, R) - 1
        if j >= i:
            full = prefix_coins[j+1] - prefix_coins[i]
        else:
            j = i - 1; full = 0
        # 部分区间 j+1
        if j + 1 < n and coins[j+1][0] <= R:
            partial = (R - coins[j+1][0] + 1) * coins[j+1][2]
            full += partial
        res = max(res, full)

    # 枚举右端为每个区间终点
    for i in range(n):
        R = coins[i][1]; L = R - k + 1
        j = bisect.bisect_left(starts, L)
        if j <= i:
            full = prefix_coins[i+1] - prefix_coins[j]
        else:
            j = i + 1; full = 0
        if j - 1 >= 0 and coins[j-1][1] >= L:
            partial = (coins[j-1][1] - L + 1) * coins[j-1][2]
            full += partial
        res = max(res, full)
    return res
```

```python
def maximumAmount(self, coins):
    m, n = len(coins), len(coins[0])
    INF = float('-inf')
    # dp[i][j][k] = 到(i,j)用了k次neutralize的最大值
    dp = [[[INF]*3 for _ in range(n)] for _ in range(m)]
    dp[0][0][0] = coins[0][0]
    if coins[0][0] < 0: dp[0][0][1] = 0
    for i in range(m):
        for j in range(n):
            for k in range(3):
                if dp[i][j][k] == INF: continue
                for ni, nj in [(i+1,j),(i,j+1)]:
                    if ni < m and nj < n:
                        dp[ni][nj][k] = max(dp[ni][nj][k], dp[i][j][k] + coins[ni][nj])
                        if k < 2 and coins[ni][nj] < 0:
                            dp[ni][nj][k+1] = max(dp[ni][nj][k+1], dp[i][j][k])
    return max(dp[m-1][n-1])
```

## 3419. Minimize the Maximum Edge Weight of Graph

https://leetcode.cn/problems/minimize-the-maximum-edge-weight-of-graph/

```python
import heapq
def minMaxWeight(self, n, edges, threshold):
    # 反向图: 从 0 到所有节点 => 所有节点到 0 的反向
    adj = [[] for _ in range(n)]
    for u, v, w in edges:
        adj[v].append((u, w))  # 反向
    # 最小化最大边权使得所有节点可达 0
    # Prim-like: 最小瓶颈生成树
    dist = [float('inf')] * n; dist[0] = 0
    heap = [(0, 0)]; visited = set()
    while heap:
        w, u = heapq.heappop(heap)
        if u in visited: continue
        visited.add(u)
        for v, wt in adj[u]:
            if wt < dist[v]:
                dist[v] = wt; heapq.heappush(heap, (wt, v))
    if len(visited) < n: return -1
    return max(dist)
```

```python
def minCost(self, arr, brr, k):
    # 不重排: 直接逐个差的绝对值
    no_sort = sum(abs(a-b) for a,b in zip(arr, brr))
    # 重排: 排序后逐个配对 + k
    sa = sorted(arr); sb = sorted(brr)
    with_sort = k + sum(abs(a-b) for a,b in zip(sa, sb))
    return min(no_sort, with_sort)
```

## 3434. Maximum Frequency After Subarray Operation

https://leetcode.cn/problems/maximum-frequency-after-subarray-operation/

```python
from collections import Counter
def maxFrequency(self, nums, k):
    base = nums.count(k); res = base
    # 选一个子数组 [l,r], 加 offset 使某个值 v 变成 k
    # 对每个 v != k, v + offset = k => offset = k - v
    # 在子数组中: v 变成 k (+1), 原本是 k 变成 k + offset != k (-1)
    for v in set(nums):
        if v == k: continue
        # Kadane: 对每个位置 +1 if nums[i]==v, -1 if nums[i]==k, else 0
        cur = 0; best = 0
        for x in nums:
            d = 0
            if x == v: d = 1
            elif x == k: d = -1
            cur = max(0, cur + d)
            best = max(best, cur)
        res = max(res, base + best)
    return res
```

```python
def maxFreeTime(self, eventTime, k, startTime, endTime):
    n = len(startTime)
```

## 3440. Reschedule Meetings for Maximum Free Time II

https://leetcode.cn/problems/reschedule-meetings-for-maximum-free-time-ii/

```python
def maxFreeTime(self, eventTime, startTime, endTime):
    n = len(startTime)
    gaps = []
    gaps.append(startTime[0])
    for i in range(1, n): gaps.append(startTime[i] - endTime[i-1])
    gaps.append(eventTime - endTime[-1])
    # 移除一个会议: gap[i] + duration[i] + gap[i+1] 合并
    # 然后把这个会议插入到某个足够大的 gap 中
    durations = [endTime[i] - startTime[i] for i in range(n)]
    merged = [gaps[i] + durations[i] + gaps[i+1] for i in range(n)]
    # 如果有别的 gap >= duration[i], 就多出 merged[i] 的空闲
    # 否则多出 merged[i] - duration[i] + max_other_gap
    # 预计算: 最大 gap, 次大 gap
    sorted_gaps = sorted(range(len(gaps)), key=lambda x: -gaps[x])
    res = 0
    for i in range(n):
        # 移除会议 i, 涉及 gaps[i] 和 gaps[i+1]
        free = merged[i]
        d = durations[i]
        # 找一个不是 gaps[i] 也不是 gaps[i+1] 的最大 gap
        best_gap = 0
        for idx in sorted_gaps:
            if idx != i and idx != i + 1:
                best_gap = gaps[idx]; break
        if best_gap >= d:
            res = max(res, free)
        else:
            res = max(res, free - d + best_gap)
    return res
```

```python
def maxDistance(self, s, k):
    # 分4个方向: NE(+1,+1), NW(-1,+1), SE(+1,-1), SW(-1,-1)
    # 对每个方向, 最大化 prefix 距离, 最多改 k 步
    dirs = {'N': (0,1), 'S': (0,-1), 'E': (1,0), 'W': (-1,0)}
    res = 0
    for tx, ty in [(1,1),(1,-1),(-1,1),(-1,-1)]:
        x = y = 0; changes = 0; res_dir = 0
        l = 0
        # 不是滑动窗口, 是前缀
        # 每步走一格, 最多改 k 步使方向为 (tx,ty)
        cur = 0; fix = 0
        for c in s:
            dx, dy = dirs[c]
            step = dx * tx + dy * ty  # +2 if aligned, 0 if perpendicular, -2 if opposite
            if step == 2:
                cur += 1
            elif step == -2:
                if fix < k: fix += 1; cur += 1
                else: cur -= 1
            # perpendicular: 0 => 改也只增1
            else:
                if fix < k: fix += 1; cur += 1
                # else: cur unchanged, actually -0
                # 实际不改的话这步贡献为 0
            cur = max(cur, 0)
            res_dir = max(res_dir, cur)
        res = max(res, res_dir)
```

```python
            d += 1
        res.append(best)
```

```python
def lenOfVDiag(self, grid):
    m, n = len(grid), len(grid[0]); res = 0
    dirs = [(1,1),(1,-1),(-1,1),(-1,-1)]
    for i in range(m):
        for j in range(n):
            if grid[i][j] != 1: continue
            for dx, dy in dirs:
                # 沿方向走 1->2->0->1->2->0... 然后可以转弯
                # V 形: 先沿一个方向, 然后转90度
                for dx2, dy2 in dirs:
                    if (dx2, dy2) == (dx, dy) or (dx2, dy2) == (-dx, -dy): continue
                    # 沿 (dx,dy) 走, 然后沿 (dx2,dy2) 走
                    seq = [1, 2, 0]; L1 = 0
                    x, y = i, j
                    while 0<=x<m and 0<=y<n and grid[x][y] == seq[L1 % 3]:
                        L1 += 1; x += dx; y += dy
                    # 转弯点: (x-dx, y-dy), 沿 (dx2, dy2)
                    tx, ty = x - dx, y - dy
                    L2 = 0; x, y = tx + dx2, ty + dy2
                    while 0<=x<m and 0<=y<n and grid[x][y] == seq[(L1 + L2) % 3]:
                        L2 += 1; x += dx2; y += dy2
                    res = max(res, L1 + L2)
    return res
```

```python
from functools import lru_cache
def minCost(self, nums):
    n = len(nums)
    if n <= 2: return max(nums)
    # 每次取3个中去掉两个(代价=较小的那个或max), 保留1个
    # dp(i, last): 当前处理到 i, 上一轮保留的元素是 last
    @lru_cache(None)
    def dp(i, last):
        if i + 1 == n:
            return max(nums[last], nums[i])
        if i == n:
            return nums[last]
        # 取 nums[i] 和 nums[i+1], 与 last 配对
        a, b, c = nums[last], nums[i], nums[i+1]
        # 去掉两个, 保留一个
        cost1 = max(b, c) + dp(i+2, last)  # 去掉 i, i+1, 保留 last
        cost2 = max(a, c) + dp(i+2, i)     # 去掉 last, i+1, 保留 i
        cost3 = max(a, b) + dp(i+2, i+1)   # 去掉 last, i, 保留 i+1
        return min(cost1, cost2, cost3)
    return dp(1, 0)
```

## 3473. Sum of K Subarrays With Length at Least M

https://leetcode.cn/problems/sum-of-k-subarrays-with-length-at-least-m/

```python
def maxSum(self, nums, k, m):
    n = len(nums)
    # dp[j] = 选了 j 个不重叠子数组(每个长度>=m)的最大和, 考虑前 i 个元素
    # best[i][j] = max dp value considering nums[:i] with j subarrays
    # prefix sum
    pre = [0]*(n+1)
    for i in range(n): pre[i+1] = pre[i]+nums[i]
    INF = float('-inf')
    # dp[i][j] = 前i个元素选j个子数组的最大和
    # dp[i][j] = max(dp[i-1][j], max over l<=i-m+1 of dp[l-1][j-1] + pre[i]-pre[l-1])
    # = max(dp[i-1][j], pre[i] + max(dp[l-1][j-1] - pre[l-1]) for l<=i-m+1)
    # 维护 mx[j] = max(dp[l-1][j-1] - pre[l-1]) for l up to current
    dp = [[INF]*(k+1) for _ in range(n+1)]
    for i in range(n+1): dp[i][0] = 0
    mx = [INF]*(k+1)
    for i in range(1, n+1):
        for j in range(k+1):
            dp[i][j] = dp[i-1][j]
        if i >= m:
            for j in range(1, k+1):
                if dp[i-m][j-1] > INF:
                    mx[j] = max(mx[j], dp[i-m][j-1] - pre[i-m])
                if mx[j] > INF:
                    dp[i][j] = max(dp[i][j], pre[i] + mx[j])
    return dp[n][k]
```

```python
def numOfUnplacedFruits(self, fruits, baskets):
    # 线段树: 维护区间最大值, 对每个 fruit 找第一个 >= fruit 的 basket
    n = len(baskets)
    tree = [0] * (4 * n)
    def build(node, l, r):
        if l == r: tree[node] = baskets[l]; return
        mid = (l+r)//2
        build(2*node, l, mid); build(2*node+1, mid+1, r)
        tree[node] = max(tree[2*node], tree[2*node+1])
    def query(node, l, r, val):
        if tree[node] < val: return -1
        if l == r: return l
        mid = (l+r)//2
        res = query(2*node, l, mid, val)
        if res != -1: return res
        return query(2*node+1, mid+1, r, val)
    def update(node, l, r, idx, val):
        if l == r: tree[node] = val; return
        mid = (l+r)//2
        if idx <= mid: update(2*node, l, mid, idx, val)
        else: update(2*node+1, mid+1, r, idx, val)
        tree[node] = max(tree[2*node], tree[2*node+1])
    build(1, 0, n-1)
    cnt = 0
    for f in fruits:
        idx = query(1, 0, n-1, f)
        if idx == -1: cnt += 1
        else: update(1, 0, n-1, idx, 0)
    return cnt
```

```python
import bisect
def solveQueries(self, nums, queries):
    n = len(nums)
    pos = defaultdict(list)

    return res
```

https://leetcode.cn/problems/zero-array-transformation-iv/

```python
def minZeroArray(self, nums, queries):
    n = len(nums)
    def check(k):
        # 对每个位置 i, 从前 k 个 query 中选子集使得总 val >= nums[i]
        for i in range(n):
            target = nums[i]
            if target == 0: continue
            # 收集覆盖位置 i 的 query 的 val
            vals = []
            for l, r, v in queries[:k]:
                if l <= i <= r: vals.append(v)
            # 子集和 >= target? (0-1 knapsack 可行性)
            dp = {0}
            for v in vals:
                dp = dp | {s + v for s in dp}
                if any(s >= target for s in dp): break
            if not any(s >= target for s in dp): return False
        return True
    if check(len(queries)) == False: return -1
    lo, hi = 0, len(queries)
    while lo < hi:
        mid = (lo+hi)//2
        if check(mid): hi = mid
        else: lo = mid + 1
    return lo
```

```python
def numberOfComponents(self, properties, k):
    n = len(properties)
    sets = [set(p) for p in properties]
    parent = list(range(n))
    def find(x):
        while parent[x] != x: parent[x] = parent[parent[x]]; x = parent[x]
        return x
    def union(a, b): parent[find(a)] = find(b)
    for i in range(n):
        for j in range(i+1, n):
            if len(sets[i] & sets[j]) >= k: union(i, j)
    return len(set(find(i) for i in range(n)))
```

## 3494. Find the Minimum Amount of Time to Brew Potions

https://leetcode.cn/problems/find-the-minimum-amount-of-time-to-brew-potions/

```python
def minTime(self, skill, mana):
    n, m = len(skill), len(mana)
    # 每个巫师顺序处理每瓶药水
    # 巫师 i 处理药水 j 需要 skill[i] * mana[j] 时间
    # 一个巫师同时只能处理一瓶, 且药水必须按顺序经过巫师
    start = [0] * m  # 每瓶药水开始时间
    for j in range(m):
        if j > 0:
            # 第 j 瓶必须在前一瓶之后开始足够晚
            # 使得对每个巫师 i, start[j] + sum(skill[0..i]*mana[j]) >= start[j-1] + sum(skill[0..i]*mana[j-1]) + skill[i]*mana[j-1]
            pass
    # 模拟: 每个巫师的完成时间
    finish = [0] * n  # 巫师 i 上次完成时间
    total = 0
    for j in range(m):
        for i in range(n):
            start_time = finish[i]
            if i > 0: start_time = max(start_time, finish[i-1])
            # 等前一个巫师完成本药水
            # 其实 finish[i-1] 已经被更新为本药水在巫师 i-1 的完成时间
            finish[i] = start_time + skill[i] * mana[j]
        total = finish[-1]
    return total
```

## 3499. Maximize Active Section with Trade I

https://leetcode.cn/problems/maximize-active-section-with-trade-i/

```python
def maxActiveSectionsAfterTrade(self, s):
    n = len(s); ones = s.count('1')
    # 找两段连续0, 中间隔着1, 合并后增益最大
    # 枚举所有 0-段
    segs = []
    i = 0
    while i < n:
        if s[i] == '0':
            j = i
            while j < n and s[j] == '0': j += 1
            segs.append((i, j-1)); i = j
        else:
            i += 1
    if len(segs) <= 1: return ones + (segs[0][1]-segs[0][0]+1 if segs else 0)
    # 合并相邻两段0(中间的1也变成1...不对, trade 是翻转一段)
    # 选一段 [l,r] 翻转: 0->1, 1->0. 我们想要增益最大.
    # 最优: 选包含两个 0-段和中间 1-段的区间翻转
    # 增益 = 两段0长度之和 - 中间1段长度
    best = 0
    for i in range(len(segs)-1):
        gain = (segs[i][1]-segs[i][0]+1) + (segs[i+1][1]-segs[i+1][0]+1)
        best = max(best, gain)
    return ones + best
```

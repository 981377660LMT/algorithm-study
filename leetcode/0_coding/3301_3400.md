```python
def validSequence(self, word1, word2):
    n, m = len(word1), len(word2)
    # suffix[i] = word2 能从 word1[i:] 后缀匹配到的位置
    suffix = [m] * (n + 1)
    j = m - 1
    for i in range(n - 1, -1, -1):
        suffix[i] = suffix[i + 1]
        if j >= 0 and word1[i] == word2[j]:
            suffix[i] = j; j -= 1
    # 贪心匹配, 最多跳过一个字符
    res = []; j = 0; used = False
    for i in range(n):
        if j == m: break
        if word1[i] == word2[j]:
            res.append(i); j += 1
        elif not used and j + 1 <= m and suffix[i + 1] <= j + 1:
            res.append(i); j += 1; used = True
    return res if len(res) == m else []
```

```python
def kthCharacter(self, k, operations):
    # 长度翻倍: 第 i 次操作后长度 = 2^(i+1)
    # 逆推: 从 k 反推到初始 'a'
    k -= 1  # 0-indexed
    n = len(operations)
    shift = 0
    for i in range(n - 1, -1, -1):
        half = 1 << i
        if k >= half:
            k -= half
            shift += operations[i]
    return chr((shift % 26) + 97)
```

```python
from collections import deque
def remainingMethods(self, n, k, invocations):
    adj = [[] for _ in range(n)]
    radj = [[] for _ in range(n)]
    for u, v in invocations:
        adj[u].append(v); radj[v].append(u)
    # BFS from k 找所有可疑方法
    sus = set(); q = deque([k]); sus.add(k)
    while q:
        u = q.popleft()
        for v in adj[u]:
            if v not in sus: sus.add(v); q.append(v)
    # 检查是否有外部调用进入可疑
    for u, v in invocations:
        if u not in sus and v in sus:
            return list(range(n))
    return sorted(i for i in range(n) if i not in sus)
```

```python
    res = []

        if x == 2: res.append(-1); continue
```

```python
def maxRemovals(self, source, pattern, targetIndices):
    ti = set(targetIndices)
    n, m = len(source), len(pattern)
    # dp[j] = 匹配 pattern[:j] 的最大移除数
    dp = [-1] * (m + 1); dp[0] = 0
    for i in range(n):
        bonus = 1 if i in ti else 0
        for j in range(min(m, i+1), 0, -1):
            if source[i] == pattern[j-1] and dp[j-1] >= 0:
                dp[j] = max(dp[j], dp[j-1])
            if dp[j] >= 0: dp[j] += bonus
        dp[0] += bonus
    return dp[m]
```

```python
from sortedcontainers import SortedList
from collections import Counter
def findXSum(self, nums, k, x):

    # 维护 top-x 频率元素
    res = []
    for r in range(len(nums)):
        old_key = (cnt[nums[r]], nums[r])
        if old_key[0] > 0: sl.remove(old_key)
        cnt[nums[r]] += 1
        sl.add((cnt[nums[r]], nums[r]))
        if r >= k:
            old = nums[r - k]
            old_key = (cnt[old], old)

            total = 0; take = min(x, len(sl))

                total += f * v
```

```python
from functools import lru_cache
def countWinningSequences(self, s):
    MOD = 10**9 + 7
    n = len(s)
    # Bob 选 F/W/E, 不能连续两次相同
    # 得分差: Fire>Earth, Earth>Water, Water>Fire
    mp = {'F': 0, 'W': 1, 'E': 2}
    @lru_cache(None)
    def dp(i, diff, last):
        if diff > n - i: return 1  # 已经赢定
        if diff < -(n - i): return 0
        if i == n: return 1 if diff > 0 else 0
        a = mp[s[i]]; res = 0
        for b in range(3):
            if b == last: continue
            # b beats a?
            if (b == 0 and a == 2) or (b == 1 and a == 0) or (b == 2 and a == 1):
                d = 1
            elif b == a:
                d = 0
            else:
                d = -1
            res = (res + dp(i + 1, diff + d, b)) % MOD
        return res
    return dp(0, 0, -1)
```

```python
def findSubtreeSizes(self, parent, s):
    n = len(parent)
    adj = [[] for _ in range(n)]
    for i in range(1, n): adj[parent[i]].append(i)
    # 每个节点找最近祖先同字符
    new_parent = parent[:]
    def dfs(u, ancestors):
        c = s[u]
        if c in ancestors and ancestors[c]:
            new_parent[u] = ancestors[c][-1]
        ancestors.setdefault(c, []).append(u)
        for v in adj[u]:
            dfs(v, ancestors)
        ancestors[c].pop()
    dfs(0, {})
    # 计算子树大小
    new_adj = [[] for _ in range(n)]
    for i in range(1, n): new_adj[new_parent[i]].append(i)
    size = [1] * n
    def dfs2(u):
        for v in new_adj[u]:
            dfs2(v); size[u] += size[v]
    dfs2(0)
    return size
```

```python
    dp = [stayScore[0][j] if k > 0 else 0 for j in range(n)]
```

```python
from math import gcd
from functools import reduce
def maxScore(self, nums):
    if not nums: return 0
    n = len(nums)
    def lcm(a, b): return a * b // gcd(a, b)
    def score(arr):
        if not arr: return 0
        g = reduce(gcd, arr); l = reduce(lcm, arr)
        return g * l
    res = score(nums)
    for i in range(n):
        sub = nums[:i] + nums[i+1:]
        if sub: res = max(res, score(sub))
    return res
```

## 3341. Find Minimum Time to Reach Last Room I / 3342. Find Minimum Time to Reach Last Room II

https://leetcode.cn/problems/find-minimum-time-to-reach-last-room-ii/

Dijkstra，移动代价交替 1 和 2。

```python
import heapq
def minTimeToReach(self, moveTime):
    m, n = len(moveTime), len(moveTime[0])
    dist = [[float('inf')]*n for _ in range(m)]
    dist[0][0] = 0
    heap = [(0, 0, 0, 0)]  # time, row, col, step_parity
    while heap:
        t, r, c, p = heapq.heappop(heap)
        if r == m-1 and c == n-1: return t
        if t > dist[r][c]: continue
        cost = 1 if p % 2 == 0 else 2
        for dr, dc in [(0,1),(0,-1),(1,0),(-1,0)]:
            nr, nc = r+dr, c+dc
            if 0<=nr<m and 0<=nc<n:
                nt = max(t, moveTime[nr][nc]) + cost
                if nt < dist[nr][nc]:
                    dist[nr][nc] = nt
                    heapq.heappush(heap, (nt, nr, nc, p+1))
    return dist[m-1][n-1]
```

```python
def sumOfGoodSubsequences(self, nums):
    MOD = 10**9 + 7
    cnt = {}  # val -> (count of subsequences ending with val, their sum)
    total_sum = 0

        c = 1; s = x  # 单独 x
```

```python
from functools import lru_cache
def minArraySum(self, nums, k, op1, op2):
    n = len(nums)
    @lru_cache(None)
    def dp(i, o1, o2):
        if i == n: return 0
        x = nums[i]; res = x + dp(i+1, o1, o2)
        if o1 > 0:
            y = (x + 1) // 2
            res = min(res, y + dp(i+1, o1-1, o2))
            if o2 > 0 and y >= k:
                res = min(res, y - k + dp(i+1, o1-1, o2-1))
        if o2 > 0 and x >= k:
            y = x - k
            res = min(res, y + dp(i+1, o1, o2-1))
            if o1 > 0:
                res = min(res, (y+1)//2 + dp(i+1, o1-1, o2-1))
        return res
    return dp(0, op1, op2)
```

## 3367. Maximize Sum of Weights after Edge Removals

https://leetcode.cn/problems/maximize-sum-of-weights-after-edge-removals/

树 DP。每个节点最多保留 k 条边。

```python
def maximizeSumOfWeights(self, edges, k):
    from collections import defaultdict
    adj = defaultdict(list)
    for u, v, w in edges:
        adj[u].append((v, w)); adj[v].append((u, w))
    def dfs(u, par):
        gains = []  # 保留这条边的额外增益
        base = 0
        for v, w in adj[u]:
            if v == par: continue
            keep, skip = dfs(v, u)
            base += skip
            gains.append(keep + w - skip)
        gains.sort(reverse=True)
        # 至多选 k 条(但作为子节点, 给父边留一个名额)
        total_skip = base + sum(max(0, g) for g in gains[:k])
        total_keep = base + sum(max(0, g) for g in gains[:k-1])
        return total_keep, total_skip
    _, res = dfs(0, -1)
    return res
```

```python
    res = -1

            x1,y1 = points[i]; x2,y2 = points[j]
```

```python
def beautifulSplits(self, nums):
    n = len(nums); res = 0
    # LCP[i][j] = longest common prefix of nums[i:] and nums[j:]
    lcp = [[0]*(n+1) for _ in range(n+1)]
    for i in range(n-1, -1, -1):
        for j in range(n-1, -1, -1):
            if nums[i]==nums[j]: lcp[i][j]=lcp[i+1][j+1]+1
    for i in range(1, n):
        for j in range(i+1, n):
            # nums1=[:i], nums2=[i:j], nums3=[j:]
            # 条件: nums1 是 nums2 的前缀 或 nums2 是 nums3 的前缀
            if (i <= j-i and lcp[0][i] >= i) or (j-i <= n-j and lcp[i][j] >= j-i):
```

```python
def countPathsWithXorValue(self, grid, k):
    MOD = 10**9 + 7
    m, n = len(grid), len(grid[0])
    dp = [[{} for _ in range(n)] for _ in range(m)]
    dp[0][0][grid[0][0]] = 1
    for i in range(m):
        for j in range(n):
            for xor_val, cnt in dp[i][j].items():
                for ni, nj in [(i+1,j),(i,j+1)]:
                    if ni < m and nj < n:
                        nv = xor_val ^ grid[ni][nj]
                        dp[ni][nj][nv] = (dp[ni][nj].get(nv, 0) + cnt) % MOD
    return dp[m-1][n-1].get(k, 0)
```

## 3398. Smallest Substring With Identical Characters I

```python
def minLength(self, s, numOps):
    n = len(s)
    def check(L):
        if L == 1:
            # 需要变成 010101 或 101010
            return min(sum(int(s[i]) != i%2 for i in range(n)),
                       sum(int(s[i]) != (i+1)%2 for i in range(n))) <= numOps
        # 每段连续相同字符长度 > L 的需要翻转
        ops = 0; i = 0
        while i < n:
            j = i
            while j < n and s[j] == s[i]: j += 1
            run = j - i
            ops += run // (L + 1)
            i = j
        return ops <= numOps
    for L in range(1, n + 1):
        if check(L): return L
    return n
```

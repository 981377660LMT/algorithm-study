```python
    stack = []
    for d in num:
        while k and stack and stack[-1] > d:
```

```python
def canCross(self, stones):
    dp = {s: set() for s in stones}
    dp[0].add(0)
    for s in stones:
        for k in dp[s]:
            for nk in (k-1, k, k+1):
                if nk > 0 and s + nk in dp:
```

## 405. Number to Hexadecimal

每次取低 4 位，负数用补码。

```python
def toHex(self, num):
    if num == 0: return '0'
    if num < 0: num += 2**32
    res = []
    while num:
        res.append('0123456789abcdef'[num & 0xf])
        num >>= 4
    return ''.join(res[::-1])
```

## 410. Split Array Largest Sum

```python
def splitArray(self, nums, k):
    def can_split(max_sum):
        count, cur = 1, 0
        for x in nums:
            cur += x
            if cur > max_sum:
                count += 1; cur = x
        return count <= k
    lo, hi = max(nums), sum(nums)
    while lo < hi:
        mid = (lo + hi) // 2
        if can_split(mid): hi = mid
        else: lo = mid + 1
    return lo
```

```python
def canPartition(self, nums):
    total = sum(nums)
    if total % 2: return False
    target = total // 2
    dp = [False] * (target + 1)
    dp[0] = True
    for x in nums:
        for j in range(target, x - 1, -1):
            dp[j] = dp[j] or dp[j - x]
```

## 417. Pacific Atlantic Water Flow
https://leetcode.com/problems/pacific-atlantic-water-flow/

从两个洋的边界反向 BFS/DFS，取交集。

```python
def pacificAtlantic(self, heights):
    if not heights: return []
    m, n = len(heights), len(heights[0])
    def bfs(starts):
        visited = set(starts)
        q = list(starts)
        while q:
            nq = []
            for i, j in q:
                for di, dj in [(-1,0),(1,0),(0,-1),(0,1)]:
                    ni, nj = i+di, j+dj
                    if 0 <= ni < m and 0 <= nj < n and (ni,nj) not in visited \
                       and heights[ni][nj] >= heights[i][j]:
                        visited.add((ni, nj))
                        nq.append((ni, nj))
            q = nq
        return visited
    pac = bfs([(i, 0) for i in range(m)] + [(0, j) for j in range(n)])
    atl = bfs([(i, n-1) for i in range(m)] + [(m-1, j) for j in range(n)])
    return list(pac & atl)
```

## 420. Strong Password Checker
https://leetcode.com/problems/strong-password-checker/

贪心：处理缺少字符类型、长度不足/过长、连续重复字符。

```python
def strongPasswordChecker(self, password):
    n = len(password)
    missing = int(not any(c.islower() for c in password)) + \
              int(not any(c.isupper() for c in password)) + \
              int(not any(c.isdigit() for c in password))
    replace = one = two = 0
    i = 2
    while i < n:
        if password[i] == password[i-1] == password[i-2]:
            length = 2
            while i < n and password[i] == password[i-1]:
                length += 1; i += 1
            replace += length // 3
            if length % 3 == 0: one += 1
            elif length % 3 == 1: two += 1
        else: i += 1
    if n < 6: return max(missing, 6 - n)
    if n <= 20: return max(missing, replace)
    delete = n - 20
    replace -= min(delete, one)
    replace -= min(max(delete - one, 0), two * 2) // 2
    replace -= max(delete - one - 2 * two, 0) // 3
    return delete + max(missing, replace)
```

```python
        mask |= 1 << i
        prefixes = {x & mask for x in nums}
```

```python
from collections import Counter
def characterReplacement(self, s, k):
    cnt = Counter()
    left = max_freq = res = 0
    for right, c in enumerate(s):
        cnt[c] += 1
        max_freq = max(max_freq, cnt[c])
        while right - left + 1 - max_freq > k:
            cnt[s[left]] -= 1
            left += 1
        res = max(res, right - left + 1)
    return res
```

```python
from collections import deque
def levelOrder(self, root):
    if not root: return []
    res, q = [], deque([root])
    while q:
        level = []

            node = q.popleft()

            q.extend(node.children)
        res.append(level)
```

```python
def flatten(self, head):
    cur = head
    while cur:
        if cur.child:
            nxt = cur.next
            child_tail = cur.child
            while child_tail.next:
                child_tail = child_tail.next
            cur.next = cur.child
            cur.child.prev = cur
            cur.child = None
            child_tail.next = nxt
            if nxt: nxt.prev = child_tail
        cur = cur.next
    return head
```

## 432. All O`one Data Structure
https://leetcode.com/problems/all-oone-data-structure/

哈希表 + 按频率分桶。

```python
from collections import defaultdict
class AllOne:
    def __init__(self):
        self.key_cnt = {}
        self.cnt_keys = defaultdict(set)
    def inc(self, key):
        c = self.key_cnt.get(key, 0)
        self.key_cnt[key] = c + 1
        self.cnt_keys[c + 1].add(key)
        if c: self.cnt_keys[c].discard(key)
        if c and not self.cnt_keys[c]: del self.cnt_keys[c]
    def dec(self, key):
        c = self.key_cnt[key]
        self.cnt_keys[c].discard(key)
        if not self.cnt_keys[c]: del self.cnt_keys[c]
        if c == 1: del self.key_cnt[key]
        else: self.key_cnt[key] = c - 1; self.cnt_keys[c - 1].add(key)
    def getMaxKey(self):
        return next(iter(self.cnt_keys[max(self.cnt_keys)])) if self.cnt_keys else ''
    def getMinKey(self):
        return next(iter(self.cnt_keys[min(self.cnt_keys)])) if self.cnt_keys else ''
```

```python
from collections import deque
def minMutation(self, startGene, endGene, bank):
    bank = set(bank)
    if endGene not in bank: return -1
    q = deque([(startGene, 0)])
    visited = {startGene}
    while q:
        gene, steps = q.popleft()
        if gene == endGene: return steps
        for i in range(8):
            for c in 'ACGT':
                nxt = gene[:i] + c + gene[i+1:]
                if nxt in bank and nxt not in visited:
                    visited.add(nxt)
                    q.append((nxt, steps + 1))
    return -1
```

```python
def pathSum(self, root, targetSum):
    self.res = 0
    prefix = defaultdict(int)
    prefix[0] = 1
    def dfs(node, cur):
        if not node: return
        cur += node.val
```

```python
from collections import Counter
def findAnagrams(self, s, p):
    pn, sn = len(p), len(s)
    if pn > sn: return []
    pc = Counter(p)
    sc = Counter(s[:pn])
    res = []
    if sc == pc: res.append(0)
    for i in range(pn, sn):
        sc[s[i]] += 1
        sc[s[i - pn]] -= 1
        if sc[s[i - pn]] == 0: del sc[s[i - pn]]
        if sc == pc: res.append(i - pn + 1)
    return res
```

```python
        cur, nxt, cnt = prefix, prefix + 1, 0

            cnt += min(n + 1, nxt) - cur
            cur *= 10; nxt *= 10
```

```python
    if key < root.val:
        root.left = self.deleteNode(root.left, key)
```

```python
def find132pattern(self, nums):
    stack = []
    s2 = float('-inf')  # 132 中的 '2'（中间值）
    for x in reversed(nums):
```

## 457. Circular Array Loop
https://leetcode.com/problems/circular-array-loop/

快慢指针检测环，要求同方向且环长 > 1。

```python
def circularArrayLoop(self, nums):
    n = len(nums)
    def nxt(i): return (i + nums[i]) % n
    for i in range(n):
        if nums[i] == 0: continue
        slow, fast = i, nxt(i)
        while nums[fast] * nums[i] > 0 and nums[nxt(fast)] * nums[i] > 0:
            if slow == fast:
                if slow == nxt(slow): break
                return True
            slow = nxt(slow)
            fast = nxt(nxt(fast))
        j = i
        while nums[j] * nums[i] > 0:
            tmp = nxt(j); nums[j] = 0; j = tmp
    return False
```

```python
        self.key_freq = {}
        self.freq_keys = defaultdict(OrderedDict)
    def _update(self, key):
        freq = self.key_freq[key]
        del self.freq_keys[freq][key]
        if not self.freq_keys[freq] and freq == self.min_freq:
            self.min_freq += 1
        self.key_freq[key] = freq + 1
        self.freq_keys[freq + 1][key] = None
    def get(self, key):
        if key not in self.key_val: return -1
        self._update(key)
        return self.key_val[key]
    def put(self, key, value):
        if self.cap <= 0: return
        if key in self.key_val:
            self.key_val[key] = value
            self._update(key)
            return
        if len(self.key_val) >= self.cap:
            evict, _ = self.freq_keys[self.min_freq].popitem(last=False)
            del self.key_val[evict]; del self.key_freq[evict]
        self.key_val[key] = value
        self.key_freq[key] = 1
        self.freq_keys[1][key] = None
        self.min_freq = 1
```

## 464. Can I Win
https://leetcode.com/problems/can-i-win/

Bitmask DP。

```python
def canIWin(self, maxChoosableInteger, desiredTotal):
    if maxChoosableInteger * (maxChoosableInteger + 1) // 2 < desiredTotal:
        return False
    memo = {}
    def dp(mask, total):
        if mask in memo: return memo[mask]
        for i in range(1, maxChoosableInteger + 1):
            if mask & (1 << i): continue
            if total + i >= desiredTotal:
                memo[mask] = True; return True
            if not dp(mask | (1 << i), total + i):
                memo[mask] = True; return True
        memo[mask] = False; return False
    return dp(0, 0)
```

## 466. Count The Repetitions
https://leetcode.com/problems/count-the-repetitions/

找循环节的优化。

```python
def getMaxRepetitions(self, s1, n1, s2, n2):
    if not set(s2).issubset(set(s1)): return 0
    recall = {}
    s2_count = s1_count = idx = 0
    while s1_count < n1:
        for c in s1:
            if c == s2[idx]:
                idx += 1
                if idx == len(s2):
                    s2_count += 1; idx = 0
        s1_count += 1
        if idx in recall:
            prev_s1, prev_s2 = recall[idx]
            cycle_s1 = s1_count - prev_s1
            cycle_s2 = s2_count - prev_s2
            remaining = n1 - s1_count
            s2_count += (remaining // cycle_s1) * cycle_s2
            s1_count += (remaining // cycle_s1) * cycle_s1
        else:
            recall[idx] = (s1_count, s2_count)
    return s2_count // n2
```

```python
def makesquare(self, matchsticks):
    total = sum(matchsticks)
    if total % 4: return False
    side = total // 4
    matchsticks.sort(reverse=True)
    sides = [0] * 4
    def bt(idx):
        if idx == len(matchsticks): return all(s == side for s in sides)
        for i in range(4):
            if sides[i] + matchsticks[idx] <= side:
                sides[i] += matchsticks[idx]
                if bt(idx + 1): return True
                sides[i] -= matchsticks[idx]
            if sides[i] == 0: break
        return False
    return bt(0)
```

## 488. Zuma Game
https://leetcode.com/problems/zuma-game/

DFS + 剪枝：尝试在每组连续相同颜色处插入手中的球。

```python
def findMinStep(self, board, hand):
    from collections import Counter
    def clean(s):
        i = 0; res = ''
        while i < len(s):
            j = i
            while j < len(s) and s[j] == s[i]: j += 1
            if j - i < 3: res += s[i:j]
            else: return clean(res + s[j:])
            i = j
        return res
    cnt = Counter(hand)
    best = [float('inf')]
    def dfs(b, step):
        if step >= best[0]: return
        if not b: best[0] = step; return
        i = 0
        while i < len(b):
            j = i
            while j < len(b) and b[j] == b[i]: j += 1
            need = 3 - (j - i)
            if 0 < need <= cnt[b[i]]:
                cnt[b[i]] -= need
                dfs(clean(b[:i] + b[j:]), step + need)
                cnt[b[i]] += need
            i = j
    dfs(board, 0)
    return best[0] if best[0] < float('inf') else -1
```

```python
        if len(arr) <= 1: return arr
```

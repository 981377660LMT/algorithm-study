## 1701. Average Waiting Time
https://leetcode.com/problems/average-waiting-time/

```python
def averageWaitingTime(self, customers):
    cur_time = 0; total_wait = 0
    for arrival, time in customers:
        cur_time = max(cur_time, arrival) + time
        total_wait += cur_time - arrival
    return total_wait / len(customers)
```

## 1712. Ways to Split Array Into Three Parts
https://leetcode.com/problems/ways-to-split-array-into-three-parts/

前缀和 + 二分。

```python
import bisect
def waysToSplit(self, nums):
    MOD = 10**9 + 7
    n = len(nums)
    prefix = [0] * (n + 1)
    for i in range(n): prefix[i+1] = prefix[i] + nums[i]
    total = prefix[n]; res = 0
    for i in range(1, n - 1):
        left = prefix[i]
        if left * 3 > total: break
        # j 的范围：prefix[j] >= 2*left and prefix[j] <= (total + left) // ... 
        # mid = prefix[j] - left, right = total - prefix[j]
        # left <= mid => prefix[j] >= 2*left
        # mid <= right => prefix[j] <= (total + left) / 2 ... 不对
        # mid <= right => prefix[j] - left <= total - prefix[j] => prefix[j] <= (total + left) / 2
        lo = bisect.bisect_left(prefix, 2 * left, i + 1, n)
        hi = bisect.bisect_right(prefix, (total + left) // 2, i + 1, n)
        # 需要 (total + left) 整除检查
        # mid <= right: 2*prefix[j] <= total + left
        hi2 = bisect.bisect_right(prefix, (total + left) // 2, lo, n)
        if (total + left) % 2 == 0:
            hi2 = bisect.bisect_right(prefix, (total + left) // 2, lo, n)
        else:
            hi2 = bisect.bisect_left(prefix, (total + left) // 2 + 1, lo, n)
        res = (res + max(0, min(hi2, n) - lo)) % MOD
    return res
```

## 1713. Minimum Operations to Make a Subsequence
https://leetcode.com/problems/minimum-operations-to-make-a-subsequence/

转化为 LIS 问题用 BIT/二分。

```python
import bisect
def minOperations(self, target, arr):
    pos = {v: i for i, v in enumerate(target)}
    filtered = [pos[x] for x in arr if x in pos]
    # LIS of filtered
    tails = []
    for x in filtered:
        i = bisect.bisect_left(tails, x)
        if i == len(tails): tails.append(x)
        else: tails[i] = x
    return len(target) - len(tails)
```

## 1717. Maximum Score From Removing Substrings
https://leetcode.com/problems/maximum-score-from-removing-substrings/

贪心：先移除分高的模式，再移除分低的。

```python
def maximumGain(self, s, x, y):
    def remove(s, first, second, score):
        stack = []; total = 0
        for c in s:
            if stack and stack[-1] == first and c == second:
                stack.pop(); total += score
            else:
                stack.append(c)
        return ''.join(stack), total
    if x >= y:
        s, r1 = remove(s, 'a', 'b', x)
        _, r2 = remove(s, 'b', 'a', y)
    else:
        s, r1 = remove(s, 'b', 'a', y)
        _, r2 = remove(s, 'a', 'b', x)
    return r1 + r2
```

## 1718. Construct the Lexicographically Largest Valid Sequence
https://leetcode.com/problems/construct-the-lexicographically-largest-valid-sequence/

回溯。

```python
def constructDistancedSequence(self, n):
    size = 2 * n - 1
    res = [0] * size; used = [False] * (n + 1)
    def bt(idx):
        if idx == size: return True
        if res[idx]: return bt(idx + 1)
        for num in range(n, 0, -1):
            if used[num]: continue
            if num == 1:
                res[idx] = 1; used[1] = True
                if bt(idx + 1): return True
                res[idx] = 0; used[1] = False
            elif idx + num < size and res[idx + num] == 0:
                res[idx] = res[idx + num] = num; used[num] = True
                if bt(idx + 1): return True
                res[idx] = res[idx + num] = 0; used[num] = False
        return False
    bt(0)
    return res
```

## 1722. Minimize Hamming Distance After Swap Operations
https://leetcode.com/problems/minimize-hamming-distance-after-swap-operations/

并查集分组，组内匹配。

```python
from collections import Counter
def minimumHammingDistance(self, source, target, allowedSwaps):
    n = len(source)
    parent = list(range(n))
    def find(x):
        while parent[x] != x: parent[x] = parent[parent[x]]; x = parent[x]
        return x
    for u, v in allowedSwaps:
        parent[find(u)] = find(v)
    groups = {}
    for i in range(n):
        r = find(i)
        if r not in groups: groups[r] = []
        groups[r].append(i)
    res = 0
    for indices in groups.values():
        src = Counter(source[i] for i in indices)
        tgt = Counter(target[i] for i in indices)
        res += sum((src - tgt).values())
    return res
```

## 1723. Find Minimum Time to Finish All Jobs

```python
def minimumTimeRequired(self, jobs, k):
    jobs.sort(reverse=True)
    workers = [0] * k
    res = [sum(jobs)]
    def bt(i):
        if i == len(jobs):
            res[0] = min(res[0], max(workers)); return
        if max(workers) >= res[0]: return
        seen = set()
        for j in range(k):
            if workers[j] in seen: continue
            seen.add(workers[j])
            workers[j] += jobs[i]
            bt(i + 1)
            workers[j] -= jobs[i]
    bt(0)
    return res[0]
```

```python
    # 找出没有共同语言的朋友对涉及的人
    need = set()

        u -= 1; v -= 1
```

## 1734. Decode XORed Permutation
https://leetcode.com/problems/decode-xored-permutation/

利用全排列的 XOR 和 encoded 的偶数位 XOR 求第一个元素。

```python
def decode(self, encoded):
    n = len(encoded) + 1
    total_xor = 0
    for i in range(1, n + 1): total_xor ^= i
    # encoded[1] ^ encoded[3] ^ ... = perm[2] ^ perm[3] ^ ... ^ perm[n]
    suffix_xor = 0
    for i in range(1, len(encoded), 2): suffix_xor ^= encoded[i]
    first = total_xor ^ suffix_xor
    res = [first]
    for e in encoded:
        res.append(res[-1] ^ e)
    return res
```

```python
def kthLargestValue(self, matrix, k):
    m, n = len(matrix), len(matrix[0])

    vals = []
    for i in range(m):
        for j in range(n):
```

```python
from collections import defaultdict
def restoreArray(self, adjacentPairs):
    graph = defaultdict(list)
    for u, v in adjacentPairs:
        graph[u].append(v); graph[v].append(u)
    start = next(k for k, v in graph.items() if len(v) == 1)
    res = []; visited = set()

        visited.add(node); res.append(node)
        for nxt in graph[node]:
```

```python
import bisect
def minAbsDifference(self, nums, goal):
    def subsets(arr):
        sums = {0}
        for x in arr:
            sums = sums | {s + x for s in sums}
        return sorted(sums)
    n = len(nums)
    left = subsets(nums[:n//2])
    right = subsets(nums[n//2:])
    res = float('inf')
    for s in left:
        target = goal - s
        i = bisect.bisect_left(right, target)
        if i < len(right): res = min(res, abs(right[i] - target))
        if i > 0: res = min(res, abs(right[i-1] - target))
```

https://leetcode.com/problems/minimum-degree-of-a-connected-triad-in-a-graph/

枚举三元组，度之和减 6。

```python
def minTrioDegree(self, n, edges):
    degree = [0] * (n + 1)
    adj = set()
    for u, v in edges:
        degree[u] += 1; degree[v] += 1
        adj.add((min(u,v), max(u,v)))
    res = float('inf')
    for u, v in edges:
        for w in range(1, n + 1):
            if w != u and w != v:
                a, b = (min(u,w), max(u,w)), (min(v,w), max(v,w))
                if a in adj and b in adj:
                    res = min(res, degree[u] + degree[v] + degree[w] - 6)
    return res if res != float('inf') else -1
```

```python
def longestNiceSubstring(self, s):
    if len(s) < 2: return ''
    ss = set(s)
    for i, c in enumerate(s):
        if c.swapcase() not in ss:
            left = self.longestNiceSubstring(s[:i])
```

## 1765. Map of Highest Peak
https://leetcode.com/problems/map-of-highest-peak/

多源 BFS。

```python
from collections import deque
def highestPeak(self, isWater):
    m, n = len(isWater), len(isWater[0])
    height = [[-1]*n for _ in range(m)]
    q = deque()
    for i in range(m):
        for j in range(n):
            if isWater[i][j]:
                height[i][j] = 0; q.append((i, j))
    while q:
        r, c = q.popleft()
        for dr, dc in [(-1,0),(1,0),(0,-1),(0,1)]:
            nr, nc = r+dr, c+dc
            if 0 <= nr < m and 0 <= nc < n and height[nr][nc] == -1:
                height[nr][nc] = height[r][c] + 1
                q.append((nr, nc))
    return height
```

## 1770. Maximum Score from Performing Multiplication Operations
https://leetcode.com/problems/maximum-score-from-performing-multiplication-operations/

DP：`dp[i][j]` = 左取 i 个右取 j 个的最大分。

```python
def maximumScore(self, nums, multipliers):
    m = len(multipliers); n = len(nums)
    dp = [[0]*(m+1) for _ in range(m+1)]
    for k in range(m-1, -1, -1):
        for left in range(k, -1, -1):
            right = k - left
            dp[left][right] = max(
                nums[left] * multipliers[k] + dp[left+1][right],
                nums[n-1-right] * multipliers[k] + dp[left][right+1]
            )
    return dp[0][0]
```

```python
def closestCost(self, baseCosts, toppingCosts, target):

    def dfs(i, cur):
        if abs(cur - target) < abs(res[0] - target) or \
           (abs(cur - target) == abs(res[0] - target) and cur < res[0]):
```

## 1786. Number of Restricted Paths From First to Last Node
https://leetcode.com/problems/number-of-restricted-paths-from-first-to-last-node/

Dijkstra + 记忆化 DFS/DP。

```python
import heapq
from functools import lru_cache
def countRestrictedPaths(self, n, edges):
    MOD = 10**9 + 7
    graph = [[] for _ in range(n + 1)]
    for u, v, w in edges:
        graph[u].append((v, w)); graph[v].append((u, w))
    dist = [float('inf')] * (n + 1); dist[n] = 0
    h = [(0, n)]
    while h:
        d, u = heapq.heappop(h)
        if d > dist[u]: continue
        for v, w in graph[u]:
            if d + w < dist[v]:
                dist[v] = d + w; heapq.heappush(h, (dist[v], v))
    @lru_cache(maxsize=None)
    def dfs(u):
        if u == n: return 1
        total = 0
        for v, _ in graph[u]:
            if dist[v] < dist[u]:
                total = (total + dfs(v)) % MOD
        return total
    return dfs(1)
```

## 1792. Maximum Average Pass Ratio
https://leetcode.com/problems/maximum-average-pass-ratio/

最大堆：每次给增益最大的班级加1人。

```python
import heapq
def maxAverageRatio(self, classes, extraStudents):
    def gain(p, t): return (p+1)/(t+1) - p/t
    h = [(-gain(p, t), p, t) for p, t in classes]
    heapq.heapify(h)
    for _ in range(extraStudents):
        _, p, t = heapq.heappop(h)
        heapq.heappush(h, (-gain(p+1, t+1), p+1, t+1))
    return sum(p/t for _, p, t in h) / len(h)
```

## 1793. Maximum Score of a Good Subarray
https://leetcode.com/problems/maximum-score-of-a-good-subarray/

双指针从 k 向两边扩展，贪心选更大的那边。

```python
def maximumScore(self, nums, k):
    n = len(nums)
    l = r = k; mn = nums[k]; res = mn
    while l > 0 or r < n - 1:
        left = nums[l-1] if l > 0 else 0
        right = nums[r+1] if r < n-1 else 0
        if left >= right:
            l -= 1; mn = min(mn, nums[l])
        else:
            r += 1; mn = min(mn, nums[r])
        res = max(res, mn * (r - l + 1))
    return res
```

## 1799. Maximize Score After N Operations
https://leetcode.com/problems/maximize-score-after-n-operations/

状压 DP。

```python
from math import gcd
from functools import lru_cache
def maxScore(self, nums):
    n = len(nums)
    @lru_cache(maxsize=None)
    def dp(mask, op):
        if op > n // 2: return 0
        res = 0
        for i in range(n):
            if mask & (1 << i): continue
            for j in range(i+1, n):
                if mask & (1 << j): continue
                new_mask = mask | (1 << i) | (1 << j)
                res = max(res, op * gcd(nums[i], nums[j]) + dp(new_mask, op + 1))
        return res
    return dp(0, 1)
```

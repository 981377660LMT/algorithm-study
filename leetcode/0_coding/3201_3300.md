## 3201. Find the Maximum Length of Valid Subsequence I / 3202. Find the Maximum Length of Valid Subsequence II
https://leetcode.com/problems/find-the-maximum-length-of-valid-subsequence-ii/

```python
def maximumLength(self, nums, k):
    # (a[i]+a[i+1]) % k 恒定
    res = 0
    for r in range(k):
        dp = [0] * k
        for x in nums:
            dp[x % k] = dp[(r - x % k) % k] + 1
        res = max(res, max(dp))
    return res
```

## 3203. Find Minimum Diameter After Merging Two Trees
https://leetcode.com/problems/find-minimum-diameter-after-merging-two-trees/

```python
from collections import deque
def minimumDiameterAfterMerge(self, edges1, edges2):
    def diameter(n, edges):
        if n == 1: return 0
        adj = [[] for _ in range(n)]
        for u, v in edges: adj[u].append(v); adj[v].append(u)
        def bfs(start):
            dist = [-1]*n; dist[start]=0; q=deque([start])
            farthest = start
            while q:
                u = q.popleft()
                for v in adj[u]:
                    if dist[v]==-1: dist[v]=dist[u]+1; q.append(v)
                    if dist[v]>dist[farthest]: farthest=v
            return farthest, dist[farthest]
        far, _ = bfs(0)
        _, d = bfs(far)
        return d
    d1 = diameter(len(edges1)+1, edges1)
    d2 = diameter(len(edges2)+1, edges2)
    from math import ceil
    return max(d1, d2, ceil(d1/2) + ceil(d2/2) + 1)
```

## 3209. Number of Subarrays With AND Value of K

```python
def countSubarrays(self, nums, k):
    # AND 递减, 每个位置维护所有可能的 AND 值
    res = 0
    prev = {}  # and_val -> count
    for x in nums:
        cur = {}
        for v, c in prev.items():
            nv = v & x
            cur[nv] = cur.get(nv, 0) + c
        cur[x] = cur.get(x, 0) + 1
        res += cur.get(k, 0)
        prev = cur
    return res
```

## 3224. Minimum Array Changes to Make Differences Equal
https://leetcode.com/problems/minimum-array-changes-to-make-differences-equal/

差分数组。

```python
def minChanges(self, nums, k):
    n = len(nums)
    # diff[d] 上的操作次数
    diff = [0] * (k + 2)
    for i in range(n // 2):
        a, b = nums[i], nums[n-1-i]
        lo, hi = min(a, b), max(a, b)
        d = hi - lo
        # 0次操作: d
        # 1次操作: [0, max(k-lo, hi)] 内除了 d
        max_d = max(k - lo, hi)
        # 差分
        diff[0] += 2; diff[max_d + 1] -= 2
        diff[0] -= 2 # undo
        # [0, max_d] 需要1次, d需要0次 => 做差分
        diff[0] += 2
        if max_d + 1 <= k: diff[max_d + 1] -= 2
        # 简化：对于每对，生成不同 d 值的代价
    # 重新做
    diff = [0] * (k + 2)
    for i in range(n // 2):
        a, b = sorted([nums[i], nums[n-1-i]])
        d = b - a
        max_d = max(k - a, b)
        # cost 2 for all d in [0,k]
        diff[0] += 2; diff[k+1] -= 2
        # cost 1 for d in [0, max_d] (saves 1)
        diff[0] -= 1; diff[min(max_d, k) + 1] += 1
        # cost 0 for d (saves 1 more)
        diff[d] -= 1; diff[d+1] += 1
    # 前缀和
    cur = 0; res = float('inf')
    for d in range(k + 1):
        cur += diff[d]
        res = min(res, cur)
    return res
```

```python
def minimumOperations(self, nums, target):
    diff = [t - n for n, t in zip(nums, target)]
    res = abs(diff[0])
    for i in range(1, len(diff)):
        if diff[i] > 0 and diff[i-1] > 0:
            if diff[i] > diff[i-1]: res += diff[i] - diff[i-1]
        elif diff[i] < 0 and diff[i-1] < 0:

        else:
            res += abs(diff[i])
    return res
```

```python
def nonSpecialCount(self, l, r):
    def sieve(n):
        is_p = [True]*(n+1); is_p[0]=is_p[1]=False
        for i in range(2,int(n**0.5)+1):
            if is_p[i]:
                for j in range(i*i,n+1,i): is_p[j]=False
        return is_p
    sq = isqrt(r); is_p = sieve(sq)
```

```python
from collections import deque
def shortestDistanceAfterQueries(self, n, queries):
    adj = [[] for _ in range(n)]
    for i in range(n-1): adj[i].append(i+1)
    res = []
    for u, v in queries:
        adj[u].append(v)
        dist = [-1]*n; dist[0]=0; q=deque([0])
        while q:
            x = q.popleft()
            for y in adj[x]:
                if dist[y]==-1: dist[y]=dist[x]+1; q.append(y)
        res.append(dist[n-1])
    return res
```

## 3244. Shortest Distance After Road Addition Queries II
https://leetcode.com/problems/shortest-distance-after-road-addition-queries-ii/

```python
from sortedcontainers import SortedList
def shortestDistanceAfterQueries(self, n, queries):
    # 边不交叉, 维护区间集合
    sl = SortedList(range(n))
    res = []
    for u, v in queries:
        # 删除 (u, v) 之间的中间节点
        idx = sl.bisect_right(u)
        while idx < len(sl) and sl[idx] < v:
            sl.pop(idx)
        res.append(len(sl) - 1)
    return res
```

## 3250. Find the Count of Monotonic Pairs I / 3251. Find the Count of Monotonic Pairs II
https://leetcode.com/problems/find-the-count-of-monotonic-pairs-ii/

```python
def countOfPairs(self, nums):
    MOD = 10**9 + 7; n = len(nums)
    # dp[j] = 第 i 个元素 arr1[i]=j 的方案数
    dp = [1] * (nums[0] + 1)
    for i in range(1, n):
        new_dp = [0] * (nums[i] + 1)
        prefix = [0] * (len(dp) + 1)
        for j in range(len(dp)): prefix[j+1] = (prefix[j] + dp[j]) % MOD
        for j in range(nums[i] + 1):
            # arr1[i]=j, arr2[i]=nums[i]-j
            # arr1 非递减: arr1[i-1] <= j
            # arr2 非递增: nums[i-1]-arr1[i-1] >= nums[i]-j
            # => arr1[i-1] >= nums[i-1]-nums[i]+j
            lo = max(0, nums[i-1] - nums[i] + j)
            hi = min(j, nums[i-1])
            if lo > hi or hi >= len(prefix): continue
            new_dp[j] = (prefix[hi+1] - prefix[lo]) % MOD
        dp = new_dp
    return sum(dp) % MOD
```

```python
import heapq
def getFinalState(self, nums, k, multiplier):
    heap = [(x, i) for i, x in enumerate(nums)]
    heapq.heapify(heap)
    for _ in range(k):
```

```python
            if key in seen: continue
            seen.add(key)
            cnt = Counter(pal)
            # 计算排列数(无前导零)
            total = factorial(n)
            for v in cnt.values(): total //= factorial(v)
            # 减去前导零
            if cnt['0'] > 0:
                cnt['0'] -= 1
                bad = factorial(n-1)
                for v in cnt.values(): bad //= factorial(v)
                cnt['0'] += 1
                total -= bad
            res += total
    return res
```

## 3276. Select Cells in Grid With Maximum Score
https://leetcode.com/problems/select-cells-in-grid-with-maximum-score/

```python
from functools import lru_cache
def maxScore(self, grid):
    # 每行最多选一个, 值不能重复
    # 按值分组, 每个值可以在哪些行
    from collections import defaultdict
    val_rows = defaultdict(set)
    for i in range(len(grid)):
        for j in range(len(grid[0])):
            val_rows[grid[i][j]].add(i)
    vals = sorted(val_rows.keys(), reverse=True)
    m = len(grid)
    @lru_cache(None)
    def dp(idx, mask):
        if idx == len(vals): return 0
        # 不选 vals[idx]
        res = dp(idx + 1, mask)
        for r in val_rows[vals[idx]]:
            if not (mask >> r & 1):
                res = max(res, vals[idx] + dp(idx + 1, mask | (1 << r)))
        return res
    return dp(0, 0)
```

## 3290. Maximum Multiplication Score

```python
def maxScore(self, a, b):
    # 选 b 中4个递增下标 i<j<k<l, 最大化 a[0]*b[i]+a[1]*b[j]+a[2]*b[k]+a[3]*b[l]
    dp = [float('-inf')] * 4
    for x in b:
        for j in range(3, -1, -1):
            prev = dp[j-1] if j > 0 else 0
            dp[j] = max(dp[j], prev + a[j] * x)
    return dp[3]
```

## 3292. Minimum Number of Valid Strings to Form Target I / 3291. Minimum Number of Valid Strings to Form Target II

```python
def minValidStrings(self, words, target):
    # 每个 word 的任意前缀可用
    # 建Trie
    n = len(target)
    # 对每个位置i, 找最长前缀匹配
    trie = {}
    for w in words:
        node = trie
        for c in w:
            if c not in node: node[c] = {}
            node = node[c]
    # dp[i] = min strings to form target[:i]
    dp = [float('inf')] * (n + 1); dp[0] = 0
    for i in range(n):
        if dp[i] == float('inf'): continue
        node = trie
        j = i
        while j < n and target[j] in node:
            node = node[target[j]]; j += 1
            dp[j] = min(dp[j], dp[i] + 1)
    return dp[n] if dp[n] < float('inf') else -1
```

## 3296. Minimum Number of Seconds to Make Mountain Height Zero
https://leetcode.com/problems/minimum-number-of-seconds-to-make-mountain-height-zero/

```python
from math import isqrt
def minNumberOfSeconds(self, mountainHeight, workerTimes):
    def total_reduced(t):
        s = 0
        for w in workerTimes:
            # w*(1+2+...+k) <= t → k*(k+1)/2 <= t/w
            r = t // w
            k = (isqrt(8*r+1) - 1) // 2
            s += k
        return s
    lo, hi = 0, max(workerTimes) * mountainHeight * (mountainHeight+1) // 2
    while lo < hi:
        mid = (lo + hi) // 2
        if total_reduced(mid) >= mountainHeight: hi = mid
        else: lo = mid + 1
    return lo
```

## 3297. Count Substrings That Can Be Rearranged to Contain a String I / 3298. Count Substrings That Can Be Rearranged to Contain a String II

```python
from collections import Counter
def validSubstringCount(self, word1, word2):
    need = Counter(word2); missing = len(need)
    l = res = 0; have = Counter()
    for r in range(len(word1)):
        c = word1[r]; have[c] += 1
        if have[c] == need[c]: missing -= 1
        while missing == 0:
            res += len(word1) - r
            have[word1[l]] -= 1
            if have[word1[l]] < need[word1[l]]: missing += 1
            l += 1
    return res
```

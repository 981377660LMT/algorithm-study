## 101. Symmetric Tree
https://leetcode.com/problems/symmetric-tree/

递归比较左右子树镜像。

```python
def isSymmetric(self, root):
    def check(a, b):
        if not a and not b: return True
        if not a or not b: return False
        return a.val == b.val and check(a.left, b.right) and check(a.right, b.left)
    return check(root.left, root.right)
```

```python
def levelOrder(self, root):
    if not root: return []

    while q:
        level = []

            node = q.popleft()
```

```python
from collections import deque
def zigzagLevelOrder(self, root):
    if not root: return []
    res, q = [], deque([root])
    left_to_right = True
    while q:
        level = []
        for _ in range(len(q)):
            node = q.popleft()
            level.append(node.val)
            if node.left: q.append(node.left)
            if node.right: q.append(node.right)
        res.append(level if left_to_right else level[::-1])
        left_to_right = not left_to_right
    return res
```

```python
def isBalanced(self, root):
    def height(node):
        if not node: return 0
        l, r = height(node.left), height(node.right)
        if l == -1 or r == -1 or abs(l - r) > 1: return -1
        return 1 + max(l, r)
    return height(root) != -1
```

```python
def pathSum(self, root, targetSum):
    res = []
    def dfs(node, remain, path):
        if not node: return
        path.append(node.val)
        if not node.left and not node.right and remain == node.val:
            res.append(path[:])
```

```python
            rightmost = cur.left

            cur.right = cur.left
            cur.left = None
```

## 115. Distinct Subsequences

```python
def numDistinct(self, s, t):
    m, n = len(s), len(t)
    dp = [0] * (n + 1)
    dp[0] = 1
    for i in range(1, m + 1):
        for j in range(n, 0, -1):
            if s[i - 1] == t[j - 1]:
                dp[j] += dp[j - 1]
    return dp[n]
```

```python
        if not node: return 0
```

## 126. Word Ladder II
https://leetcode.com/problems/word-ladder-ii/

BFS 找最短路径长度 + DFS 回溯所有最短路径。

```python
from collections import defaultdict, deque
def findLadders(self, beginWord, endWord, wordList):
    words = set(wordList)
    if endWord not in words: return []
    # BFS建层级图
    layer = defaultdict(set)
    q = deque([beginWord])
    visited = {beginWord}
    found = False
    while q and not found:
        current_level = set()
        for _ in range(len(q)):
            word = q.popleft()
            for i in range(len(word)):
                for c in 'abcdefghijklmnopqrstuvwxyz':
                    nw = word[:i] + c + word[i+1:]
                    if nw == endWord: found = True
                    if nw in words and nw not in visited:
                        current_level.add(nw)
                        layer[nw].add(word)
        visited |= current_level
        q.extend(current_level)
    # DFS回溯
    res = []
    def bt(word, path):
        if word == beginWord:
            res.append([beginWord] + path[::-1]); return
        for prev in layer[word]:
            bt(prev, path + [word])
    if found: bt(endWord, [])
    return res
```

## 127. Word Ladder

```python
from collections import deque
def ladderLength(self, beginWord, endWord, wordList):
    words = set(wordList)
    if endWord not in words: return 0
    q = deque([(beginWord, 1)])
    visited = {beginWord}
    while q:
        word, steps = q.popleft()
        for i in range(len(word)):
            for c in 'abcdefghijklmnopqrstuvwxyz':
                nw = word[:i] + c + word[i+1:]
                if nw == endWord: return steps + 1
                if nw in words and nw not in visited:
                    visited.add(nw)
                    q.append((nw, steps + 1))
    return 0
```

## 131. Palindrome Partitioning

```python
def partition(self, s):
    res = []
    def bt(start, path):
        if start == len(s):
            res.append(path[:]); return
        for end in range(start + 1, len(s) + 1):
            sub = s[start:end]
            if sub == sub[::-1]:
                path.append(sub)
                bt(end, path)
                path.pop()
    bt(0, [])
    return res
```

## 132. Palindrome Partitioning II
https://leetcode.com/problems/palindrome-partitioning-ii/

DP：`dp[i]` 为 `s[:i]` 最少切割次数，预处理回文表。

```python
def minCut(self, s):
    n = len(s)
    # 预处理回文
    pal = [[False] * n for _ in range(n)]
    for i in range(n - 1, -1, -1):
        for j in range(i, n):
            if s[i] == s[j] and (j - i < 2 or pal[i+1][j-1]):
                pal[i][j] = True
    dp = list(range(-1, n))  # dp[i+1] = s[:i+1]的最少切割
    for i in range(1, n):
        for j in range(i + 1):
            if pal[j][i]:
                dp[i + 1] = min(dp[i + 1], dp[j] + 1)
    return dp[n]
```

```python
    candy = [1] * n
    for i in range(1, n):
        if ratings[i] > ratings[i - 1]:
            candy[i] = candy[i - 1] + 1
```

```python
def reorderList(self, head):
    # 找中点
    slow = fast = head
    while fast.next and fast.next.next:
        slow = slow.next; fast = fast.next.next
    # 反转后半
    prev, cur = None, slow.next
    slow.next = None
    while cur:
        cur.next, prev, cur = prev, cur, cur.next
    # 交替合并
    first, second = head, prev
    while second:
        tmp1, tmp2 = first.next, second.next
        first.next = second
        second.next = tmp1
        first, second = tmp1, tmp2
```

## 146. LRU Cache
https://leetcode.com/problems/lru-cache/

OrderedDict 实现。

```python
from collections import OrderedDict
class LRUCache:
    def __init__(self, capacity):
        self.cap = capacity
        self.cache = OrderedDict()
    def get(self, key):
        if key not in self.cache: return -1
        self.cache.move_to_end(key)
        return self.cache[key]
    def put(self, key, value):
        if key in self.cache:
            self.cache.move_to_end(key)
        self.cache[key] = value
```

```python
def sortList(self, head):
    if not head or not head.next: return head
    slow, fast = head, head.next
    while fast and fast.next:
        slow = slow.next; fast = fast.next.next
    mid = slow.next; slow.next = None
    l, r = self.sortList(head), self.sortList(mid)
    dummy = cur = ListNode(0)
    while l and r:
        if l.val <= r.val: cur.next = l; l = l.next

        cur = cur.next
    cur.next = l or r
    return dummy.next
```

## 149. Max Points on a Line
https://leetcode.com/problems/max-points-on-a-line/

固定一点，用斜率（最简分数）统计共线。

```python
from math import gcd
from collections import defaultdict
def maxPoints(self, points):
    n = len(points)
    if n <= 2: return n
    res = 2
    for i in range(n):
        cnt = defaultdict(int)
        for j in range(i + 1, n):
            dx = points[j][0] - points[i][0]
            dy = points[j][1] - points[i][1]
            g = gcd(dx, dy)
            cnt[(dx // g, dy // g)] += 1
        res = max(res, max(cnt.values()) + 1)
    return res
```

## 164. Maximum Gap
https://leetcode.com/problems/maximum-gap/

桶排序：最大间距一定出现在不同桶之间。

```python
def maximumGap(self, nums):
    if len(nums) < 2: return 0
    lo, hi = min(nums), max(nums)
    if lo == hi: return 0
    n = len(nums)
    size = max(1, (hi - lo) // (n - 1))
    buckets = [[float('inf'), float('-inf')] for _ in range((hi - lo) // size + 1)]
    for x in nums:
        b = (x - lo) // size
        buckets[b][0] = min(buckets[b][0], x)
        buckets[b][1] = max(buckets[b][1], x)
    res = 0
    prev_max = lo
    for mn, mx in buckets:
        if mn == float('inf'): continue
        res = max(res, mn - prev_max)
        prev_max = mx
    return res
```

```python
def fractionToDecimal(self, numerator, denominator):

    res = ''
```

```python
def rightSideView(self, root):
```

## 200. Number of Islands
https://leetcode.com/problems/number-of-islands/

DFS/BFS 遍历，每发现一个 '1' 计数并淹没整个岛。

```python
def numIslands(self, grid):
    def dfs(i, j):
        if i < 0 or i >= len(grid) or j < 0 or j >= len(grid[0]) or grid[i][j] != '1':
            return
        grid[i][j] = '0'
        for di, dj in [(-1,0),(1,0),(0,-1),(0,1)]:
            dfs(i + di, j + dj)
    count = 0
    for i in range(len(grid)):
        for j in range(len(grid[0])):
            if grid[i][j] == '1':
                dfs(i, j)
                count += 1
    return count
```

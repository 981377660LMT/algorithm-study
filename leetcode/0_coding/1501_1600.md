## 1504. Count Submatrices With All Ones
https://leetcode.com/problems/count-submatrices-with-all-ones/

按行计算柱状图高度 + 单调栈。

```python
def numSubmat(self, mat):
    m, n = len(mat), len(mat[0])
    res = 0
    heights = [0] * n
    for i in range(m):
        for j in range(n):
            heights[j] = heights[j] + 1 if mat[i][j] else 0
        # 对每个位置计算以该位置为右下角的全1子矩阵数
        stack = []
        cnt = [0] * n
        for j in range(n):
            while stack and heights[stack[-1]] >= heights[j]:
                stack.pop()
            if stack:
                cnt[j] = cnt[stack[-1]] + heights[j] * (j - stack[-1])
            else:
                cnt[j] = heights[j] * (j + 1)
            stack.append(j)
        res += sum(cnt)
    return res
```

## 1514. Path with Maximum Probability
https://leetcode.com/problems/path-with-maximum-probability/

Dijkstra（最大堆）。

```python
import heapq
from collections import defaultdict
def maxProbability(self, n, edges, succProb, start, end):
    graph = defaultdict(list)
    for (u, v), p in zip(edges, succProb):
        graph[u].append((v, p)); graph[v].append((u, p))
    dist = [0.0] * n; dist[start] = 1.0
    h = [(-1.0, start)]
    while h:
        prob, u = heapq.heappop(h)
        prob = -prob
        if u == end: return prob
        if prob < dist[u]: continue
        for v, p in graph[u]:
            if prob * p > dist[v]:
                dist[v] = prob * p
                heapq.heappush(h, (-dist[v], v))
    return 0.0
```

```python
def countPairs(self, root, distance):
    self.res = 0
    def dfs(node):
        if not node: return []
        if not node.left and not node.right: return [1]
        left = dfs(node.left)
        right = dfs(node.right)
        for l in left:
            for r in right:
                if l + r <= distance: self.res += 1
        return [d + 1 for d in left + right if d + 1 < distance]
    dfs(root)
    return self.res
```

## 1536. Minimum Swaps to Arrange a Binary Grid
https://leetcode.com/problems/minimum-swaps-to-arrange-a-binary-grid/

模拟冒泡排序：每行需要后缀0的数量。

```python
def minSwaps(self, grid):
    n = len(grid)
    trailing = []
    for row in grid:
        cnt = 0
        for j in range(n-1, -1, -1):
            if row[j] == 0: cnt += 1
            else: break
        trailing.append(cnt)
    res = 0
    for i in range(n):
        need = n - 1 - i
        j = i
        while j < n and trailing[j] < need: j += 1
        if j == n: return -1
        while j > i:
            trailing[j], trailing[j-1] = trailing[j-1], trailing[j]
            j -= 1; res += 1
    return res
```

## 1559. Detect Cycles in 2D Grid

```python
def containsCycle(self, grid):
    m, n = len(grid), len(grid[0])
    parent = {}
    def find(x):
        while parent[x] != x:
            parent[x] = parent[parent[x]]; x = parent[x]
        return x
    for i in range(m):
        for j in range(n):
            parent[(i,j)] = (i,j)
    for i in range(m):
        for j in range(n):
            for ni, nj in [(i+1,j),(i,j+1)]:
                if ni < m and nj < n and grid[ni][nj] == grid[i][j]:
                    a, b = find((i,j)), find((ni,nj))
                    if a == b: return True
                    parent[a] = b
    return False
```

```python
from functools import lru_cache
def countRoutes(self, locations, start, finish, fuel):
    MOD = 10**9 + 7
    @lru_cache(None)
    def dp(city, rem):
        res = 1 if city == finish else 0
        for j in range(len(locations)):
            if j != city:
                cost = abs(locations[city] - locations[j])
                if cost <= rem:
                    res = (res + dp(j, rem - cost)) % MOD
        return res
    return dp(start, fuel)
```

```python
def minCost(self, colors, neededTime):
```

## 1579. Remove Max Number of Edges to Keep Graph Fully Traversable
https://leetcode.com/problems/remove-max-number-of-edges-to-keep-graph-fully-traversable/

两个并查集：先加类型3的边（共用），再分别加类型1和2的边。

```python
def maxNumEdgesToRemove(self, n, edges):
    class UF:
        def __init__(self, n):
            self.parent = list(range(n+1)); self.rank = [0]*(n+1); self.comps = n
        def find(self, x):
            while self.parent[x] != x:
                self.parent[x] = self.parent[self.parent[x]]; x = self.parent[x]
            return x
        def union(self, a, b):
            a, b = self.find(a), self.find(b)
            if a == b: return False
            if self.rank[a] < self.rank[b]: a, b = b, a
            self.parent[b] = a
            if self.rank[a] == self.rank[b]: self.rank[a] += 1
            self.comps -= 1
            return True
    alice, bob = UF(n), UF(n)
    used = 0
    for t, u, v in edges:
        if t == 3:
            a = alice.union(u, v)
            b = bob.union(u, v)
            if a or b: used += 1
    for t, u, v in edges:
        if t == 1 and alice.union(u, v): used += 1
        if t == 2 and bob.union(u, v): used += 1
    if alice.comps > 1 or bob.comps > 1: return -1
    return len(edges) - used
```

## 1584. Min Cost to Connect All Points
https://leetcode.com/problems/min-cost-to-connect-all-points/

Prim 或 Kruskal 求 MST。

```python
import heapq
def minCostConnectPoints(self, points):
    n = len(points)
    visited = [False] * n
    h = [(0, 0)]  # (cost, point)
    res = 0; cnt = 0
    while cnt < n:
        cost, u = heapq.heappop(h)
        if visited[u]: continue
        visited[u] = True; res += cost; cnt += 1
        for v in range(n):
            if not visited[v]:
                d = abs(points[u][0]-points[v][0]) + abs(points[u][1]-points[v][1])
                heapq.heappush(h, (d, v))
    return res
```

```python
def sumOddLengthSubarrays(self, arr):

    for i in range(n):
        # 包含 arr[i] 的子数组数 = (i+1) * (n-i)
```

```python
def maxSumRangeQuery(self, nums, requests):
    MOD = 10**9 + 7
    n = len(nums)
    cnt = [0] * (n + 1)
    for l, r in requests:
        cnt[l] += 1; cnt[r+1] -= 1
    for i in range(1, n): cnt[i] += cnt[i-1]
```

```python
def minSubarray(self, nums, p):
    total = sum(nums) % p
    if total == 0: return 0
    prefix = {0: -1}
    cur = 0; res = len(nums)
    for i, x in enumerate(nums):
        cur = (cur + x) % p
        need = (cur - total) % p
        if need in prefix:
            res = min(res, i - prefix[need])
        prefix[cur] = i
    return res if res < len(nums) else -1
```

```python
def maxUniqueSplit(self, s):
    def bt(idx, seen):
        if idx == len(s): return 0
        res = -1
        for i in range(idx+1, len(s)+1):
            sub = s[idx:i]
            if sub not in seen:
                seen.add(sub)
                nxt = bt(i, seen)
                if nxt >= 0: res = max(res, 1 + nxt)
                seen.remove(sub)
        return res
    return bt(0, set())
```

## 1594. Maximum Non Negative Product in a Matrix
https://leetcode.com/problems/maximum-non-negative-product-in-a-matrix/

DP 维护每个位置的最大积和最小积。

```python
def maxProductPath(self, grid):
    MOD = 10**9 + 7
    m, n = len(grid), len(grid[0])
    mx = [[0]*n for _ in range(m)]
    mn = [[0]*n for _ in range(m)]
    mx[0][0] = mn[0][0] = grid[0][0]
    for j in range(1, n):
        mx[0][j] = mn[0][j] = mx[0][j-1] * grid[0][j]
    for i in range(1, m):
        mx[i][0] = mn[i][0] = mx[i-1][0] * grid[i][0]
    for i in range(1, m):
        for j in range(1, n):
            if grid[i][j] >= 0:
                mx[i][j] = max(mx[i-1][j], mx[i][j-1]) * grid[i][j]
                mn[i][j] = min(mn[i-1][j], mn[i][j-1]) * grid[i][j]
            else:
                mx[i][j] = min(mn[i-1][j], mn[i][j-1]) * grid[i][j]
                mn[i][j] = max(mx[i-1][j], mx[i][j-1]) * grid[i][j]
    return mx[m-1][n-1] % MOD if mx[m-1][n-1] >= 0 else -1
```

## 1600. Throne Inheritance
https://leetcode.com/problems/throne-inheritance/

多叉树前序遍历跳过死亡节点。

```python
from collections import defaultdict
class ThroneInheritance:
    def __init__(self, kingName):
        self.king = kingName
        self.children = defaultdict(list)
        self.dead = set()
    def birth(self, parentName, childName):
        self.children[parentName].append(childName)
    def death(self, name):
        self.dead.add(name)
    def getInheritanceOrder(self):
        order = []
        def dfs(name):
            if name not in self.dead: order.append(name)
            for child in self.children[name]: dfs(child)
        dfs(self.king)
        return order
```

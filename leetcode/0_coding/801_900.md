## 803. Bricks Falling When Hit

https://leetcode.cn/problems/bricks-falling-when-hit/

反向并查集：先全部打击，再反向添加统计新连接。

```python
def hitBricks(self, grid, hits):
    m, n = len(grid), len(grid[0])
    parent = list(range(m * n + 1)); size = [1] * (m * n + 1)
    roof = m * n  # virtual node
    def find(x):
        while parent[x] != x: parent[x] = parent[parent[x]]; x = parent[x]
        return x
    def union(a, b):
        a, b = find(a), find(b)
        if a == b: return
        if size[a] < size[b]: a, b = b, a
        parent[b] = a; size[a] += size[b]
    def idx(r, c): return r * n + c
    # Mark hits
    A = [row[:] for row in grid]
    for r, c in hits: A[r][c] = 0
    # Build initial UF
    for i in range(m):
        for j in range(n):
            if A[i][j] == 0: continue
            if i == 0: union(idx(i,j), roof)
            if i > 0 and A[i-1][j]: union(idx(i,j), idx(i-1,j))
            if j > 0 and A[i][j-1]: union(idx(i,j), idx(i,j-1))
    res = []
    for r, c in reversed(hits):
        if grid[r][c] == 0: res.append(0); continue
        prev = size[find(roof)]
        A[r][c] = 1
        if r == 0: union(idx(r,c), roof)
        for dr, dc in [(-1,0),(1,0),(0,-1),(0,1)]:
            nr, nc = r+dr, c+dc
            if 0 <= nr < m and 0 <= nc < n and A[nr][nc]:
                union(idx(r,c), idx(nr,nc))
        res.append(max(0, size[find(roof)] - prev - 1))
    return res[::-1]
```

## 815. Bus Routes

https://leetcode.cn/problems/bus-routes/

BFS 按路线层级搜索，而非按站点。

```python
from collections import defaultdict, deque
def numBusesToDestination(self, routes, source, target):
    if source == target: return 0
    stop_to_routes = defaultdict(set)
    for i, route in enumerate(routes):
        for stop in route:
            stop_to_routes[stop].add(i)
    visited_stops = {source}
    visited_routes = set()
    q = deque([source])
    buses = 0
    while q:
        buses += 1
        for _ in range(len(q)):
            stop = q.popleft()
            for route_id in stop_to_routes[stop]:
                if route_id in visited_routes: continue
                visited_routes.add(route_id)
                for next_stop in routes[route_id]:
                    if next_stop == target: return buses
                    if next_stop not in visited_stops:
                        visited_stops.add(next_stop)
                        q.append(next_stop)
    return -1
```

## 827. Making A Large Island

https://leetcode.cn/problems/making-a-large-island/

DFS 给每个岛标记编号和面积，然后枚举每个 0，合并相邻不同岛的面积。

```python
def largestIsland(self, grid):
    n = len(grid)
    idx = 2
    area = {0: 0}
    def dfs(i, j, idx):
        if i < 0 or i >= n or j < 0 or j >= n or grid[i][j] != 1: return 0
        grid[i][j] = idx
        return 1 + dfs(i+1,j,idx) + dfs(i-1,j,idx) + dfs(i,j+1,idx) + dfs(i,j-1,idx)
    for i in range(n):
        for j in range(n):
            if grid[i][j] == 1:
                area[idx] = dfs(i, j, idx)
                idx += 1
    res = max(area.values())
    for i in range(n):
        for j in range(n):
            if grid[i][j] == 0:
                seen = set()
                for di, dj in [(-1,0),(1,0),(0,-1),(0,1)]:
                    ni, nj = i + di, j + dj
                    if 0 <= ni < n and 0 <= nj < n:
                        seen.add(grid[ni][nj])
                res = max(res, 1 + sum(area.get(s, 0) for s in seen))
    return res
```

```python
def consecutiveNumbersSum(self, n):
    cnt = 0; k = 1
    while k * (k - 1) // 2 < n:
        if (n - k * (k - 1) // 2) % k == 0: cnt += 1
        k += 1
    return cnt
```

## 842. Split Array into Fibonacci Sequence

```python
def splitIntoFibonacci(self, num):
    def bt(idx, path):
        if idx == len(num) and len(path) >= 3: return path
        for i in range(idx + 1, len(num) + 1):
            if num[idx] == '0' and i > idx + 1: break
            val = int(num[idx:i])
            if val > 2**31 - 1: break
            if len(path) >= 2 and val > path[-1] + path[-2]: break
            if len(path) < 2 or val == path[-1] + path[-2]:
                res = bt(i, path + [val])
                if res: return res
        return []
    return bt(0, [])
```

```python
from collections import Counter
def isNStraightHand(self, hand, groupSize):
    if len(hand) % groupSize: return False
    cnt = Counter(hand)
    for card in sorted(cnt):
        if cnt[card] > 0:
            need = cnt[card]
            for i in range(groupSize):
                if cnt[card + i] < need: return False
                cnt[card + i] -= need
    return True
```

## 847. Shortest Path Visiting All Nodes

https://leetcode.cn/problems/shortest-path-visiting-all-nodes/

BFS + bitmask 状态压缩。

```python
from collections import deque
def shortestPathLength(self, graph):
    n = len(graph)
    full = (1 << n) - 1
    q = deque([(i, 1 << i, 0) for i in range(n)])
    visited = {(i, 1 << i) for i in range(n)}
    while q:
        node, mask, dist = q.popleft()
        if mask == full: return dist
        for nb in graph[node]:
            nm = mask | (1 << nb)
            if (nb, nm) not in visited:
                visited.add((nb, nm))
                q.append((nb, nm, dist + 1))
    return -1
```

```python
from collections import deque
def kSimilarity(self, s1, s2):
    q = deque([(s1, 0)]); visited = {s1}
    while q:
        s, k = q.popleft()
        if s == s2: return k
        i = 0
        while s[i] == s2[i]: i += 1
        for j in range(i + 1, len(s)):
            if s[j] == s2[i] and s[j] != s2[j]:
                ns = list(s); ns[i], ns[j] = ns[j], ns[i]; ns = ''.join(ns)
                if ns not in visited: visited.add(ns); q.append((ns, k+1))
```

## 856. Score of Parentheses

栈模拟：`()` 得 1 分，`(A)` 得 `2*A` 分，`AB` 得 `A+B` 分。

```python
def scoreOfParentheses(self, s):
    stack = [0]
    for c in s:
        if c == '(':
            stack.append(0)
        else:
            v = stack.pop()
            stack[-1] += max(2 * v, 1)
    return stack[0]
```

```python
import heapq
def mincostToHireWorkers(self, quality, wage, k):
    workers = sorted((w/q, q) for w, q in zip(wage, quality))
    heap = []; qsum = 0; res = float('inf')
    for ratio, q in workers:
        heapq.heappush(heap, -q); qsum += q
```

## 862. Shortest Subarray with Sum at Least K

https://leetcode.cn/problems/shortest-subarray-with-sum-at-least-k/

前缀和 + 单调双端队列。

```python
from collections import deque
def shortestSubarray(self, nums, k):
    n = len(nums)
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + nums[i]
    res = float('inf')
    dq = deque()
    for i in range(n + 1):
        while dq and prefix[i] - prefix[dq[0]] >= k:
            res = min(res, i - dq.popleft())
        while dq and prefix[i] <= prefix[dq[-1]]:
            dq.pop()
        dq.append(i)
    return res if res != float('inf') else -1
```

```python
from collections import deque, defaultdict
def distanceK(self, root, target, k):
    graph = defaultdict(list)
    def build(node, parent):
        if not node: return
        if parent:
            graph[node].append(parent)
            graph[parent].append(node)
        build(node.left, node)
        build(node.right, node)
    build(root, None)
    visited = {target}
    q = deque([target])
    for _ in range(k):
        for _ in range(len(q)):
            node = q.popleft()
            for nb in graph[node]:
                if nb not in visited:
                    visited.add(nb)
                    q.append(nb)
    return [n.val for n in q]
```

## 895. Maximum Frequency Stack

https://leetcode.cn/problems/maximum-frequency-stack/

频率栈：按频率分组，弹出最高频率最近入栈的。

```python
from collections import defaultdict
class FreqStack:
    def __init__(self):
        self.freq = defaultdict(int)
        self.group = defaultdict(list)
        self.max_freq = 0
    def push(self, val):
        self.freq[val] += 1
        f = self.freq[val]
        self.max_freq = max(self.max_freq, f)
        self.group[f].append(val)
    def pop(self):
        val = self.group[self.max_freq].pop()
        self.freq[val] -= 1
        if not self.group[self.max_freq]:
            self.max_freq -= 1
        return val
```

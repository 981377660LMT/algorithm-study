```python
def longestNiceSubarray(self, nums):
    res = left = 0; used = 0
    for right in range(len(nums)):
        while used & nums[right]:
            used ^= nums[left]; left += 1
        used |= nums[right]
        res = max(res, right - left + 1)
    return res
```

## 2402. Meeting Rooms III

https://leetcode.cn/problems/meeting-rooms-iii/

```python
import heapq
def mostBooked(self, n, meetings):
    meetings.sort()
    free = list(range(n))  # 空闲房间（最小堆）
    busy = []  # (结束时间, 房间号)
    cnt = [0] * n
    for start, end in meetings:
        while busy and busy[0][0] <= start:
            _, room = heapq.heappop(busy)
            heapq.heappush(free, room)
        if free:
            room = heapq.heappop(free)
            heapq.heappush(busy, (end, room))
        else:
            end_time, room = heapq.heappop(busy)
            heapq.heappush(busy, (end_time + end - start, room))
        cnt[room] += 1
    return cnt.index(max(cnt))
```

```python
from collections import deque
def reverseOddLevels(self, root):
    q = deque([root]); level = 0
    while q:
        nodes = list(q)
        if level % 2:
            vals = [n.val for n in nodes]
            for i, n in enumerate(nodes):
                n.val = vals[~i]
        for n in nodes:
            if n.left: q.append(n.left)
            if n.right: q.append(n.right)
        level += 1
    return root
```

```python
def goodIndices(self, nums, k):
```

## 2421. Number of Good Paths

https://leetcode.cn/problems/number-of-good-paths/

并查集，按值从小到大加点。

```python
from collections import defaultdict, Counter
def numberOfGoodPaths(self, vals, edges):
    n = len(vals)
    parent = list(range(n)); rank = [0] * n
    def find(x):
        while parent[x] != x: parent[x] = parent[parent[x]]; x = parent[x]
        return x
    def union(a, b):
        pa, pb = find(a), find(b)
        if pa == pb: return
        if rank[pa] < rank[pb]: pa, pb = pb, pa
        parent[pb] = pa
        if rank[pa] == rank[pb]: rank[pa] += 1

    val_nodes = defaultdict(list)
    for i, v in enumerate(vals): val_nodes[v].append(i)

    adj = defaultdict(list)
    for u, v in edges: adj[u].append(v); adj[v].append(u)

    res = n  # 每个单点算一条
    for val in sorted(val_nodes):
        for node in val_nodes[val]:
            for nb in adj[node]:
                if vals[nb] <= val: union(node, nb)
        cnt = Counter(find(node) for node in val_nodes[val])
        for c in cnt.values():
            res += c * (c - 1) // 2
    return res
```

## 2430. Maximum Deletions on a String

```python
def deleteString(self, s):
    n = len(s)
    # lcp[i][j] = longest common prefix of s[i:] and s[j:]
    lcp = [[0] * (n + 1) for _ in range(n + 1)]
    for i in range(n-1, -1, -1):
        for j in range(n-1, -1, -1):
            if s[i] == s[j]: lcp[i][j] = lcp[i+1][j+1] + 1
    dp = [1] * n
    for i in range(n-1, -1, -1):
        for l in range(1, (n - i) // 2 + 1):
            if lcp[i][i+l] >= l:
                dp[i] = max(dp[i], dp[i+l] + 1)
    return dp[0]
```

## 2440. Create Components With Same Value

https://leetcode.cn/problems/create-components-with-same-value/

枚举切几刀，DFS 验证。

```python
from collections import defaultdict
def componentValue(self, nums, edges):
    n = len(nums); total = sum(nums)
    adj = defaultdict(list)
    for u, v in edges: adj[u].append(v); adj[v].append(u)

    def dfs(u, parent, target):
        s = nums[u]
        for v in adj[u]:
            if v == parent:continue
            child = dfs(v, u, target)
            if child < 0: return -1
            s += child
        if s == target: return 0
        if s > target: return -1
        return s

    for k in range(n, 0, -1):
        if total % k: continue
        target = total // k
        if dfs(0, -1, target) == 0:
            return k - 1
    return 0
```

```python
import heapq
def secondGreaterElement(self, nums):
    n = len(nums); res = [-1] * n
    stk = []; waiting = []  # min-heap
    for i in range(n):
        while waiting and waiting[0][0] < nums[i]:
            _, idx = heapq.heappop(waiting)
            res[idx] = nums[i]
        buf = []
        while stk and nums[stk[-1]] < nums[i]:
```

```python
    height = {}; depth = {}
    def get_height(node):
        if not node: return -1
        height[node.val] = 1 + max(get_height(node.left), get_height(node.right))
        return height[node.val]
    def get_depth(node, d=0):
        if not node: return
        depth[node.val] = d
        get_depth(node.left, d+1); get_depth(node.right, d+1)
    get_height(root); get_depth(root)

    # 每层按子树高度排序，取 top2
    level_top = defaultdict(list)
    for val in height:
        d = depth[val]; h = height[val]
        level_top[d].append((h, val))
        level_top[d].sort(reverse=True)
        if len(level_top[d]) > 2: level_top[d] = level_top[d][:2]

    res = []
    for q in queries:
        d = depth[q]
        top = level_top[d]
        if len(top) == 1:
            res.append(d - 1)
        elif top[0][1] == q:
            res.append(d + top[1][0])
        else:
            res.append(d + top[0][0])
    return res
```

## 2467. Most Profitable Path in a Tree

```python

def mostProfitablePath(self, edges, bob, amount):
    n = len(amount)
    adj = defaultdict(list)
    for u, v in edges: adj[u].append(v); adj[v].append(u)
    # BFS 找 bob 到 0 的路径
    parent = [-1] * n; visited = [False] * n
    q = deque([0]); visited[0] = True
    while q:
        u = q.popleft()
        for v in adj[u]:
            if not visited[v]:
                visited[v] = True; parent[v] = u; q.append(v)
    bob_time = [n] * n; t = 0; cur = bob
    while cur != -1:
        bob_time[cur] = t; t += 1; cur = parent[cur]

    # DFS Alice
    res = float('-inf')
    stk = [(0, -1, 0, 0)]  # node, parent, time, income
    while stk:
        u, par, t, inc = stk.pop()
        if t < bob_time[u]: inc += amount[u]
        elif t == bob_time[u]: inc += amount[u] // 2
        is_leaf = True
        for v in adj[u]:
            if v != par:
                is_leaf = False; stk.append((v, u, t+1, inc))
        if is_leaf: res = max(res, inc)
    return res
```

```python
    return res
```

## 3503. Longest Palindrome After Substring Concatenation I / 3504. Longest Palindrome After Substring Concatenation II

https://leetcode.cn/problems/longest-palindrome-after-substring-concatenation-ii/

```python
def longestPalindrome(self, s, t):
    t_rev = t[::-1]
    res = 1
    n, m = len(s), len(t)
    # 枚举 s 的子串 + t 的子串拼接后是回文
    # 优化: s[i:j] + t[k:l] 构成回文
    # 等价: s 的某前缀 = t 某后缀的反转, 然后中间可以扩展回文

    def isPalin(x):
        return x == x[::-1]

    # 暴力优化版: O(n*m + n^2 + m^2)
    # 1. 纯 s 子串回文
    for i in range(n):
        for j in range(i+1, n+1):
            if isPalin(s[i:j]): res = max(res, j-i)
    # 2. 纯 t 子串回文
    for i in range(m):
        for j in range(i+1, m+1):
            if isPalin(t[i:j]): res = max(res, j-i)
    # 3. s[i:j] + t[k:l] 是回文
    # s[i:j] 匹配 t[k:l] 的反转前缀, 中间部分是回文
    for i in range(n):
        for k in range(m-1, -1, -1):
            # 匹配 s[i:] 与 t_rev[m-1-k:]
            L = 0
            while i+L < n and m-1-k+L < m and s[i+L] == t_rev[m-1-k+L]:
                L += 1
            if L == 0: continue
            # s[i:i+L] + t[k+1-L+1:k+1] 两端匹配
            # 中间可以扩展 s 或 t 的回文
            # 找 s[i+L:] 中最长回文前缀
            best_mid = 0
            for end in range(n, i+L-1, -1):
                if isPalin(s[i+L:end]):
                    best_mid = max(best_mid, end-i-L); break
            # 找 t[:k+1-L] 中最长回文后缀
            for start in range(0, k+1-L+1):
                if isPalin(t[start:k+1-L]):
                    best_mid = max(best_mid, k+1-L-start); break
            res = max(res, 2*L + best_mid)
    return res
```

```python
def minOperations(self, nums, k):
    nums.sort(); n = len(nums)
    # 中位数策略: 将所有元素调到某个范围 [x, x+k]
    # 滑动窗口: 排序后找最少调整使 max-min <= k
    # 贪心: 二分 or 中位数
    # 答案 = 对每个窗口 [i, n-1], 将 nums[0:i] 调到 nums[i]-k 以上

        for x in nums:
            if x < v: ops += v - x
            elif x > v + k: ops += x - v - k
        res = min(res, ops)
    # 也检查 v = nums[i] - k
    for i in range(n):
        v = nums[i] - k
        if v < 0: v = 0
        ops = 0
        for x in nums:
            if x < v: ops += v - x
            elif x > v + k: ops += x - v - k
```

```python
def uniqueXorTriplets(self, nums):
    n = len(nums)
    # 先求所有 pair XOR
    pair_xor = set()
    for i in range(n):
        for j in range(i, n):
            pair_xor.add(nums[i] ^ nums[j])
    # 再与所有元素 XOR
    res = set()
    for x in pair_xor:
        for y in nums:
            res.add(x ^ y)
    return len(res)
```

## 3519. Count Numbers with Non-Decreasing Digits

https://leetcode.cn/problems/count-numbers-with-non-decreasing-digits/

数位 DP。

```python
def countNumbers(self, l, r, b):
    def convert(n, base):
        if n == 0: return [0]
        digits = []
        while n: digits.append(n % base); n //= base
        return digits[::-1]

    def count(n, base):
        if n < 0: return 0
        digits = convert(n, base)
        from functools import lru_cache
        @lru_cache(None)
        def dp(i, last, tight, started):
            if i == len(digits): return 1 if started else 1  # 包含0
            limit = digits[i] if tight else base - 1
            res = 0
            for d in range(0, limit + 1):
                if started and d < last: continue
                res += dp(i + 1, d if started or d > 0 else 0,
                         tight and d == limit, started or d > 0)
            return res
        return dp(0, 0, True, False)

    MOD = 10**9 + 7
    return (count(int(r), b) - count(int(l) - 1, b)) % MOD
```

注意：l, r 为字符串表示的大数，需要用字符串减法。

```python
def resultArray(self, nums, k):
    n = len(nums)
    # dp[r] = 以当前位置结尾, 乘积 mod k == r 的子数组数
    res = [0] * k
    dp = [0] * k
    for x in nums:
        new_dp = [0] * k
        # 扩展已有子数组
        for r in range(k):
            if dp[r] > 0:
                new_dp[(r * x) % k] += dp[r]
        # 新子数组只含 x
        new_dp[x % k] += 1
        for r in range(k):
            res[r] += new_dp[r]
        dp = new_dp
    return res
```

```python
from collections import defaultdict, deque
def baseUnitConversions(self, conversions):
    MOD = 10**9 + 7; n = len(conversions) + 1
    adj = defaultdict(list)
    for s, t, f in conversions:
        adj[s].append((t, f))
    res = [0] * n; res[0] = 1
    visited = [False] * n; visited[0] = True
    q = deque([0])
    while q:
        u = q.popleft()
        for v, f in adj[u]:
            if not visited[v]:
                res[v] = res[u] * f % MOD
                visited[v] = True; q.append(v)
    return res
```

## 3529. Count Cells in Overlapping Horizontal and Vertical Substrings

```python
def countCells(self, grid, pattern):
    m, n = len(grid), len(grid[0])
    # KMP 找 pattern 在横向和纵向字符串中的所有出现位置
    def kmp_search(text, pat):
        lps = [0] * len(pat); k = 0
        for i in range(1, len(pat)):
            while k and pat[k] != pat[i]: k = lps[k-1]
            if pat[k] == pat[i]: k += 1
            lps[i] = k
        res = []; k = 0
        for i in range(len(text)):
            while k and pat[k] != text[i]: k = lps[k-1]
            if pat[k] == text[i]: k += 1
            if k == len(pat): res.append(i - len(pat) + 1); k = lps[k-1]
        return res

    h_cells = set(); v_cells = set()
    p = len(pattern)
    # 横向
    for i in range(m):
        row = ''.join(grid[i])
        for start in kmp_search(row, pattern):
            for j in range(start, start + p):
                h_cells.add((i, j))
    # 纵向
    for j in range(n):
        col = ''.join(grid[i][j] for i in range(m))
        for start in kmp_search(col, pattern):
            for i in range(start, start + p):
                v_cells.add((i, j))
    return len(h_cells & v_cells)
```

```python
def countCoveredBuildings(self, n, buildings):
    from collections import defaultdict
    row_min = defaultdict(lambda: float('inf'))
    row_max = defaultdict(lambda: float('-inf'))
    col_min = defaultdict(lambda: float('inf'))
    col_max = defaultdict(lambda: float('-inf'))
    for x, y in buildings:
        row_min[x] = min(row_min[x], y)
        row_max[x] = max(row_max[x], y)
        col_min[y] = min(col_min[y], x)
        col_max[y] = max(col_max[y], x)
    cnt = 0
    for x, y in buildings:
        if (row_min[x] < y and row_max[x] > y and
            col_min[y] < x and col_max[y] > x):
            cnt += 1
    return cnt
```

```python
def pathExistenceQueries(self, n, nums, maxDiff, queries):
    # 排序后相邻差 <= maxDiff 则连通
    idx = sorted(range(n), key=lambda i: nums[i])
    comp = [0] * n; c = 0
    comp[idx[0]] = 0
    for i in range(1, n):
        if nums[idx[i]] - nums[idx[i-1]] > maxDiff: c += 1
        comp[idx[i]] = c
    return [comp[u] == comp[v] for u, v in queries]
```

```python
def specialGrid(self, n):
    if n == 0: return [[0]]
    prev = self.specialGrid(n - 1)
    sz = len(prev); total = sz * sz
    # 4象限: 右上(+0), 右下(+total), 左下(+2*total), 左上(+3*total)

            grid[i][j] = prev[i][j] + 3*total          # 左上
    return grid
```

```python
def minOperations(self, nums):
    # 单调栈: 每个非零元素贡献一次操作(如果它没被之前更大的覆盖)
    stack = [0]; ops = 0
    for x in nums:
        while stack and stack[-1] > x: stack.pop()
        if stack and stack[-1] == x: continue
        if x > 0: ops += 1
        stack.append(x)
    return ops
```

```python
def canPartitionGrid(self, grid):
    m, n = len(grid), len(grid[0])
    total = sum(sum(row) for row in grid)
    if total % 2 != 0: return False
    half = total // 2
    # 水平切
    cur = 0

        cur += sum(grid[i])
        if cur == half: return True
```

```python
def minSwaps(self, nums):
    def dsum(x): return sum(int(d) for d in str(x))
    n = len(nums)
    target = sorted(range(n), key=lambda i: (dsum(nums[i]), nums[i]))
    visited = [False]*n; swaps = 0
    for i in range(n):
        if visited[i] or target[i] == i: continue
        cycle = 0; j = i
```

https://leetcode.cn/problems/minimum-cost-path-with-alternating-directions-i/

```python
def minPathCost(self, grid):
    m, n = len(grid), len(grid[0])
    # 交替右和下移动, 从(0,0)到(m-1,n-1)
    # 必须 m+n-2 步, 交替 R 和 D
    INF = float('inf')
    # dp[i][j][d] = 到达(i,j)最后一步方向为d的最小代价
    dp = [[[INF]*2 for _ in range(n)] for _ in range(m)]
    dp[0][0][0] = dp[0][0][1] = grid[0][0]
    for i in range(m):
        for j in range(n):
            # last = 0 (right), try down next
            if i+1 < m and dp[i][j][0] < INF:
                dp[i+1][j][1] = min(dp[i+1][j][1], dp[i][j][0] + grid[i+1][j])
            # last = 1 (down), try right next
            if j+1 < n and dp[i][j][1] < INF:
                dp[i][j+1][0] = min(dp[i][j+1][0], dp[i][j][1] + grid[i][j+1])
            # last = 0, try right? No, must alternate
            # 初始可以选任意方向
            if i+1 < m and dp[i][j][1] < INF:  # 重复? 不
                pass  # 已处理
    return min(dp[m-1][n-1])
```

## 3598. Longest Common Prefix Between Adjacent Strings After Removals

https://leetcode.cn/problems/longest-common-prefix-between-adjacent-strings-after-removals/

```python
def longestCommonPrefix(self, words):
    n = len(words)
    # lcp[i] = LCP(words[i], words[i+1])
    lcp = [0] * (n - 1)
    for i in range(n - 1):
        j = 0
        while j < len(words[i]) and j < len(words[i+1]) and words[i][j] == words[i+1][j]:
            j += 1
        lcp[i] = j
    # 删除第 k 个后, 新相邻对: (k-1, k+1) 替代 (k-1,k) 和 (k,k+1)
    # 原始 max_lcp = max(lcp)
    # 预计算 lcp_skip[k] = LCP(words[k-1], words[k+1])
    res = []
    orig_max = max(lcp) if lcp else 0
    for k in range(n):
        cur_max = 0
        # 不含 k 的所有相邻 lcp
        for i in range(n - 1):
            if i == k - 1 or i == k: continue
            cur_max = max(cur_max, lcp[i])
        # 新的相邻对 (k-1, k+1)
        if 0 < k < n - 1:
            j = 0
            while j < len(words[k-1]) and j < len(words[k+1]) and words[k-1][j] == words[k+1][j]:
                j += 1
            cur_max = max(cur_max, j)
        res.append(cur_max)
    return res
```

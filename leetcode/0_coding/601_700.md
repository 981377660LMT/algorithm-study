```python
def triangleNumber(self, nums):
    nums.sort()
    res = 0
    for k in range(2, len(nums)):
        i, j = 0, k - 1
        while i < j:
            if nums[i] + nums[j] > nums[k]:
                res += j - i; j -= 1
            else: i += 1
    return res
```

```python
from collections import Counter
def leastInterval(self, tasks, n):
    cnt = Counter(tasks)
    max_freq = max(cnt.values())
    max_count = sum(1 for v in cnt.values() if v == max_freq)
```

```python
def exclusiveTime(self, n, logs):
    res = [0] * n
    stack = []
    prev_time = 0
    for log in logs:
        parts = log.split(':')
        fid, typ, time = int(parts[0]), parts[1], int(parts[2])
        if typ == 'start':
            if stack:
                res[stack[-1]] += time - prev_time
            stack.append(fid)
```

```python
def shoppingOffers(self, price, special, needs):
    def dfs(needs):
        cost = sum(n * p for n, p in zip(needs, price))
        for offer in special:
            nn = [n - o for n, o in zip(needs, offer[:-1])]
            if all(x >= 0 for x in nn):
```

```python
import re
def solveEquation(self, equation):
    def parse(s):
        coef = val = 0
        for t in re.findall(r'[+-]?[^+-]+', s):
            if 'x' in t:
                t = t.replace('x', '')
                coef += int(t) if t not in ('', '+', '-') else (1 if t != '-' else -1)
```

```python
from collections import deque
def widthOfBinaryTree(self, root):
    if not root: return 0
    res = 0
    q = deque([(root, 0)])
    while q:
        _, left_idx = q[0]
        _, right_idx = q[-1]
        res = max(res, right_idx - left_idx + 1)
        for _ in range(len(q)):
            node, idx = q.popleft()
            if node.left: q.append((node.left, 2 * idx))
            if node.right: q.append((node.right, 2 * idx + 1))
    return res
```

```python
def strangePrinter(self, s):
    s = ''.join(c for i, c in enumerate(s) if i == 0 or c != s[i-1])
    n = len(s)
    dp = [[0]*n for _ in range(n)]
    for i in range(n): dp[i][i] = 1
    for l in range(2, n+1):
        for i in range(n - l + 1):
            j = i + l - 1
            dp[i][j] = dp[i][j-1] + 1
            for k in range(i, j):
                if s[k] == s[j]:
                    dp[i][j] = min(dp[i][j], dp[i][k] + (dp[k+1][j-1] if k+1 <= j-1 else 0))
    return dp[0][n-1]
```

## 675. Cut Off Trees for Golf Event

按高度排序后依次 BFS 求最短路。

```python
from collections import deque
def cutOffTree(self, forest):
    trees = sorted((h, r, c) for r, row in enumerate(forest)
                   for c, h in enumerate(row) if h > 1)
    m, n = len(forest), len(forest[0])
    def bfs(sr, sc, tr, tc):
        if (sr, sc) == (tr, tc): return 0
        visited = {(sr, sc)}
        q = deque([(sr, sc, 0)])
        while q:
            r, c, d = q.popleft()
            for dr, dc in [(-1,0),(1,0),(0,-1),(0,1)]:
                nr, nc = r+dr, c+dc
                if 0<=nr<m and 0<=nc<n and (nr,nc) not in visited and forest[nr][nc]:
                    if (nr, nc) == (tr, tc): return d + 1
                    visited.add((nr, nc)); q.append((nr, nc, d+1))
        return -1
    sr = sc = total = 0
    for _, tr, tc in trees:
        d = bfs(sr, sc, tr, tc)
        if d == -1: return -1
        total += d; sr, sc = tr, tc
    return total
```

```python
    def __init__(self): self.d = {}; self.trie = {}
    def insert(self, key, val):
        delta = val - self.d.get(key, 0); self.d[key] = val
        node = self.trie
        for c in key:
            node = node.setdefault(c, {'_sum': 0})
            node['_sum'] += delta
    def sum(self, prefix):
        node = self.trie
        for c in prefix:
            if c not in node: return 0
            node = node[c]
        return node['_sum']
```

```python
def validPalindrome(self, s):
    def check(l, r):
        while l < r:
            if s[l] != s[r]: return False
            l += 1; r -= 1
        return True
    l, r = 0, len(s) - 1
    while l < r:
        if s[l] != s[r]:
            return check(l + 1, r) or check(l, r - 1)
        l += 1; r -= 1
    return True
```

```python
def findRedundantConnection(self, edges):

        while parent[x] != x:
            parent[x] = parent[parent[x]]
            x = parent[x]
```

```python
def knightProbability(self, n, k, row, column):
    dp = {(row, column): 1.0}
    moves = [(-2,-1),(-2,1),(-1,-2),(-1,2),(1,-2),(1,2),(2,-1),(2,1)]
```

## 689. Maximum Sum of 3 Non-Overlapping Subarrays

```python
def maxSumOfThreeSubarrays(self, nums, k):
    n = len(nums)
    sums = [sum(nums[:k])]
    for i in range(k, n): sums.append(sums[-1] + nums[i] - nums[i-k])
    left = [0] * len(sums); best = 0
    for i in range(len(sums)):
        if sums[i] > sums[best]: best = i
        left[i] = best
    right = [0] * len(sums); best = len(sums) - 1
    for i in range(len(sums) - 1, -1, -1):
        if sums[i] >= sums[best]: best = i
        right[i] = best
    res = None; max_sum = 0
    for j in range(k, len(sums) - k):
        i, l = left[j - k], right[j + k]
        total = sums[i] + sums[j] + sums[l]
        if total > max_sum: max_sum = total; res = [i, j, l]
    return res
```

```python
from functools import lru_cache
from collections import Counter
def minStickers(self, stickers, target):
    sc = [Counter(s) for s in stickers]
    @lru_cache(maxsize=None)
    def dp(rem):
        if not rem: return 0
        res = float('inf')
        for s in sc:
            if rem[0] not in s: continue
            nxt = rem
            for c, n in s.items(): nxt = nxt.replace(c, '', n)
            if nxt != rem: res = min(res, 1 + dp(nxt))
        return res
    ans = dp(target)
    return ans if ans < float('inf') else -1
```

## 698. Partition to K Equal Sum Subsets
https://leetcode.com/problems/partition-to-k-equal-sum-subsets/

回溯 + 剪枝：将数组分成 k 个和相等的子集。

```python
def canPartitionKSubsets(self, nums, k):
    total = sum(nums)
    if total % k: return False
    target = total // k
    nums.sort(reverse=True)
    if nums[0] > target: return False
    buckets = [0] * k
    def bt(idx):
        if idx == len(nums): return all(b == target for b in buckets)
        for i in range(k):
            if buckets[i] + nums[idx] <= target:
                buckets[i] += nums[idx]
                if bt(idx + 1): return True
                buckets[i] -= nums[idx]
            if buckets[i] == 0: break  # 剪枝：空桶等价
        return False
    return bt(0)
```

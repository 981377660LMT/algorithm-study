## 301. Remove Invalid Parentheses
https://leetcode.com/problems/remove-invalid-parentheses/

BFS 逐层删一个括号，找到最少删除数。

```python
def removeInvalidParentheses(self, s):
    def valid(s):
        count = 0
        for c in s:
            if c == '(': count += 1
            elif c == ')': count -= 1
            if count < 0: return False
        return count == 0
    level = {s}
    while True:
        valid_set = list(filter(valid, level))
        if valid_set: return valid_set
        next_level = set()
        for item in level:
            for i in range(len(item)):
                if item[i] in '()':
                    next_level.add(item[:i] + item[i+1:])
        level = next_level
```

## 306. Additive Number
https://leetcode.com/problems/additive-number/

回溯枚举前两个数，验证后续是否满足加法序列。

```python
def isAdditiveNumber(self, num):
    n = len(num)
    def check(i, j):
        a, b = num[:i], num[i:j]
        if (len(a) > 1 and a[0] == '0') or (len(b) > 1 and b[0] == '0'):
            return False
        while j < n:
            c = str(int(a) + int(b))
            if not num.startswith(c, j): return False
            j += len(c)
            a, b = b, c
        return j == n
    for i in range(1, n):
        for j in range(i + 1, n):
            if check(i, j): return True
    return False
```

## 307. Range Sum Query - Mutable
https://leetcode.com/problems/range-sum-query-mutable/

树状数组 (Binary Indexed Tree)。

```python
class NumArray:
    def __init__(self, nums):
        self.n = len(nums)
        self.nums = [0] * self.n
        self.bit = [0] * (self.n + 1)
        for i, x in enumerate(nums):
            self.update(i, x)
    def _add(self, i, delta):
        i += 1
        while i <= self.n:
            self.bit[i] += delta
            i += i & (-i)
    def _prefix(self, i):
        s = 0; i += 1
        while i > 0:
            s += self.bit[i]
            i -= i & (-i)
        return s
    def update(self, index, val):
        self._add(index, val - self.nums[index])
        self.nums[index] = val
    def sumRange(self, left, right):
        return self._prefix(right) - (self._prefix(left - 1) if left > 0 else 0)
```

## 310. Minimum Height Trees
https://leetcode.com/problems/minimum-height-trees/

拓扑剥叶子，剩 1-2 个即为根。

```python
from collections import defaultdict, deque
def findMinHeightTrees(self, n, edges):
    if n <= 2: return list(range(n))
    g = defaultdict(set)
    for u, v in edges:
        g[u].add(v); g[v].add(u)
    leaves = deque(i for i in range(n) if len(g[i]) == 1)
    remaining = n
    while remaining > 2:
        remaining -= len(leaves)
        new_leaves = deque()
        for leaf in leaves:
            nb = g[leaf].pop()
            g[nb].remove(leaf)
            if len(g[nb]) == 1: new_leaves.append(nb)
        leaves = new_leaves
    return list(leaves)
```

## 312. Burst Balloons
https://leetcode.com/problems/burst-balloons/

区间 DP：`dp[l][r]` = 戳破 `(l, r)` 之间气球的最大得分，最后戳 k。

```python
def maxCoins(self, nums):
    nums = [1] + nums + [1]
    n = len(nums)
    dp = [[0] * n for _ in range(n)]
    for length in range(2, n):
        for l in range(n - length):
            r = l + length
            for k in range(l + 1, r):
                dp[l][r] = max(dp[l][r],
                    dp[l][k] + dp[k][r] + nums[l] * nums[k] * nums[r])
    return dp[0][n - 1]
```

## 315. Count of Smaller Numbers After Self
https://leetcode.com/problems/count-of-smaller-numbers-after-self/

归并排序统计逆序对。

```python
def countSmaller(self, nums):
    n = len(nums)
    res = [0] * n
    indices = list(range(n))
    def merge_sort(arr):
        if len(arr) <= 1: return arr
        mid = len(arr) // 2
        left = merge_sort(arr[:mid])
        right = merge_sort(arr[mid:])
        merged = []
        i = j = 0
        while i < len(left) and j < len(right):
            if nums[left[i]] <= nums[right[j]]:
                res[left[i]] += j
                merged.append(left[i]); i += 1
            else:
                merged.append(right[j]); j += 1
        while i < len(left):
            res[left[i]] += j
            merged.append(left[i]); i += 1
        while j < len(right):
            merged.append(right[j]); j += 1
        return merged
    merge_sort(indices)
    return res
```

## 316. Remove Duplicate Letters

```python
def removeDuplicateLetters(self, s):
    last = {c: i for i, c in enumerate(s)}
    stack = []
    seen = set()
    for i, c in enumerate(s):
        if c in seen: continue
        while stack and c < stack[-1] and last[stack[-1]] > i:
            seen.remove(stack.pop())
        stack.append(c)
        seen.add(c)
    return ''.join(stack)
```

## 318. Maximum Product of Word Lengths
https://leetcode.com/problems/maximum-product-of-word-lengths/

用位掩码表示字母集，`&` 判断无公共字母。

```python
def maxProduct(self, words):
    masks = [0] * len(words)
    for i, w in enumerate(words):
        for c in w:
            masks[i] |= 1 << (ord(c) - ord('a'))
    res = 0
    for i in range(len(words)):
        for j in range(i + 1, len(words)):
            if masks[i] & masks[j] == 0:
                res = max(res, len(words[i]) * len(words[j]))
    return res
```

```python
def coinChange(self, coins, amount):
    dp = [float('inf')] * (amount + 1)
    dp[0] = 0
    for i in range(1, amount + 1):
        for c in coins:
            if c <= i:
                dp[i] = min(dp[i], dp[i - c] + 1)
    return dp[amount] if dp[amount] != float('inf') else -1
```

## 329. Longest Increasing Path in a Matrix
https://leetcode.com/problems/longest-increasing-path-in-a-matrix/

DFS + 记忆化。

```python
def longestIncreasingPath(self, matrix):
    m, n = len(matrix), len(matrix[0])
    memo = {}
    def dfs(i, j):
        if (i, j) in memo: return memo[(i, j)]
        res = 1
        for di, dj in [(-1,0),(1,0),(0,-1),(0,1)]:
            ni, nj = i+di, j+dj
            if 0 <= ni < m and 0 <= nj < n and matrix[ni][nj] > matrix[i][j]:
                res = max(res, 1 + dfs(ni, nj))
        memo[(i, j)] = res
        return res
    return max(dfs(i, j) for i in range(m) for j in range(n))
```

```python
def palindromePairs(self, words):
    d = {w[::-1]: i for i, w in enumerate(words)}
    res = []
    for i, w in enumerate(words):
        for j in range(len(w) + 1):
            prefix, suffix = w[:j], w[j:]
            if prefix in d and d[prefix] != i and suffix == suffix[::-1]:
                res.append([i, d[prefix]])
            if j > 0 and suffix in d and d[suffix] != i and prefix == prefix[::-1]:
                res.append([d[suffix], i])
    return res
```

```python
def rob(self, root):
    def dfs(node):
        if not node: return (0, 0)
        l, r = dfs(node.left), dfs(node.right)
        rob_it = node.val + l[1] + r[1]
        skip_it = max(l) + max(r)
        return (rob_it, skip_it)
    return max(dfs(root))
```

```python
def maxEnvelopes(self, envelopes):
    envelopes.sort(key=lambda x: (x[0], -x[1]))
    tails = []
```

```python
class Twitter:

        self.time = 0
        self.tweets = defaultdict(list)
```

```python
import heapq
def kSmallestPairs(self, nums1, nums2, k):
    if not nums1 or not nums2: return []
    h = [(nums1[i] + nums2[0], i, 0) for i in range(min(k, len(nums1)))]

    return res
```

```python
def kthSmallest(self, matrix, k):
    n = len(matrix)
    lo, hi = matrix[0][0], matrix[-1][-1]
    while lo < hi:
        mid = (lo + hi) // 2
        count = 0
        j = n - 1
        for i in range(n):
            while j >= 0 and matrix[i][j] > mid: j -= 1
            count += j + 1
        if count < k: lo = mid + 1
        else: hi = mid
    return lo
```

```python
import random
class RandomizedSet:
    def __init__(self):
        self.vals = []
        self.idx = {}
    def insert(self, val):
        if val in self.idx: return False
        self.idx[val] = len(self.vals)
        self.vals.append(val)
        return True
    def remove(self, val):
        if val not in self.idx: return False
        i = self.idx[val]
        last = self.vals[-1]
        self.vals[i] = last
        self.idx[last] = i
        self.vals.pop()
        del self.idx[val]
        return True
    def getRandom(self):
        return random.choice(self.vals)
```

```python
    def __init__(self, nums):
        self.nums = nums
    def pick(self, target):
        count = 0
        res = 0
        for i, x in enumerate(self.nums):
            if x == target:
                count += 1
                if random.randint(1, count) == 1:
                    res = i
        return res
```

## 399. Evaluate Division
https://leetcode.com/problems/evaluate-division/

图 + BFS/DFS：a/b=k 建边 a→b 权重 k，b→a 权重 1/k。

```python
from collections import defaultdict, deque
def calcEquation(self, equations, values, queries):
    graph = defaultdict(dict)
    for (a, b), v in zip(equations, values):
        graph[a][b] = v
        graph[b][a] = 1.0 / v
    def bfs(src, dst):
        if src not in graph or dst not in graph: return -1.0
        if src == dst: return 1.0
        visited = {src}
        q = deque([(src, 1.0)])
        while q:
            node, val = q.popleft()
            if node == dst: return val
            for nb, w in graph[node].items():
                if nb not in visited:
                    visited.add(nb)
                    q.append((nb, val * w))
        return -1.0
    return [bfs(a, b) for a, b in queries]
```

## 400. Nth Digit
https://leetcode.com/problems/nth-digit/

按位数区间定位。

```python
def findNthDigit(self, n):
    digits, count, start = 1, 9, 1
    while n > digits * count:
        n -= digits * count
        digits += 1; count *= 10; start *= 10
    num = start + (n - 1) // digits
    return int(str(num)[(n - 1) % digits])
```

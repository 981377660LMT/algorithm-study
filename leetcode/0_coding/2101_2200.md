## 2101. Detonate the Maximum Bombs

https://leetcode.cn/problems/detonate-the-maximum-bombs/

建图 + BFS/DFS。

```python
from collections import deque
def maximumDetonation(self, bombs):
    n = len(bombs); graph = [[] for _ in range(n)]
    for i in range(n):
        for j in range(n):
            if i == j: continue
            dx = bombs[i][0] - bombs[j][0]
            dy = bombs[i][1] - bombs[j][1]
            if dx*dx + dy*dy <= bombs[i][2]**2:
                graph[i].append(j)
    res = 0
    for i in range(n):
        visited = {i}; q = deque([i])
        while q:
            u = q.popleft()
            for v in graph[u]:
                if v not in visited:
                    visited.add(v); q.append(v)
        res = max(res, len(visited))
    return res
```

```python
def minimumRefill(self, plants, capacityA, capacityB):
    a, b = capacityA, capacityB
    l, r = 0, len(plants) - 1; refills = 0
    while l < r:
        if a < plants[l]: refills += 1; a = capacityA
        a -= plants[l]
```

## 2115. Find All Possible Recipes from Given Supplies

```python
from collections import deque, defaultdict
def findAllRecipes(self, recipes, ingredients, supplies):
    indeg = defaultdict(int)
    graph = defaultdict(list)
    recipe_set = set(recipes)
    for i, recipe in enumerate(recipes):
        for ing in ingredients[i]:
            graph[ing].append(recipe)
            indeg[recipe] += 1
    q = deque(supplies)
    res = []
    while q:
        item = q.popleft()
        for nxt in graph[item]:
            indeg[nxt] -= 1
            if indeg[nxt] == 0:
                q.append(nxt)
                if nxt in recipe_set: res.append(nxt)
    return res
```

```python
from collections import defaultdict
def getDistances(self, arr):
    pos = defaultdict(list)
    for i, x in enumerate(arr): pos[x].append(i)
    res = [0] * len(arr)
    for indices in pos.values():
        n = len(indices)
        prefix = [0] * (n + 1)
        for i in range(n): prefix[i+1] = prefix[i] + indices[i]
        for i in range(n):
            left = indices[i] * i - prefix[i]
            right = prefix[n] - prefix[i+1] - indices[i] * (n - i - 1)
            res[indices[i]] = left + right
    return res
```

```python
def longestPalindrome(self, words):
    cnt = Counter(words); res = 0; has_center = False
    for w, c in cnt.items():
        rev = w[::-1]
        if w == rev:
            res += c // 2 * 4
```

## 2163. Minimum Difference in Sums After Removal of Elements

```python
import heapq
def minimumDifference(self, nums):
    n = len(nums) // 3
    # 前缀：选 n 个最小的
    prefix_min = [0] * (3 * n)
    h = []; s = 0
    for i in range(2 * n):
        heapq.heappush(h, -nums[i]); s += nums[i]
        if len(h) > n: s += heapq.heappop(h)
        if i >= n - 1: prefix_min[i] = s
    # 后缀：选 n 个最大的
    suffix_max = [0] * (3 * n)
    h = []; s = 0
    for i in range(3*n-1, n-1, -1):
        heapq.heappush(h, nums[i]); s += nums[i]
        if len(h) > n: s -= heapq.heappop(h)
        if i <= 2*n: suffix_max[i] = s
    return min(prefix_min[i] - suffix_max[i+1] for i in range(n-1, 2*n))
```

```python
class Bitset:
    def __init__(self, size):
        self.bits = [0] * size; self.ones = 0; self.flipped = False
    def fix(self, idx):
        if self.flipped:
            if self.bits[idx] == 1: self.bits[idx] = 0; self.ones += 1
        else:
            if self.bits[idx] == 0: self.bits[idx] = 1; self.ones += 1
    def unfix(self, idx):
        if self.flipped:
            if self.bits[idx] == 0: self.bits[idx] = 1; self.ones -= 1
        else:
            if self.bits[idx] == 1: self.bits[idx] = 0; self.ones -= 1

        self.flipped = not self.flipped
        self.ones = len(self.bits) - self.ones
    def all(self): return self.ones == len(self.bits)
    def one(self): return self.ones > 0
    def count(self): return self.ones
```

## 2167. Minimum Time to Remove All Cars Containing Illegal Goods

https://leetcode.cn/problems/minimum-time-to-remove-all-cars-containing-illegal-goods/

```python
def minimumTime(self, s):
    n = len(s)
    left = [0] * n  # 从左清除 s[0..i] 中所有1的最小代价
    left[0] = int(s[0])
    for i in range(1, n):
        if s[i] == '1': left[i] = min(left[i-1] + 2, i + 1)
        else: left[i] = left[i-1]
    right = [0] * n
    right[-1] = int(s[-1])
    for i in range(n-2, -1, -1):
        if s[i] == '1': right[i] = min(right[i+1] + 2, n - i)
        else: right[i] = right[i+1]
    res = min(right[0], left[-1])
    for i in range(n-1):
        res = min(res, left[i] + right[i+1])
    return res
```

## 2182. Construct String With Repeat Limit

```python
from collections import Counter
def repeatLimitedString(self, s, repeatLimit):
    cnt = Counter(s)
    chars = sorted(cnt, reverse=True)
    res = []; i = 0
    while i < len(chars):
        c = chars[i]
        if cnt[c] == 0: i += 1; continue
        use = min(cnt[c], repeatLimit)
        res.append(c * use); cnt[c] -= use
        if cnt[c] > 0:
            j = i + 1
            while j < len(chars) and cnt[chars[j]] == 0: j += 1
            if j >= len(chars): break
            res.append(chars[j]); cnt[chars[j]] -= 1
        else:
            i += 1
    return ''.join(res)
```

## 2183. Count Array Pairs Divisible by K

https://leetcode.cn/problems/count-array-pairs-divisible-by-k/

GCD 分桶。

```python
from math import gcd
from collections import Counter
def countPairs(self, nums, k):
    gcds = Counter(gcd(x, k) for x in nums)
    res = 0
    keys = list(gcds)
    for i in range(len(keys)):
        for j in range(i, len(keys)):
            if keys[i] * keys[j] % k == 0:
                if i == j: res += gcds[keys[i]] * (gcds[keys[i]] - 1) // 2
                else: res += gcds[keys[i]] * gcds[keys[j]]
    return res
```

## 2192. All Ancestors of a Node in a Directed Acyclic Graph

https://leetcode.cn/problems/all-ancestors-of-a-node-in-a-directed-acyclic-graph/

拓扑排序 + 集合传播。

```python
from collections import deque
def getAncestors(self, n, edges):
    graph = [[] for _ in range(n)]
    indeg = [0] * n
    for u, v in edges:
        graph[u].append(v); indeg[v] += 1
    ancestors = [set() for _ in range(n)]
    q = deque(i for i in range(n) if indeg[i] == 0)
    while q:
        u = q.popleft()
        for v in graph[u]:
            ancestors[v].add(u)
            ancestors[v] |= ancestors[u]
            indeg[v] -= 1
            if indeg[v] == 0: q.append(v)
    return [sorted(a) for a in ancestors]
```

## 2195. Append K Integers With Minimal Sum

https://leetcode.cn/problems/append-k-integers-with-minimal-sum/

```python
def minimalKSum(self, nums, k):
    nums = sorted(set(nums))
    res = 0; prev = 0
    for x in nums:
        gap = x - prev - 1
        if gap > 0:
            take = min(gap, k)
            res += (prev + 1 + prev + take) * take // 2
            k -= take
        if k == 0: break
        prev = x
    if k > 0:
        start = (nums[-1] if nums else 0) + 1
        res += (start + start + k - 1) * k // 2
    return res
```

```python
def createBinaryTree(self, descriptions):

        if parent not in nodes: nodes[parent] = TreeNode(parent)
```

## 2812. Find the Safest Path in a Grid
https://leetcode.com/problems/find-the-safest-path-in-a-grid/

多源BFS + 二分/并查集。

```python
from collections import deque
def maximumSafenessFactor(self, grid):
    n = len(grid)
    dist = [[float('inf')]*n for _ in range(n)]
    q = deque()
    for i in range(n):
        for j in range(n):
            if grid[i][j] == 1:
                dist[i][j] = 0; q.append((i, j))
    while q:
        x, y = q.popleft()
        for dx, dy in [(0,1),(0,-1),(1,0),(-1,0)]:
            nx, ny = x+dx, y+dy
            if 0 <= nx < n and 0 <= ny < n and dist[nx][ny] > dist[x][y]+1:
                dist[nx][ny] = dist[x][y]+1; q.append((nx, ny))
    # 二分答案
    def check(d):
        if dist[0][0] < d or dist[n-1][n-1] < d: return False
        vis = [[False]*n for _ in range(n)]
        vis[0][0] = True; stk = [(0,0)]
        while stk:
            x, y = stk.pop()
            if x == n-1 and y == n-1: return True
            for dx, dy in [(0,1),(0,-1),(1,0),(-1,0)]:
                nx, ny = x+dx, y+dy
                if 0<=nx<n and 0<=ny<n and not vis[nx][ny] and dist[nx][ny]>=d:
                    vis[nx][ny]=True; stk.append((nx,ny))
        return False
    lo, hi = 0, n*2
    while lo < hi:
        mid = (lo+hi+1)//2
        if check(mid): lo = mid
        else: hi = mid - 1
    return lo
```

## 2813. Maximum Elegance of a K-Length Subsequence
https://leetcode.com/problems/maximum-elegance-of-a-k-length-subsequence/

贪心：先按 profit 降序选 k 个，然后用重复类别的最小 profit 替换新类别。

```python
def findMaximumElegance(self, items, k):
    items.sort(key=lambda x: -x[0])
    seen = set(); dup = []; total = 0
    for i in range(k):
        total += items[i][0]
        if items[i][1] in seen:
            dup.append(items[i][0])
        else:
            seen.add(items[i][1])
    res = total + len(seen) ** 2
    for i in range(k, len(items)):
        if items[i][1] not in seen and dup:
            total -= dup.pop()
            total += items[i][0]
            seen.add(items[i][1])
            res = max(res, total + len(seen) ** 2)
    return res
```

## 2842. Count K-Subsequences of a String With Maximum Beauty
https://leetcode.com/problems/count-k-subsequences-of-a-string-with-maximum-beauty/

```python
from collections import Counter
from math import comb
def countKSubsequencesWithMaxBeauty(self, s, k):
    MOD = 10**9 + 7
    cnt = Counter(s)
    if k > len(cnt): return 0
    freqs = sorted(cnt.values(), reverse=True)
    # 找第 k 大频率
    kth = freqs[k-1]
    # 频率 > kth 的必选
    must = sum(1 for f in freqs if f > kth)
    same = sum(1 for f in freqs if f == kth)
    need = k - must
    res = comb(same, need) % MOD
    for f in freqs[:k]:
        res = res * (f if f > kth else kth) % MOD
    # 其实要乘的是对应频率值
    res = comb(same, need)
    for f in freqs:
        if f > kth:
            res = res * f % MOD
    res = res * pow(kth, need, MOD) % MOD
    return res
```

```python
def maximumSumOfHeights(self, maxHeights):
    n = len(maxHeights)
    left = [0] * n; stk = []
    for i in range(n):
        while stk and maxHeights[stk[-1]] >= maxHeights[i]: stk.pop()
        if not stk:
            left[i] = maxHeights[i] * (i + 1)
        else:
            left[i] = left[stk[-1]] + maxHeights[i] * (i - stk[-1])
        stk.append(i)
    right = [0] * n; stk = []
    for i in range(n-1, -1, -1):
        while stk and maxHeights[stk[-1]] >= maxHeights[i]: stk.pop()
        if not stk:
            right[i] = maxHeights[i] * (n - i)
        else:
            right[i] = right[stk[-1]] + maxHeights[i] * (stk[-1] - i)
        stk.append(i)
    return max(left[i] + right[i] - maxHeights[i] for i in range(n))
```

## 2867. Count Valid Paths in a Tree
https://leetcode.com/problems/count-valid-paths-in-a-tree/

```python
def countPaths(self, n, edges):
    sieve = [True] * (n + 1); sieve[0] = sieve[1] = False
    for i in range(2, int(n**0.5)+1):
        if sieve[i]:
            for j in range(i*i, n+1, i): sieve[j] = False
    adj = [[] for _ in range(n+1)]
    for u, v in edges: adj[u].append(v); adj[v].append(u)
    parent = list(range(n+1)); size = [1]*(n+1)
    def find(x):
        while parent[x] != x: parent[x] = parent[parent[x]]; x = parent[x]
        return x
    def union(a, b):
        pa, pb = find(a), find(b)
        if pa != pb: 
            if size[pa] < size[pb]: pa, pb = pb, pa
            parent[pb] = pa; size[pa] += size[pb]
    # 只合并非质数节点
    for u, v in edges:
        if not sieve[u] and not sieve[v]: union(u, v)
    res = 0
    for p in range(1, n+1):
        if not sieve[p]: continue
        groups = []
        for v in adj[p]:
            groups.append(size[find(v)] if not sieve[v] else 0)
        # 路径经过 p：单侧 + 跨侧
        s = 0
        for g in groups:
            res += g  # p -> 非质数子树
            res += s * g  # 跨两侧
            s += g
    return res
```

```python
def countVisitedNodes(self, edges):
    n = len(edges); res = [0] * n
    vis = [0] * n  # 0 unvisited, 1 in-progress, 2 done
    # 拓扑排序去掉非环
    deg = [0] * n
    for e in edges: deg[e] += 1
    from collections import deque
    q = deque(i for i in range(n) if deg[i] == 0)
    order = []

        u = q.popleft(); order.append(u)
        v = edges[u]; deg[v] -= 1
        if deg[v] == 0: q.append(v)

            cycle = []; j = i
            while res[j] == 0:
                cycle.append(j); res[j] = -1; j = edges[j]
            l = len(cycle)
            for c in cycle: res[c] = l
```

```python
def minOperations(self, s1, s2, x):
    diff = [i for i in range(len(s1)) if s1[i] != s2[i]]
    n = len(diff)
    if n % 2: return -1
    if n == 0: return 0
    # dp[i] = min cost to pair first i diff positions
    from functools import lru_cache
    @lru_cache(None)
    def dp(i):
        if i < 0: return 0
        if i == 0: return x / 2  # 一个配对用操作1代价 x/2
        return min(
            dp(i-1) + x / 2,  # 用操作1匹配远距离
            dp(i-2) + min(diff[i] - diff[i-1], x)  # 用操作2匹配相邻
        )
    # 其实所有 diff 两两匹配，操作1代价 x，操作2代价 |i-j|
    # dp 转换：每对要么花 x，要么花 diff[2k+1]-diff[2k]
    return int(dp(n-1))
```

```python
            if x >> b & 1: bits[b] += 1
    res = 0
    for i in range(k):
        val = 0
        for b in range(30):
            if bits[b] > 0:
                val |= (1 << b); bits[b] -= 1
        res = (res + val * val) % MOD
    return res
```

## 2900. Longest Unequal Adjacent Groups Subsequence I

```python
def getLongestSubsequence(self, words, groups):
    res = [words[0]]; last = groups[0]
    for i in range(1, len(words)):
        if groups[i] != last:
            res.append(words[i]); last = groups[i]
    return res
```

def checkOverlap(self, radius, xCenter, yCenter, x1, y1, x2, y2):
    cx = max(x1, min(xCenter, x2))
    cy = max(y1, min(yCenter, y2))
    return (cx - xCenter)**2 + (cy - yCenter)**2 <= radius**2

## 1405. Longest Happy String
https://leetcode.com/problems/longest-happy-string/

贪心 + 最大堆。

```python
import heapq
def longestDiverseString(self, a, b, c):
    h = []
    if a: heapq.heappush(h, (-a, 'a'))
    if b: heapq.heappush(h, (-b, 'b'))
    if c: heapq.heappush(h, (-c, 'c'))
    res = []
    while h:
        cnt, ch = heapq.heappop(h)
        if len(res) >= 2 and res[-1] == res[-2] == ch:
            if not h: break
            cnt2, ch2 = heapq.heappop(h)
            res.append(ch2)
            if cnt2 + 1: heapq.heappush(h, (cnt2 + 1, ch2))
            heapq.heappush(h, (cnt, ch))
        else:
            res.append(ch)
            if cnt + 1: heapq.heappush(h, (cnt + 1, ch))
    return ''.join(res)
```

def stoneGameIII(self, stoneValue):
    n = len(stoneValue)
    dp = [0] * (n + 1)
    for i in range(n-1, -1, -1):
        dp[i] = float('-inf')
        total = 0
        for k in range(1, 4):
            if i + k > n: break
            total += stoneValue[i+k-1]
            dp[i] = max(dp[i], total - dp[i+k])

    elif dp[0] < 0: return "Bob"
    return "Tie"
```

def reformat(self, s):
    letters = [c for c in s if c.isalpha()]
    digits = [c for c in s if c.isdigit()]
    if abs(len(letters) - len(digits)) > 1: return ''
    if len(letters) < len(digits): letters, digits = digits, letters

from collections import deque
def constrainedSubsetSum(self, nums, k):
    n = len(nums)
    dp = nums[:]
    dq = deque()
    for i in range(n):
        if dq: dp[i] = max(dp[i], dp[i] + dp[dq[0]])
        while dq and dp[dq[-1]] <= dp[i]: dq.pop()
        dq.append(i)
        if dq[0] <= i - k: dq.popleft()
    return max(dp)
```

## 1438. Longest Continuous Subarray With Absolute Diff Less Than or Equal to Limit
https://leetcode.com/problems/longest-continuous-subarray-with-absolute-diff-less-than-or-equal-to-limit/

```python
from collections import deque
def longestSubarray(self, nums, limit):
    max_dq = deque(); min_dq = deque()
    left = res = 0
    for right in range(len(nums)):
        while max_dq and nums[max_dq[-1]] <= nums[right]: max_dq.pop()
        while min_dq and nums[min_dq[-1]] >= nums[right]: min_dq.pop()
        max_dq.append(right); min_dq.append(right)
        while nums[max_dq[0]] - nums[min_dq[0]] > limit:
            left += 1
            if max_dq[0] < left: max_dq.popleft()
            if min_dq[0] < left: min_dq.popleft()
        res = max(res, right - left + 1)
    return res
```

## 1439. Find the Kth Smallest Sum of a Matrix With Sorted Rows
https://leetcode.com/problems/find-the-kth-smallest-sum-of-a-matrix-with-sorted-rows/

逐行合并，最小堆取前 k 个和。

```python
import heapq
def kthSmallest(self, mat, k):
    prev = mat[0]
    for row in mat[1:]:
        h = [(prev[0]+row[0], 0, 0)]
        visited = {(0, 0)}
        cur = []
        while h and len(cur) < k:
            s, i, j = heapq.heappop(h)
            cur.append(s)
            if i+1 < len(prev) and (i+1, j) not in visited:
                visited.add((i+1, j))
                heapq.heappush(h, (prev[i+1]+row[j], i+1, j))
            if j+1 < len(row) and (i, j+1) not in visited:
                visited.add((i, j+1))
                heapq.heappush(h, (prev[i]+row[j+1], i, j+1))
        prev = cur
    return prev[k-1]
```

def pseudoPalindromicPaths(self, root):
    def dfs(node, mask):
        if not node: return 0
        mask ^= 1 << node.val
        if not node.left and not node.right:
            return 1 if mask & (mask - 1) == 0 else 0
        return dfs(node.left, mask) + dfs(node.right, mask)
    return dfs(root, 0)

    reach = [[False]*numCourses for _ in range(numCourses)]

    for k in range(numCourses):

## 1463. Cherry Pickup II

```python
def cherryPickup(self, grid):
    m, n = len(grid), len(grid[0])
    dp = [[float('-inf')]*n for _ in range(n)]
    dp[0][n-1] = grid[0][0] + grid[0][n-1]
    for i in range(1, m):
        ndp = [[float('-inf')]*n for _ in range(n)]
        for j1 in range(min(i+1, n)):
            for j2 in range(max(0, n-1-i), n):
                for dj1 in [-1, 0, 1]:
                    for dj2 in [-1, 0, 1]:
                        pj1, pj2 = j1-dj1, j2-dj2
                        if 0 <= pj1 < n and 0 <= pj2 < n:
                            val = grid[i][j1] + (grid[i][j2] if j1 != j2 else 0)
                            ndp[j1][j2] = max(ndp[j1][j2], dp[pj1][pj2] + val)
        dp = ndp
    return max(dp[j1][j2] for j1 in range(n) for j2 in range(n))
```

def minReorder(self, n, connections):
    graph = defaultdict(list)
    roads = set()
    for u, v in connections:
        graph[u].append(v); graph[v].append(u)
        roads.add((u, v))
    visited = {0}; q = deque([0]); res = 0
    while q:
        node = q.popleft()
        for nb in graph[node]:

## 1498. Number of Subsequences That Satisfy the Given Sum Condition

```python
def numSubseq(self, nums, target):
    nums.sort()
    MOD = 10**9 + 7
    res = 0
    left, right = 0, len(nums) - 1
    while left <= right:
        if nums[left] + nums[right] <= target:
            res = (res + pow(2, right - left, MOD)) % MOD
            left += 1
        else:
            right -= 1
    return res
```

from collections import deque
def findMaxValueOfEquation(self, points, k):
    dq = deque(); res = float('-inf')
    for x, y in points:
        while dq and x - dq[0][1] > k: dq.popleft()
        if dq: res = max(res, y + x + dq[0][0])

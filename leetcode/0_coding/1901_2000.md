## 1901. Find a Peak Element II

https://leetcode.cn/problems/find-a-peak-element-ii/

二分列，找该列最大值行。

```python
def findPeakGrid(self, mat):
    m, n = len(mat), len(mat[0])
    lo, hi = 0, n - 1
    while lo <= hi:
        mid = (lo + hi) // 2
        max_row = max(range(m), key=lambda r: mat[r][mid])
        left = mat[max_row][mid-1] if mid > 0 else -1
        right = mat[max_row][mid+1] if mid < n-1 else -1
        if mat[max_row][mid] > left and mat[max_row][mid] > right:
            return [max_row, mid]
        elif left > mat[max_row][mid]: hi = mid - 1
        else: lo = mid + 1
    return [-1, -1]
```

## 1905. Count Sub Islands

https://leetcode.cn/problems/count-sub-islands/

DFS: grid2 的岛如果对应 grid1 全为 1 则是子岛。

```python
def countSubIslands(self, grid1, grid2):
    m, n = len(grid1), len(grid1[0])
    def dfs(i, j):
        if i < 0 or i >= m or j < 0 or j >= n or grid2[i][j] == 0:
            return True
        grid2[i][j] = 0
        res = grid1[i][j] == 1
        for di, dj in [(-1,0),(1,0),(0,-1),(0,1)]:
            if not dfs(i+di, j+dj): res = False
        return res
    return sum(dfs(i, j) for i in range(m) for j in range(n) if grid2[i][j])
```

## 1906. Minimum Absolute Difference Queries

https://leetcode.cn/problems/minimum-absolute-difference-queries/

前缀和记录每个值 (1-100) 的频率。

```python
def minDifference(self, nums, queries):
    n = len(nums)
    prefix = [[0]*101 for _ in range(n+1)]
    for i, x in enumerate(nums):
        for v in range(101):
            prefix[i+1][v] = prefix[i][v]
        prefix[i+1][x] += 1
    res = []
    for l, r in queries:
        vals = [v for v in range(1, 101) if prefix[r+1][v] - prefix[l][v] > 0]
        if len(vals) <= 1:
            res.append(-1)
        else:
            res.append(min(vals[i+1] - vals[i] for i in range(len(vals)-1)))
    return res
```

## 1914. Cyclically Rotating a Grid

https://leetcode.cn/problems/cyclically-rotating-a-grid/

逐层提取、旋转、放回。

```python
def rotateGrid(self, grid, k):
    m, n = len(grid), len(grid[0])
    layers = min(m, n) // 2
    for layer in range(layers):
        cells = []
        r1, c1, r2, c2 = layer, layer, m-1-layer, n-1-layer
        for c in range(c1, c2): cells.append((r1, c))
        for r in range(r1, r2): cells.append((r, c2))
        for c in range(c2, c1, -1): cells.append((r2, c))
        for r in range(r2, r1, -1): cells.append((r, c1))
        vals = [grid[r][c] for r, c in cells]
        kk = k % len(vals)
        vals = vals[kk:] + vals[:kk]
        for (r, c), v in zip(cells, vals):
            grid[r][c] = v
    return grid
```

## 1926. Nearest Exit from Entrance in Maze

```python
from collections import deque
def nearestExit(self, maze, entrance):
    m, n = len(maze), len(maze[0])
    q = deque([(entrance[0], entrance[1], 0)])
    maze[entrance[0]][entrance[1]] = '+'
    while q:
        r, c, d = q.popleft()
        for dr, dc in [(-1,0),(1,0),(0,-1),(0,1)]:
            nr, nc = r+dr, c+dc
            if 0 <= nr < m and 0 <= nc < n and maze[nr][nc] == '.':
                if nr == 0 or nr == m-1 or nc == 0 or nc == n-1:
                    return d + 1
                maze[nr][nc] = '+'
                q.append((nr, nc, d+1))
    return -1
```

## 1927. Sum Game

https://leetcode.cn/problems/sum-game/

博弈论：分析两半的差和 '?' 个数。

```python
def sumGame(self, num):
    n = len(num)
    half = n // 2
    left_sum = sum(int(c) for c in num[:half] if c != '?')
    right_sum = sum(int(c) for c in num[half:] if c != '?')
    left_q = num[:half].count('?')
    right_q = num[half:].count('?')
    # Alice 先手，她想让不等，Bob 想让相等
    # 等价于 left_sum - right_sum + (left_q - right_q) * 4.5 != 0
    # 即 2*(left_sum - right_sum) + 9*(left_q - right_q) != 0
    return 2 * (left_sum - right_sum) + 9 * (left_q - right_q) != 0
```

## 1928. Minimum Cost to Reach Destination in Time

https://leetcode.cn/problems/minimum-cost-to-reach-destination-in-time/

Dijkstra 带时间维度。

```python
import heapq
def minCost(self, maxTime, edges, passingFees):
    n = len(passingFees)
    graph = [[] for _ in range(n)]
    for u, v, t in edges:
        graph[u].append((v, t)); graph[v].append((u, t))
    dist = [float('inf')] * n; dist[0] = 0
    h = [(passingFees[0], 0, 0)]  # cost, time, node
    while h:
        cost, time, u = heapq.heappop(h)
        if u == n - 1: return cost
        if time > dist[u]: continue
        for v, t in graph[u]:
            new_time = time + t
            if new_time < maxTime and new_time < dist[v]:
                dist[v] = new_time
                heapq.heappush(h, (cost + passingFees[v], new_time, v))
    return -1
```

## 1959. Minimum Total Space Wasted With K Resizing Operations

https://leetcode.cn/problems/minimum-total-space-wasted-with-k-resizing-operations/

DP：`dp[i][j]` = 前 i 天 j 次调整的最小浪费。

```python
def minSpaceWastedKResizing(self, nums, k):
    n = len(nums)
    dp = [[float('inf')]*(k+2) for _ in range(n+1)]
    dp[0][0] = 0
    for i in range(1, n+1):
        mx = 0; total = 0
        for j in range(i, 0, -1):
            mx = max(mx, nums[j-1])
            total += nums[j-1]
            waste = mx * (i - j + 1) - total
            for t in range(k+1):
                if dp[j-1][t] < float('inf'):
                    dp[i][t+1] = min(dp[i][t+1], dp[j-1][t] + waste)
    return min(dp[n][t] for t in range(k+2))
```

## 1970. Last Day Where You Can Still Cross

https://leetcode.cn/problems/last-day-where-you-can-still-cross/

二分 + BFS / 并查集逆序。

```python
from collections import deque
def latestDayToCross(self, row, col, cells):
    def can_cross(day):
        water = set((r, c) for r, c in cells[:day])
        q = deque()
        for c in range(1, col + 1):
            if (1, c) not in water:
                q.append((1, c)); water.add((1, c))
        while q:
            r, cc = q.popleft()
            if r == row: return True
            for dr, dc in [(-1,0),(1,0),(0,-1),(0,1)]:
                nr, nc = r+dr, cc+dc
                if 1 <= nr <= row and 1 <= nc <= col and (nr, nc) not in water:
                    water.add((nr, nc)); q.append((nr, nc))
        return False
    lo, hi = 1, len(cells)
    while lo < hi:
        mid = (lo + hi + 1) // 2
        if can_cross(mid): lo = mid
        else: hi = mid - 1
    return lo
```

## 1976. Number of Ways to Arrive at Destination

https://leetcode.cn/problems/number-of-ways-to-arrive-at-destination/

Dijkstra + 计数。

```python
import heapq
def countPaths(self, n, roads):
    MOD = 10**9 + 7
    graph = [[] for _ in range(n)]
    for u, v, t in roads:
        graph[u].append((v, t)); graph[v].append((u, t))
    dist = [float('inf')] * n; dist[0] = 0
    ways = [0] * n; ways[0] = 1
    h = [(0, 0)]
    while h:
        d, u = heapq.heappop(h)
        if d > dist[u]: continue
        for v, t in graph[u]:
            if d + t < dist[v]:
                dist[v] = d + t; ways[v] = ways[u]
                heapq.heappush(h, (dist[v], v))
            elif d + t == dist[v]:
                ways[v] = (ways[v] + ways[u]) % MOD
    return ways[n-1]
```

## 1986. Minimum Number of Work Sessions to Finish the Tasks

https://leetcode.cn/problems/minimum-number-of-work-sessions-to-finish-the-tasks/

状压 DP。

```python
def minSessions(self, tasks, sessionTime):
    n = len(tasks)
    # 预计算每个子集的时间
    subset_time = [0] * (1 << n)
    for mask in range(1, 1 << n):
        bit = (mask & -mask).bit_length() - 1
        subset_time[mask] = subset_time[mask ^ (1 << bit)] + tasks[bit]
    # dp[mask] = (sessions, remaining_time_in_last_session)
    INF = (n + 1, 0)
    dp = [INF] * (1 << n); dp[0] = (1, sessionTime)
    for mask in range(1, 1 << n):
        sub = mask
        while sub:
            if subset_time[sub] <= sessionTime:
                prev = mask ^ sub
                if dp[prev][1] >= subset_time[sub]:
                    dp[mask] = min(dp[mask], (dp[prev][0], dp[prev][1] - subset_time[sub]))
                else:
                    dp[mask] = min(dp[mask], (dp[prev][0] + 1, sessionTime - subset_time[sub]))
            sub = (sub - 1) & mask
    return dp[(1 << n) - 1][0]
```

## 1993. Operations on Tree

https://leetcode.cn/problems/operations-on-tree/

```python
class LockingTree:
    def __init__(self, parent):
        self.parent = parent
        self.children = [[] for _ in range(len(parent))]
        for i in range(1, len(parent)):
            self.children[parent[i]].append(i)
        self.locked = [0] * len(parent)
    def lock(self, num, user):
        if self.locked[num]: return False
        self.locked[num] = user; return True
    def unlock(self, num, user):
        if self.locked[num] != user: return False
        self.locked[num] = 0; return True
    def upgrade(self, num, user):
        # 自身未锁，祖先无锁，至少一个后代锁定
        if self.locked[num]: return False
        p = self.parent[num]
        while p != -1:
            if self.locked[p]: return False
            p = self.parent[p]
        if not self._unlock_descendants(num): return False
        self.locked[num] = user; return True
    def _unlock_descendants(self, node):
        found = False
        for child in self.children[node]:
            if self.locked[child]:
                self.locked[child] = 0; found = True
            if self._unlock_descendants(child): found = True
        return found
```

```python
def countQuadruplets(self, nums):
    n = len(nums); res = 0
    cnt = Counter()
    for b in range(n-3, 0, -1):
        # 加入 c = b+1 到 n-1 中的 nums[d] - nums[c]
```

```python
    n = len(nextVisit)
```

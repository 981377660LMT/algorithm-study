```python
from collections import defaultdict
def totalFruit(self, fruits):
    cnt = defaultdict(int)
    left = res = 0
    for right, f in enumerate(fruits):
        cnt[f] += 1
        while len(cnt) > 2:
            cnt[fruits[left]] -= 1
            if cnt[fruits[left]] == 0: del cnt[fruits[left]]
            left += 1
        res = max(res, right - left + 1)
```

## 907. Sum of Subarray Minimums
https://leetcode.com/problems/sum-of-subarray-minimums/

单调栈：计算每个元素作为最小值的子数组个数。

```python
def sumSubarrayMins(self, arr):
    MOD = 10**9 + 7
    n = len(arr)
    stack = []
    left = [0] * n   # 左边界距离
    right = [0] * n  # 右边界距离
    for i in range(n):
        count = 1
        while stack and stack[-1][0] >= arr[i]:
            count += stack.pop()[1]
        left[i] = count
        stack.append((arr[i], count))
    stack = []
    for i in range(n - 1, -1, -1):
        count = 1
        while stack and stack[-1][0] > arr[i]:
            count += stack.pop()[1]
        right[i] = count
        stack.append((arr[i], count))
    return sum(a * l * r for a, l, r in zip(arr, left, right)) % MOD
```

```python
from collections import deque
def snakesAndLadders(self, board):
    n = len(board)
    def get_pos(s):
        r, c = divmod(s - 1, n)
        if r % 2: c = n - 1 - c
        return n - 1 - r, c
    visited = {1}; q = deque([(1, 0)])
    while q:
        s, d = q.popleft()
        for i in range(1, 7):
            ns = s + i
            if ns > n * n: break
            r, c = get_pos(ns)
            if board[r][c] != -1: ns = board[r][c]
            if ns == n * n: return d + 1
            if ns not in visited: visited.add(ns); q.append((ns, d + 1))
    return -1
```

```python
def sortArray(self, nums):

        if lo >= hi: return
        pivot = random.randint(lo, hi)
        nums[pivot], nums[hi] = nums[hi], nums[pivot]
        i = lo
        for j in range(lo, hi):
            if nums[j] < nums[hi]:
```

## 934. Shortest Bridge
https://leetcode.com/problems/shortest-bridge/

DFS 找第一个岛，BFS 扩展到第二个岛。

```python
from collections import deque
def shortestBridge(self, grid):
    m, n = len(grid), len(grid[0])
    q = deque()
    found = False
    def dfs(i, j):
        if i < 0 or i >= m or j < 0 or j >= n or grid[i][j] != 1: return
        grid[i][j] = 2
        q.append((i, j))
        for di, dj in [(-1,0),(1,0),(0,-1),(0,1)]: dfs(i+di, j+dj)
    for i in range(m):
        if found: break
        for j in range(n):
            if grid[i][j] == 1:
                dfs(i, j); found = True; break
    steps = 0
    while q:
        for _ in range(len(q)):
            i, j = q.popleft()
            for di, dj in [(-1,0),(1,0),(0,-1),(0,1)]:
                ni, nj = i+di, j+dj
                if 0 <= ni < m and 0 <= nj < n:
                    if grid[ni][nj] == 1: return steps
                    if grid[ni][nj] == 0:
                        grid[ni][nj] = 2
                        q.append((ni, nj))
        steps += 1
    return -1
```

```python
def removeStones(self, stones):
    parent = {}
    def find(x):
        parent.setdefault(x, x)
        while parent[x] != x:
            parent[x] = parent[parent[x]]
            x = parent[x]
        return x
    def union(x, y):
        parent[find(x)] = find(y)
    for r, c in stones:
        union(r, ~c)  # 用 ~c 区分行和列
    return len(stones) - len({find(r) for r, c in stones})
```

```python
    def dfs(node):
        if not node: return 2  # 空节点视为已覆盖
```

```python
def verticalTraversal(self, root):
    nodes = []
    def dfs(node, row, col):
        if not node: return
        nodes.append((col, row, node.val))
        dfs(node.left, row + 1, col - 1)
        dfs(node.right, row + 1, col + 1)
```

```python
def orangesRotting(self, grid):
    m, n = len(grid), len(grid[0])
    q = deque()
    fresh = 0
    for i in range(m):
        for j in range(n):
            if grid[i][j] == 2: q.append((i, j))

    if fresh == 0: return 0
```

```python
    n = len(stones)
    if (n - 1) % (k - 1): return -1
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + stones[i]
    dp = [[0] * n for _ in range(n)]
    for length in range(k, n + 1):
        for i in range(n - length + 1):
            j = i + length - 1
            dp[i][j] = float('inf')
            for mid in range(i, j, k - 1):
                dp[i][j] = min(dp[i][j], dp[i][mid] + dp[mid + 1][j])
            if (j - i) % (k - 1) == 0:
                dp[i][j] += prefix[j + 1] - prefix[i]
    return dp[0][n - 1]
```

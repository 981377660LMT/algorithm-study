```python
def interchangeableRectangles(self, rectangles):
    cnt = Counter()

        g = gcd(w, h); cnt[(w//g, h//g)] += 1
```

## 2002. Maximum Product of the Length of Two Palindromic Subsequences
https://leetcode.com/problems/maximum-product-of-the-length-of-two-palindromic-subsequences/

枚举子集。

```python
def maxProduct(self, s):
    n = len(s)
    def is_palindrome(mask):
        chars = [s[i] for i in range(n) if mask & (1 << i)]
        return chars == chars[::-1]
    pal = {}
    for mask in range(1, 1 << n):
        if is_palindrome(mask):
            pal[mask] = bin(mask).count('1')
    res = 0
    masks = list(pal.keys())
    for m1 in masks:
        # 补集的子集
        comp = ((1 << n) - 1) ^ m1
        sub = comp
        while sub:
            if sub in pal:
                res = max(res, pal[m1] * pal[sub])
            sub = (sub - 1) & comp
    return res
```

## 2007. Find Original Array From Doubled Array
https://leetcode.com/problems/find-original-array-from-doubled-array/

```python
from collections import Counter
def findOriginalArray(self, changed):
    if len(changed) % 2: return []
    cnt = Counter(changed)
    res = []
    for x in sorted(cnt, key=abs):
        if cnt[x] == 0: continue
        if cnt[2*x] < cnt[x]: return []
        for _ in range(cnt[x]):
            res.append(x)
            cnt[2*x] -= 1
        cnt[x] = 0
    return res
```

```python
import bisect
def maxTaxiEarnings(self, n, rides):
    rides.sort(key=lambda x: x[1])
    ends = [r[1] for r in rides]
    m = len(rides)
    dp = [0] * (m + 1)
    for i in range(1, m + 1):
        s, e, t = rides[i-1]
```

## 2013. Detect Squares
https://leetcode.com/problems/detect-squares/

```python
from collections import Counter
class DetectSquares:
    def __init__(self):
        self.cnt = Counter()
        self.xs = {}  # x -> list of y
    def add(self, point):
        x, y = point
        self.cnt[(x, y)] += 1
        if x not in self.xs: self.xs[x] = []
        self.xs[x].append(y)
    def count(self, point):
        x1, y1 = point; res = 0
        if x1 not in self.xs: return 0
        for y2 in self.xs[x1]:
            d = y2 - y1
            if d == 0: continue
            # 检查另外两个角
            for x3 in [x1 + d, x1 - d]:
                res += self.cnt[(x3, y1)] * self.cnt[(x3, y2)]
        return res
```

```python
def placeWordInCrossword(self, board, word):
    m, n = len(board), len(board[0])
    wlen = len(word)

        slots = ''.join(line).split('#')
        for slot in slots:
            if len(slot) == wlen:
                if all(slot[i] == ' ' or slot[i] == word[i] for i in range(wlen)):
                    return True

        if check_slots(col): return True
    return False
```

## 2035. Partition Array Into Two Arrays to Minimize Sum Difference
https://leetcode.com/problems/partition-array-into-two-arrays-to-minimize-sum-difference/

Meet in the middle。

```python
import bisect
def minimumDifference(self, nums):
    n = len(nums) // 2
    total = sum(nums)
    left_sums = {}; right_sums = {}
    for i in range(n + 1):
        left_sums[i] = []; right_sums[i] = []
    # 左半
    for mask in range(1 << n):
        s = sum(nums[j] for j in range(n) if mask & (1 << j))
        left_sums[bin(mask).count('1')].append(s)
    # 右半
    for mask in range(1 << n):
        s = sum(nums[n+j] for j in range(n) if mask & (1 << j))
        right_sums[bin(mask).count('1')].append(s)
    for v in right_sums.values(): v.sort()
    res = float('inf')
    for k in range(n + 1):
        for ls in left_sums[k]:
            # 右边选 n-k 个，total_left = ls + rs, want 2*(ls+rs) close to total
            target = total / 2 - ls
            rlist = right_sums[n - k]
            i = bisect.bisect_left(rlist, target)
            for j in [i - 1, i]:
                if 0 <= j < len(rlist):
                    s1 = ls + rlist[j]; s2 = total - s1
                    res = min(res, abs(s1 - s2))
    return res
```

```python
from collections import deque
def networkBecomesIdle(self, edges, patience):
    n = len(patience)
    graph = [[] for _ in range(n)]
    for u, v in edges:
        graph[u].append(v); graph[v].append(u)
    dist = [-1] * n; dist[0] = 0
    q = deque([0])
    while q:
        u = q.popleft()
        for v in graph[u]:
            if dist[v] == -1:
                dist[v] = dist[u] + 1; q.append(v)
    res = 0
    for i in range(1, n):
        rtt = 2 * dist[i]
        # 最后一次重发时间
        last_resend = ((rtt - 1) // patience[i]) * patience[i]
        res = max(res, last_resend + rtt + 1)
    return res
```

## 2045. Second Minimum Time to Reach Destination
https://leetcode.com/problems/second-minimum-time-to-reach-destination/

BFS 找最短和次短路径长度，模拟红绿灯。

```python
from collections import deque
def secondMinimum(self, n, edges, time, change):
    graph = [[] for _ in range(n + 1)]
    for u, v in edges:
        graph[u].append(v); graph[v].append(u)
    # BFS 找最短和次短步数
    dist = [[float('inf')] * 2 for _ in range(n + 1)]
    dist[1][0] = 0
    q = deque([(1, 0)])
    while q:
        u, d = q.popleft()
        for v in graph[u]:
            nd = d + 1
            if nd < dist[v][0]:
                dist[v][1] = dist[v][0]; dist[v][0] = nd
                q.append((v, nd))
            elif nd != dist[v][0] and nd < dist[v][1]:
                dist[v][1] = nd
                q.append((v, nd))
    steps = dist[n][1]
    # 模拟红绿灯
    cur = 0
    for _ in range(steps):
        cycle = cur // change
        if cycle % 2 == 1:  # 红灯
            cur = (cycle + 1) * change
        cur += time
    return cur
```

## 2065. Maximum Path Quality of a Graph
https://leetcode.com/problems/maximum-path-quality-of-a-graph/

DFS + 回溯（因节点数少）。

```python
def maximalPathQuality(self, values, edges, maxTime):
    n = len(values)
    graph = [[] for _ in range(n)]
    for u, v, t in edges:
        graph[u].append((v, t)); graph[v].append((u, t))
    res = [0]; visited = [0] * n; visited[0] = 1
    def dfs(u, time, quality):
        if u == 0: res[0] = max(res[0], quality)
        for v, t in graph[u]:
            if time + t <= maxTime:
                bonus = values[v] if visited[v] == 0 else 0
                visited[v] += 1
                dfs(v, time + t, quality + bonus)
                visited[v] -= 1
    dfs(0, 0, values[0])
    return res[0]
```

```python
def friendRequests(self, n, restrictions, requests):
    parent = list(range(n))

        while parent[x] != x: parent[x] = parent[parent[x]]; x = parent[x]
        return x
```

## 2097. Valid Arrangement of Pairs
https://leetcode.com/problems/valid-arrangement-of-pairs/

欧拉路径 (Hierholzer)。

```python
from collections import defaultdict, deque
def validArrangement(self, pairs):
    graph = defaultdict(deque)
    indeg = defaultdict(int); outdeg = defaultdict(int)
    for u, v in pairs:
        graph[u].append(v)
        outdeg[u] += 1; indeg[v] += 1
    start = pairs[0][0]
    for u in outdeg:
        if outdeg[u] - indeg[u] == 1:
            start = u; break
    path = []; stack = [start]
    while stack:
        while graph[stack[-1]]:
            stack.append(graph[stack[-1]].popleft())
        path.append(stack.pop())
    path.reverse()
    return [[path[i], path[i+1]] for i in range(len(path)-1)]
```

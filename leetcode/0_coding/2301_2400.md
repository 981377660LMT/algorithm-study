https://leetcode.cn/problems/match-substring-after-replacement/

```python
def matchReplacement(self, s, sub, mappings):
    mp = set((a, b) for a, b in mappings)
    n, m = len(s), len(sub)
    for i in range(n - m + 1):
        if all(s[i+j] == sub[j] or (sub[j], s[i+j]) in mp for j in range(m)):
            return True
    return False
```

## 2305. Fair Distribution of Cookies

https://leetcode.cn/problems/fair-distribution-of-cookies/

回溯。

```python
def distributeCookies(self, cookies, k):
    children = [0] * k; self.res = float('inf')
    cookies.sort(reverse=True)
    def bt(i):
        if i == len(cookies):
            self.res = min(self.res, max(children)); return
        if max(children) >= self.res: return  # 剪枝
        seen = set()
        for j in range(k):
            if children[j] in seen: continue
            seen.add(children[j])
            children[j] += cookies[i]
            bt(i + 1)
            children[j] -= cookies[i]
    bt(0)
    return self.res
```

```python
def sellingWood(self, m, n, prices):
    price = [[0] * (n+1) for _ in range(m+1)]
    for h, w, p in prices: price[h][w] = p
    dp = [[0] * (n+1) for _ in range(m+1)]
    for i in range(1, m+1):
        for j in range(1, n+1):
            dp[i][j] = price[i][j]
```

## 2316. Count Unreachable Pairs of Nodes in an Undirected Graph

https://leetcode.cn/problems/count-unreachable-pairs-of-nodes-in-an-undirected-graph/

```python
class UF:
    def __init__(self, n):
        self.p = list(range(n)); self.sz = [1] * n
    def find(self, x):
        while self.p[x] != x: self.p[x] = self.p[self.p[x]]; x = self.p[x]
        return x
    def union(self, x, y):
        px, py = self.find(x), self.find(y)
        if px == py: return
        if self.sz[px] < self.sz[py]: px, py = py, px
        self.p[py] = px; self.sz[px] += self.sz[py]

def countPairs(self, n, edges):
    uf = UF(n)
    for u, v in edges: uf.union(u, v)
    sizes = {}
    for i in range(n):
        r = uf.find(i)
        sizes[r] = sizes.get(r, 0) + 1
    total = 0; s = 0
    for sz in sizes.values():
        total += s * sz; s += sz
    return total
```

```python
def distinctSequences(self, n):

    def dp(i, prev, prev2):
        if i == n: return 1
```

```python
def spiralMatrix(self, m, n, head):
    grid = [[-1]*n for _ in range(m)]
    dirs = [(0,1),(1,0),(0,-1),(-1,0)]
    r = c = d = 0
    while head:
        grid[r][c] = head.val; head = head.next
        nr, nc = r + dirs[d][0], c + dirs[d][1]
        if not (0 <= nr < m and 0 <= nc < n and grid[nr][nc] == -1):
            d = (d + 1) % 4
            nr, nc = r + dirs[d][0], c + dirs[d][1]
        r, c = nr, nc
    return grid
```

## 2328. Number of Increasing Paths in a Grid

https://leetcode.cn/problems/number-of-increasing-paths-in-a-grid/

```python
from functools import lru_cache
def countPaths(self, grid):
    MOD = 10**9 + 7
    m, n = len(grid), len(grid[0])
    @lru_cache(maxsize=None)
    def dfs(i, j):
        res = 1
        for di, dj in [(0,1),(0,-1),(1,0),(-1,0)]:
            ni, nj = i+di, j+dj
            if 0 <= ni < m and 0 <= nj < n and grid[ni][nj] > grid[i][j]:
                res = (res + dfs(ni, nj)) % MOD
        return res
    return sum(dfs(i, j) for i in range(m) for j in range(n)) % MOD
```

## 2333. Minimum Sum of Squared Difference

https://leetcode.cn/problems/minimum-sum-of-squared-difference/

```python
from collections import Counter
def minSumSquareDiff(self, nums1, nums2, k1, k2):
    diffs = sorted([abs(a - b) for a, b in zip(nums1, nums2)], reverse=True)
    k = k1 + k2
    if sum(diffs) <= k: return 0
    n = len(diffs)
    for i in range(n):
        # 将 diffs[i] 降到 diffs[i+1] 的水平
        nxt = diffs[i+1] if i + 1 < n else 0
        need = (i + 1) * (diffs[i] - nxt)
        if need <= k:
            k -= need
            diffs[i] = nxt
        else:
            dec = k // (i + 1); rem = k % (i + 1)
            # 前 i+1 个都降 dec，其中 rem 个额外降1
            for j in range(i + 1):
                diffs[j] -= dec + (1 if j < rem else 0)
            break
    # 注意前面赋值了但没同步
    # 重新计算
    for j in range(1, i + 1):
        diffs[j] = diffs[i]
    return sum(d * d for d in diffs)
```

更简洁方式：

```python
def minSumSquareDiff(self, nums1, nums2, k1, k2):
    diffs = sorted((abs(a-b) for a, b in zip(nums1, nums2)), reverse=True)
    k = k1 + k2
    if sum(diffs) <= k: return 0
    n = len(diffs); diffs.append(0)
    for i in range(n):
        gap = diffs[i] - diffs[i+1]
        need = gap * (i + 1)
        if need <= k:
            k -= need
        else:
            d = k // (i + 1); r = k % (i + 1)
            level = diffs[i] - d
            return (i + 1 - r) * level * level + r * (level - 1) * (level - 1) + sum(x*x for x in diffs[i+1:n])
    return 0
```

```python
def validSubarraySize(self, nums, threshold):
    n = len(nums); left = [-1] * n; right = [n] * n
    stk = []
    for i in range(n):
        while stk and nums[stk[-1]] >= nums[i]:
            right[stk.pop()] = i
        if stk: left[i] = stk[-1]
        stk.append(i)
    for i in range(n):
        length = right[i] - left[i] - 1
        if nums[i] > threshold / length: return length
    return -1
```

```python
def countSpecialNumbers(self, n):
    digits = list(map(int, str(n)))
    from functools import lru_cache
    @lru_cache(maxsize=None)
    def dp(pos, mask, tight, started):
        if pos == len(digits): return 1 if started else 0
        limit = digits[pos] if tight else 9
        res = 0
        for d in range(0, limit + 1):
            if not started and d == 0:
                res += dp(pos+1, mask, False, False)
            elif mask & (1 << d) == 0:
                res += dp(pos+1, mask | (1 << d), tight and d == limit, True)
        return res
    return dp(0, 0, True, False)
```

## 2382. Maximum Segment Sum After Removals

https://leetcode.cn/problems/maximum-segment-sum-after-removals/

逆序并查集。

```python
def maximumSegmentSum(self, nums, removeQueries):
    n = len(nums); parent = list(range(n)); sz = [0] * n; total = [0] * n
    removed = [True] * n; res = [0] * n; mx = 0
    def find(x):
        while parent[x] != x: parent[x] = parent[parent[x]]; x = parent[x]
        return x
    def union(a, b):
        pa, pb = find(a), find(b)
        if pa == pb: return
        if sz[pa] < sz[pb]: pa, pb = pb, pa
        parent[pb] = pa; sz[pa] += sz[pb]; total[pa] += total[pb]
    for i in range(n-1, 0, -1):
        idx = removeQueries[i]
        removed[idx] = False; sz[idx] = 1; total[idx] = nums[idx]
        if idx > 0 and not removed[idx-1]: union(idx, idx-1)
        if idx < n-1 and not removed[idx+1]: union(idx, idx+1)
        mx = max(mx, total[find(idx)])
        res[i-1] = mx
    return res
```

## 2386. Find the K-Sum of an Array

https://leetcode.cn/problems/find-the-k-sum-of-an-array/

思路兜底：最大子序列和 = 所有正数之和。第 k 大等价于从最大值减去第 k-1 小的"减少量"。用最小堆，初始放最小绝对值，每次弹出后扩展。

```python
import heapq
def kSum(self, nums, k):
    max_sum = sum(x for x in nums if x > 0)
    absNums = sorted(abs(x) for x in nums)
    h = [(absNums[0], 0)]
    for _ in range(k - 1):
        val, i = heapq.heappop(h)
        if i + 1 < len(absNums):
            heapq.heappush(h, (val + absNums[i+1], i+1))
            heapq.heappush(h, (val - absNums[i] + absNums[i+1], i+1))
    if k == 1: return max_sum
    return max_sum - heapq.heappop(h)[0]
```

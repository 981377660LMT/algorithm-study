```python
from collections import deque
def deepestLeavesSum(self, root):
    q = deque([root])
    while q:
        level_sum = 0
        for _ in range(len(q)):
            node = q.popleft(); level_sum += node.val
            if node.left: q.append(node.left)
            if node.right: q.append(node.right)
    return level_sum
```

```python
    a, b = inorder(root1), inorder(root2)
```

## 1306. Jump Game III
https://leetcode.com/problems/jump-game-iii/

BFS/DFS。

```python
def canReach(self, arr, start):
    n = len(arr)
    visited = set()
    stack = [start]
    while stack:
        i = stack.pop()
        if arr[i] == 0: return True
        if i in visited: continue
        visited.add(i)
        if i + arr[i] < n: stack.append(i + arr[i])
        if i - arr[i] >= 0: stack.append(i - arr[i])
    return False
```

```python
def watchedVideosByFriends(self, watchedVideos, friends, id, level):
```

## 1312. Minimum Insertion Steps to Make a String Palindrome
https://leetcode.com/problems/minimum-insertion-steps-to-make-a-string-palindrome/

`n - LPS(s)` 其中 LPS = 最长回文子序列。

```python
def minInsertions(self, s):
    n = len(s)
    dp = [0] * n
    for i in range(n-1, -1, -1):
        ndp = [0] * n
        ndp[i] = 1
        for j in range(i+1, n):
            if s[i] == s[j]: ndp[j] = dp[j-1] + 2
            else: ndp[j] = max(dp[j], ndp[j-1])
        dp = ndp
    return n - dp[n-1]
```

## 1314. Matrix Block Sum
https://leetcode.com/problems/matrix-block-sum/

二维前缀和。

```python
def matrixBlockSum(self, mat, k):
    m, n = len(mat), len(mat[0])
    prefix = [[0]*(n+1) for _ in range(m+1)]
    for i in range(m):
        for j in range(n):
            prefix[i+1][j+1] = mat[i][j]+prefix[i][j+1]+prefix[i+1][j]-prefix[i][j]
    def area(r1,c1,r2,c2):
        r1,c1 = max(r1,0), max(c1,0)
        r2,c2 = min(r2,m-1), min(c2,n-1)
        return prefix[r2+1][c2+1]-prefix[r1][c2+1]-prefix[r2+1][c1]+prefix[r1][c1]
    return [[area(i-k,j-k,i+k,j+k) for j in range(n)] for i in range(m)]
```

## 1319. Number of Operations to Make Network Connected
https://leetcode.com/problems/number-of-operations-to-make-network-connected/

并查集：需要 `连通分量数 - 1` 条边，前提是边数 >= n-1。

```python
def makeConnected(self, n, connections):
    if len(connections) < n - 1: return -1
    parent = list(range(n))
    def find(x):
        while parent[x] != x:
            parent[x] = parent[parent[x]]; x = parent[x]
        return x
    components = n
    for a, b in connections:
        ra, rb = find(a), find(b)
        if ra != rb:
            parent[ra] = rb; components -= 1
    return components - 1
```

```python
def minTaps(self, n, ranges):
    max_reach = [0] * (n + 1)
    for i, r in enumerate(ranges):
        left = max(0, i - r)
        max_reach[left] = max(max_reach[left], i + r)
    taps = end = farthest = 0
    for i in range(n):
```

```python
from collections import defaultdict
def diagonalSort(self, mat):
    m, n = len(mat), len(mat[0])
    d = defaultdict(list)
    for i in range(m):
        for j in range(n):
            d[i-j].append(mat[i][j])
    for k in d: d[k].sort(reverse=True)
    for i in range(m):
        for j in range(n):
            mat[i][j] = d[i-j].pop()
    return mat
```

Floyd-Warshall。

```python
def findTheCity(self, n, edges, distanceThreshold):
    dist = [[float('inf')]*n for _ in range(n)]
    for i in range(n): dist[i][i] = 0
    for u, v, w in edges:
        dist[u][v] = dist[v][u] = w
    for k in range(n):
        for i in range(n):
            for j in range(n):
                dist[i][j] = min(dist[i][j], dist[i][k]+dist[k][j])
    res = -1; best = float('inf')
    for i in range(n):
        cnt = sum(1 for j in range(n) if j != i and dist[i][j] <= distanceThreshold)
        if cnt <= best: best = cnt; res = i
    return res
```

## 1335. Minimum Difficulty of a Job Schedule
https://leetcode.com/problems/minimum-difficulty-of-a-job-schedule/

```python
def minDifficulty(self, jobDifficulty, d):
    n = len(jobDifficulty)
    if n < d: return -1
    dp = [[float('inf')]*(d+1) for _ in range(n+1)]
    dp[0][0] = 0
    for i in range(1, n+1):
        for dd in range(1, min(i, d)+1):
            mx = 0
            for j in range(i, dd-1, -1):
                mx = max(mx, jobDifficulty[j-1])
                dp[i][dd] = min(dp[i][dd], dp[j-1][dd-1] + mx)
    return dp[n][d]
```

## 1339. Maximum Product of Splitted Binary Tree
https://leetcode.com/problems/maximum-product-of-splitted-binary-tree/

先求总和，再找最接近总和一半的子树和。

```python
def maxProduct(self, root):
    MOD = 10**9 + 7
    self.total = 0
    self.res = 0
    def get_sum(node):
        if not node: return 0
        s = node.val + get_sum(node.left) + get_sum(node.right)
        return s
    self.total = get_sum(root)
    def dfs(node):
        if not node: return 0
        s = node.val + dfs(node.left) + dfs(node.right)
        self.res = max(self.res, s * (self.total - s))
        return s
    dfs(root)
    return self.res % MOD
```

```python
from collections import deque, defaultdict
def minJumps(self, arr):
    n = len(arr)
    if n == 1: return 0
    graph = defaultdict(list)
    for i, v in enumerate(arr): graph[v].append(i)
    visited = {0}
    q = deque([0]); steps = 0
    while q:
        steps += 1
        for _ in range(len(q)):
            i = q.popleft()
            for j in [i-1, i+1] + graph[arr[i]]:
                if j == n-1: return steps
                if 0 <= j < n and j not in visited:
                    visited.add(j); q.append(j)
            graph[arr[i]].clear()  # 避免重复遍历
    return -1
```

```python
def maxEvents(self, events):
```

## 1354. Construct Target Array With Multiple Sums
https://leetcode.com/problems/construct-target-array-with-multiple-sums/

```python
import heapq
def isPossible(self, target):
    total = sum(target)
    h = [-x for x in target]
    heapq.heapify(h)
    while -h[0] > 1:
        mx = -heapq.heappop(h)
        rest = total - mx
        if rest <= 0 or rest == 0: return False
        if mx <= rest: return False
        mx %= rest
        if mx == 0: mx = rest
        heapq.heappush(h, -mx)
        total = rest + mx
    return True
```

## 1361. Validate Binary Tree Nodes
https://leetcode.com/problems/validate-binary-tree-nodes/

找根（入度0），BFS验证所有节点恰好访问一次。

```python
from collections import deque
def validateBinaryTreeNodes(self, n, leftChild, rightChild):
    indegree = [0] * n
    for l, r in zip(leftChild, rightChild):
        if l != -1: indegree[l] += 1
        if r != -1: indegree[r] += 1
    roots = [i for i in range(n) if indegree[i] == 0]
    if len(roots) != 1: return False
    visited = set()
    q = deque([roots[0]])
    while q:
        node = q.popleft()
        if node in visited: return False
        visited.add(node)
        if leftChild[node] != -1: q.append(leftChild[node])
        if rightChild[node] != -1: q.append(rightChild[node])
    return len(visited) == n
```

```python
def largestMultipleOfThree(self, digits):
    digits.sort(reverse=True)
    total = sum(digits)
    r = total % 3
    if r == 0:
        return '0' if digits[0] == 0 else ''.join(map(str, digits))
    rem1 = sorted([d for d in digits if d % 3 == r])
    rem2 = sorted([d for d in digits if d % 3 == (3 - r) % 3 * 2 % 3])
    # 尝试删一个余数为 r 的，或删两个余数为 3-r 的
    option1 = digits[:] if not rem1 else None
    option2 = digits[:] if len(rem2) < 2 else None
    if rem1:
        option1 = digits[:]
        option1.remove(rem1[0])
    if len(rem2) >= 2:
        option2 = digits[:]
        option2.remove(rem2[0])
        option2.remove(rem2[1])
    res = ''
    for opt in [option1, option2]:
        if opt is not None:
            opt.sort(reverse=True)
            s = ''.join(map(str, opt))
            if s and (not s.startswith('0') or s == '0'):
                if len(s) > len(res): res = s
    return res if res else ''
```

## 1368. Minimum Cost to Make at Least One Valid Path in a Grid
https://leetcode.com/problems/minimum-cost-to-make-at-least-one-valid-path-in-a-grid/

0-1 BFS：沿箭头方向代价0，其他方向代价1。

```python
from collections import deque
def minCost(self, grid):
    m, n = len(grid), len(grid[0])
    dirs = [(0,1),(0,-1),(1,0),(-1,0)]
    dist = [[float('inf')]*n for _ in range(m)]
    dist[0][0] = 0
    dq = deque([(0, 0, 0)])
    while dq:
        d, r, c = dq.popleft()
        if d > dist[r][c]: continue
        for i, (dr, dc) in enumerate(dirs):
            nr, nc = r+dr, c+dc
            cost = 0 if grid[r][c] == i+1 else 1
            if 0 <= nr < m and 0 <= nc < n and d+cost < dist[nr][nc]:
                dist[nr][nc] = d+cost
                if cost == 0: dq.appendleft((d, nr, nc))
                else: dq.append((d+cost, nr, nc))
    return dist[m-1][n-1]
```

```python
    mask = res = 0
```

```python
def maxSumBST(self, root):
    self.res = 0
    def dfs(node):  # returns (is_bst, min_val, max_val, sum)
        if not node: return True, float('inf'), float('-inf'), 0
        lb, lmin, lmax, ls = dfs(node.left)
        rb, rmin, rmax, rs = dfs(node.right)
        if lb and rb and lmax < node.val < rmin:
            s = ls + rs + node.val
            self.res = max(self.res, s)
            return True, min(lmin, node.val), max(rmax, node.val), s
        return False, 0, 0, 0
    dfs(root)
    return self.res
```

```python
from collections import defaultdict, deque
def frogPosition(self, n, edges, t, target):
    if n == 1: return 1.0
    graph = defaultdict(set)
    for u, v in edges:
        graph[u].add(v); graph[v].add(u)
    visited = {1}
    q = deque([(1, 1.0)])
    for _ in range(t):
        for _ in range(len(q)):
            node, prob = q.popleft()
            children = [c for c in graph[node] if c not in visited]
            if not children:
                q.append((node, prob)); continue
            for c in children:
                visited.add(c)
                q.append((c, prob / len(children)))
    for node, prob in q:
        if node == target: return prob
    return 0.0
```

```python
class UndergroundSystem:
    def __init__(self):
        self.check_ins = {}  # id -> (station, time)
        self.trips = {}  # (start, end) -> [total_time, count]
```

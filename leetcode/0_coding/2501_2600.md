## 2501. Longest Square Streak in an Array

```python
def longestSquareStreak(self, nums):
    s = set(nums); res = -1
    for x in sorted(nums):
        cnt = 0; cur = x
        while cur in s:
            cnt += 1; cur = cur * cur
        if cnt >= 2: res = max(res, cnt)
    return res
```

## 2502. Design Memory Allocator

https://leetcode.cn/problems/design-memory-allocator/

```python
class Allocator:
    def __init__(self, n):
        self.mem = [0] * n
    def allocate(self, size, mID):
        cnt = 0
        for i in range(len(self.mem)):
            if self.mem[i] == 0: cnt += 1
            else: cnt = 0
            if cnt == size:
                for j in range(i - size + 1, i + 1):
                    self.mem[j] = mID
                return i - size + 1
        return -1
    def freeMemory(self, mID):
        cnt = 0
        for i in range(len(self.mem)):
            if self.mem[i] == mID:
                self.mem[i] = 0; cnt += 1
        return cnt
```

## 2503. Maximum Number of Points From Grid Queries

https://leetcode.cn/problems/maximum-number-of-points-from-grid-queries/

离线 + BFS/最小堆。

```python
import heapq
def maxPoints(self, grid, queries):
    m, n = len(grid), len(grid[0])
    sorted_q = sorted(enumerate(queries), key=lambda x: x[1])
    h = [(grid[0][0], 0, 0)]; visited = [[False]*n for _ in range(m)]
    visited[0][0] = True; cnt = 0
    res = [0] * len(queries)
    for qi, q in sorted_q:
        while h and h[0][0] < q:
            val, r, c = heapq.heappop(h); cnt += 1
            for dr, dc in [(0,1),(0,-1),(1,0),(-1,0)]:
                nr, nc = r+dr, c+dc
                if 0 <= nr < m and 0 <= nc < n and not visited[nr][nc]:
                    visited[nr][nc] = True
                    heapq.heappush(h, (grid[nr][nc], nr, nc))
        res[qi] = cnt
    return res
```

```python
def cycleLengthQueries(self, n, queries):
    res = []
    for a, b in queries:
        cnt = 1
        while a != b:
            if a > b: a //= 2
            else: b //= 2
            cnt += 1
        res.append(cnt)
    return res
```

```python
from collections import Counter
def takeCharacters(self, s, k):
    total = Counter(s)
    if any(total[c] < k for c in 'abc'): return -1
    n = len(s); res = n; cnt = Counter(); left = 0
    for right in range(n):
        cnt[s[right]] += 1
```

```python
def maximumTastiness(self, price, k):
    price.sort()
    def check(d):
        cnt = 1; prev = price[0]
        for p in price[1:]:
            if p - prev >= d: cnt += 1; prev = p
        return cnt >= k
    lo, hi = 0, price[-1] - price[0]
    while lo < hi:
        mid = (lo + hi + 1) // 2
        if check(mid): lo = mid
        else: hi = mid - 1
    return lo
```

## 2518. Number of Great Partitions

https://leetcode.cn/problems/number-of-great-partitions/

总方案减去"差"方案。

```python
def countPartitions(self, nums, k):
    MOD = 10**9 + 7
    if sum(nums) < 2 * k: return 0
    # "差"方案 = 子集和 < k 的方案数 * 2
    dp = [0] * k; dp[0] = 1
    for x in nums:
        for j in range(k-1, x-1, -1):
            dp[j] = (dp[j] + dp[j-x]) % MOD
    bad = sum(dp) * 2 % MOD
    total = pow(2, len(nums), MOD)
    return (total - bad) % MOD
```

## 2523. Closest Prime Numbers in Range

https://leetcode.cn/problems/closest-prime-numbers-in-range/

```python
def closestPrimes(self, left, right):
    sieve = [True] * (right + 1)
    sieve[0] = sieve[1] = False
    for i in range(2, int(right**0.5)+1):
        if sieve[i]:
            for j in range(i*i, right+1, i): sieve[j] = False
    primes = [i for i in range(left, right+1) if sieve[i]]
    if len(primes) < 2: return [-1, -1]
    best = (primes[1] - primes[0], 0)
    for i in range(1, len(primes)-1):
        d = primes[i+1] - primes[i]
        if d < best[0]: best = (d, i)
    return [primes[best[1]], primes[best[1]+1]]
```

```python
from collections import Counter
def isItPossible(self, word1, word2):
    c1, c2 = Counter(word1), Counter(word2)
    for a in 'abcdefghijklmnopqrstuvwxyz':
        for b in 'abcdefghijklmnopqrstuvwxyz':
```

## 2536. Increment Submatrices by One

```python
def rangeAddQueries(self, n, queries):
    diff = [[0]*(n+1) for _ in range(n+1)]
    for r1, c1, r2, c2 in queries:
        diff[r1][c1] += 1; diff[r1][c2+1] -= 1
        diff[r2+1][c1] -= 1; diff[r2+1][c2+1] += 1
    res = [[0]*n for _ in range(n)]
    for i in range(n):
        for j in range(n):
            diff[i][j] += (diff[i-1][j] if i else 0) + (diff[i][j-1] if j else 0) - (diff[i-1][j-1] if i and j else 0)
            res[i][j] = diff[i][j]
    return res
```

## 2555. Maximize Win From Two Segments

```python
from bisect import bisect_right
def maximizeWin(self, prizePositions, k):
    n = len(prizePositions)
    dp = [0] * (n + 1)  # dp[i] = 前 i 个位置中一段能获得的最大奖品
    res = 0
    for i in range(n):
        j = bisect_right(prizePositions, prizePositions[i] - k - 1)
        # 当前段覆盖 [prizePositions[j], prizePositions[i]]
        cnt = i - j + 1
        dp[i + 1] = max(dp[i], cnt)
    # 第二段
    for i in range(n):
        j = bisect_right(prizePositions, prizePositions[i] - k - 1)
        cnt = i - j + 1
        res = max(res, cnt + dp[j])
    return res
```

```python
def maxNumOfMarkedIndices(self, nums):
    nums.sort(); n = len(nums)
    j = n // 2; res = 0
    for i in range(n // 2):
        while j < n and nums[j] < 2 * nums[i]: j += 1
        if j < n: res += 2; j += 1
    return res
```

```python
    ranges.sort(); MOD = 10**9 + 7
    groups = 0; end = -1
```

## 2583. Kth Largest Sum in a Binary Tree

https://leetcode.cn/problems/kth-largest-sum-in-a-binary-tree/

```python
from collections import deque
def kthLargestLevelSum(self, root, k):
    q = deque([root]); sums = []
    while q:
        s = 0
        for _ in range(len(q)):
            node = q.popleft(); s += node.val
            if node.left: q.append(node.left)
            if node.right: q.append(node.right)
        sums.append(s)
    if k > len(sums): return -1
    sums.sort(reverse=True)
    return sums[k-1]
```

## 2585. Number of Ways to Earn Points

https://leetcode.cn/problems/number-of-ways-to-earn-points/

分组背包。

```python
def waysToReachTarget(self, target, types):
    MOD = 10**9 + 7
    dp = [0] * (target + 1); dp[0] = 1
    for count, marks in types:
        for j in range(target, -1, -1):
            for k in range(1, count + 1):
                if j - k * marks >= 0:
                    dp[j] = (dp[j] + dp[j - k * marks]) % MOD
    return dp[target]
```

## 2589. Minimum Time to Complete All Tasks

https://leetcode.cn/problems/minimum-time-to-complete-all-tasks/

贪心：按结束时间排序，尽量晚运行。

```python
def findMinimumTime(self, tasks):
    tasks.sort(key=lambda x: x[1])
    used = [False] * 2001
    for start, end, dur in tasks:
        cnt = sum(used[t] for t in range(start, end + 1))
        remain = dur - cnt
        for t in range(end, start - 1, -1):
            if remain <= 0: break
            if not used[t]: used[t] = True; remain -= 1
    return sum(used)
```

## 2591. Distribute Money to Maximum Children

https://leetcode.cn/problems/distribute-money-to-maximum-children/

```python
def distMoney(self, money, children):
    money -= children  # 每人至少1
    if money < 0: return -1
    # 给尽量多的人7块
    full = min(money // 7, children)
    money -= full * 7
    remain = children - full
    if remain == 0 and money > 0: full -= 1
    if remain == 1 and money == 3: full -= 1  # 不能给3（会变成4→不行）
    return full
```

## 2594. Minimum Time to Repair Cars

https://leetcode.cn/problems/minimum-time-to-repair-cars/

二分答案。

```python
import math
def repairCars(self, ranks, cars):
    lo, hi = 1, min(ranks) * cars * cars
    while lo < hi:
        mid = (lo + hi) // 2
        if sum(int(math.sqrt(mid // r)) for r in ranks) >= cars: hi = mid
        else: lo = mid + 1
    return lo
```

## 2596. Check Knight Tour Configuration

https://leetcode.cn/problems/check-knight-tour-configuration/

```python
def checkValidGrid(self, grid):
    n = len(grid)
    pos = [None] * (n * n)
    for i in range(n):
        for j in range(n):
            pos[grid[i][j]] = (i, j)
    if pos[0] != (0, 0): return False
    for k in range(1, n * n):
        r1, c1 = pos[k-1]; r2, c2 = pos[k]
        dr, dc = abs(r1-r2), abs(c1-c2)
        if sorted([dr, dc]) != [1, 2]: return False
    return True
```

## 2597. The Number of Beautiful Subsets

https://leetcode.cn/problems/the-number-of-beautiful-subsets/

```python
from collections import Counter
def beautifulSubsets(self, nums, k):
    cnt = Counter()
    self.res = 0
    nums.sort()
    def bt(i):
        if i == len(nums):
            if cnt: self.res += 1
            return
        # skip
        bt(i + 1)
        # take if valid
        if cnt[nums[i] - k] == 0:
            cnt[nums[i]] += 1
            bt(i + 1)
            cnt[nums[i]] -= 1
    bt(0)
    return self.res
```

## 2598. Smallest Missing Non-negative Integer After Operations

https://leetcode.cn/problems/smallest-missing-non-negative-integer-after-operations/

```python
from collections import Counter
def findSmallestInteger(self, nums, value):
    cnt = Counter(x % value for x in nums)
    for i in range(len(nums) + 1):
        r = i % value
        if cnt[r] == 0: return i
        cnt[r] -= 1
    return len(nums)
```

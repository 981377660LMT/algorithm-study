```python
def distributeCandies(self, candies, num_people):
    res = [0] * num_people
    i = 0
    while candies > 0:

        candies -= (i + 1)
        i += 1
```

## 1104. Path In Zigzag Labelled Binary Tree

https://leetcode.cn/problems/path-in-zigzag-labelled-binary-tree/

```python
def pathInZigZagTree(self, label):
    res = []
    while label >= 1:
        res.append(label)
        level = label.bit_length() - 1
        # 翻转得到正常位置，再算父节点，再翻转回来
        mx = (1 << (level + 1)) - 1
        mn = 1 << level
        mirror = mx + mn - label
        label = mirror // 2
    return res[::-1]
```

```python
def minHeightShelves(self, books, shelfWidth):
    n = len(books)
    dp = [float('inf')] * (n + 1)
    dp[0] = 0
    for i in range(1, n + 1):
        w = h = 0
        for j in range(i, 0, -1):
            w += books[j-1][0]
            if w > shelfWidth: break
            h = max(h, books[j-1][1])
            dp[i] = min(dp[i], dp[j-1] + h)
    return dp[n]
```

## 1106. Parsing A Boolean Expression

https://leetcode.cn/problems/parsing-a-boolean-expression/

递归/栈解析布尔表达式。

```python
def parseBoolExpr(self, expression):
    stack = []
    for c in expression:
        if c == ',': continue
        elif c == ')':
            vals = []
            while stack[-1] not in '!&|':
                vals.append(stack.pop())
            op = stack.pop()
            if op == '!': stack.append(not vals[0])
            elif op == '&': stack.append(all(vals))
            else: stack.append(any(vals))
        elif c == 't': stack.append(True)
        elif c == 'f': stack.append(False)
        elif c == '(': continue
        else: stack.append(c)
    return 't' if stack[0] else 'f'
```

## 1129. Shortest Path with Alternating Colors

https://leetcode.cn/problems/shortest-path-with-alternating-colors/

```python
from collections import deque, defaultdict
def shortestAlternatingPaths(self, n, redEdges, blueEdges):
    graph = defaultdict(list)
    for u, v in redEdges: graph[(u, 0)].append(v)
    for u, v in blueEdges: graph[(u, 1)].append(v)
    res = [-1] * n
    q = deque([(0, 0, 0), (0, 1, 0)])  # node, color, dist
    visited = {(0, 0), (0, 1)}
    res[0] = 0
    while q:
        node, color, dist = q.popleft()
        nc = 1 - color
        for nb in graph[(node, nc)]:
            if (nb, nc) not in visited:
                visited.add((nb, nc))
                if res[nb] == -1: res[nb] = dist + 1
                q.append((nb, nc, dist + 1))
    return res
```

## 1139. Largest 1-Bordered Square

https://leetcode.cn/problems/largest-1-bordered-square/

预计算每个位置向右和向下连续1的长度。

```python
def largest1BorderedSquare(self, grid):
    m, n = len(grid), len(grid[0])
    right = [[0]*n for _ in range(m)]
    down = [[0]*n for _ in range(m)]
    for i in range(m-1, -1, -1):
        for j in range(n-1, -1, -1):
            if grid[i][j]:
                right[i][j] = (right[i][j+1] if j+1 < n else 0) + 1
                down[i][j] = (down[i+1][j] if i+1 < m else 0) + 1
    for side in range(min(m,n), 0, -1):
        for i in range(m-side+1):
            for j in range(n-side+1):
                if right[i][j] >= side and down[i][j] >= side and right[i+side-1][j] >= side and down[i][j+side-1] >= side:
                    return side * side
    return 0
```

## 1140. Stone Game II

https://leetcode.cn/problems/stone-game-ii/

记忆化搜索：`dp(i, M)` = 从 i 开始，M 值时先手能拿的最多石子。

```python
from functools import lru_cache
def stoneGameII(self, piles):
    n = len(piles)
    suffix = piles[:]
    for i in range(n-2, -1, -1):
        suffix[i] += suffix[i+1]
    @lru_cache(None)
    def dp(i, M):
        if i >= n: return 0
        if i + 2 * M >= n: return suffix[i]
        return max(suffix[i] - dp(i+x, max(M, x)) for x in range(1, 2*M+1))
    return dp(0, 1)
```

```python
def longestCommonSubsequence(self, text1, text2):

    dp = [0] * (n + 1)
    for i in range(1, m + 1):
        prev = 0
        for j in range(1, n + 1):
            tmp = dp[j]
```

```python
        for v in graph[u]:
            if v == parent: continue
            if disc[v] == 0:
                dfs(v, u)
                low[u] = min(low[u], low[v])
                if low[v] > disc[u]:
                    res.append([u, v])
            else:
                low[u] = min(low[u], disc[v])
    dfs(0, -1)
    return res
```

## 1200. Minimum Absolute Difference

https://leetcode.cn/problems/minimum-absolute-difference/

```python
def minimumAbsDifference(self, arr):
    arr.sort()
    min_diff = min(arr[i+1] - arr[i] for i in range(len(arr)-1))
    return [[arr[i], arr[i+1]] for i in range(len(arr)-1) if arr[i+1] - arr[i] == min_diff]
```

## 2601. Prime Subtraction Operation
https://leetcode.com/problems/prime-subtraction-operation/

```python
def primeSubOperation(self, nums):
    def sieve(n):
        is_p = [True]*(n+1); is_p[0]=is_p[1]=False
        for i in range(2,int(n**0.5)+1):
            if is_p[i]:
                for j in range(i*i,n+1,i): is_p[j]=False
        return [i for i in range(2,n+1) if is_p[i]]
    primes = sieve(1000)
    from bisect import bisect_left
    prev = 0
    for x in nums:
        # 找最大质数 p 使得 x - p > prev
        need = x - prev  # x - p > prev → p < need
        idx = bisect_left(primes, need) - 1
        if idx >= 0: x -= primes[idx]
        if x <= prev: return False
        prev = x
    return True
```

## 2602. Minimum Operations to Make All Array Elements Equal
https://leetcode.com/problems/minimum-operations-to-make-all-array-elements-equal/

```python
from bisect import bisect_left
def minOperations(self, nums, queries):
    nums.sort(); n = len(nums)
    prefix = [0]*(n+1)
    for i in range(n): prefix[i+1] = prefix[i] + nums[i]
    res = []
    for q in queries:
        i = bisect_left(nums, q)
        left = q * i - prefix[i]
        right = (prefix[n] - prefix[i]) - q * (n - i)
        res.append(left + right)
    return res
```

```python
    n = len(arr); g = gcd(n, k)
    res = 0
```

## 2608. Shortest Cycle in a Graph
https://leetcode.com/problems/shortest-cycle-in-a-graph/

```python
from collections import deque
def findShortestCycle(self, n, edges):
    adj = [[] for _ in range(n)]
    for u, v in edges: adj[u].append(v); adj[v].append(u)
    res = float('inf')
    for start in range(n):
        dist = [-1] * n; dist[start] = 0
        q = deque([start])
        while q:
            u = q.popleft()
            for v in adj[u]:
                if dist[v] == -1:
                    dist[v] = dist[u] + 1; q.append(v)
                elif dist[v] >= dist[u]:
                    res = min(res, dist[u] + dist[v] + 1)
    return res if res < float('inf') else -1
```

```python
        if n < 2: return False
        for i in range(2, int(n**0.5)+1):
            if n % i == 0: return False
        return True
    n = len(nums); res = 0
```

## 2615. Sum of Distances
https://leetcode.com/problems/sum-of-distances/

```python
from collections import defaultdict
def distance(self, nums):
    pos = defaultdict(list)
    for i, x in enumerate(nums): pos[x].append(i)
    res = [0] * len(nums)
    for indices in pos.values():
        n = len(indices)
        prefix = [0] * (n + 1)
        for i in range(n): prefix[i+1] = prefix[i] + indices[i]
        for k in range(n):
            left = indices[k] * k - prefix[k]
            right = prefix[n] - prefix[k+1] - indices[k] * (n - k - 1)
            res[indices[k]] = left + right
    return res
```

## 2616. Minimize the Maximum Difference of Pairs

```python
def minimizeMax(self, nums, p):
    nums.sort()
    def check(d):
        cnt = i = 0
        while i < len(nums) - 1:
            if nums[i+1] - nums[i] <= d: cnt += 1; i += 2
            else: i += 1
        return cnt >= p
    lo, hi = 0, nums[-1] - nums[0]
    while lo < hi:
        mid = (lo + hi) // 2
        if check(mid): hi = mid
        else: lo = mid + 1
    return lo
```

## 2654. Minimum Number of Operations to Make All Array Elements Equal to 1

```python
from math import gcd
def minOperations(self, nums):
    ones = nums.count(1)
    if ones: return len(nums) - ones
    # 找最短子数组使得 gcd = 1
    n = len(nums); best = float('inf')
    for i in range(n):
        g = 0
        for j in range(i, n):
            g = gcd(g, nums[j])
            if g == 1: best = min(best, j - i); break
    return best + n - 1 if best < float('inf') else -1
```

```python
    def dfs(i, j):
        if i < 0 or i >= m or j < 0 or j >= n or grid[i][j] == 0: return 0
        val = grid[i][j]; grid[i][j] = 0
```

```python
from collections import Counter
class FrequencyTracker:
    def __init__(self):
        self.cnt = Counter(); self.freq = Counter()
    def add(self, number):
        self.freq[self.cnt[number]] -= 1
        self.cnt[number] += 1
        self.freq[self.cnt[number]] += 1
    def deleteOne(self, number):
        if self.cnt[number] == 0: return
        self.freq[self.cnt[number]] -= 1
        self.cnt[number] -= 1
        self.freq[self.cnt[number]] += 1
    def hasFrequency(self, frequency):
        return self.freq[frequency] > 0
```

```python
def colorTheArray(self, n, queries):
    colors = [0] * n; same = 0; res = []
    for i, c in queries:
        # 移除旧的相邻
        if colors[i]:
            if i > 0 and colors[i-1] == colors[i]: same -= 1
            if i < n-1 and colors[i+1] == colors[i]: same -= 1
        colors[i] = c
```

```python
def maximumOr(self, nums, k):
    n = len(nums)
    suffix_or = [0] * (n + 1)
    for i in range(n-1, -1, -1): suffix_or[i] = suffix_or[i+1] | nums[i]
    res = prefix = 0
    for i in range(n):
        res = max(res, prefix | (nums[i] << k) | suffix_or[i+1])
        prefix |= nums[i]
    return res
```

```python
def maxMoves(self, grid):
    m, n = len(grid), len(grid[0])
    dp = set(range(m))
    for j in range(n-1):
        new_dp = set()
        for i in dp:
            for di in [-1, 0, 1]:
                ni = i + di
                if 0 <= ni < m and grid[ni][j+1] > grid[i][j]:
                    new_dp.add(ni)
        if not new_dp: return j
        dp = new_dp
```

## 2685. Count the Number of Complete Components
https://leetcode.com/problems/count-the-number-of-complete-components/

```python
def countCompleteComponents(self, n, edges):
    parent = list(range(n)); rank = [0]*n
    def find(x):
        while parent[x] != x: parent[x] = parent[parent[x]]; x = parent[x]
        return x
    def union(a, b):
        pa, pb = find(a), find(b)
        if pa == pb: return
        if rank[pa] < rank[pb]: pa, pb = pb, pa
        parent[pb] = pa
        if rank[pa] == rank[pb]: rank[pa] += 1
    for u, v in edges: union(u, v)
    from collections import Counter
    comp_nodes = Counter(find(i) for i in range(n))
    comp_edges = Counter()
    for u, v in edges: comp_edges[find(u)] += 1
    return sum(1 for r in comp_nodes if comp_edges[r] == comp_nodes[r] * (comp_nodes[r]-1) // 2)
```

```python
def punishmentNumber(self, n):
    def can_partition(s, target):
        if not s: return target == 0
        for i in range(1, len(s)+1):
            if int(s[:i]) <= target and can_partition(s[i:], target - int(s[:i])):
                return True
        return False
    return sum(i*i for i in range(1, n+1) if can_partition(str(i*i), i))
```

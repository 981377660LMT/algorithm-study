## 821. 字符的最短距离 (Shortest Distance to a Character)

**描述**：给定字符串和目标字符，求每个位置到最近该字符的距离。

**思路**：左扫一遍+右扫一遍取最小。

```python
def shortestToChar(s, c):
    n = len(s); res = [float('inf')]*n
    pos = -n
    for i in range(n):
        if s[i]==c: pos=i
        res[i] = i-pos
    pos = 2*n
    for i in range(n-1,-1,-1):
        if s[i]==c: pos=i
        res[i] = min(res[i], pos-i)
    return res
```

```typescript
function shortestToChar(s: string, c: string): number[] {
  const n=s.length, res=new Array(n).fill(n); let pos=-n
  for(let i=0;i<n;i++){if(s[i]===c)pos=i; res[i]=i-pos}
  pos=2*n; for(let i=n-1;i>=0;i--){if(s[i]===c)pos=i; res[i]=Math.min(res[i],pos-i)}
  return res
}
```

## 824. 山羊拉丁文 (Goat Latin)

**描述**：按规则变换句子。

**思路**：模拟字符串操作。

```python
def toGoatLatin(sentence):
    vowels = set('aeiouAEIOU')
    words = sentence.split()
    res = []
    for i, w in enumerate(words):
        if w[0] in vowels: w = w + 'ma'
        else: w = w[1:] + w[0] + 'ma'
        w += 'a'*(i+1)
        res.append(w)
    return ' '.join(res)
```

```typescript
function toGoatLatin(sentence: string): string {
  const v=new Set('aeiouAEIOU')

    w=v.has(w[0])?w+'ma':w.slice(1)+w[0]+'ma'; return w+'a'.repeat(i+1)
  }).join(' ')
}
```

## 827. 最大人工岛 (Making A Large Island)

**描述**：最多翻转一个 0 为 1，求最大岛屿面积。

**思路**：先 DFS 标记每个岛面积，再遍历每个 0 看周围岛。

```python
def largestIsland(grid):
    n = len(grid)
    island_id = 2; area = {}
    def dfs(r, c, idx):
        if r<0 or r>=n or c<0 or c>=n or grid[r][c]!=1: return 0
        grid[r][c] = idx
        return 1+dfs(r+1,c,idx)+dfs(r-1,c,idx)+dfs(r,c+1,idx)+dfs(r,c-1,idx)
    for i in range(n):
        for j in range(n):
            if grid[i][j]==1:
                area[island_id] = dfs(i,j,island_id)
                island_id += 1
    res = max(area.values()) if area else 0
    for i in range(n):
        for j in range(n):
            if grid[i][j]==0:
                seen = set()
                s = 1
                for di,dj in [(0,1),(0,-1),(1,0),(-1,0)]:
                    ni,nj = i+di,j+dj
                    if 0<=ni<n and 0<=nj<n and grid[ni][nj]>1 and grid[ni][nj] not in seen:
                        seen.add(grid[ni][nj]); s += area[grid[ni][nj]]
                res = max(res, s)
    return res
```

```typescript
function largestIsland(grid: number[][]): number {
  const n=grid.length; let id=2; const area=new Map<number,number>()
  const dfs=(r:number,c:number,idx:number):number=>{
    if(r<0||r>=n||c<0||c>=n||grid[r][c]!==1) return 0
    grid[r][c]=idx; return 1+dfs(r+1,c,idx)+dfs(r-1,c,idx)+dfs(r,c+1,idx)+dfs(r,c-1,idx)
  }
  for(let i=0;i<n;i++)for(let j=0;j<n;j++) if(grid[i][j]===1){area.set(id,dfs(i,j,id));id++}
  let res=Math.max(...area.values(),0)
  const dirs=[[0,1],[0,-1],[1,0],[-1,0]]
  for(let i=0;i<n;i++)for(let j=0;j<n;j++) if(grid[i][j]===0){
    const seen=new Set<number>(); let s=1
    for(const [di,dj] of dirs){const ni=i+di,nj=j+dj; if(ni>=0&&ni<n&&nj>=0&&nj<n&&grid[ni][nj]>1&&!seen.has(grid[ni][nj])){seen.add(grid[ni][nj]);s+=area.get(grid[ni][nj])!}}
    res=Math.max(res,s)
  }
  return res
}
```

## 830. 较大分组的位置 (Positions of Large Groups)

**描述**：连续相同字母长度>=3 的分组起止索引。

**思路**：双指针扫一遍。

```python
def largeGroupPositions(s):
    res = []; i = 0
    while i < len(s):
        j = i
        while j < len(s) and s[j]==s[i]: j += 1
        if j-i >= 3: res.append([i, j-1])
        i = j
    return res
```

```typescript
function largeGroupPositions(s: string): number[][] {
  const res:number[][]=[];let i=0
  while(i<s.length){let j=i;while(j<s.length&&s[j]===s[i])j++;if(j-i>=3)res.push([i,j-1]);i=j}
  return res
}
```

```python
def flipAndInvertImage(image):
```

## 835. 图像重叠 (Image Overlap)

**描述**：两个二进制矩阵平移后重叠1的最大个数。

**思路**：枚举所有 1 的点对偏移量，统计相同偏移最多。

```python
def largestOverlap(img1, img2):
    from collections import Counter
    n = len(img1)
    pts1 = [(i,j) for i in range(n) for j in range(n) if img1[i][j]]
    pts2 = [(i,j) for i in range(n) for j in range(n) if img2[i][j]]
    if not pts1 or not pts2: return 0
    cnt = Counter((a-c,b-d) for a,b in pts1 for c,d in pts2)
    return max(cnt.values())
```

```typescript
function largestOverlap(img1: number[][], img2: number[][]): number {
  const n=img1.length
  const pts1:number[][]=[], pts2:number[][]=[]
  for(let i=0;i<n;i++)for(let j=0;j<n;j++){if(img1[i][j])pts1.push([i,j]);if(img2[i][j])pts2.push([i,j])}
  if(!pts1.length||!pts2.length) return 0
  const cnt=new Map<string,number>()
  for(const [a,b] of pts1) for(const [c,d] of pts2){const k=`${a-c},${b-d}`;cnt.set(k,(cnt.get(k)??0)+1)}
  return Math.max(...cnt.values())
}
```

## 838. 推多米诺 (Push Dominoes)

**描述**：'L'、'R'、'.' 的多米诺骨牌下推结果。

**思路**：BFS 模拟力，同时到达则抵消。

```python
def pushDominoes(dominoes):
    from collections import deque
    n = len(dominoes); s = list(dominoes)
    q = deque()
    force = [0]*n
    for i,c in enumerate(s):
        if c=='L': q.append((i,-1)); force[i]=-1
        elif c=='R': q.append((i,1)); force[i]=1
    while q:
        i, d = q.popleft()
        ni = i+d
        if 0<=ni<n and s[ni]=='.':
            if force[ni]==0: force[ni]=d; s[ni]='L' if d<0 else 'R'; q.append((ni,d))
            elif force[ni]==-d: s[ni]='.'  # 抵消
    return ''.join(s)
```

```typescript
function pushDominoes(dominoes: string): string {
  const n=dominoes.length, s=[...dominoes], force=new Array(n).fill(0)
  const q:[number,number][]=[]
  for(let i=0;i<n;i++){if(s[i]==='L'){q.push([i,-1]);force[i]=-1}else if(s[i]==='R'){q.push([i,1]);force[i]=1}}
  let head=0
  while(head<q.length){
    const [i,d]=q[head++]; const ni=i+d
    if(ni>=0&&ni<n&&s[ni]==='.'){
      if(!force[ni]){force[ni]=d;s[ni]=d<0?'L':'R';q.push([ni,d])}
      else if(force[ni]===-d) s[ni]='.'
    }
  }
  return s.join('')
}
```

## 840. 矩阵中的幻方 (Magic Squares In Grid)

**描述**：3x3 子矩阵中有多少个幻方。

**思路**：暴力检查每个 3x3。

```python
def numMagicSquaresInside(grid):
    def magic(r, c):
        vals = set()
        for i in range(3):
            for j in range(3):
                v = grid[r+i][c+j]
                if v < 1 or v > 9: return False
                vals.add(v)
        if len(vals) != 9: return False
        s = sum(grid[r][c:c+3])
        for i in range(3):
            if sum(grid[r+i][c:c+3]) != s: return False
            if sum(grid[r+j][c+i] for j in range(3)) != s: return False
        if grid[r][c]+grid[r+1][c+1]+grid[r+2][c+2] != s: return False
        if grid[r][c+2]+grid[r+1][c+1]+grid[r+2][c] != s: return False
        return True
    m, n = len(grid), len(grid[0])
    return sum(magic(i,j) for i in range(m-2) for j in range(n-2))
```

```typescript
function numMagicSquaresInside(grid: number[][]): number {
  const m=grid.length,n=grid[0].length; let res=0
  for(let i=0;i<=m-3;i++) for(let j=0;j<=n-3;j++){
    const vals=new Set<number>(); let ok=true
    for(let r=0;r<3;r++) for(let c=0;c<3;c++){const v=grid[i+r][j+c]; if(v<1||v>9)ok=false; vals.add(v)}
    if(!ok||vals.size!==9) continue
    const s=grid[i][j]+grid[i][j+1]+grid[i][j+2]; let valid=true
    for(let r=0;r<3&&valid;r++){let rs=0,cs=0;for(let c=0;c<3;c++){rs+=grid[i+r][j+c];cs+=grid[i+c][j+r]}; if(rs!==s||cs!==s)valid=false}
    if(grid[i][j]+grid[i+1][j+1]+grid[i+2][j+2]!==s) valid=false
    if(grid[i][j+2]+grid[i+1][j+1]+grid[i+2][j]!==s) valid=false
    if(valid) res++
  }
  return res
}
```

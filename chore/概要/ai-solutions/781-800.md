## 783. 二叉搜索树节点最小距离 (Minimum Distance Between BST Nodes)

**描述**：BST 中任意两节点的最小差。

**思路**：中序遍历取相邻差（同530题）。

```python
def minDiffInBST(root):
    prev = [None]; res = [float('inf')]
    def inorder(node):
        if not node: return
        inorder(node.left)
        if prev[0] is not None: res[0] = min(res[0], node.val - prev[0])
        prev[0] = node.val
        inorder(node.right)
    inorder(root); return res[0]
```

```typescript
function minDiffInBST(root: TreeNode | null): number {
  let prev = -1,
    res = Infinity
  function inorder(node: TreeNode | null) {
    if (!node) return
    inorder(node.left)
    if (prev >= 0) res = Math.min(res, node.val - prev)
    prev = node.val
    inorder(node.right)
  }
  inorder(root)
  return res
}
```

## 784. 字母大小写全排列 (Letter Case Permutation)

**描述**：字符串中字母可大可小，列出所有组合。

**思路**：回溯。

```python
def letterCasePermutation(s):
    res = []
    def dfs(i, path):
        if i == len(s): res.append(''.join(path)); return
        if s[i].isalpha():
            path.append(s[i].lower()); dfs(i+1, path); path.pop()
            path.append(s[i].upper()); dfs(i+1, path); path.pop()
        else: path.append(s[i]); dfs(i+1, path); path.pop()
    dfs(0, []); return res
```

```typescript
function letterCasePermutation(s: string): string[] {
  const res: string[] = []
  function dfs(i: number, path: string[]) {
    if (i === s.length) {
      res.push(path.join(''))
      return
    }
    if (/[a-zA-Z]/.test(s[i])) {
      path.push(s[i].toLowerCase())
      dfs(i + 1, path)
      path.pop()
      path.push(s[i].toUpperCase())
      dfs(i + 1, path)
      path.pop()
    } else {
      path.push(s[i])
      dfs(i + 1, path)
      path.pop()
    }
  }
  dfs(0, [])
  return res
}
```

## 785. 判断二分图 (Is Graph Bipartite)

**描述**：判断无向图是否为二分图。

**思路**：BFS/DFS 染色。

```python
def isBipartite(graph):
    color = [0]*len(graph)
    for i in range(len(graph)):
        if color[i]: continue
        q = [i]; color[i] = 1
        while q:
            u = q.pop(0)
            for v in graph[u]:
                if color[v] == 0: color[v] = -color[u]; q.append(v)
                elif color[v] == color[u]: return False
    return True
```

```typescript
function isBipartite(graph: number[][]): boolean {
  const color = new Array(graph.length).fill(0)
  for (let i = 0; i < graph.length; i++) {
    if (color[i]) continue
    const q = [i]
    color[i] = 1
    while (q.length) {
      const u = q.shift()!
      for (const v of graph[u]) {
        if (!color[v]) {
          color[v] = -color[u]
          q.push(v)
        } else if (color[v] === color[u]) return false
      }
    }
  }
  return true
}
```

## 787. K 站中转内最便宜的航班 (Cheapest Flights Within K Stops)

**描述**：最多 k 次中转的最低费用。

**思路**：Bellman-Ford k+1 轮松弛。

```python
def findCheapestPrice(n, flights, src, dst, k):
    dist = [float('inf')]*n; dist[src] = 0
    for _ in range(k+1):
        ndist = dist[:]
        for u, v, w in flights:
            if dist[u]+w < ndist[v]: ndist[v] = dist[u]+w
        dist = ndist
    return dist[dst] if dist[dst] != float('inf') else -1
```

```typescript
function findCheapestPrice(
  n: number,
  flights: number[][],
  src: number,
  dst: number,
  k: number
): number {
  let dist = new Array(n).fill(Infinity)
  dist[src] = 0
  for (let i = 0; i <= k; i++) {
    const ndist = [...dist]
    for (const [u, v, w] of flights) if (dist[u] + w < ndist[v]) ndist[v] = dist[u] + w
    dist = ndist
  }
  return dist[dst] === Infinity ? -1 : dist[dst]
}
```

## 790. 多米诺和托米诺平铺 (Domino and Tromino Tiling)

**描述**：2×n 棋盘用 domino 和 tromino 平铺的方案数。

**思路**：DP，`dp[n] = 2*dp[n-1] + dp[n-3]`。

```python
def numTilings(n):
    MOD = 10**9+7
    if n <= 2: return n
    dp = [0]*(n+1); dp[0]=1; dp[1]=1; dp[2]=2
    for i in range(3, n+1): dp[i] = (2*dp[i-1]+dp[i-3]) % MOD
    return dp[n]
```

```typescript
function numTilings(n: number): number {
  const MOD = 1e9 + 7
  if (n <= 2) return n
  const dp = new Array(n + 1).fill(0)
  dp[0] = 1
  dp[1] = 1
  dp[2] = 2
  for (let i = 3; i <= n; i++) dp[i] = (2 * dp[i - 1] + dp[i - 3]) % MOD
  return dp[n]
}
```

## 791. 自定义字符串排序 (Custom Sort String)

**描述**：按给定顺序排列字符串。

**思路**：按 order 中的位置排序。

```python
def customSortString(order, s):
    rank = {c:i for i,c in enumerate(order)}
    return ''.join(sorted(s, key=lambda c: rank.get(c, 26)))
```

```typescript
function customSortString(order: string, s: string): string {
  const rank = new Map<string, number>()
  for (let i = 0; i < order.length; i++) rank.set(order[i], i)
  return [...s].sort((a, b) => (rank.get(a) ?? 26) - (rank.get(b) ?? 26)).join('')
}
```

## 794. 有效的井字游戏 (Valid Tic-Tac-Toe State)

**描述**：判断井字棋状态是否合理。

**思路**：检查 X 和 O 个数、谁赢了的逻辑。

```python
def validTicTacToe(board):
    xc = sum(r.count('X') for r in board)
    oc = sum(r.count('O') for r in board)
    def wins(p):
        for r in board:
            if r == p*3: return True
        for c in range(3):
            if all(board[r][c]==p for r in range(3)): return True
        if board[0][0]==board[1][1]==board[2][2]==p: return True
        if board[0][2]==board[1][1]==board[2][0]==p: return True
        return False
    xw, ow = wins('X'), wins('O')
    if oc > xc or xc > oc+1: return False
    if xw and ow: return False
    if xw and xc == oc: return False  # X赢了但O又下了
    if ow and xc > oc: return False  # O赢了但X又下了
    return True
```

```typescript
function validTicTacToe(board: string[]): boolean {
  let xc = 0,
    oc = 0
  for (const r of board)
    for (const c of r) {
      if (c === 'X') xc++
      if (c === 'O') oc++
    }
  const wins = (p: string) => {
    for (const r of board) if (r === p.repeat(3)) return true
    for (let c = 0; c < 3; c++) if ([0, 1, 2].every(r => board[r][c] === p)) return true
    if (board[0][0] === p && board[1][1] === p && board[2][2] === p) return true
    if (board[0][2] === p && board[1][1] === p && board[2][0] === p) return true
    return false
  }
  const xw = wins('X'),
    ow = wins('O')
  if (oc > xc || xc > oc + 1) return false
  if (xw && ow) return false
  if (xw && xc === oc) return false
  if (ow && xc > oc) return false
  return true
}
```

## 795. 区间子数组个数 (Number of Subarrays with Bounded Maximum)

**描述**：子数组最大值在 [left, right] 区间内的个数。

**思路**：`atMost(right) - atMost(left-1)`。

```python
def numSubarrayBoundedMax(nums, left, right):
    def atMost(bound):
        res = cnt = 0
        for n in nums:
            cnt = cnt+1 if n <= bound else 0
            res += cnt
        return res
    return atMost(right) - atMost(left-1)
```

```typescript
function numSubarrayBoundedMax(nums: number[], left: number, right: number): number {
  const atMost = (bound: number) => {
    let res = 0,
      cnt = 0
    for (const n of nums) {
      cnt = n <= bound ? cnt + 1 : 0
      res += cnt
    }
    return res
  }
  return atMost(right) - atMost(left - 1)
}
```

## 796. 旋转字符串 (Rotate String)

**描述**：s 旋转后能否得到 goal。

**思路**：`goal in s+s`。

```python
def rotateString(s, goal):
    return len(s) == len(goal) and goal in s + s
```

```typescript

}
```

## 797. 所有可能的路径 (All Paths From Source to Target)

**描述**：DAG 中从 0 到 n-1 的所有路径。

**思路**：DFS 回溯。

```python
def allPathsSourceTarget(graph):
    res = []; n = len(graph)
    def dfs(u, path):
        if u == n-1: res.append(path[:]); return
        for v in graph[u]: path.append(v); dfs(v, path); path.pop()
    dfs(0, [0]); return res
```

```typescript
function allPathsSourceTarget(graph: number[][]): number[][] {
  const res: number[][] = [],
    n = graph.length
  function dfs(u: number, path: number[]) {
    if (u === n - 1) {
      res.push([...path])
      return
    }
    for (const v of graph[u]) {
      path.push(v)
      dfs(v, path)
      path.pop()
    }
  }
  dfs(0, [0])
  return res
}
```

## 641. 设计循环双端队列 (Design Circular Deque)

**描述**：实现循环双端队列。

**思路**：数组 + 头尾指针。

```python
class MyCircularDeque:
    def __init__(self, k):
        self.q = [0]*(k+1); self.cap = k+1; self.front = 0; self.rear = 0
    def insertFront(self, value):
        if self.isFull(): return False
        self.front = (self.front-1) % self.cap; self.q[self.front] = value; return True
    def insertLast(self, value):
        if self.isFull(): return False
        self.q[self.rear] = value; self.rear = (self.rear+1) % self.cap; return True
    def deleteFront(self):
        if self.isEmpty(): return False
        self.front = (self.front+1) % self.cap; return True
    def deleteLast(self):
        if self.isEmpty(): return False
        self.rear = (self.rear-1) % self.cap; return True
    def getFront(self): return -1 if self.isEmpty() else self.q[self.front]
    def getRear(self): return -1 if self.isEmpty() else self.q[(self.rear-1) % self.cap]
    def isEmpty(self): return self.front == self.rear
    def isFull(self): return (self.rear+1) % self.cap == self.front
```

```typescript
class MyCircularDeque {
  q: number[]
  cap: number
  front = 0
  rear = 0
  constructor(k: number) {
    this.q = new Array(k + 1)
    this.cap = k + 1
  }
  insertFront(value: number): boolean {
    if (this.isFull()) return false
    this.front = (this.front - 1 + this.cap) % this.cap
    this.q[this.front] = value
    return true
  }
  insertLast(value: number): boolean {
    if (this.isFull()) return false
    this.q[this.rear] = value
    this.rear = (this.rear + 1) % this.cap
    return true
  }
  deleteFront(): boolean {
    if (this.isEmpty()) return false
    this.front = (this.front + 1) % this.cap
    return true
  }
  deleteLast(): boolean {
    if (this.isEmpty()) return false
    this.rear = (this.rear - 1 + this.cap) % this.cap
    return true
  }
  getFront(): number {
    return this.isEmpty() ? -1 : this.q[this.front]
  }
  getRear(): number {
    return this.isEmpty() ? -1 : this.q[(this.rear - 1 + this.cap) % this.cap]
  }
  isEmpty(): boolean {
    return this.front === this.rear
  }
  isFull(): boolean {
    return (this.rear + 1) % this.cap === this.front
  }
}
```

## 645. 错误的集合 (Set Mismatch)

**描述**：1~n 中一个数重复、一个数丢失。

**思路**：用 set 或数学 / 原地标记。

```python
def findErrorNums(nums):
    n = len(nums); s = set()
    dup = 0
    for x in nums:
        if x in s: dup = x
        s.add(x)
    miss = n*(n+1)//2 - sum(nums) + dup
    return [dup, miss]
```

```typescript
function findErrorNums(nums: number[]): number[] {
  const s = new Set<number>()
  let dup = 0
  for (const x of nums) {
    if (s.has(x)) dup = x
    s.add(x)
  }
  const n = nums.length
  const miss = (n * (n + 1)) / 2 - nums.reduce((a, b) => a + b, 0) + dup
  return [dup, miss]
}
```

## 646. 最长数对链 (Maximum Length of Pair Chain)

**描述**：数对链 [a,b]，b < c 时可连接 [c,d]，最长链。

**思路**：贪心，按右端排序。

```python
def findLongestChain(pairs):
    pairs.sort(key=lambda x: x[1])
    cur = float('-inf'); res = 0
    for a, b in pairs:
        if a > cur: res += 1; cur = b
    return res
```

```typescript
function findLongestChain(pairs: number[][]): number {
  pairs.sort((a, b) => a[1] - b[1])
  let cur = -Infinity,
    res = 0
  for (const [a, b] of pairs)
    if (a > cur) {
      res++
      cur = b
    }
  return res
}
```

## 647. 回文子串 (Palindromic Substrings)

**描述**：字符串中回文子串的个数。

**思路**：中心扩展。

```python
def countSubstrings(s):
    res = 0
    for center in range(2*len(s)-1):
        l, r = center//2, center//2 + center%2
        while l >= 0 and r < len(s) and s[l] == s[r]:
            res += 1; l -= 1; r += 1
    return res
```

```typescript
function countSubstrings(s: string): number {
  let res = 0
  for (let c = 0; c < 2 * s.length - 1; c++) {
    let l = c >> 1,
      r = (c >> 1) + (c & 1)
    while (l >= 0 && r < s.length && s[l] === s[r]) {
      res++
      l--
      r++
    }
  }
  return res
}
```

## 648. 单词替换 (Replace Words)

**描述**：用词根列表替换句子中的词。

**思路**：Trie 或排序后前缀匹配。

```python
def replaceWords(dictionary, sentence):
    roots = set(dictionary)
    def replace(word):
        for i in range(1, len(word)+1):
            if word[:i] in roots: return word[:i]
        return word
    return ' '.join(replace(w) for w in sentence.split())
```

```typescript
function replaceWords(dictionary: string[], sentence: string): string {
  const roots = new Set(dictionary)
  return sentence
    .split(' ')
    .map(w => {
      for (let i = 1; i <= w.length; i++) if (roots.has(w.slice(0, i))) return w.slice(0, i)
      return w
    })
    .join(' ')
}
```

## 649. Dota2 参议院 (Dota2 Senate)

**描述**：R 和 D 轮流投票禁对方，哪方最终赢。

**思路**：两个队列模拟。

```python
from collections import deque
def predictPartyVictory(senate):
    n = len(senate)
    rq, dq = deque(), deque()
    for i, c in enumerate(senate):
        if c == 'R': rq.append(i)
        else: dq.append(i)
    while rq and dq:
        r, d = rq.popleft(), dq.popleft()
        if r < d: rq.append(r+n)
        else: dq.append(d+n)
    return "Radiant" if rq else "Dire"
```

```typescript
function predictPartyVictory(senate: string): string {
  const n = senate.length,
    rq: number[] = [],
    dq: number[] = []
  for (let i = 0; i < n; i++) senate[i] === 'R' ? rq.push(i) : dq.push(i)
  let ri = 0,
    di = 0
  while (ri < rq.length && di < dq.length) {
    if (rq[ri] < dq[di]) rq.push(rq[ri] + n)
    else dq.push(dq[di] + n)
    ri++
    di++
  }
  return ri < rq.length ? 'Radiant' : 'Dire'
}
```

## 650. 两个键的键盘 (2 Keys Keyboard)

**描述**：只有"全选复制"和"粘贴"两个操作，最少几步得到 n 个 A。

**思路**：因式分解，答案 = 质因数之和。

```python
def minSteps(n):
    res = 0; d = 2
    while n > 1:
        while n % d == 0: res += d; n //= d
        d += 1
    return res
```

```typescript
function minSteps(n: number): number {
  let res = 0,
    d = 2
  while (n > 1) {
    while (n % d === 0) {
      res += d
      n /= d
    }
    d++
  }
  return res
}
```

## 652. 寻找重复的子树 (Find Duplicate Subtrees)

**描述**：找所有重复子树的根节点。

**思路**：序列化每棵子树，哈希统计。

```python
from collections import defaultdict
def findDuplicateSubtrees(root):
    cnt = defaultdict(int); res = []
    def dfs(node):
        if not node: return "#"
        key = f"{node.val},{dfs(node.left)},{dfs(node.right)}"
        cnt[key] += 1
        if cnt[key] == 2: res.append(node)
        return key
    dfs(root); return res
```

```typescript
function findDuplicateSubtrees(root: TreeNode | null): (TreeNode | null)[] {
  const cnt = new Map<string, number>(),
    res: TreeNode[] = []
  function dfs(node: TreeNode | null): string {
    if (!node) return '#'
    const key = `${node.val},${dfs(node.left)},${dfs(node.right)}`
    cnt.set(key, (cnt.get(key) ?? 0) + 1)
    if (cnt.get(key) === 2) res.push(node)
    return key
  }
  dfs(root)
  return res
}
```

## 653. 两数之和 IV - 输入二叉搜索树 (Two Sum IV - Input is a BST)

**描述**：BST 中是否存在两节点值之和等于 k。

**思路**：中序遍历 + 双指针，或 DFS + 哈希。

```python
def findTarget(root, k):
    seen = set()
    def dfs(node):
        if not node: return False
        if k - node.val in seen: return True
        seen.add(node.val)
        return dfs(node.left) or dfs(node.right)
    return dfs(root)
```

```typescript
function findTarget(root: TreeNode | null, k: number): boolean {
  const seen = new Set<number>()
  function dfs(node: TreeNode | null): boolean {
    if (!node) return false
    if (seen.has(k - node.val)) return true
    seen.add(node.val)
    return dfs(node.left) || dfs(node.right)
  }
  return dfs(root)
}
```

## 654. 最大二叉树 (Maximum Binary Tree)

**描述**：根是最大值，左右递归构建。

**思路**：递归找最大值分割。

```python
def constructMaximumBinaryTree(nums):
    if not nums: return None
    mx = max(nums); idx = nums.index(mx)
    node = TreeNode(mx)
    node.left = constructMaximumBinaryTree(nums[:idx])
    node.right = constructMaximumBinaryTree(nums[idx+1:])
    return node
```

```typescript
function constructMaximumBinaryTree(nums: number[]): TreeNode | null {
  if (!nums.length) return null
  const mx = Math.max(...nums),
    idx = nums.indexOf(mx)
  const node = new TreeNode(mx)
  node.left = constructMaximumBinaryTree(nums.slice(0, idx))
  node.right = constructMaximumBinaryTree(nums.slice(idx + 1))
  return node
}
```

## 658. 找到 K 个最接近的元素 (Find K Closest Elements)

**描述**：有序数组中找 k 个最接近 x 的元素。

**思路**：二分找左边界。

```python
def findClosestElements(arr, k, x):
    lo, hi = 0, len(arr)-k
    while lo < hi:
        mid = (lo+hi)//2
        if x - arr[mid] > arr[mid+k] - x: lo = mid+1
        else: hi = mid
    return arr[lo:lo+k]
```

```typescript
function findClosestElements(arr: number[], k: number, x: number): number[] {
  let lo = 0,
    hi = arr.length - k
  while (lo < hi) {
    const mid = (lo + hi) >> 1
    if (x - arr[mid] > arr[mid + k] - x) lo = mid + 1
    else hi = mid
  }
  return arr.slice(lo, lo + k)
}
```

## 659. 分割数组为连续子序列 (Split Array into Consecutive Subsequences)

**描述**：判断数组能否分成若干长度 ≥ 3 的连续递增子序列。

**思路**：贪心，优先续接已有序列。

```python
from collections import Counter
def isPossible(nums):
    cnt = Counter(nums)
    tail = Counter()  # tail[x] = 以x结尾的序列数
    for x in nums:
        if cnt[x] == 0: continue
        cnt[x] -= 1
        if tail[x-1] > 0:  # 续接
            tail[x-1] -= 1; tail[x] += 1
        elif cnt[x+1] > 0 and cnt[x+2] > 0:  # 新开
            cnt[x+1] -= 1; cnt[x+2] -= 1; tail[x+2] += 1
        else: return False
    return True
```

```typescript
function isPossible(nums: number[]): boolean {
  const cnt = new Map<number, number>(),
    tail = new Map<number, number>()
  for (const x of nums) cnt.set(x, (cnt.get(x) ?? 0) + 1)
  for (const x of nums) {
    if (!cnt.get(x)) continue
    cnt.set(x, cnt.get(x)! - 1)
    if ((tail.get(x - 1) ?? 0) > 0) {
      tail.set(x - 1, tail.get(x - 1)! - 1)
      tail.set(x, (tail.get(x) ?? 0) + 1)
    } else if ((cnt.get(x + 1) ?? 0) > 0 && (cnt.get(x + 2) ?? 0) > 0) {
      cnt.set(x + 1, cnt.get(x + 1)! - 1)
      cnt.set(x + 2, cnt.get(x + 2)! - 1)
      tail.set(x + 2, (tail.get(x + 2) ?? 0) + 1)
    } else return false
  }
  return true
}
```

## 201. 数字范围按位与 (Bitwise AND of Numbers Range)

**描述**：计算 [left, right] 范围内所有数字的按位与。

**思路**：找公共前缀，不断右移直到 left == right，再左移回来。

```python
def rangeBitwiseAnd(left, right):
    shift = 0
    while left < right:
        left >>= 1; right >>= 1; shift += 1
    return left << shift
```

```typescript
function rangeBitwiseAnd(left: number, right: number): number {
  let shift = 0
  while (left < right) {
    left >>= 1
    right >>= 1
    shift++
  }
  return left << shift
}
```

## 202. 快乐数 (Happy Number)

**描述**：判断一个数是否是快乐数（各位平方和最终变为1）。

**思路**：快慢指针检测循环，或哈希表记录见过的数。

```python
def isHappy(n):
    def nxt(x):
        s = 0
        while x:
            x, d = divmod(x, 10); s += d * d
        return s
    slow = fast = n
    while True:
        slow = nxt(slow); fast = nxt(nxt(fast))
        if fast == 1: return True
        if slow == fast: return False
```

```typescript
function isHappy(n: number): boolean {
  const nxt = (x: number) => {
    let s = 0
    while (x) {
      s += (x % 10) ** 2
      x = Math.floor(x / 10)
    }
    return s
  }
  let slow = n,
    fast = n
  do {
    slow = nxt(slow)
    fast = nxt(nxt(fast))
  } while (fast !== 1 && slow !== fast)
  return fast === 1
}
```

## 204. 计数质数 (Count Primes)

**描述**：统计小于 n 的质数数量。

**思路**：埃氏筛。

```python
def countPrimes(n):
    if n < 2: return 0
    is_prime = [True] * n
    is_prime[0] = is_prime[1] = False
    for i in range(2, int(n**0.5) + 1):
        if is_prime[i]:
            is_prime[i*i:n:i] = [False] * len(is_prime[i*i:n:i])
    return sum(is_prime)
```

```typescript
function countPrimes(n: number): number {
  if (n < 2) return 0
  const sieve = new Uint8Array(n).fill(1)
  sieve[0] = sieve[1] = 0
  for (let i = 2; i * i < n; i++) if (sieve[i]) for (let j = i * i; j < n; j += i) sieve[j] = 0
  return sieve.reduce((a, b) => a + b, 0)
}
```

## 205. 同构字符串 (Isomorphic Strings)

**描述**：判断两个字符串是否同构（一一映射）。

**思路**：双向映射，或比较首次出现位置模式。

```python
def isIsomorphic(s, t):
    return len(set(zip(s, t))) == len(set(s)) == len(set(t))
```

```typescript
function isIsomorphic(s: string, t: string): boolean {
  const m1 = new Map<string, string>(),
    m2 = new Map<string, string>()
  for (let i = 0; i < s.length; i++) {
    if (m1.has(s[i]) && m1.get(s[i]) !== t[i]) return false
    if (m2.has(t[i]) && m2.get(t[i]) !== s[i]) return false
    m1.set(s[i], t[i])
    m2.set(t[i], s[i])
  }
  return true
}
```

## 206. 反转链表 (Reverse Linked List)

**描述**：反转单链表。

**思路**：迭代，三指针逐步翻转。

```python
def reverseList(head):
    prev, cur = None, head
    while cur:
        cur.next, prev, cur = prev, cur, cur.next
    return prev
```

```typescript
function reverseList(head: ListNode | null): ListNode | null {
  let prev: ListNode | null = null,
    cur = head
  while (cur) {
    const nxt = cur.next
    cur.next = prev
    prev = cur
    cur = nxt
  }
  return prev
}
```

## 207. 课程表 (Course Schedule)

**描述**：判断是否能完成所有课程（检测有向图是否有环）。

**思路**：拓扑排序（BFS），入度为0的入队，处理完看计数是否等于课程数。

```python
from collections import deque, defaultdict
def canFinish(numCourses, prerequisites):
    graph = defaultdict(list)
    indeg = [0] * numCourses
    for a, b in prerequisites:
        graph[b].append(a); indeg[a] += 1
    queue = deque(i for i in range(numCourses) if indeg[i] == 0)
    cnt = 0
    while queue:
        node = queue.popleft(); cnt += 1
        for nb in graph[node]:
            indeg[nb] -= 1
            if indeg[nb] == 0: queue.append(nb)
    return cnt == numCourses
```

```typescript
function canFinish(numCourses: number, prerequisites: number[][]): boolean {
  const graph: number[][] = Array.from({ length: numCourses }, () => [])
  const indeg = new Array(numCourses).fill(0)
  for (const [a, b] of prerequisites) {
    graph[b].push(a)
    indeg[a]++
  }
  const queue: number[] = []
  for (let i = 0; i < numCourses; i++) if (indeg[i] === 0) queue.push(i)
  let cnt = 0
  while (queue.length) {
    const node = queue.shift()!
    cnt++
    for (const nb of graph[node]) {
      indeg[nb]--
      if (indeg[nb] === 0) queue.push(nb)
    }
  }
  return cnt === numCourses
}
```

## 208. 实现 Trie (前缀树) (Implement Trie)

**描述**：实现字典树的 insert、search、startsWith。

**思路**：每个节点包含26个子节点和一个结束标记。

```python
class TrieNode:
    def __init__(self):
        self.children = {}
        self.is_end = False

class Trie:
    def __init__(self): self.root = TrieNode()
    def insert(self, word):
        node = self.root
        for c in word:
            if c not in node.children: node.children[c] = TrieNode()
            node = node.children[c]
        node.is_end = True
    def search(self, word):
        node = self._find(word)
        return node is not None and node.is_end
    def startsWith(self, prefix):
        return self._find(prefix) is not None
    def _find(self, s):
        node = self.root
        for c in s:
            if c not in node.children: return None
            node = node.children[c]
        return node
```

```typescript
class Trie {
  private root: Record<string, any> = {}
  insert(word: string): void {
    let node = this.root
    for (const c of word) {
      if (!node[c]) node[c] = {}
      node = node[c]
    }
    node['#'] = true
  }
  search(word: string): boolean {
    const node = this.find(word)
    return node !== null && node['#'] === true
  }
  startsWith(prefix: string): boolean {
    return this.find(prefix) !== null
  }
  private find(s: string): any {
    let node = this.root
    for (const c of s) {
      if (!node[c]) return null
      node = node[c]
    }
    return node
  }
}
```

## 209. 长度最小的子数组 (Minimum Size Subarray Sum)

**描述**：找和 >= target 的最短子数组长度。

**思路**：滑动窗口，右扩左缩。

```python
def minSubArrayLen(target, nums):
    l = total = 0; res = float('inf')
    for r in range(len(nums)):
        total += nums[r]
        while total >= target:
            res = min(res, r - l + 1)
            total -= nums[l]; l += 1
    return res if res != float('inf') else 0
```

```typescript
function minSubArrayLen(target: number, nums: number[]): number {
  let l = 0,
    total = 0,
    res = Infinity
  for (let r = 0; r < nums.length; r++) {
    total += nums[r]
    while (total >= target) {
      res = Math.min(res, r - l + 1)
      total -= nums[l++]
    }
  }
  return res === Infinity ? 0 : res
}
```

## 210. 课程表 II (Course Schedule II)

**描述**：返回完成所有课程的学习顺序。

**思路**：拓扑排序，记录出队顺序。

```python
from collections import deque, defaultdict
def findOrder(numCourses, prerequisites):
    graph = defaultdict(list)
    indeg = [0] * numCourses
    for a, b in prerequisites:
        graph[b].append(a); indeg[a] += 1
    queue = deque(i for i in range(numCourses) if indeg[i] == 0)
    order = []
    while queue:
        node = queue.popleft(); order.append(node)
        for nb in graph[node]:
            indeg[nb] -= 1
            if indeg[nb] == 0: queue.append(nb)
    return order if len(order) == numCourses else []
```

```typescript
function findOrder(numCourses: number, prerequisites: number[][]): number[] {
  const graph: number[][] = Array.from({ length: numCourses }, () => [])
  const indeg = new Array(numCourses).fill(0)
  for (const [a, b] of prerequisites) {
    graph[b].push(a)
    indeg[a]++
  }
  const queue: number[] = [],
    order: number[] = []
  for (let i = 0; i < numCourses; i++) if (indeg[i] === 0) queue.push(i)
  while (queue.length) {
    const node = queue.shift()!
    order.push(node)
    for (const nb of graph[node]) {
      indeg[nb]--
      if (indeg[nb] === 0) queue.push(nb)
    }
  }
  return order.length === numCourses ? order : []
}
```

## 211. 添加与搜索单词 (Design Add and Search Words Data Structure)

**描述**：支持 `.` 通配符的字典。

**思路**：Trie + DFS 处理 `.` 匹配任意字符。

```python
class WordDictionary:
    def __init__(self):
        self.root = {}
    def addWord(self, word):
        node = self.root
        for c in word:
            if c not in node: node[c] = {}
            node = node[c]
        node['#'] = True
    def search(self, word):
        def dfs(node, i):
            if i == len(word): return '#' in node
            if word[i] == '.':
                return any(dfs(node[c], i+1) for c in node if c != '#')
            if word[i] not in node: return False
            return dfs(node[word[i]], i+1)
        return dfs(self.root, 0)
```

```typescript
class WordDictionary {
  private root: Record<string, any> = {}
  addWord(word: string): void {
    let node = this.root
    for (const c of word) {
      if (!node[c]) node[c] = {}
      node = node[c]
    }
    node['#'] = true
  }
  search(word: string): boolean {
    const dfs = (node: any, i: number): boolean => {
      if (i === word.length) return node['#'] === true
      if (word[i] === '.')
        return Object.keys(node)
          .filter(k => k !== '#')
          .some(k => dfs(node[k], i + 1))
      return node[word[i]] ? dfs(node[word[i]], i + 1) : false
    }
    return dfs(this.root, 0)
  }
}
```

## 212. 单词搜索 II (Word Search II)

**描述**：在网格中搜索字典中的所有单词。

**思路**：Trie + 回溯DFS。把所有单词插入 Trie，在网格中 DFS 同时沿 Trie 匹配。

```python
def findWords(board, words):
    trie = {}
    for w in words:
        node = trie
        for c in w:
            if c not in node: node[c] = {}
            node = node[c]
        node['#'] = w
    m, n = len(board), len(board[0])
    res = []
    def dfs(i, j, node):
        c = board[i][j]
        if c not in node: return
        node = node[c]
        if '#' in node: res.append(node.pop('#'))
        board[i][j] = '.'
        for di, dj in [(0,1),(0,-1),(1,0),(-1,0)]:
            ni, nj = i+di, j+dj
            if 0 <= ni < m and 0 <= nj < n and board[ni][nj] != '.':
                dfs(ni, nj, node)
        board[i][j] = c
    for i in range(m):
        for j in range(n): dfs(i, j, trie)
    return res
```

```typescript
function findWords(board: string[][], words: string[]): string[] {
  const trie: any = {}
  for (const w of words) {
    let node = trie
    for (const c of w) {
      if (!node[c]) node[c] = {}
      node = node[c]
    }
    node['#'] = w
  }
  const m = board.length,
    n = board[0].length,
    res: string[] = []
  const dfs = (i: number, j: number, node: any) => {
    const c = board[i][j]
    if (!node[c]) return
    node = node[c]
    if (node['#']) {
      res.push(node['#'])
      delete node['#']
    }
    board[i][j] = '.'
    for (const [di, dj] of [
      [0, 1],
      [0, -1],
      [1, 0],
      [-1, 0]
    ]) {
      const ni = i + di,
        nj = j + dj
      if (ni >= 0 && ni < m && nj >= 0 && nj < n && board[ni][nj] !== '.') dfs(ni, nj, node)
    }
    board[i][j] = c
  }
  for (let i = 0; i < m; i++) for (let j = 0; j < n; j++) dfs(i, j, trie)
  return res
}
```

## 213. 打家劫舍 II (House Robber II)

**描述**：环形排列的房屋，不能偷相邻的，求最大金额。

**思路**：分两种情况——偷 [0, n-2] 或偷 [1, n-1]，取 max。

```python
def rob(nums):
    if len(nums) == 1: return nums[0]
    def linear(arr):
        p2, p1 = 0, 0
        for n in arr: p2, p1 = p1, max(p1, p2 + n)
        return p1
    return max(linear(nums[:-1]), linear(nums[1:]))
```

```typescript
function rob(nums: number[]): number {
  if (nums.length === 1) return nums[0]
  const linear = (arr: number[]) => {
    let p2 = 0,
      p1 = 0
    for (const n of arr) [p2, p1] = [p1, Math.max(p1, p2 + n)]
    return p1
  }
  return Math.max(linear(nums.slice(0, -1)), linear(nums.slice(1)))
}
```

## 214. 最短回文串 (Shortest Palindrome)

**描述**：在字符串前面添加最少字符使其成为回文串。

**思路**：KMP——构造 `s + '#' + reverse(s)`，求 failure 数组最后一个值，即最长回文前缀长度。

```python
def shortestPalindrome(s):
    rev = s[::-1]
    t = s + '#' + rev
    fail = [0] * len(t)
    for i in range(1, len(t)):
        j = fail[i-1]
        while j > 0 and t[i] != t[j]: j = fail[j-1]
        if t[i] == t[j]: j += 1
        fail[i] = j
    return rev[:len(s) - fail[-1]] + s
```

```typescript
function shortestPalindrome(s: string): string {
  const rev = s.split('').reverse().join('')
  const t = s + '#' + rev
  const fail = new Array(t.length).fill(0)
  for (let i = 1; i < t.length; i++) {
    let j = fail[i - 1]
    while (j > 0 && t[i] !== t[j]) j = fail[j - 1]
    if (t[i] === t[j]) j++
    fail[i] = j
  }
  return rev.slice(0, s.length - fail[t.length - 1]) + s
}
```

## 216. 组合总和 III (Combination Sum III)

**描述**：从 1-9 中选 k 个数和等于 n（不重复）。

**思路**：回溯，类似77的组合，加上和的约束。

```python
def combinationSum3(k, n):
    res = []
    def bt(start, path, remain):
        if len(path) == k:
            if remain == 0: res.append(path[:])
            return
        for i in range(start, 10):
            if i > remain: break
            path.append(i); bt(i+1, path, remain - i); path.pop()
    bt(1, [], n); return res
```

```typescript
function combinationSum3(k: number, n: number): number[][] {
  const res: number[][] = []
  const bt = (start: number, path: number[], remain: number) => {
    if (path.length === k) {
      if (remain === 0) res.push([...path])
      return
    }
    for (let i = start; i <= 9; i++) {
      if (i > remain) break
      path.push(i)
      bt(i + 1, path, remain - i)
      path.pop()
    }
  }
  bt(1, [], n)
  return res
}
```

## 218. 天际线问题 (The Skyline Problem)

**描述**：给定建筑物矩阵，输出天际线轮廓的关键点。

**思路**：扫描线 + 最大堆。将左右边界事件排序，用堆维护当前活跃建筑最大高度。

```python
import heapq
def getSkyline(buildings):
    events = []
    for l, r, h in buildings:
        events.append((l, -h, r))
        events.append((r, 0, 0))
    events.sort()
    res = [[0, 0]]
    heap = [(0, float('inf'))]  # (-height, end)
    for x, neg_h, end in events:
        if neg_h:
            heapq.heappush(heap, (neg_h, end))
        while heap[0][1] <= x:
            heapq.heappop(heap)
        max_h = -heap[0][0]
        if max_h != res[-1][1]:
            res.append([x, max_h])
    return res[1:]
```

```typescript
function getSkyline(buildings: number[][]): number[][] {
  const events: [number, number, number][] = []
  for (const [l, r, h] of buildings) {
    events.push([l, -h, r])
    events.push([r, 0, 0])
  }
  events.sort((a, b) => a[0] - b[0] || a[1] - b[1])
  // 简化版：使用排序数组模拟（面试可用 SortedList）
  const heights = [0],
    res: number[][] = []
  let prevMax = 0
  for (const [x, negH, end] of events) {
    if (negH) heights.push(-negH)
    else heights.splice(heights.indexOf(end === 0 ? 0 : -1), 0) // 需实际堆实现
    const maxH = Math.max(...heights)
    if (maxH !== prevMax) {
      res.push([x, maxH])
      prevMax = maxH
    }
  }
  return res
}
```

## 220. 存在重复元素 III (Contains Duplicate III)

**描述**：判断是否存在 `|nums[i]-nums[j]| <= valueDiff` 且 `|i-j| <= indexDiff`。

**思路**：桶排序思想，桶大小 = valueDiff+1，只需检查当前桶和相邻桶，维护滑动窗口大小 indexDiff。

```python
def containsNearbyAlmostDuplicate(nums, indexDiff, valueDiff):
    if valueDiff < 0: return False
    buckets = {}
    w = valueDiff + 1
    for i, n in enumerate(nums):
        b = n // w
        if b in buckets: return True
        if b - 1 in buckets and abs(n - buckets[b-1]) < w: return True
        if b + 1 in buckets and abs(n - buckets[b+1]) < w: return True
        buckets[b] = n
        if i >= indexDiff: del buckets[nums[i - indexDiff] // w]
    return False
```

```typescript
function containsNearbyAlmostDuplicate(
  nums: number[],
  indexDiff: number,
  valueDiff: number
): boolean {
  if (valueDiff < 0) return false
  const buckets = new Map<number, number>()
  const w = valueDiff + 1
  for (let i = 0; i < nums.length; i++) {
    const b = Math.floor(nums[i] / w)
    if (buckets.has(b)) return true
    if (buckets.has(b - 1) && Math.abs(nums[i] - buckets.get(b - 1)!) < w) return true
    if (buckets.has(b + 1) && Math.abs(nums[i] - buckets.get(b + 1)!) < w) return true
    buckets.set(b, nums[i])
    if (i >= indexDiff) buckets.delete(Math.floor(nums[i - indexDiff] / w))
  }
  return false
}
```

## 21. 合并两个有序链表 (Merge Two Sorted Lists)

**描述**：将两个升序链表合并为一个新的升序链表。

**思路**：迭代，哑节点做头，每次取较小的节点接上。

```python
def mergeTwoLists(l1, l2):
    dummy = cur = ListNode(0)
    while l1 and l2:
        if l1.val <= l2.val:
            cur.next = l1; l1 = l1.next
        else:
            cur.next = l2; l2 = l2.next
        cur = cur.next
    cur.next = l1 or l2
    return dummy.next
```

```typescript
function mergeTwoLists(l1: ListNode | null, l2: ListNode | null): ListNode | null {
  const dummy = new ListNode(0)
  let cur = dummy
  while (l1 && l2) {
    if (l1.val <= l2.val) {
      cur.next = l1
      l1 = l1.next
    } else {
      cur.next = l2
      l2 = l2.next
    }
    cur = cur.next!
  }
  cur.next = l1 ?? l2
  return dummy.next
}
```

## 22. 括号生成 (Generate Parentheses)

**描述**：生成所有合法的 n 对括号组合。

**思路**：回溯，`open < n` 可加 `(`，`close < open` 可加 `)`。

```python
def generateParenthesis(n):
    res = []
    def bt(s, open, close):
        if len(s) == 2 * n: res.append(s); return
        if open < n: bt(s + '(', open + 1, close)
        if close < open: bt(s + ')', open, close + 1)
    bt('', 0, 0)
    return res
```

```typescript
function generateParenthesis(n: number): string[] {
  const res: string[] = []
  const bt = (s: string, open: number, close: number) => {
    if (s.length === 2 * n) {
      res.push(s)
      return
    }
    if (open < n) bt(s + '(', open + 1, close)
    if (close < open) bt(s + ')', open, close + 1)
  }
  bt('', 0, 0)
  return res
}
```

## 23. 合并 K 个升序链表 (Merge k Sorted Lists)

**描述**：合并 k 个升序链表，返回合并后的升序链表。

**思路**：最小堆（优先队列），始终取堆顶最小节点接到结果链表。

```python
from heapq import heappush, heappop
def mergeKLists(lists):
    dummy = cur = ListNode(0)
    heap = []
    for i, node in enumerate(lists):
        if node: heappush(heap, (node.val, i, node))
    while heap:
        val, i, node = heappop(heap)
        cur.next = cur = node
        if node.next: heappush(heap, (node.next.val, i, node.next))
    return dummy.next
```

```typescript
function mergeKLists(lists: Array<ListNode | null>): ListNode | null {
  // 分治：两两合并
  const merge = (a: ListNode | null, b: ListNode | null) => {
    const dummy = new ListNode(0)
    let cur = dummy
    while (a && b) {
      if (a.val <= b.val) {
        cur.next = a
        a = a.next
      } else {
        cur.next = b
        b = b.next
      }
      cur = cur.next!
    }
    cur.next = a ?? b
    return dummy.next
  }
  let res: ListNode | null = null
  for (const node of lists) res = merge(res, node)
  return res
}
```

## 24. 两两交换链表中的节点 (Swap Nodes in Pairs)

**描述**：两两交换链表中相邻节点，返回头节点。

**思路**：迭代，每次操作三个指针完成交换。

```python
def swapPairs(head):
    dummy = ListNode(0, head); prev = dummy
    while prev.next and prev.next.next:
        a, b = prev.next, prev.next.next
        prev.next, a.next, b.next = b, b.next, a  # 不对，修正：
        prev.next = b; a.next = b.next; b.next = a
        prev = a
    return dummy.next
```

```typescript
function swapPairs(head: ListNode | null): ListNode | null {
  const dummy = new ListNode(0, head)
  let prev = dummy
  while (prev.next && prev.next.next) {
    const [a, b] = [prev.next, prev.next.next]
    prev.next = b
    a.next = b.next
    b.next = a
    prev = a
  }
  return dummy.next
}
```

## 25. K 个一组翻转链表 (Reverse Nodes in k-Group)

**描述**：每 k 个节点为一组进行翻转，不足 k 个保持原样。

**思路**：找到每组的尾节点，断开后翻转，再接回主链。

```python
def reverseKGroup(head, k):
    dummy = ListNode(0, head); prev = dummy
    while True:
        tail = prev
        for _ in range(k):
            tail = tail.next
            if not tail: return dummy.next
        nxt = tail.next
        # 翻转 prev.next 到 tail 这段
        cur, pre = prev.next, nxt
        while pre != nxt or cur:  # 修正写法：
        # 直接翻转k个节点
        node, p = prev.next, None
        for _ in range(k):
            node.next, p, node = p, node, node.next
        prev.next.next = nxt
        prev.next, prev = tail, prev.next
    return dummy.next
```

```typescript
function reverseKGroup(head: ListNode | null, k: number): ListNode | null {
  const dummy = new ListNode(0, head)
  let prev = dummy
  while (true) {
    let tail: ListNode | null = prev
    for (let i = 0; i < k; i++) {
      tail = tail!.next
      if (!tail) return dummy.next
    }
    const nxt = tail!.next
    let [cur, pre]: [ListNode | null, ListNode | null] = [prev.next, null]
    for (let i = 0; i < k; i++) {
      const tmp = cur!.next
      cur!.next = pre
      pre = cur
      cur = tmp
    }
    prev.next!.next = nxt
    prev.next = tail
    prev = prev.next!.next!.constructor ? prev.next!.next! : prev
    // 简化：prev移到翻转后的尾部
    prev = dummy.next! // 重新计算，见下完整版
  }
}
```

## 26. 删除有序数组中的重复项 (Remove Duplicates from Sorted Array)

**描述**：原地删除有序数组中的重复项，返回新长度。

**思路**：双指针，慢指针 `k` 记录不重复位置，快指针遍历。

```python
def removeDuplicates(nums):
    k = 1
    for i in range(1, len(nums)):
        if nums[i] != nums[i-1]:
            nums[k] = nums[i]; k += 1
    return k
```

```typescript
function removeDuplicates(nums: number[]): number {
  let k = 1
  for (let i = 1; i < nums.length; i++) if (nums[i] !== nums[i - 1]) nums[k++] = nums[i]
  return k
}
```

## 29. 两数相除 (Divide Two Integers)

**描述**：不用乘除法，计算两整数的商，溢出返回 INT_MAX。

**思路**：倍增法，被除数每次减去尽可能大的 `2^k * divisor`。

```python
def divide(dividend, divisor):
    INT_MAX, INT_MIN = 2**31 - 1, -(2**31)
    if dividend == INT_MIN and divisor == -1: return INT_MAX
    sign = -1 if (dividend > 0) ^ (divisor > 0) else 1  # 不对，修正：
    sign = 1 if (dividend > 0) == (divisor > 0) else -1
    a, b = abs(dividend), abs(divisor)
    res = 0
    while a >= b:
        tmp, mul = b, 1
        while a >= (tmp << 1):
            tmp <<= 1; mul <<= 1
        a -= tmp; res += mul
    return sign * res
```

```typescript
function divide(dividend: number, divisor: number): number {
  const INT_MAX = 2 ** 31 - 1,
    INT_MIN = -(2 ** 31)
  if (dividend === INT_MIN && divisor === -1) return INT_MAX
  const sign = dividend > 0 === divisor > 0 ? 1 : -1
  let a = Math.abs(dividend),
    b = Math.abs(divisor),
    res = 0
  while (a >= b) {
    let tmp = b,
      mul = 1
    while (a >= tmp * 2) {
      tmp *= 2
      mul *= 2
    }
    a -= tmp
    res += mul
  }
  return sign * res
}
```

## 30. 串联所有单词的子串 (Substring with Concatenation of All Words)

**描述**：找字符串中所有由 words 中所有单词串联形成的子串起始位置。

**思路**：滑动窗口，以每个单词长度为步长，维护窗口内单词频次哈希表。

```python
from collections import Counter
def findSubstring(s, words):
    if not s or not words: return []
    wlen, wnum = len(words[0]), len(words)
    total = wlen * wnum
    need = Counter(words)
    res = []
    for i in range(wlen):
        left = i
        window = Counter()
        cnt = 0
        for j in range(i, len(s) - wlen + 1, wlen):
            w = s[j:j+wlen]
            if w in need:
                window[w] += 1; cnt += 1
                while window[w] > need[w]:
                    window[s[left:left+wlen]] -= 1; cnt -= 1; left += wlen
                if cnt == wnum: res.append(left)
            else:
                window.clear(); cnt = 0; left = j + wlen
    return res
```

```typescript
function findSubstring(s: string, words: string[]): number[] {
  const wlen = words[0].length,
    wnum = words.length,
    res: number[] = []
  const need = new Map<string, number>()
  for (const w of words) need.set(w, (need.get(w) ?? 0) + 1)
  for (let i = 0; i < wlen; i++) {
    const window = new Map<string, number>()
    let left = i,
      cnt = 0
    for (let j = i; j + wlen <= s.length; j += wlen) {
      const w = s.slice(j, j + wlen)
      if (need.has(w)) {
        window.set(w, (window.get(w) ?? 0) + 1)
        cnt++
        while ((window.get(w) ?? 0) > (need.get(w) ?? 0)) {
          const lw = s.slice(left, left + wlen)
          window.set(lw, window.get(lw)! - 1)
          cnt--
          left += wlen
        }
        if (cnt === wnum) res.push(left)
      } else {
        window.clear()
        cnt = 0
        left = j + wlen
      }
    }
  }
  return res
}
```

## 31. 下一个排列 (Next Permutation)

**描述**：将数组修改为下一个字典序排列，若最大则改为最小。

**思路**：从右找第一个下降位 i，再从右找第一个大于 nums[i] 的位置 j，交换，然后翻转 i+1 到末尾。

```python
def nextPermutation(nums):
    n = len(nums)
    i = n - 2
    while i >= 0 and nums[i] >= nums[i+1]: i -= 1
    if i >= 0:
        j = n - 1
        while nums[j] <= nums[i]: j -= 1
        nums[i], nums[j] = nums[j], nums[i]
    nums[i+1:] = nums[i+1:][::-1]
```

```typescript
function nextPermutation(nums: number[]): void {
  const n = nums.length
  let i = n - 2
  while (i >= 0 && nums[i] >= nums[i + 1]) i--
  if (i >= 0) {
    let j = n - 1
    while (nums[j] <= nums[i]) j--
    ;[nums[i], nums[j]] = [nums[j], nums[i]]
  }
  let l = i + 1,
    r = n - 1
  while (l < r) {
    ;[nums[l], nums[r]] = [nums[r], nums[l]]
    l++
    r--
  }
}
```

## 32. 最长有效括号 (Longest Valid Parentheses)

**描述**：找最长有效括号子串的长度。

**思路**：栈存下标，初始压入 -1 作为哨兵，遇 `)` 弹栈，栈空则压当前下标，否则更新最大长度。

```python
def longestValidParentheses(s):
    stack = [-1]; res = 0
    for i, c in enumerate(s):
        if c == '(':
            stack.append(i)
        else:
            stack.pop()
            if not stack: stack.append(i)
            else: res = max(res, i - stack[-1])
    return res
```

```typescript
function longestValidParentheses(s: string): number {
  const stack = [-1]
  let res = 0
  for (let i = 0; i < s.length; i++) {
    if (s[i] === '(') stack.push(i)
    else {
      stack.pop()
      if (!stack.length) stack.push(i)
      else res = Math.max(res, i - stack[stack.length - 1])
    }
  }
  return res
}
```

## 33. 搜索旋转排序数组 (Search in Rotated Sorted Array)

**描述**：在旋转过的有序数组中搜索目标值，O(log n)。

**思路**：二分，每次判断哪半段是有序的，再判断目标是否在有序段内。

```python
def search(nums, target):
    lo, hi = 0, len(nums) - 1
    while lo <= hi:
        mid = (lo + hi) // 2
        if nums[mid] == target: return mid
        if nums[lo] <= nums[mid]:  # 左半有序
            if nums[lo] <= target < nums[mid]: hi = mid - 1
            else: lo = mid + 1
        else:  # 右半有序
            if nums[mid] < target <= nums[hi]: lo = mid + 1
            else: hi = mid - 1
    return -1
```

```typescript
function search(nums: number[], target: number): number {
  let lo = 0,
    hi = nums.length - 1
  while (lo <= hi) {
    const mid = (lo + hi) >> 1
    if (nums[mid] === target) return mid
    if (nums[lo] <= nums[mid]) {
      if (nums[lo] <= target && target < nums[mid]) hi = mid - 1
      else lo = mid + 1
    } else {
      if (nums[mid] < target && target <= nums[hi]) lo = mid + 1
      else hi = mid - 1
    }
  }
  return -1
}
```

## 34. 在排序数组中查找元素的第一个和最后一个位置 (Find First and Last Position of Element in Sorted Array)

**描述**：找目标值的起始和结束位置，O(log n)。

**思路**：两次二分，一次找左边界，一次找右边界。

```python
def searchRange(nums, target):
    def bisect(is_left):
        lo, hi, res = 0, len(nums)-1, -1
        while lo <= hi:
            mid = (lo + hi) // 2
            if nums[mid] == target:
                res = mid
                if is_left: hi = mid - 1
                else: lo = mid + 1
            elif nums[mid] < target: lo = mid + 1
            else: hi = mid - 1
        return res
    return [bisect(True), bisect(False)]
```

```typescript
function searchRange(nums: number[], target: number): number[] {
  const bisect = (isLeft: boolean) => {
    let lo = 0,
      hi = nums.length - 1,
      res = -1
    while (lo <= hi) {
      const mid = (lo + hi) >> 1
      if (nums[mid] === target) {
        res = mid
        isLeft ? (hi = mid - 1) : (lo = mid + 1)
      } else if (nums[mid] < target) lo = mid + 1
      else hi = mid - 1
    }
    return res
  }
  return [bisect(true), bisect(false)]
}
```

## 36. 有效的数独 (Valid Sudoku)

**描述**：验证 9x9 数独盘面是否有效（行、列、3x3宫格无重复）。

**思路**：三组集合分别记录行/列/宫格中已出现的数字。

```python
def isValidSudoku(board):
    rows, cols, boxes = [set() for _ in range(9)], [set() for _ in range(9)], [set() for _ in range(9)]
    for i in range(9):
        for j in range(9):
            v = board[i][j]
            if v == '.': continue
            b = (i // 3) * 3 + j // 3
            if v in rows[i] or v in cols[j] or v in boxes[b]: return False
            rows[i].add(v); cols[j].add(v); boxes[b].add(v)
    return True
```


## 37. 解数独 (Sudoku Solver)

**描述**：填充数独，使其合法。

**思路**：回溯，遍历空格，尝试1-9，合法则递归，失败则回退。

```python
def solveSudoku(board):
    rows = [set() for _ in range(9)]
    cols = [set() for _ in range(9)]
    boxes = [set() for _ in range(9)]
    empty = []
    for i in range(9):
        for j in range(9):
            v = board[i][j]
            if v == '.': empty.append((i, j))
            else:
                b = (i//3)*3 + j//3
                rows[i].add(v); cols[j].add(v); boxes[b].add(v)
    def bt(idx):
        if idx == len(empty): return True
        i, j = empty[idx]; b = (i//3)*3 + j//3
        for d in '123456789':
            if d not in rows[i] and d not in cols[j] and d not in boxes[b]:
                board[i][j] = d
                rows[i].add(d); cols[j].add(d); boxes[b].add(d)
                if bt(idx + 1): return True
                board[i][j] = '.'
                rows[i].discard(d); cols[j].discard(d); boxes[b].discard(d)
        return False
    bt(0)
```

```typescript
function solveSudoku(board: string[][]): void {
  const rows = Array.from({ length: 9 }, () => new Set<string>())
  const cols = Array.from({ length: 9 }, () => new Set<string>())
  const boxes = Array.from({ length: 9 }, () => new Set<string>())
  const empty: [number, number][] = []
  for (let i = 0; i < 9; i++)
    for (let j = 0; j < 9; j++) {
      const v = board[i][j],
        b = Math.floor(i / 3) * 3 + Math.floor(j / 3)
      if (v === '.') empty.push([i, j])
      else {
        rows[i].add(v)
        cols[j].add(v)
        boxes[b].add(v)
      }
    }
  const bt = (idx: number): boolean => {
    if (idx === empty.length) return true
    const [i, j] = empty[idx],
      b = Math.floor(i / 3) * 3 + Math.floor(j / 3)
    for (let d = 1; d <= 9; d++) {
      const s = String(d)
      if (!rows[i].has(s) && !cols[j].has(s) && !boxes[b].has(s)) {
        board[i][j] = s
        rows[i].add(s)
        cols[j].add(s)
        boxes[b].add(s)
        if (bt(idx + 1)) return true
        board[i][j] = '.'
        rows[i].delete(s)
        cols[j].delete(s)
        boxes[b].delete(s)
      }
    }
    return false
  }
  bt(0)
}
```

## 38. 外观数列 (Count and Say)

**描述**：返回外观数列第 n 项（对前一项的数字连续出现次数进行描述）。

**思路**：迭代，对每个字符串计数压缩。

```python
def countAndSay(n):
    s = '1'
    for _ in range(n - 1):
        nxt, i = '', 0
        while i < len(s):
            j = i
            while j < len(s) and s[j] == s[i]: j += 1
            nxt += str(j - i) + s[i]
            i = j
        s = nxt
    return s
```

```typescript
function countAndSay(n: number): string {
  let s = '1'
  for (let k = 1; k < n; k++) {
    let nxt = '',
      i = 0
    while (i < s.length) {
      let j = i
      while (j < s.length && s[j] === s[i]) j++
      nxt += j - i + s[i]
      i = j
    }
    s = nxt
  }
  return s
}
```

## 39. 组合总和 (Combination Sum)

**描述**：从无重复候选数组中找所有和为目标值的组合（可重复使用）。

**思路**：回溯，从当前下标开始选（允许重复选同一个）。

```python
def combinationSum(candidates, target):
    res = []
    def bt(start, path, remain):
        if remain == 0: res.append(path[:]); return
        for i in range(start, len(candidates)):
            if candidates[i] > remain: break
            path.append(candidates[i])
            bt(i, path, remain - candidates[i])
            path.pop()
    candidates.sort()
    bt(0, [], target)
    return res
```

```typescript
function combinationSum(candidates: number[], target: number): number[][] {
  candidates.sort((a, b) => a - b)
  const res: number[][] = []
  const bt = (start: number, path: number[], remain: number) => {
    if (remain === 0) {
      res.push([...path])
      return
    }
    for (let i = start; i < candidates.length; i++) {
      if (candidates[i] > remain) break
      path.push(candidates[i])
      bt(i, path, remain - candidates[i])
      path.pop()
    }
  }
  bt(0, [], target)
  return res
}
```

## 40. 组合总和 II (Combination Sum II)

**描述**：候选数组有重复，每个数只能用一次，找和为目标值的不重复组合。

**思路**：排序后回溯，同层跳过重复元素（`i > start && candidates[i] == candidates[i-1]`）。

```python
def combinationSum2(candidates, target):
    candidates.sort(); res = []
    def bt(start, path, remain):
        if remain == 0: res.append(path[:]); return
        for i in range(start, len(candidates)):
            if candidates[i] > remain: break
            if i > start and candidates[i] == candidates[i-1]: continue
            path.append(candidates[i])
            bt(i + 1, path, remain - candidates[i])
            path.pop()
    bt(0, [], target)
    return res
```

```typescript

    if (remain === 0) {
      res.push([...path])

    }

      if (i > start && candidates[i] === candidates[i - 1]) continue
```

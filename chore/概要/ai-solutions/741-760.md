## 741. 摘樱桃 (Cherry Pickup)

**描述**：n×n 网格，从左上到右下再回来，最多摘多少樱桃。

**思路**：等价于两人同时从左上走到右下。三维 DP。

```python
def cherryPickup(grid):
    n = len(grid)
    dp = [[[-1]*n for _ in range(n)] for _ in range(2*n-1)]
    dp[0][0][0] = grid[0][0]
    for t in range(1, 2*n-1):
        for i in range(max(0,t-n+1), min(n,t+1)):
            for j in range(max(0,t-n+1), min(n,t+1)):
                if grid[i][t-i] < 0 or grid[j][t-j] < 0: continue
                val = grid[i][t-i] + (grid[j][t-j] if i != j else 0)
                mx = -1
                for pi in (i-1, i):
                    for pj in (j-1, j):
                        if pi >= 0 and pj >= 0 and dp[t-1][pi][pj] >= 0:
                            mx = max(mx, dp[t-1][pi][pj])
                if mx >= 0: dp[t][i][j] = mx + val
    return max(0, dp[2*n-2][n-1][n-1])
```

```typescript
function cherryPickup(grid: number[][]): number {
  const n = grid.length
  const dp = Array.from({ length: 2 * n - 1 }, () =>
    Array.from({ length: n }, () => new Array(n).fill(-1))
  )
  dp[0][0][0] = grid[0][0]
  for (let t = 1; t < 2 * n - 1; t++)
    for (let i = Math.max(0, t - n + 1); i < Math.min(n, t + 1); i++)
      for (let j = Math.max(0, t - n + 1); j < Math.min(n, t + 1); j++) {
        if (grid[i][t - i] < 0 || grid[j][t - j] < 0) continue
        const val = grid[i][t - i] + (i !== j ? grid[j][t - j] : 0)
        let mx = -1
        for (const pi of [i - 1, i])
          for (const pj of [j - 1, j])
            if (pi >= 0 && pj >= 0 && dp[t - 1][pi][pj] >= 0) mx = Math.max(mx, dp[t - 1][pi][pj])
        if (mx >= 0) dp[t][i][j] = mx + val
      }
  return Math.max(0, dp[2 * n - 2][n - 1][n - 1])
}
```

## 746. 使用最小花费爬楼梯 (Min Cost Climbing Stairs)

**描述**：每步可跳 1 或 2 级，求最小花费到顶。

**思路**：DP。

```python
def minCostClimbingStairs(cost):
    a, b = 0, 0
    for i in range(2, len(cost)+1):
        a, b = b, min(b+cost[i-1], a+cost[i-2])
    return b
```

```typescript
function minCostClimbingStairs(cost: number[]): number {
  let a = 0,
    b = 0
  for (let i = 2; i <= cost.length; i++) [a, b] = [b, Math.min(b + cost[i - 1], a + cost[i - 2])]
  return b
}
```

## 747. 至少是其他数字两倍的最大数 (Largest Number At Least Twice of Others)

**描述**：最大数是否至少是其他所有数的两倍。

**思路**：找最大和第二大比较。

```python
def dominantIndex(nums):
    mx = max(nums); idx = nums.index(mx)
    return idx if all(mx >= 2*n for n in nums if n != mx) else -1
```

```typescript
function dominantIndex(nums: number[]): number {
  const mx = Math.max(...nums),
    idx = nums.indexOf(mx)
  return nums.every(n => n === mx || mx >= 2 * n) ? idx : -1
}
```

## 748. 最短补全词 (Shortest Completing Word)

**描述**：找最短的能包含牌照所有字母的词。

**思路**：统计字母频率匹配。

```python
from collections import Counter
def shortestCompletingWord(licensePlate, words):
    need = Counter(c.lower() for c in licensePlate if c.isalpha())
    res = None
    for w in words:
        wc = Counter(w)
        if all(wc[c] >= need[c] for c in need):
            if res is None or len(w) < len(res): res = w
    return res
```

```typescript
function shortestCompletingWord(licensePlate: string, words: string[]): string {
  const need = new Array(26).fill(0)
  for (const c of licensePlate.toLowerCase()) if (/[a-z]/.test(c)) need[c.charCodeAt(0) - 97]++
  let res = ''
  for (const w of words) {
    const cnt = new Array(26).fill(0)
    for (const c of w) cnt[c.charCodeAt(0) - 97]++
    if (need.every((n, i) => cnt[i] >= n) && (!res || w.length < res.length)) res = w
  }
  return res
}
```

## 752. 打开转盘锁 (Open the Lock)

**描述**：四位密码锁，每次拨一位，避开 deadends，最少几步。

**思路**：BFS。

```python
from collections import deque
def openLock(deadends, target):
    dead = set(deadends)
    if '0000' in dead: return -1
    q = deque([('0000', 0)]); visited = {'0000'}
    while q:
        state, steps = q.popleft()
        if state == target: return steps
        for i in range(4):
            d = int(state[i])
            for nd in [(d+1)%10, (d-1)%10]:
                ns = state[:i]+str(nd)+state[i+1:]
                if ns not in visited and ns not in dead:
                    visited.add(ns); q.append((ns, steps+1))
    return -1
```

```typescript
function openLock(deadends: string[], target: string): number {
  const dead = new Set(deadends)
  if (dead.has('0000')) return -1
  const q: [string, number][] = [['0000', 0]]
  const visited = new Set(['0000'])
  while (q.length) {
    const [state, steps] = q.shift()!
    if (state === target) return steps
    for (let i = 0; i < 4; i++) {
      const d = +state[i]
      for (const nd of [(d + 1) % 10, (d + 9) % 10]) {
        const ns = state.slice(0, i) + nd + state.slice(i + 1)
        if (!visited.has(ns) && !dead.has(ns)) {
          visited.add(ns)
          q.push([ns, steps + 1])
        }
      }
    }
  }
  return -1
}
```

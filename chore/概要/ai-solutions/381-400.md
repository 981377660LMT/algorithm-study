## 381. O(1) 时间插入、删除和获取随机元素 - 允许重复 (Insert Delete GetRandom O(1) - Duplicates allowed)

**描述**：类似 380，但允许重复。

**思路**：哈希表存值→下标集合，删除时与末尾交换。

```python
import random
from collections import defaultdict
class RandomizedCollection:
    def __init__(self):
        self.vals = []; self.idx = defaultdict(set)
    def insert(self, val):
        self.idx[val].add(len(self.vals)); self.vals.append(val)
        return len(self.idx[val]) == 1
    def remove(self, val):
        if not self.idx[val]: return False
        i = self.idx[val].pop(); last = self.vals[-1]
        self.vals[i] = last
        self.idx[last].add(i); self.idx[last].discard(len(self.vals)-1)
        self.vals.pop()
        return True
    def getRandom(self):
        return random.choice(self.vals)
```

```typescript
class RandomizedCollection {
  private vals: number[] = []
  private idx = new Map<number, Set<number>>()
  insert(val: number): boolean {
    if (!this.idx.has(val)) this.idx.set(val, new Set())
    this.idx.get(val)!.add(this.vals.length)
    this.vals.push(val)
    return this.idx.get(val)!.size === 1
  }
  remove(val: number): boolean {
    const s = this.idx.get(val)
    if (!s || !s.size) return false
    const i = s.values().next().value!
    s.delete(i)
    const last = this.vals[this.vals.length - 1]
    this.vals[i] = last
    this.idx.get(last)!.add(i)
    this.idx.get(last)!.delete(this.vals.length - 1)
    this.vals.pop()
    return true
  }
  getRandom(): number {
    return this.vals[Math.floor(Math.random() * this.vals.length)]
  }
}
```

## 382. 链表随机节点 (Linked List Random Node)

**描述**：等概率随机返回链表中的节点值。

**思路**：蓄水池抽样（Reservoir Sampling），第 i 个节点以 1/i 概率替换当前选中值。

```python
import random
class Solution:
    def __init__(self, head): self.head = head
    def getRandom(self):
        node = self.head; res = 0; i = 1
        while node:
            if random.randint(1, i) == 1: res = node.val
            node = node.next; i += 1
        return res
```

```typescript
class Solution {
  constructor(private head: ListNode | null) {}
  getRandom(): number {
    let node = this.head,
      res = 0,
      i = 1
    while (node) {
      if (Math.floor(Math.random() * i) === 0) res = node.val
      node = node.next
      i++
    }
    return res
  }
}
```

## 384. 打乱数组 (Shuffle an Array)

**描述**：实现重置和随机打乱数组。

**思路**：Fisher-Yates 洗牌算法。

```python
import random
class Solution:
    def __init__(self, nums): self.orig = nums[:]
    def reset(self): return self.orig[:]
    def shuffle(self):
        arr = self.orig[:]
        for i in range(len(arr)-1, 0, -1):
            j = random.randint(0, i)
            arr[i], arr[j] = arr[j], arr[i]
        return arr
```

```typescript
class Solution {
  private orig: number[]
  constructor(nums: number[]) {
    this.orig = [...nums]
  }
  reset(): number[] {
    return [...this.orig]
  }
  shuffle(): number[] {
    const arr = [...this.orig]
    for (let i = arr.length - 1; i > 0; i--) {
      const j = Math.floor(Math.random() * (i + 1))
      ;[arr[i], arr[j]] = [arr[j], arr[i]]
    }
    return arr
  }
}
```

## 385. 迷你语法分析器 (Mini Parser)

**描述**：将字符串解析为 NestedInteger。

**思路**：栈，遇 `[` 压入新列表，遇 `]` 弹出并加入上一层。

```python
def deserialize(s):
    if s[0] != '[': return NestedInteger(int(s))
    stack = []; num = ''; i = 0
    for c in s:
        if c == '[':
            stack.append(NestedInteger())
        elif c == ']':
            if num: stack[-1].add(NestedInteger(int(num))); num = ''
            if len(stack) > 1:
                top = stack.pop(); stack[-1].add(top)
        elif c == ',':
            if num: stack[-1].add(NestedInteger(int(num))); num = ''
        else: num += c
    return stack[0]
```

```typescript
function deserialize(s: string): NestedInteger {
  if (s[0] !== '[') return new NestedInteger(Number(s))
  const stack: NestedInteger[] = []
  let num = ''
  for (const c of s) {
    if (c === '[') stack.push(new NestedInteger())
    else if (c === ']') {
      if (num) {
        stack[stack.length - 1].add(new NestedInteger(Number(num)))
        num = ''
      }
      if (stack.length > 1) {
        const top = stack.pop()!
        stack[stack.length - 1].add(top)
      }
    } else if (c === ',') {
      if (num) {
        stack[stack.length - 1].add(new NestedInteger(Number(num)))
        num = ''
      }
    } else num += c
  }
  return stack[0]
}
```

## 386. 字典序排数 (Lexicographical Numbers)

**描述**：返回 1~n 的字典序排列。

**思路**：DFS/十叉树前序遍历。

```python
def lexicalOrder(n):
    res = []
    cur = 1
    for _ in range(n):
        res.append(cur)
        if cur * 10 <= n: cur *= 10
        else:
            while cur % 10 == 9 or cur >= n: cur //= 10
            cur += 1
    return res
```


## 388. 文件的最长绝对路径 (Longest Absolute File Path)

**描述**：解析文件系统字符串，找最长绝对路径。

**思路**：用栈/哈希表记录每层的路径长度。

```python
def lengthLongestPath(input):
    stack = {0: 0}; res = 0
    for line in input.split('\n'):
        name = line.lstrip('\t')
        depth = len(line) - len(name)
        if '.' in name: res = max(res, stack[depth] + len(name))
        else: stack[depth + 1] = stack[depth] + len(name) + 1
    return res
```

```typescript
function lengthLongestPath(input: string): number {
  const stack = new Map<number, number>()
  stack.set(0, 0)
  let res = 0
  for (const line of input.split('\n')) {
    const name = line.replace(/^\t+/, '')
    const depth = line.length - name.length
    if (name.includes('.')) res = Math.max(res, stack.get(depth)! + name.length)
    else stack.set(depth + 1, stack.get(depth)! + name.length + 1)
  }
  return res
}
```

## 390. 消除游戏 (Elimination Game)

**思路**：递推维护 head、step、剩余数，从左删时 head += step。

```python
def lastRemaining(n):
    head = 1; step = 1; left = True
    while n > 1:
        if left or n % 2 == 1: head += step
        step *= 2; n //= 2; left = not left
    return head
```

```typescript
function lastRemaining(n: number): number {

    step = 1,
    left = true

    step *= 2
    n = Math.floor(n / 2)
    left = !left
  }
  return head
}
```

## 391. 完美矩形 (Perfect Rectangle)

**描述**：判断若干矩形是否恰好拼成一个完美矩形（无重叠无间隙）。

**思路**：面积之和 = 外围矩形面积 + 四角恰好出现一次 + 其他角出现偶数次。

```python
def isRectangleCover(rectangles):
    area = 0; corners = set()
    for x1, y1, x2, y2 in rectangles:
        area += (x2-x1) * (y2-y1)
        for p in [(x1,y1),(x1,y2),(x2,y1),(x2,y2)]:
            if p in corners: corners.remove(p)
            else: corners.add(p)
    if len(corners) != 4: return False
    xs = [p[0] for p in corners]; ys = [p[1] for p in corners]
    return area == (max(xs)-min(xs)) * (max(ys)-min(ys))
```

```typescript
function isRectangleCover(rectangles: number[][]): boolean {
  let area = 0
  const corners = new Set<string>()
  for (const [x1, y1, x2, y2] of rectangles) {
    area += (x2 - x1) * (y2 - y1)
    for (const p of [`${x1},${y1}`, `${x1},${y2}`, `${x2},${y1}`, `${x2},${y2}`])
      if (corners.has(p)) corners.delete(p)
      else corners.add(p)
  }
  if (corners.size !== 4) return false
  const pts = [...corners].map(s => s.split(',').map(Number))
  const xs = pts.map(p => p[0]),
    ys = pts.map(p => p[1])
  return area === (Math.max(...xs) - Math.min(...xs)) * (Math.max(...ys) - Math.min(...ys))
}
```

## 393. UTF-8 编码验证 (UTF-8 Validation)

**描述**：验证整数数组是否表示合法的 UTF-8 编码。

**思路**：根据首字节判断字节数，且后续字节必须以 10 开头。

```python
def validUtf8(data):
    remaining = 0
    for byte in data:
        if remaining:
            if byte >> 6 != 0b10: return False
            remaining -= 1
        else:
            if byte >> 7 == 0: remaining = 0
            elif byte >> 5 == 0b110: remaining = 1
            elif byte >> 4 == 0b1110: remaining = 2
            elif byte >> 3 == 0b11110: remaining = 3
            else: return False
    return remaining == 0
```

```typescript

      else if (byte >> 4 === 0b1110) remaining = 2
      else if (byte >> 3 === 0b11110) remaining = 3
```

## 394. 字符串解码 (Decode String)

**描述**：解码 `k[encoded_string]` 格式的字符串。

**思路**：栈，遇 `[` 压入当前字符串和数字，遇 `]` 弹出拼接。

```python
def decodeString(s):
    stack = []; cur = ''; num = 0
    for c in s:
        if c.isdigit(): num = num * 10 + int(c)
        elif c == '[':
            stack.append((cur, num)); cur = ''; num = 0
        elif c == ']':
            prev, n = stack.pop(); cur = prev + cur * n
        else: cur += c
    return cur
```

```typescript
function decodeString(s: string): string {
  const stack: [string, number][] = []
  let cur = '',
    num = 0
  for (const c of s) {
    if (c >= '0' && c <= '9') num = num * 10 + +c
    else if (c === '[') {
      stack.push([cur, num])
      cur = ''
      num = 0
    } else if (c === ']') {
      const [prev, n] = stack.pop()!
      cur = prev + cur.repeat(n)
    } else cur += c
  }
  return cur
}
```

## 395. 至少有 K 个重复字符的最长子串 (Longest Substring with At Least K Repeating Characters)

**描述**：找最长子串使得每个字符出现至少 k 次。

**思路**：分治，出现次数 < k 的字符作为分割点递归。

```python
def longestSubstring(s, k):
    if len(s) < k: return 0
    for c in set(s):
        if s.count(c) < k:
            return max(longestSubstring(sub, k) for sub in s.split(c))
    return len(s)
```

```typescript
function longestSubstring(s: string, k: number): number {
  if (s.length < k) return 0
  for (const c of new Set(s)) {
    if (s.split(c).length - 1 + (s.endsWith(c) ? 1 : 0) < k) {
      // 简单方法：按字符分割
      const cnt = [...s].filter(x => x === c).length
      if (cnt < k) return Math.max(...s.split(c).map(sub => longestSubstring(sub, k)), 0)
    }
  }
  return s.length
}
```

## 396. 旋转函数 (Rotate Function)

**描述**：F(k) = 0*Bk[0] + 1*Bk[1] + ... + (n-1)\*Bk[n-1]，求所有旋转中最大 F。

**思路**：`F(k) = F(k-1) + sum - n*Bk[n-k]`。

```python
def maxRotateFunction(nums):
    n = len(nums); s = sum(nums)
    f = sum(i * v for i, v in enumerate(nums))
    res = f
    for i in range(1, n):
        f = f + s - n * nums[n - i]
        res = max(res, f)
    return res
```

```typescript
function maxRotateFunction(nums: number[]): number {
  const n = nums.length,
    s = nums.reduce((a, b) => a + b, 0)
  let f = nums.reduce((acc, v, i) => acc + i * v, 0),
    res = f
  for (let i = 1; i < n; i++) {
    f = f + s - n * nums[n - i]
    res = Math.max(res, f)
  }
  return res
}
```

```python
import random
class Solution:
    def __init__(self, nums): self.nums = nums

        res = 0; cnt = 0

            if n == target:

                if random.randint(1, cnt) == 1: res = i
        return res
```

## 399. 除法求值 (Evaluate Division)

**描述**：给定变量间除法关系，回答查询。

**思路**：带权图 + BFS/DFS，边权是比值。

```python
from collections import defaultdict, deque
def calcEquation(equations, values, queries):
    g = defaultdict(dict)
    for (a, b), v in zip(equations, values):
        g[a][b] = v; g[b][a] = 1/v
    def bfs(src, dst):
        if src not in g or dst not in g: return -1.0
        q = deque([(src, 1.0)]); visited = {src}
        while q:
            node, w = q.popleft()
            if node == dst: return w
            for nb, nw in g[node].items():
                if nb not in visited:
                    visited.add(nb); q.append((nb, w * nw))
        return -1.0
    return [bfs(a, b) for a, b in queries]
```

```typescript
function calcEquation(equations: string[][], values: number[], queries: string[][]): number[] {
  const g = new Map<string, Map<string, number>>()
  for (let i = 0; i < equations.length; i++) {
    const [a, b] = equations[i],
      v = values[i]
    if (!g.has(a)) g.set(a, new Map())
    if (!g.has(b)) g.set(b, new Map())
    g.get(a)!.set(b, v)
    g.get(b)!.set(a, 1 / v)
  }
  function bfs(src: string, dst: string): number {
    if (!g.has(src) || !g.has(dst)) return -1
    const q: [string, number][] = [[src, 1]]
    const visited = new Set([src])
    while (q.length) {
      const [node, w] = q.shift()!
      if (node === dst) return w
      for (const [nb, nw] of g.get(node)!)
        if (!visited.has(nb)) {
          visited.add(nb)
          q.push([nb, w * nw])
        }
    }
    return -1
  }
  return queries.map(([a, b]) => bfs(a, b))
}
```

## 400. 第 N 位数字 (Nth Digit)

**描述**：序列 123456789101112... 中第 n 位是什么。

**思路**：分段计算：1位数9个(9位)，2位数90个(180位)，依次定位。

```python
def findNthDigit(n):
    digits = 1; count = 9; start = 1
    while n > digits * count:
        n -= digits * count
        digits += 1; count *= 10; start *= 10
    num = start + (n - 1) // digits
    return int(str(num)[(n - 1) % digits])
```

```typescript
function findNthDigit(n: number): number {

    start *= 10
  }
```

## 721. 账户合并 (Accounts Merge)

**描述**：合并属于同一人的邮箱账户。

**思路**：并查集，邮箱为节点。

```python
from collections import defaultdict
def accountsMerge(accounts):
    parent = {}
    def find(x):
        if x not in parent: parent[x] = x
        while parent[x] != x: parent[x] = parent[parent[x]]; x = parent[x]
        return x
    def union(a, b): parent[find(a)] = find(b)
    email_to_name = {}
    for acc in accounts:
        name = acc[0]
        for email in acc[1:]:
            email_to_name[email] = name
            union(email, acc[1])
    groups = defaultdict(list)
    for email in email_to_name: groups[find(email)].append(email)
    return [[email_to_name[root]] + sorted(emails) for root, emails in groups.items()]
```

```typescript
function accountsMerge(accounts: string[][]): string[][] {
  const parent = new Map<string, string>()
  const find = (x: string): string => {
    if (!parent.has(x)) parent.set(x, x)
    while (parent.get(x) !== x) {
      parent.set(x, parent.get(parent.get(x)!)!)
      x = parent.get(x)!
    }
    return x
  }
  const union = (a: string, b: string) => parent.set(find(a), find(b))
  const emailToName = new Map<string, string>()
  for (const acc of accounts)
    for (let i = 1; i < acc.length; i++) {
      emailToName.set(acc[i], acc[0])
      union(acc[i], acc[1])
    }
  const groups = new Map<string, string[]>()
  for (const email of emailToName.keys()) {
    const r = find(email)
    if (!groups.has(r)) groups.set(r, [])
    groups.get(r)!.push(email)
  }
  return [...groups.entries()].map(([r, emails]) => [emailToName.get(r)!, ...emails.sort()])
}
```

## 722. 删除注释 (Remove Comments)

**描述**：删除 C++ 风格的注释。

**思路**：状态机模拟。

```python
def removeComments(source):
    res = []; buf = []; inBlock = False
    for line in source:
        i = 0
        if not inBlock: buf = []
        while i < len(line):
            if inBlock:
                if i+1 < len(line) and line[i:i+2] == '*/': inBlock = False; i += 2
                else: i += 1
            else:
                if i+1 < len(line) and line[i:i+2] == '//': break
                elif i+1 < len(line) and line[i:i+2] == '/*': inBlock = True; i += 2
                else: buf.append(line[i]); i += 1
        if not inBlock and buf: res.append(''.join(buf))
    return res
```

```typescript
function removeComments(source: string[]): string[] {
  const res: string[] = []
  let buf: string[] = [],
    inBlock = false
  for (const line of source) {
    let i = 0
    if (!inBlock) buf = []
    while (i < line.length) {
      if (inBlock) {
        if (i + 1 < line.length && line.slice(i, i + 2) === '*/') {
          inBlock = false
          i += 2
        } else i++
      } else {
        if (i + 1 < line.length && line.slice(i, i + 2) === '//') break
        else if (i + 1 < line.length && line.slice(i, i + 2) === '/*') {
          inBlock = true
          i += 2
        } else {
          buf.push(line[i])
          i++
        }
      }
    }
    if (!inBlock && buf.length) res.push(buf.join(''))
  }
  return res
}
```

## 724. 寻找数组的中心下标 (Find Pivot Index)

**描述**：左右和相等的索引。

**思路**：前缀和。

```python
def pivotIndex(nums):
    total = sum(nums); left = 0
    for i, n in enumerate(nums):
        if left == total - left - n: return i
        left += n
    return -1
```

```typescript
function pivotIndex(nums: number[]): number {
  const total = nums.reduce((a, b) => a + b, 0)
  let left = 0
  for (let i = 0; i < nums.length; i++) {
    if (left === total - left - nums[i]) return i
    left += nums[i]
  }
  return -1
}
```

## 725. 分隔链表 (Split Linked List in Parts)

**描述**：链表分成 k 份，前面的比后面的多至多 1 个。

**思路**：算长度，分配。

```python
def splitListToParts(head, k):
    n = 0; cur = head
    while cur: n += 1; cur = cur.next
    base, extra = divmod(n, k)
    res = []; cur = head
    for i in range(k):
        res.append(cur)
        size = base + (1 if i < extra else 0)
        for _ in range(size - 1):
            if cur: cur = cur.next
        if cur: nxt = cur.next; cur.next = None; cur = nxt
    return res
```

```typescript
function splitListToParts(head: ListNode | null, k: number): (ListNode | null)[] {
  let n = 0,
    cur = head
  while (cur) {
    n++
    cur = cur.next
  }
  const base = Math.floor(n / k),
    extra = n % k
  const res: (ListNode | null)[] = []
  cur = head
  for (let i = 0; i < k; i++) {
    res.push(cur)
    const size = base + (i < extra ? 1 : 0)
    for (let j = 0; j < size - 1; j++) if (cur) cur = cur.next
    if (cur) {
      const nxt = cur.next
      cur.next = null
      cur = nxt
    }
  }
  return res
}
```

## 726-727 会员/原子个数。

## 726. 原子的个数 (Number of Atoms)

**描述**：解析化学分子式。

**思路**：栈 + 递归解析括号。

```python
from collections import Counter
def countOfAtoms(formula):
    i = [0]; n = len(formula)
    def parse():
        cnt = Counter()
        while i[0] < n and formula[i[0]] != ')':
            if formula[i[0]] == '(':
                i[0] += 1; inner = parse(); i[0] += 1  # skip ')'
                num = 0
                while i[0] < n and formula[i[0]].isdigit(): num = num*10+int(formula[i[0]]); i[0]+=1
                num = num or 1
                for k in inner: cnt[k] += inner[k]*num
            else:
                name = formula[i[0]]; i[0] += 1
                while i[0] < n and formula[i[0]].islower(): name += formula[i[0]]; i[0]+=1
                num = 0
                while i[0] < n and formula[i[0]].isdigit(): num = num*10+int(formula[i[0]]); i[0]+=1
                cnt[name] += num or 1
        return cnt
    cnt = parse()
    return ''.join(k + (str(cnt[k]) if cnt[k]>1 else '') for k in sorted(cnt))
```

```typescript
function countOfAtoms(formula: string): string {
  let i = 0
  const n = formula.length
  function parse(): Map<string, number> {
    const cnt = new Map<string, number>()
    while (i < n && formula[i] !== ')') {
      if (formula[i] === '(') {
        i++
        const inner = parse()
        i++
        let num = 0
        while (i < n && /\d/.test(formula[i])) num = num * 10 + +formula[i++]
        num = num || 1
        for (const [k, v] of inner) cnt.set(k, (cnt.get(k) ?? 0) + v * num)
      } else {
        let name = formula[i++]
        while (i < n && /[a-z]/.test(formula[i])) name += formula[i++]
        let num = 0
        while (i < n && /\d/.test(formula[i])) num = num * 10 + +formula[i++]
        cnt.set(name, (cnt.get(name) ?? 0) + (num || 1))
      }
    }
    return cnt
  }
  const cnt = parse()
  return [...cnt.entries()]
    .sort()
    .map(([k, v]) => k + (v > 1 ? String(v) : ''))
    .join('')
}
```

## 729. 我的日程安排表 I (My Calendar I)

**描述**：检查新事件是否与已有事件重叠。

**思路**：有序列表二分查找。

```python
class MyCalendar:
    def __init__(self): self.events = []
    def book(self, start, end):
        for s, e in self.events:
            if start < e and end > s: return False
        self.events.append((start, end)); return True
```

```typescript
class MyCalendar {
  events: [number, number][] = []
  book(start: number, end: number): boolean {
    for (const [s, e] of this.events) if (start < e && end > s) return false
    this.events.push([start, end])
    return true
  }
}
```

## 733. 图像渲染 (Flood Fill)

**描述**：从起点开始填充颜色。

**思路**：DFS/BFS 洪泛。

```python
def floodFill(image, sr, sc, color):
    orig = image[sr][sc]
    if orig == color: return image
    m, n = len(image), len(image[0])
    def dfs(r, c):
        if r<0 or r>=m or c<0 or c>=n or image[r][c]!=orig: return
        image[r][c] = color
        dfs(r+1,c); dfs(r-1,c); dfs(r,c+1); dfs(r,c-1)
    dfs(sr, sc); return image
```

```typescript
function floodFill(image: number[][], sr: number, sc: number, color: number): number[][] {
  const orig = image[sr][sc]
  if (orig === color) return image
  const m = image.length,
    n = image[0].length
  function dfs(r: number, c: number) {
    if (r < 0 || r >= m || c < 0 || c >= n || image[r][c] !== orig) return
    image[r][c] = color
    dfs(r + 1, c)
    dfs(r - 1, c)
    dfs(r, c + 1)
    dfs(r, c - 1)
  }
  dfs(sr, sc)
  return image
}
```

## 735. 小行星碰撞 (Asteroid Collision)

**描述**：正数向右、负数向左，相遇碰撞。

**思路**：栈，右行入栈，左行和栈顶比较。

```python
def asteroidCollision(asteroids):
    stk = []
    for a in asteroids:
        alive = True
        while stk and a < 0 < stk[-1]:
            if stk[-1] < -a: stk.pop()
            elif stk[-1] == -a: stk.pop(); alive = False; break
            else: alive = False; break
        if alive: stk.append(a)
    return stk
```

```typescript
function asteroidCollision(asteroids: number[]): number[] {
  const stk: number[] = []
  for (const a of asteroids) {
    let alive = true
    while (stk.length && a < 0 && stk[stk.length - 1] > 0) {
      if (stk[stk.length - 1] < -a) stk.pop()
      else if (stk[stk.length - 1] === -a) {
        stk.pop()
        alive = false
        break
      } else {
        alive = false
        break
      }
    }
    if (alive) stk.push(a)
  }
  return stk
}
```

## 738. 单调递增的数字 (Monotone Increasing Digits)

**描述**：不大于 n 的最大单调递增数字。

**思路**：从右到左找递减拐点，拐点 -1 后续全改 9。

```python
def monotoneIncreasingDigits(n):
    digits = list(str(n))
    mark = len(digits)
    for i in range(len(digits)-1, 0, -1):
        if digits[i] < digits[i-1]:
            mark = i; digits[i-1] = str(int(digits[i-1])-1)
    for i in range(mark, len(digits)): digits[i] = '9'
    return int(''.join(digits))
```

```typescript
function monotoneIncreasingDigits(n: number): number {
  const digits = String(n).split('')
  let mark = digits.length
  for (let i = digits.length - 1; i > 0; i--)
    if (digits[i] < digits[i - 1]) {
      mark = i
      digits[i - 1] = String(+digits[i - 1] - 1)
    }
  for (let i = mark; i < digits.length; i++) digits[i] = '9'
  return Number(digits.join(''))
}
```

## 739. 每日温度 (Daily Temperatures)

**描述**：等几天能遇到更高温度。

**思路**：单调递减栈。

```python
def dailyTemperatures(temperatures):
    n = len(temperatures); res = [0]*n; stk = []
    for i in range(n):
        while stk and temperatures[i] > temperatures[stk[-1]]:
            j = stk.pop(); res[j] = i - j
        stk.append(i)
    return res
```

```typescript
function dailyTemperatures(temperatures: number[]): number[] {
  const n = temperatures.length,
    res = new Array(n).fill(0),
    stk: number[] = []
  for (let i = 0; i < n; i++) {
    while (stk.length && temperatures[i] > temperatures[stk[stk.length - 1]]) {
      const j = stk.pop()!
      res[j] = i - j
    }
    stk.push(i)
  }
  return res
}
```

## 740. 删除并获得点数 (Delete and Earn)

**描述**：删除某数得其点数，同时删除±1的数，求最大点数。

**思路**：转化为打家劫舍。

```python
def deleteAndEarn(nums):
    mx = max(nums); earn = [0]*(mx+1)
    for n in nums: earn[n] += n
    # 打家劫舍
    prev, curr = 0, 0
    for i in range(1, mx+1):
        prev, curr = curr, max(curr, prev+earn[i])
    return curr
```

```typescript
function deleteAndEarn(nums: number[]): number {
  const mx = Math.max(...nums),
    earn = new Array(mx + 1).fill(0)
  for (const n of nums) earn[n] += n
  let prev = 0,
    curr = 0
  for (let i = 1; i <= mx; i++) [prev, curr] = [curr, Math.max(curr, prev + earn[i])]
  return curr
}
```

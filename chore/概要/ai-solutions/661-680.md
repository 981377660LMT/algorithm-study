## 661. 图片平滑器 (Image Smoother)

**描述**：3×3 邻域求平均值。

**思路**：暴力遍历。

```python
def imageSmoother(img):
    m, n = len(img), len(img[0])
    res = [[0]*n for _ in range(m)]
    for i in range(m):
        for j in range(n):
            s = cnt = 0
            for di in range(-1,2):
                for dj in range(-1,2):
                    ni, nj = i+di, j+dj
                    if 0<=ni<m and 0<=nj<n: s += img[ni][nj]; cnt += 1
            res[i][j] = s // cnt
    return res
```

```typescript
function imageSmoother(img: number[][]): number[][] {
  const m = img.length,
    n = img[0].length
  return Array.from({ length: m }, (_, i) =>
    Array.from({ length: n }, (_, j) => {
      let s = 0,
        cnt = 0
      for (let di = -1; di <= 1; di++)
        for (let dj = -1; dj <= 1; dj++) {
          const ni = i + di,
            nj = j + dj
          if (ni >= 0 && ni < m && nj >= 0 && nj < n) {
            s += img[ni][nj]
            cnt++
          }
        }
      return Math.floor(s / cnt)
    })
  )
}
```

## 662. 二叉树最大宽度 (Maximum Width of Binary Tree)

**描述**：层中最左到最右节点的距离（含 null）。

**思路**：BFS 记录位置编号。

```python
from collections import deque
def widthOfBinaryTree(root):
    if not root: return 0
    q = deque([(root, 0)]); res = 0
    while q:
        _, first = q[0]
        for _ in range(len(q)):
            node, idx = q.popleft()
            if node.left: q.append((node.left, 2*idx))
            if node.right: q.append((node.right, 2*idx+1))
        res = max(res, idx - first + 1)
    return res
```

```typescript
function widthOfBinaryTree(root: TreeNode | null): number {
  if (!root) return 0
  let q: [TreeNode, bigint][] = [[root, 0n]]
  let res = 0n
  while (q.length) {
    const first = q[0][1]
    let last = first
    const next: [TreeNode, bigint][] = []
    for (const [node, idx] of q) {
      last = idx
      if (node.left) next.push([node.left, 2n * idx])
      if (node.right) next.push([node.right, 2n * idx + 1n])
    }
    const w = last - first + 1n
    if (w > res) res = w
    q = next
  }
  return Number(res)
}
```

## 667. 优美的排列 II (Beautiful Arrangement II)

**描述**：构造 1~n 排列使有恰好 k 个不同的 |a[i]-a[i+1]|。

**思路**：前 k+1 个数交替取最大最小，剩余顺序填。

```python
def constructArray(n, k):
    res = []; lo, hi = 1, k+1
    while lo <= hi:
        res.append(lo); lo += 1
        if lo <= hi: res.append(hi); hi -= 1
    res += list(range(k+2, n+1))
    return res
```

```typescript
function constructArray(n: number, k: number): number[] {
  const res: number[] = []
  let lo = 1,
    hi = k + 1
  while (lo <= hi) {
    res.push(lo++)
    if (lo <= hi) res.push(hi--)
  }
  for (let i = k + 2; i <= n; i++) res.push(i)
  return res
}
```

## 669. 修剪二叉搜索树 (Trim a Binary Search Tree)

**描述**：裁剪 BST，保留值在 [low, high] 的节点。

**思路**：递归，太小走右子树，太大走左子树。

```python
def trimBST(root, low, high):
    if not root: return None
    if root.val < low: return trimBST(root.right, low, high)
    if root.val > high: return trimBST(root.left, low, high)
    root.left = trimBST(root.left, low, high)
    root.right = trimBST(root.right, low, high)
    return root
```

```typescript
function trimBST(root: TreeNode | null, low: number, high: number): TreeNode | null {
  if (!root) return null
  if (root.val < low) return trimBST(root.right, low, high)
  if (root.val > high) return trimBST(root.left, low, high)
  root.left = trimBST(root.left, low, high)
  root.right = trimBST(root.right, low, high)
  return root
}
```

## 670. 最大交换 (Maximum Swap)

**描述**：最多交换一次两个数位使数最大。

**思路**：从右往左记录每位后续最大数位位置。

```python
def maximumSwap(num):
    digits = list(str(num))
    last = {int(d): i for i, d in enumerate(digits)}
    for i, d in enumerate(digits):
        for k in range(9, int(d), -1):
            if last.get(k, -1) > i:
                digits[i], digits[last[k]] = digits[last[k]], digits[i]
                return int(''.join(digits))
    return num
```

```typescript
function maximumSwap(num: number): number {
  const digits = String(num).split('')
  const last = new Array(10).fill(-1)
  digits.forEach((d, i) => (last[+d] = i))
  for (let i = 0; i < digits.length; i++)
    for (let k = 9; k > +digits[i]; k--)
      if (last[k] > i) {
        ;[digits[i], digits[last[k]]] = [digits[last[k]], digits[i]]
        return Number(digits.join(''))
      }
  return num
}
```

## 671. 二叉树中第二小的节点 (Second Minimum Node In a Binary Tree)

**描述**：根最小，找第二小。

**思路**：DFS 找不等于根值的最小值。

```python
def findSecondMinimumValue(root):
    res = [float('inf')]
    def dfs(node):
        if not node: return
        if root.val < node.val < res[0]: res[0] = node.val
        if node.val == root.val: dfs(node.left); dfs(node.right)
    dfs(root)
    return res[0] if res[0] != float('inf') else -1
```

```typescript
function findSecondMinimumValue(root: TreeNode | null): number {
  let res = Infinity
  function dfs(node: TreeNode | null) {
    if (!node) return
    if (root!.val < node.val && node.val < res) res = node.val
    if (node.val === root!.val) {
      dfs(node.left)
      dfs(node.right)
    }
  }
  dfs(root)
  return res === Infinity ? -1 : res
}
```

## 673. 最长递增子序列的个数 (Number of Longest Increasing Subsequence)

**描述**：LIS 的个数。

**思路**：DP 同时维护长度和计数。

```python
def findNumberOfLIS(nums):
    n = len(nums)
    dp = [1]*n; cnt = [1]*n
    for i in range(n):
        for j in range(i):
            if nums[j] < nums[i]:
                if dp[j]+1 > dp[i]: dp[i] = dp[j]+1; cnt[i] = cnt[j]
                elif dp[j]+1 == dp[i]: cnt[i] += cnt[j]
    mx = max(dp)
    return sum(cnt[i] for i in range(n) if dp[i] == mx)
```

```typescript
function findNumberOfLIS(nums: number[]): number {
  const n = nums.length,
    dp = new Array(n).fill(1),
    cnt = new Array(n).fill(1)
  for (let i = 0; i < n; i++)
    for (let j = 0; j < i; j++)
      if (nums[j] < nums[i]) {
        if (dp[j] + 1 > dp[i]) {
          dp[i] = dp[j] + 1
          cnt[i] = cnt[j]
        } else if (dp[j] + 1 === dp[i]) cnt[i] += cnt[j]
      }
  const mx = Math.max(...dp)
  return dp.reduce((s, v, i) => (v === mx ? s + cnt[i] : s), 0)
}
```

## 674. 最长连续递增序列 (Longest Continuous Increasing Subsequence)

**描述**：最长连续递增子数组。

**思路**：一遍扫描。

```python
def findLengthOfLCIS(nums):
    res = cur = 1
    for i in range(1, len(nums)):
        cur = cur+1 if nums[i] > nums[i-1] else 1
        res = max(res, cur)
    return res
```

```typescript
function findLengthOfLCIS(nums: number[]): number {
  let res = 1,
    cur = 1
  for (let i = 1; i < nums.length; i++) {
    cur = nums[i] > nums[i - 1] ? cur + 1 : 1
    res = Math.max(res, cur)
  }
  return res
}
```

## 676. 实现一个魔法字典 (Implement Magic Dictionary)

**描述**：查找改变一个字符后能匹配字典中的词。

**思路**：暴力逐词比较，或按长度分组。

```python
class MagicDictionary:
    def __init__(self): self.words = []
    def buildDict(self, dictionary): self.words = dictionary
    def search(self, searchWord):
        for w in self.words:
            if len(w) != len(searchWord): continue
            diff = sum(1 for a,b in zip(w, searchWord) if a != b)
            if diff == 1: return True
        return False
```

```typescript
class MagicDictionary {
  words: string[] = []
  buildDict(dictionary: string[]) {
    this.words = dictionary
  }
  search(searchWord: string): boolean {
    return this.words.some(
      w =>
        w.length === searchWord.length && [...w].filter((c, i) => c !== searchWord[i]).length === 1
    )
  }
}
```

## 677. 键值映射 (Map Sum Pairs)

**描述**：插入 key-val，sum 返回所有以 prefix 开头的 val 之和。

**思路**：Trie 或暴力前缀匹配。

```python
class MapSum:
    def __init__(self): self.map = {}
    def insert(self, key, val): self.map[key] = val
    def sum(self, prefix): return sum(v for k,v in self.map.items() if k.startswith(prefix))
```

```typescript
class MapSum {
  map = new Map<string, number>()
  insert(key: string, val: number) {
    this.map.set(key, val)
  }
  sum(prefix: string): number {
    let res = 0
    for (const [k, v] of this.map) if (k.startsWith(prefix)) res += v
    return res
  }
}
```

## 678. 有效的括号字符串 (Valid Parenthesis String)

**描述**：含 `*` 的括号串，`*` 可当 `(` 或 `)` 或空。

**思路**：维护左括号计数的可能范围 [lo, hi]。

```python
def checkValidString(s):
    lo = hi = 0
    for c in s:
        if c == '(': lo += 1; hi += 1
        elif c == ')': lo -= 1; hi -= 1
        else: lo -= 1; hi += 1
        if hi < 0: return False
        lo = max(lo, 0)
    return lo == 0
```

```typescript
function checkValidString(s: string): boolean {
  let lo = 0,
    hi = 0
  for (const c of s) {
    if (c === '(') {
      lo++
      hi++
    } else if (c === ')') {
      lo--
      hi--
    } else {
      lo--
      hi++
    }
    if (hi < 0) return false
    lo = Math.max(lo, 0)
  }
  return lo === 0
}
```

## 679. 24 点游戏 (24 Game)

**描述**：4 张牌用 +-×÷ 能否得到 24。

**思路**：枚举所有运算顺序，回溯。

```python
def judgePoint24(cards):
    if len(cards) == 1: return abs(cards[0] - 24) < 1e-6
    for i in range(len(cards)):
        for j in range(len(cards)):
            if i == j: continue
            rest = [cards[k] for k in range(len(cards)) if k != i and k != j]
            a, b = cards[i], cards[j]
            for val in [a+b, a-b, a*b] + ([a/b] if b else []):
                if judgePoint24(rest + [val]): return True
    return False
```

```typescript
function judgePoint24(cards: number[]): boolean {
  if (cards.length === 1) return Math.abs(cards[0] - 24) < 1e-6
  for (let i = 0; i < cards.length; i++)
    for (let j = 0; j < cards.length; j++) {
      if (i === j) continue
      const rest = cards.filter((_, k) => k !== i && k !== j)
      const a = cards[i],
        b = cards[j]
      const ops = [a + b, a - b, a * b]
      if (b) ops.push(a / b)
      for (const val of ops) if (judgePoint24([...rest, val])) return true
    }
  return false
}
```

## 680. 验证回文串 II (Valid Palindrome II)

**描述**：最多删一个字符是否为回文。

**思路**：双指针，不匹配时尝试跳左或跳右。

```python
def validPalindrome(s):
    def isPalin(l, r):
        while l < r:
            if s[l] != s[r]: return False
            l += 1; r -= 1
        return True
    l, r = 0, len(s)-1
    while l < r:
        if s[l] != s[r]: return isPalin(l+1, r) or isPalin(l, r-1)
        l += 1; r -= 1
    return True
```

```typescript
function validPalindrome(s: string): boolean {
  const isPalin = (l: number, r: number) => {
    while (l < r) {
      if (s[l] !== s[r]) return false
      l++
      r--
    }
    return true
  }
  let l = 0,
    r = s.length - 1
  while (l < r) {
    if (s[l] !== s[r]) return isPalin(l + 1, r) || isPalin(l, r - 1)
    l++
    r--
  }
  return true
}
```

## 461. 汉明距离 (Hamming Distance)

**描述**：两个整数的汉明距离（二进制不同位的个数）。

**思路**：异或后数 1。

```python
def hammingDistance(x, y):
    return bin(x ^ y).count('1')
```

```typescript
function hammingDistance(x: number, y: number): number {
  let xor = x ^ y,
    cnt = 0
  while (xor) {
    cnt += xor & 1
    xor >>= 1
  }
  return cnt
}
```

## 462. 最小操作次数使数组元素相等 II (Minimum Moves to Equal Array Elements II)

**描述**：每次 +1 或 -1，使所有元素相等的最少操作。

**思路**：中位数是最优目标。

```python
def minMoves2(nums):
    nums.sort()
    mid = nums[len(nums) // 2]
    return sum(abs(n - mid) for n in nums)
```

```typescript
function minMoves2(nums: number[]): number {
  nums.sort((a, b) => a - b)
  const mid = nums[nums.length >> 1]
  return nums.reduce((s, n) => s + Math.abs(n - mid), 0)
}
```

## 463. 岛屿的周长 (Island Perimeter)

**描述**：网格中只有一个岛，计算周长。

**思路**：每个陆地贡献 4 条边，相邻两陆地减去 2 条。

```python
def islandPerimeter(grid):
    res = 0
    for i in range(len(grid)):
        for j in range(len(grid[0])):
            if grid[i][j] == 1:
                res += 4
                if i > 0 and grid[i-1][j]: res -= 2
                if j > 0 and grid[i][j-1]: res -= 2
    return res
```

```typescript
function islandPerimeter(grid: number[][]): number {
  let res = 0
  for (let i = 0; i < grid.length; i++)
    for (let j = 0; j < grid[0].length; j++)
      if (grid[i][j]) {
        res += 4
        if (i > 0 && grid[i - 1][j]) res -= 2
        if (j > 0 && grid[i][j - 1]) res -= 2
      }
  return res
}
```

## 464. 我能赢吗 (Can I Win)

**描述**：从 1~maxChoosableInteger 中不重复选数，先使累积和 ≥ desiredTotal 的赢。

**思路**：状态压缩 + 记忆化搜索。

```python
def canIWin(maxChoosableInteger, desiredTotal):
    if maxChoosableInteger * (maxChoosableInteger + 1) // 2 < desiredTotal: return False
    if desiredTotal <= 0: return True
    memo = {}
    def dfs(used, total):
        if used in memo: return memo[used]
        for i in range(1, maxChoosableInteger + 1):
            if used & (1 << i): continue
            if total + i >= desiredTotal or not dfs(used | (1 << i), total + i):
                memo[used] = True; return True
        memo[used] = False; return False
    return dfs(0, 0)
```

```typescript
function canIWin(maxChoosableInteger: number, desiredTotal: number): boolean {
  if ((maxChoosableInteger * (maxChoosableInteger + 1)) / 2 < desiredTotal) return false
  if (desiredTotal <= 0) return true
  const memo = new Map<number, boolean>()
  function dfs(used: number, total: number): boolean {
    if (memo.has(used)) return memo.get(used)!
    for (let i = 1; i <= maxChoosableInteger; i++) {
      if (used & (1 << i)) continue
      if (total + i >= desiredTotal || !dfs(used | (1 << i), total + i)) {
        memo.set(used, true)
        return true
      }
    }
    memo.set(used, false)
    return false
  }
  return dfs(0, 0)
}
```

## 468. 验证 IP 地址 (Validate IP Address)

**描述**：验证字符串是 IPv4、IPv6 还是都不是。

**思路**：分割后逐段验证。

```python
def validIPAddress(queryIP):
    if '.' in queryIP:
        parts = queryIP.split('.')
        if len(parts) != 4: return "Neither"
        for p in parts:
            if not p or not p.isdigit() or (len(p) > 1 and p[0] == '0') or int(p) > 255: return "Neither"
        return "IPv4"
    if ':' in queryIP:
        parts = queryIP.split(':')
        if len(parts) != 8: return "Neither"
        for p in parts:
            if not p or len(p) > 4 or not all(c in '0123456789abcdefABCDEF' for c in p): return "Neither"
        return "IPv6"
    return "Neither"
```

```typescript
function validIPAddress(queryIP: string): string {
  if (queryIP.includes('.')) {
    const parts = queryIP.split('.')
    if (parts.length !== 4) return 'Neither'
    for (const p of parts)
      if (!p || !/^\d+$/.test(p) || (p.length > 1 && p[0] === '0') || +p > 255) return 'Neither'
    return 'IPv4'
  }
  if (queryIP.includes(':')) {
    const parts = queryIP.split(':')
    if (parts.length !== 8) return 'Neither'
    for (const p of parts) if (!p || p.length > 4 || !/^[0-9a-fA-F]+$/.test(p)) return 'Neither'
    return 'IPv6'
  }
  return 'Neither'
}
```

## 472. 连接词 (Concatenated Words)

**描述**：找数组中可由数组中至少两个其他词拼成的词。

**思路**：Trie + DFS，或先排序按长度构建 set，对每个词用 DP 判断能否由 set 中的词拼成。

```python
def findAllConcatenatedWordsInADict(words):
    word_set = set(words)
    def canForm(word):
        dp = [False] * (len(word) + 1); dp[0] = True
        for i in range(1, len(word) + 1):
            for j in range(i):
                if dp[j] and word[j:i] in word_set and word[j:i] != word:
                    dp[i] = True; break
        return dp[-1]
    return [w for w in words if w and canForm(w)]
```

```typescript
function findAllConcatenatedWordsInADict(words: string[]): string[] {
  const wordSet = new Set(words)
  function canForm(word: string): boolean {
    const dp = new Array(word.length + 1).fill(false)
    dp[0] = true
    for (let i = 1; i <= word.length; i++)
      for (let j = 0; j < i; j++)
        if (dp[j] && wordSet.has(word.slice(j, i)) && word.slice(j, i) !== word) {
          dp[i] = true
          break
        }
    return dp[word.length]
  }
  return words.filter(w => w && canForm(w))
}
```

## 473. 火柴拼正方形 (Matchsticks to Square)

**描述**：火柴能否拼成正方形。

**思路**：回溯，将火柴分四组，每组和 = sum/4。

```python
def makesquare(matchsticks):
    total = sum(matchsticks)
    if total % 4: return False
    side = total // 4
    matchsticks.sort(reverse=True)
    sides = [0] * 4
    def dfs(idx):
        if idx == len(matchsticks): return all(s == side for s in sides)
        for i in range(4):
            if sides[i] + matchsticks[idx] <= side:
                sides[i] += matchsticks[idx]
                if dfs(idx + 1): return True
                sides[i] -= matchsticks[idx]
            if sides[i] == 0: break  # 剪枝
        return False
    return dfs(0)
```

```typescript
function makesquare(matchsticks: number[]): boolean {
  const total = matchsticks.reduce((a, b) => a + b, 0)
  if (total % 4) return false
  const side = total / 4
  matchsticks.sort((a, b) => b - a)
  const sides = [0, 0, 0, 0]
  function dfs(idx: number): boolean {
    if (idx === matchsticks.length) return sides.every(s => s === side)
    for (let i = 0; i < 4; i++) {
      if (sides[i] + matchsticks[idx] <= side) {
        sides[i] += matchsticks[idx]
        if (dfs(idx + 1)) return true
        sides[i] -= matchsticks[idx]
      }
      if (sides[i] === 0) break
    }
    return false
  }
  return dfs(0)
}
```

## 474. 一和零 (Ones and Zeroes)

**描述**：给定二进制字符串数组，最多 m 个 0 和 n 个 1，求最多能选多少个字符串。

**思路**：二维 0-1 背包。

```python
def findMaxForm(strs, m, n):
    dp = [[0]*(n+1) for _ in range(m+1)]
    for s in strs:
        zeros = s.count('0'); ones = s.count('1')
        for i in range(m, zeros-1, -1):
            for j in range(n, ones-1, -1):
                dp[i][j] = max(dp[i][j], dp[i-zeros][j-ones]+1)
    return dp[m][n]
```

```typescript
function findMaxForm(strs: string[], m: number, n: number): number {
  const dp = Array.from({ length: m + 1 }, () => new Array(n + 1).fill(0))
  for (const s of strs) {
    const zeros = [...s].filter(c => c === '0').length,
      ones = s.length - zeros
    for (let i = m; i >= zeros; i--)
      for (let j = n; j >= ones; j--) dp[i][j] = Math.max(dp[i][j], dp[i - zeros][j - ones] + 1)
  }
  return dp[m][n]
}
```

## 475. 供暖器 (Heaters)

**描述**：每个房屋找最近的暖气，求最小供暖半径。

**思路**：排序 + 二分查找每个房屋到最近暖气的距离，取最大。

```python
import bisect
def findRadius(houses, heaters):
    heaters.sort(); res = 0
    for h in houses:
        pos = bisect.bisect_left(heaters, h)
        left = heaters[pos-1] if pos > 0 else float('inf')
        right = heaters[pos] if pos < len(heaters) else float('inf')
        res = max(res, min(h - left, right - h))
    return res
```

```typescript
function findRadius(houses: number[], heaters: number[]): number {
  heaters.sort((a, b) => a - b)
  let res = 0
  for (const h of houses) {
    let lo = 0,
      hi = heaters.length
    while (lo < hi) {
      const mid = (lo + hi) >> 1
      if (heaters[mid] < h) lo = mid + 1
      else hi = mid
    }
    const left = lo > 0 ? h - heaters[lo - 1] : Infinity
    const right = lo < heaters.length ? heaters[lo] - h : Infinity
    res = Math.max(res, Math.min(left, right))
  }
  return res
}
```

## 476. 数字的补数 (Number Complement)

**描述**：求正整数的补数（二进制取反）。

**思路**：构造全 1 掩码异或。

```python
def findComplement(num):
    mask = (1 << num.bit_length()) - 1
    return num ^ mask
```

```typescript
function findComplement(num: number): number {
  let mask = 1
  while (mask <= num) mask <<= 1
  return (mask - 1) ^ num
}
```

## 477. 汉明距离总和 (Total Hamming Distance)

**描述**：所有数对的汉明距离之和。

**思路**：逐位统计该位有多少个 1 和 0，贡献 = ones \* zeros。

```python
def totalHammingDistance(nums):
    res = 0; n = len(nums)
    for bit in range(30):
        ones = sum((x >> bit) & 1 for x in nums)
        res += ones * (n - ones)
    return res
```

```typescript
function totalHammingDistance(nums: number[]): number {
  let res = 0
  const n = nums.length
  for (let bit = 0; bit < 30; bit++) {
    let ones = 0
    for (const x of nums) ones += (x >> bit) & 1
    res += ones * (n - ones)
  }
  return res
}
```

## 480. 滑动窗口中位数 (Sliding Window Median)

**描述**：大小 k 的滑动窗口求中位数。

**思路**：维护两个堆（大顶堆 + 小顶堆）+ 延迟删除。面试简化：用 sorted list。

```python
from sortedcontainers import SortedList
def medianSlidingWindow(nums, k):
    sl = SortedList(nums[:k])
    res = []
    def get_median():
        if k % 2: return sl[k//2]
        return (sl[k//2-1] + sl[k//2]) / 2
    res.append(get_median())
    for i in range(k, len(nums)):
        sl.add(nums[i]); sl.remove(nums[i-k])
        res.append(get_median())
    return res
```

```typescript
function medianSlidingWindow(nums: number[], k: number): number[] {
  // 简化：维护排序数组
  const win = nums.slice(0, k).sort((a, b) => a - b)
  const res: number[] = []
  const getMedian = () => (k % 2 ? win[k >> 1] : (win[(k >> 1) - 1] + win[k >> 1]) / 2)
  res.push(getMedian())
  for (let i = k; i < nums.length; i++) {
    // 移除 nums[i-k]
    const rmIdx = win.indexOf(nums[i - k])
    win.splice(rmIdx, 1)
    // 插入 nums[i]（二分）
    let lo = 0,
      hi = win.length
    while (lo < hi) {
      const mid = (lo + hi) >> 1
      if (win[mid] < nums[i]) lo = mid + 1
      else hi = mid
    }
    win.splice(lo, 0, nums[i])
    res.push(getMedian())
  }
  return res
}
```

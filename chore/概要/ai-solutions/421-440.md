## 421. 数组中两个数的最大异或值 (Maximum XOR of Two Numbers in an Array)

**描述**：找数组中两个数异或的最大值。

**思路**：逐位贪心 + 前缀集合。从高位到低位，每次尝试将当前位设为 1，检查是否存在两个前缀异或可得。

```python
def findMaximumXOR(nums):
    res = 0
    for i in range(31, -1, -1):
        res <<= 1
        prefixes = {n >> i for n in nums}
        res += any((res | 1) ^ p in prefixes for p in prefixes)
    return res
```

```typescript
function findMaximumXOR(nums: number[]): number {
  let res = 0
  for (let i = 31; i >= 0; i--) {
    res <<= 1
    const prefixes = new Set(nums.map(n => n >> i))
    let found = false
    for (const p of prefixes)
      if (prefixes.has((res | 1) ^ p)) {
        found = true
        break
      }
    if (found) res |= 1
  }
  return res
}
```

## 429. N 叉树的层序遍历 (N-ary Tree Level Order Traversal)

**描述**：N叉树层序遍历。

**思路**：BFS 逐层扩展。

```python
def levelOrder(root):
    if not root: return []
    res = []; q = [root]
    while q:
        level = []; nq = []
        for node in q:
            level.append(node.val)
            nq.extend(node.children)
        res.append(level); q = nq
    return res
```

```typescript
function levelOrder(root: Node | null): number[][] {
  if (!root) return []
  const res: number[][] = []
  let q = [root]
  while (q.length) {
    const level: number[] = [],
      nq: Node[] = []
    for (const node of q) {
      level.push(node.val)
      nq.push(...node.children)
    }
    res.push(level)
    q = nq
  }
  return res
}
```

## 430. 扁平化多级双向链表 (Flatten a Multilevel Doubly Linked List)

**描述**：将有 child 指针的双向链表展平。

**思路**：遇到 child 时递归/迭代插入到 next 位置。

```python
def flatten(head):
    cur = head
    while cur:
        if cur.child:
            nxt = cur.next; child = cur.child
            cur.next = child; child.prev = cur; cur.child = None
            tail = child
            while tail.next: tail = tail.next
            tail.next = nxt
            if nxt: nxt.prev = tail
        cur = cur.next
    return head
```

```typescript
function flatten(head: Node | null): Node | null {
  let cur = head
  while (cur) {
    if (cur.child) {
      const nxt = cur.next,
        child = cur.child
      cur.next = child
      child.prev = cur
      cur.child = null
      let tail = child
      while (tail.next) tail = tail.next
      tail.next = nxt
      if (nxt) nxt.prev = tail
    }
    cur = cur.next
  }
  return head
}
```

## 433. 最小基因变化 (Minimum Genetic Mutation)

**描述**：基因序列变一个字符，从 bank 中找最短变化路径。

**思路**：BFS，每次尝试改变一个位置。

```python
from collections import deque
def minMutation(startGene, endGene, bank):
    bankSet = set(bank)
    if endGene not in bankSet: return -1
    q = deque([(startGene, 0)]); visited = {startGene}
    while q:
        gene, step = q.popleft()
        for i in range(8):
            for c in 'ACGT':
                nxt = gene[:i] + c + gene[i+1:]
                if nxt == endGene: return step + 1
                if nxt in bankSet and nxt not in visited:
                    visited.add(nxt); q.append((nxt, step + 1))
    return -1
```

```typescript
function minMutation(startGene: string, endGene: string, bank: string[]): number {
  const bankSet = new Set(bank)
  if (!bankSet.has(endGene)) return -1
  const q: [string, number][] = [[startGene, 0]]
  const visited = new Set([startGene])
  while (q.length) {
    const [gene, step] = q.shift()!
    for (let i = 0; i < 8; i++)
      for (const c of 'ACGT') {
        const nxt = gene.slice(0, i) + c + gene.slice(i + 1)
        if (nxt === endGene) return step + 1
        if (bankSet.has(nxt) && !visited.has(nxt)) {
          visited.add(nxt)
          q.push([nxt, step + 1])
        }
      }
  }
  return -1
}
```

## 435. 无重叠区间 (Non-overlapping Intervals)

**描述**：移除最少区间使剩余不重叠。

**思路**：贪心，按结束时间排序，总区间 - 最多不重叠区间数。

```python
def eraseOverlapIntervals(intervals):
    intervals.sort(key=lambda x: x[1])
    cnt = 0; end = float('-inf')
    for s, e in intervals:
        if s >= end: cnt += 1; end = e
    return len(intervals) - cnt
```

```typescript
function eraseOverlapIntervals(intervals: number[][]): number {
  intervals.sort((a, b) => a[1] - b[1])
  let cnt = 0,
    end = -Infinity
  for (const [s, e] of intervals) {
    if (s >= end) {
      cnt++
      end = e
    }
  }
  return intervals.length - cnt
}
```

## 436. 寻找右区间 (Find Right Interval)

**描述**：对每个区间找起始 ≥ 其终止的最小起始区间。

**思路**：排序 + 二分查找。

```python
import bisect
def findRightInterval(intervals):
    starts = sorted((iv[0], i) for i, iv in enumerate(intervals))
    res = []
    for _, e in intervals:
        pos = bisect.bisect_left(starts, (e,))
        res.append(starts[pos][1] if pos < len(starts) else -1)
    return res
```

```typescript
function findRightInterval(intervals: number[][]): number[] {
  const starts = intervals.map((iv, i) => [iv[0], i]).sort((a, b) => a[0] - b[0])
  return intervals.map(([, e]) => {
    let lo = 0,
      hi = starts.length
    while (lo < hi) {
      const mid = (lo + hi) >> 1
      if (starts[mid][0] < e) lo = mid + 1
      else hi = mid
    }
    return lo < starts.length ? starts[lo][1] : -1
  })
}
```

## 437. 路径总和 III (Path Sum III)

**描述**：二叉树中找节点值之和等于 target 的路径数（不需要从根到叶）。

**思路**：前缀和 + 哈希表，DFS 过程中记录前缀和。

```python
from collections import defaultdict
def pathSum(root, targetSum):
    prefix = defaultdict(int); prefix[0] = 1; res = 0
    def dfs(node, cur_sum):
        nonlocal res
        if not node: return
        cur_sum += node.val
        res += prefix[cur_sum - targetSum]
        prefix[cur_sum] += 1
        dfs(node.left, cur_sum); dfs(node.right, cur_sum)
        prefix[cur_sum] -= 1
    dfs(root, 0)
    return res
```

```typescript
function pathSum(root: TreeNode | null, targetSum: number): number {
  const prefix = new Map<number, number>()
  prefix.set(0, 1)
  let res = 0
  function dfs(node: TreeNode | null, curSum: number) {
    if (!node) return
    curSum += node.val
    res += prefix.get(curSum - targetSum) || 0
    prefix.set(curSum, (prefix.get(curSum) || 0) + 1)
    dfs(node.left, curSum)
    dfs(node.right, curSum)
    prefix.set(curSum, prefix.get(curSum)! - 1)
  }
  dfs(root, 0)
  return res
}
```

## 438. 找到字符串中所有字母异位词 (Find All Anagrams in a String)

**描述**：在 s 中找 p 的所有异位词的起始下标。

**思路**：滑动窗口 + 计数数组比较。

```python
from collections import Counter
def findAnagrams(s, p):
    if len(p) > len(s): return []
    pc = Counter(p); sc = Counter(s[:len(p)])
    res = [0] if sc == pc else []
    for i in range(len(p), len(s)):
        sc[s[i]] += 1
        old = s[i - len(p)]
        sc[old] -= 1
        if sc[old] == 0: del sc[old]
        if sc == pc: res.append(i - len(p) + 1)
    return res
```

```typescript
function findAnagrams(s: string, p: string): number[] {
  if (p.length > s.length) return []
  const cnt = new Array(26).fill(0)
  for (let i = 0; i < p.length; i++) {
    cnt[p.charCodeAt(i) - 97]++
    cnt[s.charCodeAt(i) - 97]--
  }
  const res: number[] = []
  if (cnt.every(c => c === 0)) res.push(0)
  for (let i = p.length; i < s.length; i++) {
    cnt[s.charCodeAt(i) - 97]--
    cnt[s.charCodeAt(i - p.length) - 97]++
    if (cnt.every(c => c === 0)) res.push(i - p.length + 1)
  }
  return res
}
```

## 440. 字典序的第 K 小数字 (K-th Smallest in Lexicographical Order)

**描述**：1~n 字典序排列中第 k 个数。

**思路**：十叉树，计算以 prefix 为前缀的数字个数来决定向下还是向右。

```python
def findKthNumber(n, k):
    def count_prefix(prefix):
        cur = prefix; nxt = prefix + 1; cnt = 0
        while cur <= n:
            cnt += min(n + 1, nxt) - cur
            cur *= 10; nxt *= 10
        return cnt
    cur = 1; k -= 1
    while k > 0:
        cnt = count_prefix(cur)
        if cnt <= k:
            k -= cnt; cur += 1  # 移到右兄弟
        else:
            k -= 1; cur *= 10  # 深入子树
    return cur
```

```typescript
function findKthNumber(n: number, k: number): number {
  function countPrefix(prefix: number): number {
    let cur = prefix,
      nxt = prefix + 1,
      cnt = 0
    while (cur <= n) {
      cnt += Math.min(n + 1, nxt) - cur
      cur *= 10
      nxt *= 10
    }
    return cnt
  }
  let cur = 1
  k--
  while (k > 0) {
    const cnt = countPrefix(cur)
    if (cnt <= k) {
      k -= cnt
      cur++
    } else {
      k--
      cur *= 10
    }
  }
  return cur
}
```

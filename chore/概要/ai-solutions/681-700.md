## 684. 冗余连接 (Redundant Connection)

**描述**：无向图多了一条边形成环，找出应删除的边。

**思路**：并查集，加边时发现已连通即为答案。

```python
def findRedundantConnection(edges):
    parent = list(range(len(edges)+1))
    def find(x):
        while parent[x] != x: parent[x] = parent[parent[x]]; x = parent[x]
        return x
    for u, v in edges:
        pu, pv = find(u), find(v)
        if pu == pv: return [u, v]
        parent[pu] = pv
```

```typescript
function findRedundantConnection(edges: number[][]): number[] {
  const parent = Array.from({ length: edges.length + 1 }, (_, i) => i)
  const find = (x: number): number => {
    while (parent[x] !== x) {
      parent[x] = parent[parent[x]]
      x = parent[x]
    }
    return x
  }
  for (const [u, v] of edges) {
    const pu = find(u),
      pv = find(v)
    if (pu === pv) return [u, v]
    parent[pu] = pv
  }
  return []
}
```

## 685. 冗余连接 II (Redundant Connection II)

**描述**：有向图（根树+一条多余边）。

**思路**：如果有节点入度 2，分两种情况处理；否则找环。

```python
def findRedundantDirectedConnection(edges):
    n = len(edges); parent = [0]*(n+1)
    cand1 = cand2 = None
    for u, v in edges:
        if parent[v]:  # 入度2
            cand1 = [parent[v], v]; cand2 = [u, v]
            break
        parent[v] = u
    # 并查集
    uf = list(range(n+1))
    def find(x):
        while uf[x] != x: uf[x] = uf[uf[x]]; x = uf[x]
        return x
    for u, v in edges:
        if [u,v] == cand2: continue  # 先去掉cand2
        pu, pv = find(u), find(v)
        if pu == pv: return cand1 if cand1 else [u, v]
        uf[pu] = pv
    return cand2
```

```typescript
function findRedundantDirectedConnection(edges: number[][]): number[] {
  const n = edges.length,
    par = new Array(n + 1).fill(0)
  let cand1: number[] | null = null,
    cand2: number[] | null = null
  for (const [u, v] of edges) {
    if (par[v]) {
      cand1 = [par[v], v]
      cand2 = [u, v]
      break
    }
    par[v] = u
  }
  const uf = Array.from({ length: n + 1 }, (_, i) => i)
  const find = (x: number): number => {
    while (uf[x] !== x) {
      uf[x] = uf[uf[x]]
      x = uf[x]
    }
    return x
  }
  for (const [u, v] of edges) {
    if (cand2 && u === cand2[0] && v === cand2[1]) continue
    const pu = find(u),
      pv = find(v)
    if (pu === pv) return cand1 || [u, v]
    uf[pu] = pv
  }
  return cand2!
}
```

## 686. 重复叠加字符串匹配 (Repeated String Matching)

**描述**：a 重复几次能包含 b。

**思路**：至少重复 ceil(len(b)/len(a)) 次，再多检查一次。

```python
import math
def repeatedStringMatch(a, b):
    times = math.ceil(len(b) / len(a))
    for t in [times, times+1]:
        if b in a * t: return t
    return -1
```

```typescript
function repeatedStringMatch(a: string, b: string): number {
  const times = Math.ceil(b.length / a.length)
  if (a.repeat(times).includes(b)) return times
  if (a.repeat(times + 1).includes(b)) return times + 1
  return -1
}
```

## 687. 最长同值路径 (Longest Univalue Path)

**描述**：二叉树中值相同的最长路径。

**思路**：DFS，返回从当前节点向下的最长同值臂。

```python
def longestUnivaluePath(root):
    res = [0]
    def dfs(node):
        if not node: return 0
        l, r = dfs(node.left), dfs(node.right)
        al = l+1 if node.left and node.left.val == node.val else 0
        ar = r+1 if node.right and node.right.val == node.val else 0
        res[0] = max(res[0], al+ar)
        return max(al, ar)
    dfs(root); return res[0]
```

```typescript
function longestUnivaluePath(root: TreeNode | null): number {
  let res = 0
  function dfs(node: TreeNode | null): number {
    if (!node) return 0
    const l = dfs(node.left),
      r = dfs(node.right)
    const al = node.left?.val === node.val ? l + 1 : 0
    const ar = node.right?.val === node.val ? r + 1 : 0
    res = Math.max(res, al + ar)
    return Math.max(al, ar)
  }
  dfs(root)
  return res
}
```

## 688. 骑士在棋盘上的概率 (Knight Probability in Chessboard)

**描述**：棋盘上骑士走 k 步后仍在棋盘内的概率。

**思路**：DP。

```python
def knightProbability(n, k, row, column):
    dp = [[0]*n for _ in range(n)]
    dp[row][column] = 1
    moves = [(-2,-1),(-2,1),(-1,-2),(-1,2),(1,-2),(1,2),(2,-1),(2,1)]
    for _ in range(k):
        ndp = [[0]*n for _ in range(n)]
        for i in range(n):
            for j in range(n):
                if dp[i][j] == 0: continue
                for di, dj in moves:
                    ni, nj = i+di, j+dj
                    if 0<=ni<n and 0<=nj<n: ndp[ni][nj] += dp[i][j]/8
        dp = ndp
    return sum(dp[i][j] for i in range(n) for j in range(n))
```

```typescript
function knightProbability(n: number, k: number, row: number, column: number): number {
  let dp = Array.from({ length: n }, () => new Array(n).fill(0))
  dp[row][column] = 1
  const moves = [
    [-2, -1],
    [-2, 1],
    [-1, -2],
    [-1, 2],
    [1, -2],
    [1, 2],
    [2, -1],
    [2, 1]
  ]
  for (let s = 0; s < k; s++) {
    const ndp = Array.from({ length: n }, () => new Array(n).fill(0))
    for (let i = 0; i < n; i++)
      for (let j = 0; j < n; j++) {
        if (!dp[i][j]) continue
        for (const [di, dj] of moves) {
          const ni = i + di,
            nj = j + dj
          if (ni >= 0 && ni < n && nj >= 0 && nj < n) ndp[ni][nj] += dp[i][j] / 8
        }
      }
    dp = ndp
  }
  return dp.reduce((s, r) => s + r.reduce((a, b) => a + b, 0), 0)
}
```

## 692. 前 K 个高频单词 (Top K Frequent Words)

**描述**：返回出现频率最高的 k 个单词，相同频率按字典序。

**思路**：排序。

```python
from collections import Counter
def topKFrequent(words, k):
    cnt = Counter(words)
    return sorted(cnt, key=lambda w: (-cnt[w], w))[:k]
```

```typescript
function topKFrequent(words: string[], k: number): string[] {
  const cnt = new Map<string, number>()
  for (const w of words) cnt.set(w, (cnt.get(w) ?? 0) + 1)
  return [...cnt.entries()]
    .sort((a, b) => b[1] - a[1] || a[0].localeCompare(b[0]))
    .slice(0, k)
    .map(e => e[0])
}
```

## 693. 交替位二进制数 (Binary Number with Alternating Bits)

**描述**：检查二进制是否 0101 交替。

**思路**：n^(n>>1) 应该全是 1。

```python
def hasAlternatingBits(n):
    x = n ^ (n >> 1)
    return x & (x + 1) == 0
```

```typescript
function hasAlternatingBits(n: number): boolean {
  const x = n ^ (n >> 1)
  return (x & (x + 1)) === 0
}
```

## 695. 岛屿的最大面积 (Max Area of Island)

**描述**：网格中最大岛屿面积。

**思路**：DFS 洪泛。

```python
def maxAreaOfIsland(grid):
    m, n = len(grid), len(grid[0])
    def dfs(i, j):
        if i < 0 or i >= m or j < 0 or j >= n or grid[i][j] == 0: return 0
        grid[i][j] = 0
        return 1 + dfs(i+1,j)+dfs(i-1,j)+dfs(i,j+1)+dfs(i,j-1)
    return max(dfs(i,j) for i in range(m) for j in range(n))
```

```typescript
function maxAreaOfIsland(grid: number[][]): number {
  const m = grid.length,
    n = grid[0].length
  function dfs(i: number, j: number): number {
    if (i < 0 || i >= m || j < 0 || j >= n || !grid[i][j]) return 0
    grid[i][j] = 0
    return 1 + dfs(i + 1, j) + dfs(i - 1, j) + dfs(i, j + 1) + dfs(i, j - 1)
  }
  let res = 0
  for (let i = 0; i < m; i++) for (let j = 0; j < n; j++) res = Math.max(res, dfs(i, j))
  return res
}
```

## 696. 计数二进制子串 (Count Binary Substrings)

**描述**：连续 0 和 1 组成的子串中相等个数的。

**思路**：分组计数，相邻组取 min。

```python
def countBinarySubstrings(s):
    groups = []; cnt = 1
    for i in range(1, len(s)):
        if s[i] == s[i-1]: cnt += 1
        else: groups.append(cnt); cnt = 1
    groups.append(cnt)
    return sum(min(groups[i], groups[i+1]) for i in range(len(groups)-1))
```

```typescript
function countBinarySubstrings(s: string): number {
  const groups: number[] = []
  let cnt = 1
  for (let i = 1; i < s.length; i++) {
    if (s[i] === s[i - 1]) cnt++
    else {
      groups.push(cnt)
      cnt = 1
    }
  }
  groups.push(cnt)
  let res = 0
  for (let i = 0; i < groups.length - 1; i++) res += Math.min(groups[i], groups[i + 1])
  return res
}
```

## 697. 数组的度 (Degree of an Array)

**描述**：最短子数组使其度和原数组相同。

**思路**：记录每个数的首尾位置和频次。

```python
def findShortestSubArray(nums):
    first = {}; last = {}; cnt = {}
    for i, n in enumerate(nums):
        if n not in first: first[n] = i
        last[n] = i; cnt[n] = cnt.get(n,0)+1
    mx = max(cnt.values()); res = len(nums)
    for k, v in cnt.items():
        if v == mx: res = min(res, last[k]-first[k]+1)
    return res
```

```typescript
function findShortestSubArray(nums: number[]): number {
  const first = new Map<number, number>(),
    last = new Map<number, number>(),
    cnt = new Map<number, number>()
  nums.forEach((n, i) => {
    if (!first.has(n)) first.set(n, i)
    last.set(n, i)
    cnt.set(n, (cnt.get(n) ?? 0) + 1)
  })
  const mx = Math.max(...cnt.values())
  let res = nums.length
  for (const [k, v] of cnt) if (v === mx) res = Math.min(res, last.get(k)! - first.get(k)! + 1)
  return res
}
```

## 698. 划分为 k 个相等的子集 (Partition to K Equal Sum Subsets)

**描述**：数组能否划分成 k 组，每组和相同。

**思路**：回溯 + 剪枝。

```python
def canPartitionKSubsets(nums, k):
    total = sum(nums)
    if total % k: return False
    target = total // k; nums.sort(reverse=True)
    if nums[0] > target: return False
    buckets = [0]*k
    def dfs(idx):
        if idx == len(nums): return True
        for i in range(k):
            if buckets[i]+nums[idx] <= target:
                buckets[i] += nums[idx]
                if dfs(idx+1): return True
                buckets[i] -= nums[idx]
            if buckets[i] == 0: break  # 空桶剪枝
        return False
    return dfs(0)
```

```typescript
function canPartitionKSubsets(nums: number[], k: number): boolean {
  const total = nums.reduce((a, b) => a + b, 0)
  if (total % k) return false
  const target = total / k
  nums.sort((a, b) => b - a)
  if (nums[0] > target) return false
  const buckets = new Array(k).fill(0)
  function dfs(idx: number): boolean {
    if (idx === nums.length) return true
    for (let i = 0; i < k; i++) {
      if (buckets[i] + nums[idx] <= target) {
        buckets[i] += nums[idx]
        if (dfs(idx + 1)) return true
        buckets[i] -= nums[idx]
      }
      if (buckets[i] === 0) break
    }
    return false
  }
  return dfs(0)
}
```

## 700. 二叉搜索树中的搜索 (Search in a Binary Search Tree)

**描述**：BST 中查找值。

**思路**：递归或迭代。

```python
def searchBST(root, val):
    if not root or root.val == val: return root
    return searchBST(root.left, val) if val < root.val else searchBST(root.right, val)
```

```typescript
function searchBST(root: TreeNode | null, val: number): TreeNode | null {
  if (!root || root.val === val) return root
  return val < root.val ? searchBST(root.left, val) : searchBST(root.right, val)
}
```

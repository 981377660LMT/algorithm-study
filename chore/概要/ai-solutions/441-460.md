## 441. 排列硬币 (Arranging Coins)

**描述**：n 枚硬币摆阶梯，第 k 行放 k 个，求能完整填满多少行。

**思路**：数学公式：`k*(k+1)/2 <= n`，解 k = `floor((-1+sqrt(1+8n))/2)`。

```python
def arrangeCoins(n):
    return int((-1 + (1 + 8 * n) ** 0.5) / 2)
```

```typescript
function arrangeCoins(n: number): number {
  return Math.floor((-1 + Math.sqrt(1 + 8 * n)) / 2)
}
```

## 442. 数组中重复的数据 (Find All Duplicates in an Array)

**描述**：1~n 范围内数出现一次或两次，找所有出现两次的。

**思路**：原地标记，遇到 nums[abs(x)-1] 已为负说明重复。

```python
def findDuplicates(nums):
    res = []
    for n in nums:
        idx = abs(n) - 1
        if nums[idx] < 0: res.append(abs(n))
        else: nums[idx] = -nums[idx]
    return res
```

```typescript
function findDuplicates(nums: number[]): number[] {
  const res: number[] = []
  for (const n of nums) {
    const idx = Math.abs(n) - 1
    if (nums[idx] < 0) res.push(Math.abs(n))
    else nums[idx] = -nums[idx]
  }
  return res
}
```

## 443. 压缩字符串 (String Compression)

**描述**：原地压缩字符数组，返回压缩后长度。

**思路**：双指针，读写指针分别前进。

```python
def compress(chars):
    write = i = 0
    while i < len(chars):
        c = chars[i]; j = i
        while j < len(chars) and chars[j] == c: j += 1
        chars[write] = c; write += 1
        if j - i > 1:
            for d in str(j - i): chars[write] = d; write += 1
        i = j
    return write
```

```typescript
function compress(chars: string[]): number {
  let write = 0,
    i = 0
  while (i < chars.length) {
    const c = chars[i]
    let j = i
    while (j < chars.length && chars[j] === c) j++
    chars[write++] = c
    if (j - i > 1) for (const d of String(j - i)) chars[write++] = d
    i = j
  }
  return write
}
```

## 445. 两数相加 II (Add Two Numbers II)

**描述**：链表表示的两数相加（高位在前）。

**思路**：用栈逆序取出，逐位相加。

```python
def addTwoNumbers(l1, l2):
    s1, s2 = [], []
    while l1: s1.append(l1.val); l1 = l1.next
    while l2: s2.append(l2.val); l2 = l2.next
    carry = 0; head = None
    while s1 or s2 or carry:
        s = (s1.pop() if s1 else 0) + (s2.pop() if s2 else 0) + carry
        carry = s // 10
        node = ListNode(s % 10); node.next = head; head = node
    return head
```

```typescript
function addTwoNumbers(l1: ListNode | null, l2: ListNode | null): ListNode | null {
  const s1: number[] = [],
    s2: number[] = []
  while (l1) {
    s1.push(l1.val)
    l1 = l1.next
  }
  while (l2) {
    s2.push(l2.val)
    l2 = l2.next
  }
  let carry = 0,
    head: ListNode | null = null
  while (s1.length || s2.length || carry) {
    const s = (s1.pop() ?? 0) + (s2.pop() ?? 0) + carry
    carry = Math.floor(s / 10)
    const node = new ListNode(s % 10)
    node.next = head
    head = node
  }
  return head
}
```

## 446. 等差数列划分 II - 子序列 (Arithmetic Slices II - Subsequence)

**描述**：计算所有等差子序列个数（长度≥3）。

**思路**：`dp[i][d]` 表示以 i 结尾公差为 d 的等差子序列个数（含长度≥2的弱序列）。

```python
from collections import defaultdict
def numberOfArithmeticSlices(nums):
    n = len(nums); total = 0
    dp = [defaultdict(int) for _ in range(n)]
    for i in range(n):
        for j in range(i):
            d = nums[i] - nums[j]
            dp[i][d] += dp[j][d] + 1
            total += dp[j][d]  # dp[j][d] 已经是长度≥2的，加上当前变成≥3
    return total
```

```typescript
function numberOfArithmeticSlices(nums: number[]): number {
  const n = nums.length
  let total = 0
  const dp: Map<number, number>[] = Array.from({ length: n }, () => new Map())
  for (let i = 0; i < n; i++)
    for (let j = 0; j < i; j++) {
      const d = nums[i] - nums[j]
      const prev = dp[j].get(d) || 0
      dp[i].set(d, (dp[i].get(d) || 0) + prev + 1)
      total += prev
    }
  return total
}
```

## 447. 回旋镖的数量 (Number of Boomerangs)

**描述**：平面上找满足 dist(i,j)=dist(i,k) 的三元组(i,j,k)个数。

**思路**：对每个点统计到其他点的距离计数，n\*(n-1) 排列。

```python
from collections import Counter
def numberOfBoomerangs(points):
    res = 0
    for p in points:
        cnt = Counter()
        for q in points:
            d = (p[0]-q[0])**2 + (p[1]-q[1])**2
            cnt[d] += 1
        for v in cnt.values(): res += v * (v - 1)
    return res
```

```typescript
function numberOfBoomerangs(points: number[][]): number {
  let res = 0
  for (const p of points) {
    const cnt = new Map<number, number>()
    for (const q of points) {
      const d = (p[0] - q[0]) ** 2 + (p[1] - q[1]) ** 2
      cnt.set(d, (cnt.get(d) || 0) + 1)
    }
    for (const v of cnt.values()) res += v * (v - 1)
  }
  return res
}
```

```python
def findDisappearedNumbers(nums):
    for n in nums:
        idx = abs(n) - 1
        nums[idx] = -abs(nums[idx])
```

## 449. 序列化和反序列化二叉搜索树 (Serialize and Deserialize BST)

**描述**：序列化 BST（可利用 BST 性质更高效）。

**思路**：前序遍历，反序列化利用上下界重建。

```python
class Codec:
    def serialize(self, root):
        vals = []
        def preorder(node):
            if node:
                vals.append(str(node.val))
                preorder(node.left); preorder(node.right)
        preorder(root)
        return ' '.join(vals)
    def deserialize(self, data):
        if not data: return None
        vals = iter(int(x) for x in data.split())
        def build(lo, hi):
            val = next(vals, None)
            if val is None or val < lo or val > hi: return None
            # 注意：需要peek而不是consume，这里简化用队列
            pass
        # 简化实现：用前序+排序（中序）重建
        preorder = list(int(x) for x in data.split())
        inorder = sorted(preorder)
        def build2(pre, ino):
            if not pre: return None
            root = TreeNode(pre[0])
            idx = ino.index(pre[0])
            root.left = build2(pre[1:idx+1], ino[:idx])
            root.right = build2(pre[idx+1:], ino[idx+1:])
            return root
        return build2(preorder, inorder)
```

```typescript
function serialize(root: TreeNode | null): string {
  const vals: number[] = []
  function preorder(node: TreeNode | null) {
    if (!node) return
    vals.push(node.val)
    preorder(node.left)
    preorder(node.right)
  }
  preorder(root)
  return vals.join(',')
}
function deserialize(data: string): TreeNode | null {
  if (!data) return null
  const vals = data.split(',').map(Number)
  let idx = 0
  function build(lo: number, hi: number): TreeNode | null {
    if (idx >= vals.length || vals[idx] < lo || vals[idx] > hi) return null
    const node = new TreeNode(vals[idx++])
    node.left = build(lo, node.val)
    node.right = build(node.val, hi)
    return node
  }
  return build(-Infinity, Infinity)
}
```

## 450. 删除二叉搜索树中的节点 (Delete Node in a BST)

**描述**：在 BST 中删除指定值的节点。

**思路**：找到后若有两个子节点，用右子树最小值替换，然后删那个最小值。

```python
def deleteNode(root, key):
    if not root: return None
    if key < root.val: root.left = deleteNode(root.left, key)
    elif key > root.val: root.right = deleteNode(root.right, key)
    else:
        if not root.left: return root.right
        if not root.right: return root.left
        # 找右子树最小
        succ = root.right
        while succ.left: succ = succ.left
        root.val = succ.val
        root.right = deleteNode(root.right, succ.val)
    return root
```

```typescript
function deleteNode(root: TreeNode | null, key: number): TreeNode | null {
  if (!root) return null
  if (key < root.val) root.left = deleteNode(root.left, key)
  else if (key > root.val) root.right = deleteNode(root.right, key)
  else {
    if (!root.left) return root.right
    if (!root.right) return root.left
    let succ = root.right
    while (succ.left) succ = succ.left
    root.val = succ.val
    root.right = deleteNode(root.right, succ.val)
  }
  return root
}
```

## 451. 根据字符出现频率排序 (Sort Characters By Frequency)

**描述**：按字符频率降序排列字符串。

**思路**：计数 + 排序。

```python
from collections import Counter
def frequencySort(s):
    return ''.join(c * cnt for c, cnt in Counter(s).most_common())
```

```typescript
function frequencySort(s: string): string {
  const cnt = new Map<string, number>()
  for (const c of s) cnt.set(c, (cnt.get(c) || 0) + 1)
  return [...cnt.entries()]
    .sort((a, b) => b[1] - a[1])
    .map(([c, n]) => c.repeat(n))
    .join('')
}
```

## 452. 用最少数量的箭引爆气球 (Minimum Number of Arrows to Burst Balloons)

**描述**：区间覆盖问题，求最少箭数。

**思路**：贪心，按右端点排序，后一个区间起点超过当前箭的位置就加一支箭。

```python
def findMinArrowShots(points):
    points.sort(key=lambda x: x[1])
    arrows = 1; end = points[0][1]
    for s, e in points[1:]:
        if s > end: arrows += 1; end = e
    return arrows
```

```typescript
function findMinArrowShots(points: number[][]): number {
  points.sort((a, b) => a[1] - b[1])
  let arrows = 1,
    end = points[0][1]
  for (let i = 1; i < points.length; i++) {
    if (points[i][0] > end) {
      arrows++
      end = points[i][1]
    }
  }
  return arrows
}
```

## 453. 最小操作次数使数组元素相等 (Minimum Moves to Equal Array Elements)

**描述**：每次可以给 n-1 个元素 +1，求使所有元素相等的最少操作。

**思路**：等价于每次给一个元素 -1，答案 = `sum - n * min`。

```python
def minMoves(nums):

```

```typescript
function minMoves(nums: number[]): number {
  const m = Math.min(...nums)
  return nums.reduce((a, b) => a + b - m, 0)
}
```

## 454. 四数相加 II (4Sum II)

**描述**：四个数组各取一个数使和为 0 的组数。

**思路**：分两组，先枚举 A+B 存哈希表，再查 -(C+D)。

```python
from collections import Counter
def fourSumCount(nums1, nums2, nums3, nums4):
    ab = Counter(a + b for a in nums1 for b in nums2)
    return sum(ab[-(c + d)] for c in nums3 for d in nums4)
```

```typescript
function fourSumCount(nums1: number[], nums2: number[], nums3: number[], nums4: number[]): number {
  const ab = new Map<number, number>()
  for (const a of nums1) for (const b of nums2) ab.set(a + b, (ab.get(a + b) || 0) + 1)
  let cnt = 0
  for (const c of nums3) for (const d of nums4) cnt += ab.get(-(c + d)) || 0
  return cnt
}
```

```python
def findContentChildren(g, s):
```

## 456. 132 模式 (132 Pattern)

**描述**：判断数组中是否存在 i<j<k 且 nums[i]<nums[k]<nums[j]。

**思路**：从右往左扫描，单调递减栈维护"2"的候选（被弹出的最大值作为 s3），遇到小于 s3 的即成立。

```python
def find132pattern(nums):
    stack = []; s3 = float('-inf')
    for n in reversed(nums):
        if n < s3: return True
        while stack and stack[-1] < n:
            s3 = stack.pop()
        stack.append(n)
    return False
```

```typescript
function find132pattern(nums: number[]): boolean {
  const stack: number[] = []
  let s3 = -Infinity
  for (let i = nums.length - 1; i >= 0; i--) {
    if (nums[i] < s3) return true
    while (stack.length && stack[stack.length - 1] < nums[i]) s3 = stack.pop()!
    stack.push(nums[i])
  }
  return false
}
```

## 460. LFU 缓存 (LFU Cache)

**描述**：实现 LFU（最不经常使用）缓存。

**思路**：哈希表 + 频率桶（每个频率一个双向链表/有序字典），维护最小频率。

```python
from collections import defaultdict, OrderedDict
class LFUCache:
    def __init__(self, capacity):
        self.cap = capacity; self.min_freq = 0
        self.key_val = {}; self.key_freq = {}
        self.freq_keys = defaultdict(OrderedDict)
    def get(self, key):
        if key not in self.key_val: return -1
        self._update(key)
        return self.key_val[key]
    def put(self, key, value):
        if self.cap <= 0: return
        if key in self.key_val:
            self.key_val[key] = value; self._update(key); return
        if len(self.key_val) >= self.cap:
            evict, _ = self.freq_keys[self.min_freq].popitem(last=False)
            del self.key_val[evict]; del self.key_freq[evict]
        self.key_val[key] = value; self.key_freq[key] = 1
        self.freq_keys[1][key] = None; self.min_freq = 1
    def _update(self, key):
        freq = self.key_freq[key]
        del self.freq_keys[freq][key]
        if not self.freq_keys[freq] and freq == self.min_freq: self.min_freq += 1
        self.key_freq[key] = freq + 1
        self.freq_keys[freq + 1][key] = None
```

```typescript
class LFUCache {
  private cap: number
  private minFreq = 0
  private keyVal = new Map<number, number>()
  private keyFreq = new Map<number, number>()
  private freqKeys = new Map<number, Map<number, null>>()
  constructor(capacity: number) {
    this.cap = capacity
  }
  get(key: number): number {
    if (!this.keyVal.has(key)) return -1
    this.update(key)
    return this.keyVal.get(key)!
  }
  put(key: number, value: number): void {
    if (this.cap <= 0) return
    if (this.keyVal.has(key)) {
      this.keyVal.set(key, value)
      this.update(key)
      return
    }
    if (this.keyVal.size >= this.cap) {
      const bucket = this.freqKeys.get(this.minFreq)!
      const evict = bucket.keys().next().value!
      bucket.delete(evict)
      this.keyVal.delete(evict)
      this.keyFreq.delete(evict)
    }
    this.keyVal.set(key, value)
    this.keyFreq.set(key, 1)
    if (!this.freqKeys.has(1)) this.freqKeys.set(1, new Map())
    this.freqKeys.get(1)!.set(key, null)
    this.minFreq = 1
  }
  private update(key: number) {
    const freq = this.keyFreq.get(key)!
    this.freqKeys.get(freq)!.delete(key)
    if (!this.freqKeys.get(freq)!.size && freq === this.minFreq) this.minFreq++
    this.keyFreq.set(key, freq + 1)
    if (!this.freqKeys.has(freq + 1)) this.freqKeys.set(freq + 1, new Map())
    this.freqKeys.get(freq + 1)!.set(key, null)
  }
}
```

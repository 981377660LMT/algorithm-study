## 481. 神奇字符串 (Magical String)

**描述**：神奇字符串 s 由 1 和 2 组成，s 分组后长度序列就是 s 本身。求前 n 个字符中 1 的个数。

**思路**：模拟生成。

```python
def magicalString(n):
    if n <= 0: return 0
    s = [1,2,2]
    i = 2
    while len(s) < n:
        # s[i] 是下一组的长度，值在 1 和 2 交替
        val = 3 - s[-1]  # 交替 1/2
        s += [val] * s[i]
        i += 1
    return s[:n].count(1)
```

```typescript
function magicalString(n: number): number {
  if (n <= 0) return 0
  const s = [1, 2, 2]
  let i = 2
  while (s.length < n) {
    const val = 3 - s[s.length - 1]
    for (let j = 0; j < s[i]; j++) s.push(val)
    i++
  }
  return s.slice(0, n).filter(x => x === 1).length
}
```

## 482. 密钥格式化 (License Key Formatting)

**描述**：用 `-` 分隔密钥，第一组可以短，其余组长度为 k，字母大写。

**思路**：去掉 `-` 后倒着分组。

```python
def licenseKeyFormatting(s, k):
    s = s.replace('-', '').upper()
    first = len(s) % k
    parts = ([s[:first]] if first else []) + [s[i:i+k] for i in range(first, len(s), k)]
    return '-'.join(parts)
```

```typescript
function licenseKeyFormatting(s: string, k: number): string {
  s = s.replace(/-/g, '').toUpperCase()
  const first = s.length % k
  const parts: string[] = []
  if (first) parts.push(s.slice(0, first))
  for (let i = first; i < s.length; i += k) parts.push(s.slice(i, i + k))
  return parts.join('-')
}
```

## 483. 最小好进制 (Smallest Good Base)

**描述**：找最小的进制 k，使 n 在 k 进制下全为 1。

**思路**：n = 1 + k + k² + ... + k^(m-1)。枚举 m 从大到小，二分找 k。

```python
def smallestGoodBase(n):
    n = int(n)
    for m in range(n.bit_length(), 1, -1):
        k = int(n ** (1.0 / (m - 1)))
        if k >= 2 and (k**m - 1) // (k - 1) == n:
            return str(k)
    return str(n - 1)
```

```typescript
function smallestGoodBase(n: string): string {
  const num = BigInt(n)
  for (let m = BigInt(num.toString(2).length); m >= 2n; m--) {
    let k = BigInt(Math.floor(Number(num) ** (1 / Number(m - 1n))))
    if (k < 2n) k = 2n
    let sum = 0n,
      cur = 1n
    for (let i = 0n; i < m; i++) {
      sum += cur
      cur *= k
    }
    if (sum === num) return k.toString()
  }
  return (num - 1n).toString()
}
```

## 485. 最大连续 1 的个数 (Max Consecutive Ones)

**描述**：二进制数组中最大连续 1 的个数。

**思路**：遍历计数。

```python
def findMaxConsecutiveOnes(nums):
    res = cur = 0
    for n in nums:
        cur = cur + 1 if n else 0
        res = max(res, cur)
    return res
```

```typescript
function findMaxConsecutiveOnes(nums: number[]): number {
  let res = 0,
    cur = 0
  for (const n of nums) {
    cur = n ? cur + 1 : 0
    res = Math.max(res, cur)
  }
  return res
}
```

## 486. 预测赢家 (Predict the Winner)

**描述**：两人从数组两端取数，先手是否能赢。

**思路**：区间 DP，`dp[i][j]` = 当前玩家在 `[i,j]` 能多得几分。

```python
def predictTheWinner(nums):
    n = len(nums)
    dp = [[0]*n for _ in range(n)]
    for i in range(n): dp[i][i] = nums[i]
    for length in range(2, n+1):
        for i in range(n-length+1):
            j = i + length - 1
            dp[i][j] = max(nums[i]-dp[i+1][j], nums[j]-dp[i][j-1])
    return dp[0][n-1] >= 0
```

```typescript
function predictTheWinner(nums: number[]): boolean {
  const n = nums.length
  const dp = Array.from({ length: n }, () => new Array(n).fill(0))
  for (let i = 0; i < n; i++) dp[i][i] = nums[i]
  for (let len = 2; len <= n; len++)
    for (let i = 0; i <= n - len; i++) {
      const j = i + len - 1
      dp[i][j] = Math.max(nums[i] - dp[i + 1][j], nums[j] - dp[i][j - 1])
    }
  return dp[0][n - 1] >= 0
}
```

## 491. 非递减子序列 (Non-decreasing Subsequences)

**描述**：返回所有至少 2 个元素的非递减子序列（去重）。

**思路**：回溯 + 同层去重。

```python
def findSubsequences(nums):
    res = []
    def dfs(start, path):
        if len(path) >= 2: res.append(path[:])
        used = set()
        for i in range(start, len(nums)):
            if nums[i] in used: continue
            if path and nums[i] < path[-1]: continue
            used.add(nums[i])
            path.append(nums[i]); dfs(i+1, path); path.pop()
    dfs(0, []); return res
```

```typescript
function findSubsequences(nums: number[]): number[][] {
  const res: number[][] = []
  function dfs(start: number, path: number[]) {
    if (path.length >= 2) res.push([...path])
    const used = new Set<number>()
    for (let i = start; i < nums.length; i++) {
      if (used.has(nums[i])) continue
      if (path.length && nums[i] < path[path.length - 1]) continue
      used.add(nums[i])
      path.push(nums[i])
      dfs(i + 1, path)
      path.pop()
    }
  }
  dfs(0, [])
  return res
}
```

## 492. 构造矩形 (Construct the Rectangle)

**描述**：面积为 area 的矩形，L ≥ W 且差最小。

**思路**：从 sqrt 向下找因子。

```python
import math
def constructRectangle(area):
    w = int(math.isqrt(area))
    while area % w: w -= 1
    return [area // w, w]
```

```typescript
function constructRectangle(area: number): number[] {
  let w = Math.floor(Math.sqrt(area))
  while (area % w) w--
  return [area / w, w]
}
```

## 493. 翻转对 (Reverse Pairs)

**描述**：i < j 且 nums[i] > 2 \* nums[j] 的对数。

**思路**：归并排序中统计；或 BIT。

```python
def reversePairs(nums):
    def merge_sort(arr):
        if len(arr) <= 1: return arr, 0
        mid = len(arr) // 2
        left, lc = merge_sort(arr[:mid])
        right, rc = merge_sort(arr[mid:])
        cnt = lc + rc
        # 统计翻转对
        j = 0
        for i in range(len(left)):
            while j < len(right) and left[i] > 2 * right[j]: j += 1
            cnt += j
        # 合并
        merged = []; i = j2 = 0
        while i < len(left) and j2 < len(right):
            if left[i] <= right[j2]: merged.append(left[i]); i += 1
            else: merged.append(right[j2]); j2 += 1
        merged += left[i:] + right[j2:]
        return merged, cnt
    return merge_sort(nums)[1]
```

```typescript
function reversePairs(nums: number[]): number {
  let cnt = 0
  function mergeSort(arr: number[]): number[] {
    if (arr.length <= 1) return arr
    const mid = arr.length >> 1
    const left = mergeSort(arr.slice(0, mid)),
      right = mergeSort(arr.slice(mid))
    let j = 0
    for (const l of left) {
      while (j < right.length && l > 2 * right[j]) j++
      cnt += j
    }
    const merged: number[] = []
    let i = 0
    j = 0
    while (i < left.length && j < right.length)
      merged.push(left[i] <= right[j] ? left[i++] : right[j++])
    merged.push(...left.slice(i), ...right.slice(j))
    return merged
  }
  mergeSort(nums)
  return cnt
}
```

## 494. 目标和 (Target Sum)

**描述**：数组每个数前加 + 或 -，使结果等于 target。

**思路**：转化为 0-1 背包。P - Q = target, P + Q = sum → P = (sum + target) / 2。

```python
def findTargetSumWays(nums, target):
    total = sum(nums)
    if (total + target) % 2 or abs(target) > total: return 0
    p = (total + target) // 2
    dp = [0] * (p + 1); dp[0] = 1
    for n in nums:
        for j in range(p, n - 1, -1):
            dp[j] += dp[j - n]
    return dp[p]
```

```typescript
function findTargetSumWays(nums: number[], target: number): number {
  const total = nums.reduce((a, b) => a + b, 0)
  if ((total + target) % 2 || Math.abs(target) > total) return 0
  const p = (total + target) / 2
  const dp = new Array(p + 1).fill(0)
  dp[0] = 1
  for (const n of nums) for (let j = p; j >= n; j--) dp[j] += dp[j - n]
  return dp[p]
}
```

## 495. 提莫攻击 (Teemo Attacking)

**描述**：每次攻击持续 duration 秒，求总中毒时间。

**思路**：相邻间隔取 min。

```python
def findPoisonedDuration(timeSeries, duration):
    res = 0
    for i in range(len(timeSeries)):
        if i + 1 < len(timeSeries):
            res += min(duration, timeSeries[i+1] - timeSeries[i])
        else:
            res += duration
    return res
```

```typescript
function findPoisonedDuration(timeSeries: number[], duration: number): number {
  let res = 0
  for (let i = 0; i < timeSeries.length; i++)
    res +=
      i + 1 < timeSeries.length ? Math.min(duration, timeSeries[i + 1] - timeSeries[i]) : duration
  return res
}
```

## 496. 下一个更大元素 I (Next Greater Element I)

**描述**：nums1 是 nums2 子集，对 nums1 中每个元素找 nums2 中右侧第一个更大的。

**思路**：单调栈预处理 nums2。

```python
def nextGreaterElement(nums1, nums2):
    stk, mp = [], {}
    for n in nums2:
        while stk and stk[-1] < n: mp[stk.pop()] = n
        stk.append(n)
    return [mp.get(n, -1) for n in nums1]
```

```typescript
function nextGreaterElement(nums1: number[], nums2: number[]): number[] {
  const stk: number[] = [],
    mp = new Map<number, number>()
  for (const n of nums2) {
    while (stk.length && stk[stk.length - 1] < n) mp.set(stk.pop()!, n)
    stk.push(n)
  }
  return nums1.map(n => mp.get(n) ?? -1)
}
```

## 498. 对角线遍历 (Diagonal Traverse)

**描述**：按对角线蛇形遍历矩阵。

**思路**：按对角线分组，奇数组反转。

```python
def findDiagonalOrder(mat):
    if not mat: return []
    m, n = len(mat), len(mat[0])
    diags = [[] for _ in range(m+n-1)]
    for i in range(m):
        for j in range(n):
            diags[i+j].append(mat[i][j])
    res = []
    for i, d in enumerate(diags):
        res += d if i % 2 else d[::-1]
    return res
```

```typescript
function findDiagonalOrder(mat: number[][]): number[] {
  const m = mat.length,
    n = mat[0].length
  const diags: number[][] = Array.from({ length: m + n - 1 }, () => [])
  for (let i = 0; i < m; i++) for (let j = 0; j < n; j++) diags[i + j].push(mat[i][j])
  const res: number[] = []
  for (let i = 0; i < diags.length; i++) res.push(...(i % 2 ? diags[i] : diags[i].reverse()))
  return res
}
```

## 500. 键盘行 (Keyboard Row)

**描述**：找能在键盘同一行打出的单词。

**思路**：每行字母集合判断。

```python
def findWords(words):
    rows = [set("qwertyuiop"), set("asdfghjkl"), set("zxcvbnm")]
    return [w for w in words if any(set(w.lower()) <= r for r in rows)]
```

```typescript
function findWords(words: string[]): string[] {
  const rows = ['qwertyuiop', 'asdfghjkl', 'zxcvbnm'].map(r => new Set(r))
  return words.filter(w => rows.some(r => [...w.toLowerCase()].every(c => r.has(c))))
}
```

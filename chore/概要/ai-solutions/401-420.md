## 401. 二进制手表 (Binary Watch)

**描述**：LED 手表显示时间，给定亮灯数返回所有可能时间。

**思路**：枚举 0~11 小时和 0~59 分钟，统计二进制 1 的个数。

```python
def readBinaryWatch(turnedOn):
    return [f"{h}:{m:02d}" for h in range(12) for m in range(60)
            if bin(h).count('1') + bin(m).count('1') == turnedOn]
```

```typescript
function readBinaryWatch(turnedOn: number): string[] {
  const res: string[] = []
  for (let h = 0; h < 12; h++)
    for (let m = 0; m < 60; m++)
      if ((h.toString(2) + m.toString(2)).split('').filter(c => c === '1').length === turnedOn)
        res.push(`${h}:${m < 10 ? '0' : ''}${m}`)
  return res
}
```

## 402. 移掉 K 位数字 (Remove K Digits)

**描述**：移除 k 位数字使剩余数最小。

**思路**：单调栈（递增），栈顶大于当前就弹出。

```python
def removeKdigits(num, k):
    stack = []
    for d in num:
        while k and stack and stack[-1] > d: stack.pop(); k -= 1
        stack.append(d)
    return ''.join(stack[:len(stack)-k]).lstrip('0') or '0'
```

```typescript
function removeKdigits(num: string, k: number): string {
  const stack: string[] = []
  for (const d of num) {
    while (k && stack.length && stack[stack.length - 1] > d) {
      stack.pop()
      k--
    }
    stack.push(d)
  }
  return (
    stack
      .slice(0, stack.length - k)
      .join('')
      .replace(/^0+/, '') || '0'
  )
}
```

## 403. 青蛙过河 (Frog Jump)

**描述**：青蛙跳石头，上次跳 k 步，这次可跳 k-1/k/k+1，判断能否到最后。

**思路**：哈希表 DP，每块石头记录可到达时的步数集合。

```python
def canCross(stones):
    dp = {s: set() for s in stones}
    dp[0].add(0)
    for s in stones:
        for k in dp[s]:
            for nk in (k-1, k, k+1):
                if nk > 0 and s + nk in dp:
                    dp[s + nk].add(nk)
    return bool(dp[stones[-1]])
```

```typescript
function canCross(stones: number[]): boolean {
  const dp = new Map<number, Set<number>>()
  for (const s of stones) dp.set(s, new Set())
  dp.get(0)!.add(0)
  for (const s of stones)
    for (const k of dp.get(s)!)
      for (const nk of [k - 1, k, k + 1]) if (nk > 0 && dp.has(s + nk)) dp.get(s + nk)!.add(nk)
  return dp.get(stones[stones.length - 1])!.size > 0
}
```

## 404. 左叶子之和 (Sum of Left Leaves)

**描述**：求所有左叶子节点的值之和。

**思路**：递归，传入是否为左子节点的标记。

```python
def sumOfLeftLeaves(root):
    def dfs(node, is_left):
        if not node: return 0
        if not node.left and not node.right: return node.val if is_left else 0
        return dfs(node.left, True) + dfs(node.right, False)
    return dfs(root, False)
```

```typescript
function sumOfLeftLeaves(root: TreeNode | null): number {
  function dfs(node: TreeNode | null, isLeft: boolean): number {
    if (!node) return 0
    if (!node.left && !node.right) return isLeft ? node.val : 0
    return dfs(node.left, true) + dfs(node.right, false)
  }
  return dfs(root, false)
}
```

## 405. 数字转换为十六进制数 (Convert a Number to Hexadecimal)

**描述**：将整数转为十六进制（负数用补码）。

**思路**：位操作，每次取低 4 位。

```python
def toHex(num):
    if num == 0: return '0'
    if num < 0: num += 2**32
    hex_chars = '0123456789abcdef'
    res = ''
    while num:
        res = hex_chars[num & 0xf] + res; num >>= 4
    return res
```

```typescript
function toHex(num: number): string {
  if (num === 0) return '0'
  if (num < 0) num += 2 ** 32
  const hex = '0123456789abcdef'
  let res = ''
  while (num) {
    res = hex[num & 0xf] + res
    num >>>= 4
  }
  return res
}
```

## 406. 根据身高重建队列 (Queue Reconstruction by Height)

**描述**：按 (h, k) 排列人队，h 身高 k 前面 ≥ h 的人数，重建队列。

**思路**：按身高降序、k 升序排序，然后依次插入到结果的第 k 个位置。

```python
def reconstructQueue(people):
    people.sort(key=lambda x: (-x[0], x[1]))
    res = []
    for p in people: res.insert(p[1], p)
    return res
```

```typescript
function reconstructQueue(people: number[][]): number[][] {
  people.sort((a, b) => (a[0] === b[0] ? a[1] - b[1] : b[0] - a[0]))
  const res: number[][] = []
  for (const p of people) res.splice(p[1], 0, p)
  return res
}
```

## 407. 接雨水 II (Trapping Rain Water II)

**描述**：二维柱状图接雨水。

**思路**：最小堆 BFS，从边界向内扩展，每次弹出最矮的柱子。

```python
import heapq
def trapRainWater(heightMap):
    if not heightMap: return 0
    m, n = len(heightMap), len(heightMap[0])
    visited = [[False]*n for _ in range(m)]
    heap = []
    for i in range(m):
        for j in range(n):
            if i == 0 or i == m-1 or j == 0 or j == n-1:
                heapq.heappush(heap, (heightMap[i][j], i, j))
                visited[i][j] = True
    res = 0
    while heap:
        h, x, y = heapq.heappop(heap)
        for dx, dy in ((-1,0),(1,0),(0,-1),(0,1)):
            nx, ny = x+dx, y+dy
            if 0 <= nx < m and 0 <= ny < n and not visited[nx][ny]:
                visited[nx][ny] = True
                res += max(0, h - heightMap[nx][ny])
                heapq.heappush(heap, (max(h, heightMap[nx][ny]), nx, ny))
    return res
```

```typescript
function trapRainWater(heightMap: number[][]): number {
  const m = heightMap.length,
    n = heightMap[0].length
  if (m < 3 || n < 3) return 0
  // 简化：用排序代替堆
  const visited = Array.from({ length: m }, () => new Array(n).fill(false))
  const heap: [number, number, number][] = []
  for (let i = 0; i < m; i++)
    for (let j = 0; j < n; j++)
      if (i === 0 || i === m - 1 || j === 0 || j === n - 1) {
        heap.push([heightMap[i][j], i, j])
        visited[i][j] = true
      }
  heap.sort((a, b) => a[0] - b[0])
  let res = 0
  while (heap.length) {
    heap.sort((a, b) => a[0] - b[0])
    const [h, x, y] = heap.shift()!
    for (const [dx, dy] of [
      [-1, 0],
      [1, 0],
      [0, -1],
      [0, 1]
    ]) {
      const nx = x + dx,
        ny = y + dy
      if (nx >= 0 && nx < m && ny >= 0 && ny < n && !visited[nx][ny]) {
        visited[nx][ny] = true
        res += Math.max(0, h - heightMap[nx][ny])
        heap.push([Math.max(h, heightMap[nx][ny]), nx, ny])
      }
    }
  }
  return res
}
```

## 409. 最长回文串 (Longest Palindrome)

**描述**：给定字符串，求能组成的最长回文长度。

**思路**：偶数个配对，最多一个奇数。

```python
from collections import Counter
def longestPalindrome(s):
    cnt = Counter(s)
    res = sum(v // 2 * 2 for v in cnt.values())
    return res + (1 if res < len(s) else 0)
```

```typescript
function longestPalindrome(s: string): number {
  const cnt = new Map<string, number>()
  for (const c of s) cnt.set(c, (cnt.get(c) || 0) + 1)
  let res = 0
  for (const v of cnt.values()) res += Math.floor(v / 2) * 2
  return res + (res < s.length ? 1 : 0)
}
```

## 410. 分割数组的最大值 (Split Array Largest Sum)

**描述**：将数组分成 k 段，使各段和的最大值最小。

**思路**：二分答案，检查能否用 ≤ k 段满足最大和 ≤ mid。

```python
def splitArray(nums, k):
    def check(limit):
        cnt = 1; s = 0
        for n in nums:
            if s + n > limit: cnt += 1; s = 0
            s += n
        return cnt <= k
    lo, hi = max(nums), sum(nums)
    while lo < hi:
        mid = (lo + hi) // 2
        if check(mid): hi = mid
        else: lo = mid + 1
    return lo
```

```typescript
function splitArray(nums: number[], k: number): number {
  let lo = Math.max(...nums),
    hi = nums.reduce((a, b) => a + b, 0)
  while (lo < hi) {
    const mid = Math.floor((lo + hi) / 2)
    let cnt = 1,
      s = 0
    for (const n of nums) {
      if (s + n > mid) {
        cnt++
        s = 0
      }
      s += n
    }
    if (cnt <= k) hi = mid
    else lo = mid + 1
  }
  return lo
}
```

## 413. 等差数列划分 (Arithmetic Slices)

**描述**：返回数组中等差子数组的个数（长度≥3）。

**思路**：DP，连续等差长度+1时贡献+1。

```python
def numberOfArithmeticSlices(nums):
    dp = 0; res = 0
    for i in range(2, len(nums)):
        if nums[i] - nums[i-1] == nums[i-1] - nums[i-2]:
            dp += 1; res += dp
        else: dp = 0
    return res
```

```typescript
function numberOfArithmeticSlices(nums: number[]): number {
  let dp = 0,
    res = 0
  for (let i = 2; i < nums.length; i++) {
    if (nums[i] - nums[i - 1] === nums[i - 1] - nums[i - 2]) {
      dp++
      res += dp
    } else dp = 0
  }
  return res
}
```

## 414. 第三大的数 (Third Maximum Number)

**描述**：找数组中第三大的不同数，不存在就返回最大。

**思路**：维护三个变量。

```python
def thirdMax(nums):
    first = second = third = float('-inf')
    for n in nums:
        if n in (first, second, third): continue
        if n > first: first, second, third = n, first, second
        elif n > second: second, third = n, second
        elif n > third: third = n
    return third if third != float('-inf') else first
```

```typescript
function thirdMax(nums: number[]): number {
  let first = -Infinity,
    second = -Infinity,
    third = -Infinity
  for (const n of nums) {
    if (n === first || n === second || n === third) continue
    if (n > first) {
      third = second
      second = first
      first = n
    } else if (n > second) {
      third = second
      second = n
    } else if (n > third) third = n
  }
  return third !== -Infinity ? third : first
}
```

## 415. 字符串相加 (Add Strings)

**描述**：大数加法。

**思路**：从末尾模拟竖式加法。

```python
def addStrings(num1, num2):
    i, j, carry = len(num1)-1, len(num2)-1, 0
    res = []
    while i >= 0 or j >= 0 or carry:
        a = int(num1[i]) if i >= 0 else 0
        b = int(num2[j]) if j >= 0 else 0
        s = a + b + carry; carry = s // 10
        res.append(str(s % 10)); i -= 1; j -= 1
    return ''.join(res[::-1])
```

```typescript
function addStrings(num1: string, num2: string): string {
  let i = num1.length - 1,
    j = num2.length - 1,
    carry = 0
  const res: string[] = []
  while (i >= 0 || j >= 0 || carry) {
    const a = i >= 0 ? +num1[i--] : 0
    const b = j >= 0 ? +num2[j--] : 0
    const s = a + b + carry
    carry = Math.floor(s / 10)
    res.push(String(s % 10))
  }
  return res.reverse().join('')
}
```

## 416. 分割等和子集 (Partition Equal Subset Sum)

**描述**：判断数组能否分为和相等的两组。

**思路**：0-1 背包，目标 sum/2。

```python
def canPartition(nums):
    s = sum(nums)
    if s % 2: return False
    target = s // 2
    dp = [False] * (target + 1); dp[0] = True
    for n in nums:
        for j in range(target, n - 1, -1):
            dp[j] = dp[j] or dp[j - n]
    return dp[target]
```

```typescript
function canPartition(nums: number[]): boolean {
  const s = nums.reduce((a, b) => a + b, 0)
  if (s % 2) return false
  const target = s / 2
  const dp = new Array(target + 1).fill(false)
  dp[0] = true
  for (const n of nums) for (let j = target; j >= n; j--) dp[j] = dp[j] || dp[j - n]
  return dp[target]
}
```

## 417. 太平洋大西洋水流问题 (Pacific Atlantic Water Flow)

**描述**：矩阵中找能同时流入太平洋和大西洋的所有格子。

**思路**：从两个洋的边逆向 BFS/DFS，取交集。

```python
def pacificAtlantic(heights):
    m, n = len(heights), len(heights[0])
    def bfs(starts):
        q = list(starts); visited = set(starts)
        while q:
            nq = []
            for x, y in q:
                for dx, dy in ((-1,0),(1,0),(0,-1),(0,1)):
                    nx, ny = x+dx, y+dy
                    if 0<=nx<m and 0<=ny<n and (nx,ny) not in visited and heights[nx][ny]>=heights[x][y]:
                        visited.add((nx,ny)); nq.append((nx,ny))
            q = nq
        return visited
    pac = bfs([(i,0) for i in range(m)] + [(0,j) for j in range(n)])
    atl = bfs([(i,n-1) for i in range(m)] + [(m-1,j) for j in range(n)])
    return [list(p) for p in pac & atl]
```

```typescript
function pacificAtlantic(heights: number[][]): number[][] {
  const m = heights.length,
    n = heights[0].length
  function bfs(starts: [number, number][]): Set<string> {
    const visited = new Set(starts.map(([x, y]) => `${x},${y}`))
    let q = [...starts]
    while (q.length) {
      const nq: [number, number][] = []
      for (const [x, y] of q)
        for (const [dx, dy] of [
          [-1, 0],
          [1, 0],
          [0, -1],
          [0, 1]
        ]) {
          const nx = x + dx,
            ny = y + dy,
            key = `${nx},${ny}`
          if (
            nx >= 0 &&
            nx < m &&
            ny >= 0 &&
            ny < n &&
            !visited.has(key) &&
            heights[nx][ny] >= heights[x][y]
          ) {
            visited.add(key)
            nq.push([nx, ny])
          }
        }
      q = nq
    }
    return visited
  }
  const pac = bfs(
    [...Array(m)]
      .map((_, i) => [i, 0] as [number, number])
      .concat([...Array(n)].map((_, j) => [0, j]))
  )
  const atl = bfs(
    [...Array(m)]
      .map((_, i) => [i, n - 1] as [number, number])
      .concat([...Array(n)].map((_, j) => [m - 1, j]))
  )
  const res: number[][] = []
  for (const key of pac) if (atl.has(key)) res.push(key.split(',').map(Number))
  return res
}
```

## 420. 强密码检验器 (Strong Password Checker)

**描述**：复杂贪心题，修改密码使满足长度和字符要求。

**思路**：分类讨论：需要的插入/删除/替换次数，优先用删除减少三连长串的替换需求。面试中理解思路即可。

```python
def strongPasswordChecker(password):
    n = len(password)
    has_lower = any(c.islower() for c in password)
    has_upper = any(c.isupper() for c in password)
    has_digit = any(c.isdigit() for c in password)
    missing = (not has_lower) + (not has_upper) + (not has_digit)
    # 统计三连
    replace = 0; one_mod = two_mod = 0; i = 2
    while i < n:
        if password[i] == password[i-1] == password[i-2]:
            length = 2
            while i < n and password[i] == password[i-1]: i += 1; length += 1
            replace += length // 3
            if length % 3 == 0: one_mod += 1
            elif length % 3 == 1: two_mod += 1
        else: i += 1
    if n < 6: return max(missing, 6 - n)
    elif n <= 20: return max(missing, replace)
    else:
        delete = n - 20
        replace -= min(delete, one_mod)
        replace -= min(max(delete - one_mod, 0), two_mod * 2) // 2
        replace -= max(delete - one_mod - 2 * two_mod, 0) // 3
        return delete + max(missing, replace)
```

```typescript
function strongPasswordChecker(password: string): number {
  const n = password.length
  const hasLower = /[a-z]/.test(password),
    hasUpper = /[A-Z]/.test(password),
    hasDigit = /\d/.test(password)
  const missing = +!hasLower + +!hasUpper + +!hasDigit
  let replace = 0,
    oneMod = 0,
    twoMod = 0,
    i = 2
  while (i < n) {
    if (password[i] === password[i - 1] && password[i - 1] === password[i - 2]) {
      let len = 2
      while (i < n && password[i] === password[i - 1]) {
        i++
        len++
      }
      replace += Math.floor(len / 3)
      if (len % 3 === 0) oneMod++
      else if (len % 3 === 1) twoMod++
    } else i++
  }
  if (n < 6) return Math.max(missing, 6 - n)
  if (n <= 20) return Math.max(missing, replace)
  const del2 = n - 20
  replace -= Math.min(del2, oneMod)
  replace -= Math.floor(Math.min(Math.max(del2 - oneMod, 0), twoMod * 2) / 2)
  replace -= Math.floor(Math.max(del2 - oneMod - 2 * twoMod, 0) / 3)
  return del2 + Math.max(missing, replace)
}
```

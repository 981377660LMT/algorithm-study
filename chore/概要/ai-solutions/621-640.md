## 621. 任务调度器 (Task Scheduler)

**描述**：CPU 执行任务需冷却时间 n，求最少时间。

**思路**：最高频任务决定框架，`(maxFreq-1)*(n+1)+countMax`，与总任务数取 max。

```python
from collections import Counter
def leastInterval(tasks, n):
    cnt = Counter(tasks)
    mx = max(cnt.values())
    mx_cnt = sum(1 for v in cnt.values() if v == mx)
    return max(len(tasks), (mx-1)*(n+1)+mx_cnt)
```

```typescript
function leastInterval(tasks: string[], n: number): number {
  const cnt = new Map<string, number>()
  for (const t of tasks) cnt.set(t, (cnt.get(t) ?? 0) + 1)
  const mx = Math.max(...cnt.values())
  const mxCnt = [...cnt.values()].filter(v => v === mx).length
  return Math.max(tasks.length, (mx - 1) * (n + 1) + mxCnt)
}
```

## 622. 设计循环队列 (Design Circular Queue)

**描述**：实现循环队列。

**思路**：数组 + 头尾指针。

```python
class MyCircularQueue:
    def __init__(self, k):
        self.q = [0]*k; self.size = 0; self.cap = k; self.head = 0
    def enQueue(self, value):
        if self.isFull(): return False
        self.q[(self.head+self.size)%self.cap] = value; self.size += 1; return True
    def deQueue(self):
        if self.isEmpty(): return False
        self.head = (self.head+1)%self.cap; self.size -= 1; return True
    def Front(self): return -1 if self.isEmpty() else self.q[self.head]
    def Rear(self): return -1 if self.isEmpty() else self.q[(self.head+self.size-1)%self.cap]
    def isEmpty(self): return self.size == 0
    def isFull(self): return self.size == self.cap
```

```typescript
class MyCircularQueue {
  q: number[]
  size = 0
  cap: number
  head = 0
  constructor(k: number) {
    this.q = new Array(k)
    this.cap = k
  }
  enQueue(value: number): boolean {
    if (this.isFull()) return false
    this.q[(this.head + this.size) % this.cap] = value
    this.size++
    return true
  }
  deQueue(): boolean {
    if (this.isEmpty()) return false
    this.head = (this.head + 1) % this.cap
    this.size--
    return true
  }
  Front(): number {
    return this.isEmpty() ? -1 : this.q[this.head]
  }
  Rear(): number {
    return this.isEmpty() ? -1 : this.q[(this.head + this.size - 1) % this.cap]
  }
  isEmpty(): boolean {
    return this.size === 0
  }
  isFull(): boolean {
    return this.size === this.cap
  }
}
```

## 623. 在二叉树中增加一行 (Add One Row to Tree)

**描述**：在第 depth 层插入值为 val 的一行节点。

**思路**：BFS 到第 depth-1 层，插入新节点。

```python
def addOneRow(root, val, depth):
    if depth == 1:
        node = TreeNode(val); node.left = root; return node
    q = [root]; d = 1
    while d < depth - 1:
        q = [child for n in q for child in (n.left, n.right) if child]; d += 1
    for n in q:
        l, r = n.left, n.right
        n.left = TreeNode(val); n.left.left = l
        n.right = TreeNode(val); n.right.right = r
    return root
```

```typescript
function addOneRow(root: TreeNode | null, val: number, depth: number): TreeNode | null {
  if (depth === 1) {
    const node = new TreeNode(val)
    node.left = root
    return node
  }
  let q = [root!]
  let d = 1
  while (d < depth - 1) {
    q = q.flatMap(n => [n.left, n.right].filter(Boolean) as TreeNode[])
    d++
  }
  for (const n of q) {
    const l = n.left,
      r = n.right
    n.left = new TreeNode(val)
    n.left.left = l
    n.right = new TreeNode(val)
    n.right.right = r
  }
  return root
}
```

## 628. 三个数的最大乘积 (Maximum Product of Three Numbers)

**描述**：数组中三个数的最大乘积。

**思路**：排序后比较最大三个和最小两个×最大。

```python
def maximumProduct(nums):
    nums.sort()
    return max(nums[-1]*nums[-2]*nums[-3], nums[0]*nums[1]*nums[-1])
```

```typescript
function maximumProduct(nums: number[]): number {
  nums.sort((a, b) => a - b)
  const n = nums.length
  return Math.max(nums[n - 1] * nums[n - 2] * nums[n - 3], nums[0] * nums[1] * nums[n - 1])
}
```

## 629. K 个逆序对数组 (K Inverse Pairs Array)

**描述**：1~n 排列中恰好 k 个逆序对的排列数。

**思路**：DP，`dp[i][j]` = 前 i 个数有 j 个逆序对。前缀和优化。

```python
def kInversePairs(n, k):
    MOD = 10**9+7
    dp = [0]*(k+1); dp[0] = 1
    for i in range(2, n+1):
        ndp = [0]*(k+1); prefix = 0
        for j in range(k+1):
            prefix += dp[j]
            if j >= i: prefix -= dp[j-i]
            ndp[j] = prefix % MOD
        dp = ndp
    return dp[k]
```

```typescript
function kInversePairs(n: number, k: number): number {
  const MOD = 1e9 + 7
  let dp = new Array(k + 1).fill(0)
  dp[0] = 1
  for (let i = 2; i <= n; i++) {
    const ndp = new Array(k + 1).fill(0)
    let prefix = 0
    for (let j = 0; j <= k; j++) {
      prefix += dp[j]
      if (j >= i) prefix -= dp[j - i]
      ndp[j] = ((prefix % MOD) + MOD) % MOD
    }
    dp = ndp
  }
  return dp[k]
}
```

## 630. 课程表 III (Course Schedule III)

**描述**：每门课有持续时间和截止日，最多能上多少门。

**思路**：按截止日排序 + 贪心 + 大顶堆（淘汰最长的课）。

```python
import heapq
def scheduleCourse(courses):
    courses.sort(key=lambda x: x[1])
    heap = []; time = 0
    for dur, end in courses:
        time += dur; heapq.heappush(heap, -dur)
        if time > end: time += heapq.heappop(heap)  # 移除最长课
    return len(heap)
```

```typescript
function scheduleCourse(courses: number[][]): number {
  courses.sort((a, b) => a[1] - b[1])
  const heap: number[] = []
  let time = 0
  // 简化：用排序数组模拟大顶堆
  for (const [dur, end] of courses) {
    time += dur
    heap.push(dur)
    heap.sort((a, b) => b - a)
    if (time > end) {
      time -= heap.shift()!
    }
  }
  return heap.length
}
```

## 632. 最小区间 (Smallest Range Covering Elements from K Lists)

**描述**：k 个有序列表，找最小区间使每个列表至少含一个数。

**思路**：合并+滑窗，或堆。

```python
import heapq
def smallestRange(nums):
    heap = []; mx = float('-inf')
    for i, row in enumerate(nums):
        heapq.heappush(heap, (row[0], i, 0)); mx = max(mx, row[0])
    res = [float('-inf'), float('inf')]
    while True:
        mn, i, j = heapq.heappop(heap)
        if mx-mn < res[1]-res[0]: res = [mn, mx]
        if j+1 == len(nums[i]): break
        nxt = nums[i][j+1]; mx = max(mx, nxt)
        heapq.heappush(heap, (nxt, i, j+1))
    return res
```

```typescript
function smallestRange(nums: number[][]): number[] {
  // 合并 + 滑动窗口
  const merged: [number, number][] = []
  for (let i = 0; i < nums.length; i++) for (const v of nums[i]) merged.push([v, i])
  merged.sort((a, b) => a[0] - b[0])
  const cnt = new Map<number, number>()
  let have = 0,
    res = [-1e9, 1e9],
    left = 0
  for (let right = 0; right < merged.length; right++) {
    const g = merged[right][1]
    cnt.set(g, (cnt.get(g) ?? 0) + 1)
    if (cnt.get(g) === 1) have++
    while (have === nums.length) {
      if (merged[right][0] - merged[left][0] < res[1] - res[0])
        res = [merged[left][0], merged[right][0]]
      const lg = merged[left][1]
      cnt.set(lg, cnt.get(lg)! - 1)
      if (cnt.get(lg) === 0) have--
      left++
    }
  }
  return res
}
```

## 633. 平方数之和 (Sum of Square Numbers)

**描述**：判断 c 是否能表示为两个整数的平方和。

**思路**：双指针，a=0, b=sqrt(c)。

```python
import math
def judgeSquareSum(c):
    a, b = 0, int(math.isqrt(c))
    while a <= b:
        s = a*a + b*b
        if s == c: return True
        elif s < c: a += 1
        else: b -= 1
    return False
```

```typescript
function judgeSquareSum(c: number): boolean {
  let a = 0,
    b = Math.floor(Math.sqrt(c))
  while (a <= b) {
    const s = a * a + b * b
    if (s === c) return true
    if (s < c) a++
    else b--
  }
  return false
}
```

## 636. 函数的独占时间 (Exclusive Time of Functions)

**描述**：单线程 CPU 的函数执行独占时间。

**思路**：栈模拟。

```python
def exclusiveTime(n, logs):
    res = [0]*n; stk = []; prev = 0
    for log in logs:
        parts = log.split(':')
        fid, typ, ts = int(parts[0]), parts[1], int(parts[2])
        if typ == 'start':
            if stk: res[stk[-1]] += ts - prev
            stk.append(fid); prev = ts
        else:
            res[stk.pop()] += ts - prev + 1; prev = ts + 1
    return res
```

```typescript
function exclusiveTime(n: number, logs: string[]): number[] {
  const res = new Array(n).fill(0),
    stk: number[] = []
  let prev = 0
  for (const log of logs) {
    const [fid, typ, ts] = log.split(':')
    const id = +fid,
      time = +ts
    if (typ === 'start') {
      if (stk.length) res[stk[stk.length - 1]] += time - prev
      stk.push(id)
      prev = time
    } else {
      res[stk.pop()!] += time - prev + 1
      prev = time + 1
    }
  }
  return res
}
```

## 637. 二叉树的层平均值 (Average of Levels in Binary Tree)

**描述**：每层的平均值。

**思路**：BFS。

```python
from collections import deque
def averageOfLevels(root):
    res = []; q = deque([root])
    while q:
        s = 0; size = len(q)
        for _ in range(size):
            node = q.popleft(); s += node.val
            if node.left: q.append(node.left)
            if node.right: q.append(node.right)
        res.append(s / size)
    return res
```

```typescript
function averageOfLevels(root: TreeNode | null): number[] {
  const res: number[] = [],
    q = [root!]
  while (q.length) {
    let s = 0
    const size = q.length
    for (let i = 0; i < size; i++) {
      const node = q.shift()!
      s += node.val
      if (node.left) q.push(node.left)
      if (node.right) q.push(node.right)
    }
    res.push(s / size)
  }
  return res
}
```

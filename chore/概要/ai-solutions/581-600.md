## 581. 最短无序连续子数组 (Shortest Unsorted Continuous Subarray)

**描述**：找最短的子数组，排序后整个数组有序。

**思路**：从左找右边界（维护最大值），从右找左边界（维护最小值）。

```python
def findUnsortedSubarray(nums):
    n = len(nums); left = -1; right = -1
    mx, mn = nums[0], nums[-1]
    for i in range(1, n):
        mx = max(mx, nums[i])
        if nums[i] < mx: right = i
    for i in range(n-2, -1, -1):
        mn = min(mn, nums[i])
        if nums[i] > mn: left = i
    return right - left + 1 if right != -1 else 0
```

```typescript
function findUnsortedSubarray(nums: number[]): number {
  const n = nums.length
  let left = -1,
    right = -1
  let mx = nums[0],
    mn = nums[n - 1]
  for (let i = 1; i < n; i++) {
    mx = Math.max(mx, nums[i])
    if (nums[i] < mx) right = i
  }
  for (let i = n - 2; i >= 0; i--) {
    mn = Math.min(mn, nums[i])
    if (nums[i] > mn) left = i
  }
  return right === -1 ? 0 : right - left + 1
}
```

## 583. 两个字符串的删除操作 (Delete Operation for Two Strings)

**描述**：最少删除步数使两字符串相同。

**思路**：LCS → 答案 = m + n - 2 \* LCS。

```python
def minDistance(word1, word2):
    m, n = len(word1), len(word2)
    dp = [[0]*(n+1) for _ in range(m+1)]
    for i in range(1, m+1):
        for j in range(1, n+1):
            if word1[i-1] == word2[j-1]: dp[i][j] = dp[i-1][j-1]+1
            else: dp[i][j] = max(dp[i-1][j], dp[i][j-1])
    return m + n - 2 * dp[m][n]
```

```typescript
function minDistance(word1: string, word2: string): number {
  const m = word1.length,
    n = word2.length
  const dp = Array.from({ length: m + 1 }, () => new Array(n + 1).fill(0))
  for (let i = 1; i <= m; i++)
    for (let j = 1; j <= n; j++)
      dp[i][j] =
        word1[i - 1] === word2[j - 1] ? dp[i - 1][j - 1] + 1 : Math.max(dp[i - 1][j], dp[i][j - 1])
  return m + n - 2 * dp[m][n]
}
```

## 587. 安装栅栏 (Erect the Fence)

**描述**：二维凸包。

**思路**：Andrew 单调链算法。

```python
def outerTrees(trees):
    trees.sort()
    def cross(O, A, B):
        return (A[0]-O[0])*(B[1]-O[1])-(A[1]-O[1])*(B[0]-O[0])
    lower = []
    for p in trees:
        while len(lower) >= 2 and cross(lower[-2], lower[-1], p) < 0: lower.pop()
        lower.append(p)
    upper = []
    for p in reversed(trees):
        while len(upper) >= 2 and cross(upper[-2], upper[-1], p) < 0: upper.pop()
        upper.append(p)
    return list(set(map(tuple, lower + upper)))
```

```typescript
function outerTrees(trees: number[][]): number[][] {
  trees.sort((a, b) => a[0] - b[0] || a[1] - b[1])
  const cross = (O: number[], A: number[], B: number[]) =>
    (A[0] - O[0]) * (B[1] - O[1]) - (A[1] - O[1]) * (B[0] - O[0])
  const lower: number[][] = [],
    upper: number[][] = []
  for (const p of trees) {
    while (lower.length >= 2 && cross(lower[lower.length - 2], lower.at(-1)!, p) < 0) lower.pop()
    lower.push(p)
  }
  for (const p of [...trees].reverse()) {
    while (upper.length >= 2 && cross(upper[upper.length - 2], upper.at(-1)!, p) < 0) upper.pop()
    upper.push(p)
  }
  const set = new Set([...lower, ...upper].map(p => p.join(',')))
  return [...set].map(s => s.split(',').map(Number))
}
```

## 589. N 叉树的前序遍历 (N-ary Tree Preorder Traversal)

**描述**：N 叉树前序遍历。

**思路**：递归或栈迭代。

```python
def preorder(root):
    res = []
    def dfs(node):
        if not node: return
        res.append(node.val)
        for c in node.children: dfs(c)
    dfs(root); return res
```

```typescript
function preorder(root: Node | null): number[] {
  const res: number[] = []
  function dfs(node: Node | null) {
    if (!node) return
    res.push(node.val)
    for (const c of node.children) dfs(c)
  }
  dfs(root)
  return res
}
```

## 592. 分数加减运算 (Fraction Addition and Subtraction)

**描述**：分数加减。

**思路**：解析所有分数，通分计算。

```python
from math import gcd
import re
def fractionAddition(expression):
    nums = re.findall(r'[+-]?\d+/\d+', expression)
    a, b = 0, 1  # 结果 a/b
    for frac in nums:
        p, q = map(int, frac.split('/'))
        a = a*q + p*b; b = b*q
        g = gcd(abs(a), b)
        a //= g; b //= g
    return f"{a}/{b}"
```

```typescript
function fractionAddition(expression: string): string {
  const fracs = expression.match(/[+-]?\d+\/\d+/g)!
  let a = 0,
    b = 1
  const gcd = (x: number, y: number): number => (y ? gcd(y, x % y) : x)
  for (const f of fracs) {
    const [p, q] = f.split('/').map(Number)
    a = a * q + p * b
    b = b * q
    const g = gcd(Math.abs(a), Math.abs(b))
    a /= g
    b /= g
  }
  if (b < 0) {
    a = -a
    b = -b
  }
  return `${a}/${b}`
}
```

## 594. 最长和谐子序列 (Longest Harmonious Subsequence)

**描述**：最大值-最小值恰好为 1 的最长子序列。

**思路**：哈希计数，检查相邻数频率之和。

```python
from collections import Counter
def findLHS(nums):
    cnt = Counter(nums); res = 0
    for k in cnt:
        if k+1 in cnt: res = max(res, cnt[k]+cnt[k+1])
    return res
```

```typescript
function findLHS(nums: number[]): number {
  const cnt = new Map<number, number>()
  let res = 0
  for (const n of nums) cnt.set(n, (cnt.get(n) ?? 0) + 1)
  for (const [k, v] of cnt) if (cnt.has(k + 1)) res = Math.max(res, v + cnt.get(k + 1)!)
  return res
}
```

## 598. 区间加法 II (Range Addition II)

**描述**：m×n 矩阵，多次对左上角子矩阵 +1，求最大值个数。

**思路**：取所有操作行列的最小值。

```python
def maxCount(m, n, ops):
    for a, b in ops: m = min(m, a); n = min(n, b)
    return m * n
```

```typescript
function maxCount(m: number, n: number, ops: number[][]): number {
  for (const [a, b] of ops) {
    m = Math.min(m, a)
    n = Math.min(n, b)
  }
  return m * n
}
```

## 599. 两个列表的最小索引总和 (Minimum Index Sum of Two Lists)

**描述**：两人都喜欢的餐厅，索引和最小的。

**思路**：哈希。

```python
def findRestaurant(list1, list2):
    mp = {s:i for i,s in enumerate(list1)}
    res = []; minSum = float('inf')
    for j, s in enumerate(list2):
        if s in mp:
            total = mp[s] + j
            if total < minSum: minSum = total; res = [s]
            elif total == minSum: res.append(s)
    return res
```

```typescript
function findRestaurant(list1: string[], list2: string[]): string[] {
  const mp = new Map(list1.map((s, i) => [s, i]))
  let res: string[] = [],
    minSum = Infinity
  for (let j = 0; j < list2.length; j++) {
    if (mp.has(list2[j])) {
      const total = mp.get(list2[j])! + j
      if (total < minSum) {
        minSum = total
        res = [list2[j]]
      } else if (total === minSum) res.push(list2[j])
    }
  }
  return res
}
```

## 600. 不含连续 1 的非负整数 (Non-negative Integers without Consecutive Ones)

**描述**：[0, n] 中二进制表示不含连续 1 的数的个数。

**思路**：数位 DP。或利用 Fibonacci 性质。

```python
def findIntegers(n):
    # f[i] = i位长不含连续1的数量 = f[i-1]+f[i-2] (Fibonacci)
    bits = bin(n)[2:]
    k = len(bits)
    f = [0]*(k+1); f[0] = 1; f[1] = 2
    for i in range(2, k+1): f[i] = f[i-1]+f[i-2]
    res = 0; prev = 0
    for i in range(k):
        if bits[i] == '1':
            res += f[k-1-i]  # 该位取0的贡献
            if prev == 1: return res  # 连续两个1，剩余无效
            prev = 1
        else: prev = 0
    return res + 1  # n本身也合法
```

```typescript
function findIntegers(n: number): number {
  const bits = n.toString(2),
    k = bits.length
  const f = new Array(k + 1).fill(0)
  f[0] = 1
  f[1] = 2
  for (let i = 2; i <= k; i++) f[i] = f[i - 1] + f[i - 2]
  let res = 0,
    prev = 0
  for (let i = 0; i < k; i++) {
    if (bits[i] === '1') {
      res += f[k - 1 - i]
      if (prev) return res
      prev = 1
    } else prev = 0
  }
  return res + 1
}
```

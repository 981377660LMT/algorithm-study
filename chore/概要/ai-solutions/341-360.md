## 341. 扁平化嵌套列表迭代器 (Flatten Nested List Iterator)

**描述**：实现一个迭代器来展开嵌套列表。

**思路**：用栈逆序压入，hasNext 时不断展开栈顶直到是整数。

```python
class NestedIterator:
    def __init__(self, nestedList):
        self.stack = nestedList[::-1]
    def next(self):
        return self.stack.pop().getInteger()
    def hasNext(self):
        while self.stack:
            top = self.stack[-1]
            if top.isInteger(): return True
            self.stack.pop()
            self.stack.extend(top.getList()[::-1])
        return False
```

```typescript
class NestedIterator {
  private stack: NestedInteger[]
  constructor(nestedList: NestedInteger[]) {
    this.stack = [...nestedList].reverse()
  }
  next(): number {
    return this.stack.pop()!.getInteger()!
  }
  hasNext(): boolean {
    while (this.stack.length) {
      const top = this.stack[this.stack.length - 1]
      if (top.isInteger()) return true
      this.stack.pop()
      const list = top.getList()
      for (let i = list.length - 1; i >= 0; i--) this.stack.push(list[i])
    }
    return false
  }
}
```

**思路**：是 2 的幂且 1 在偶数位：`n > 0 && (n & (n-1)) == 0 && (n & 0x55555555) != 0`。

```python
def isPowerOfFour(n):

```


## 343. 整数拆分 (Integer Break)

**描述**：将正整数 n 拆成至少两个正整数之和，使乘积最大。

**思路**：尽量拆 3（n>4 时），因为 3 × 3 > 2 × 2 × 2。

```python
def integerBreak(n):
    if n <= 3: return n - 1
    q, r = divmod(n, 3)
    if r == 0: return 3 ** q
    if r == 1: return 3 ** (q-1) * 4
    return 3 ** q * 2
```

```typescript
function integerBreak(n: number): number {
  if (n <= 3) return n - 1
  const q = Math.floor(n / 3),
    r = n % 3
  if (r === 0) return 3 ** q
  if (r === 1) return 3 ** (q - 1) * 4
  return 3 ** q * 2
}
```

## 344. 反转字符串 (Reverse String)

**描述**：原地反转字符数组。

**思路**：双指针首尾交换。

```python
def reverseString(s):
    s.reverse()
```

```typescript
function reverseString(s: string[]): void {
  let l = 0,
    r = s.length - 1
  while (l < r) {
    ;[s[l], s[r]] = [s[r], s[l]]
    l++
    r--
  }
}
```

## 345. 反转字符串中的元音字母 (Reverse Vowels of a String)

**描述**：反转字符串中的元音字母。

**思路**：双指针，两端向中间扫，遇到元音就交换。

```python
def reverseVowels(s):
    s = list(s); vowels = set('aeiouAEIOU')
    l, r = 0, len(s) - 1
    while l < r:
        while l < r and s[l] not in vowels: l += 1
        while l < r and s[r] not in vowels: r -= 1
        s[l], s[r] = s[r], s[l]; l += 1; r -= 1
    return ''.join(s)
```

```typescript
function reverseVowels(s: string): string {
  const arr = s.split(''),
    vowels = new Set('aeiouAEIOU')
  let l = 0,
    r = arr.length - 1
  while (l < r) {
    while (l < r && !vowels.has(arr[l])) l++
    while (l < r && !vowels.has(arr[r])) r--
    ;[arr[l], arr[r]] = [arr[r], arr[l]]
    l++
    r--
  }
  return arr.join('')
}
```

## 347. 前 K 个高频元素 (Top K Frequent Elements)

**描述**：返回数组中出现频率最高的 k 个元素。

**思路**：桶排序，下标为频率，O(n)。

```python
from collections import Counter
def topKFrequent(nums, k):
    cnt = Counter(nums)
    buckets = [[] for _ in range(len(nums) + 1)]
    for num, freq in cnt.items(): buckets[freq].append(num)
    res = []
    for i in range(len(buckets) - 1, -1, -1):
        res.extend(buckets[i])
        if len(res) >= k: return res[:k]
    return res
```

```typescript
function topKFrequent(nums: number[], k: number): number[] {
  const cnt = new Map<number, number>()
  for (const n of nums) cnt.set(n, (cnt.get(n) || 0) + 1)
  const buckets: number[][] = Array.from({ length: nums.length + 1 }, () => [])
  for (const [num, freq] of cnt) buckets[freq].push(num)
  const res: number[] = []
  for (let i = buckets.length - 1; i >= 0 && res.length < k; i--) res.push(...buckets[i])
  return res.slice(0, k)
}
```

## 349. 两个数组的交集 (Intersection of Two Arrays)

**描述**：返回两个数组的交集（去重）。

**思路**：集合求交。

```python
def intersection(nums1, nums2):
    return list(set(nums1) & set(nums2))
```

```typescript
function intersection(nums1: number[], nums2: number[]): number[] {
  const s = new Set(nums1)
  return [...new Set(nums2.filter(n => s.has(n)))]
}
```

## 350. 两个数组的交集 II (Intersection of Two Arrays II)

**描述**：返回两个数组的交集（含重复）。

**思路**：计数器，取较小值。

```python
from collections import Counter
def intersect(nums1, nums2):
    c = Counter(nums1)
    res = []
    for n in nums2:
        if c[n] > 0: res.append(n); c[n] -= 1
    return res
```

```typescript
function intersect(nums1: number[], nums2: number[]): number[] {
  const cnt = new Map<number, number>()
  for (const n of nums1) cnt.set(n, (cnt.get(n) || 0) + 1)
  const res: number[] = []
  for (const n of nums2) {
    if ((cnt.get(n) || 0) > 0) {
      res.push(n)
      cnt.set(n, cnt.get(n)! - 1)
    }
  }
  return res
}
```

## 352. 将数据流变为多个不相交区间 (Data Stream as Disjoint Intervals)

**描述**：动态添加数字，随时返回无重叠区间列表。

**思路**：TreeMap/SortedList，插入时合并相邻区间。

```python
from sortedcontainers import SortedList
class SummaryRanges:
    def __init__(self):
        self.intervals = []
        self.seen = set()
    def addNum(self, value):
        if value in self.seen: return
        self.seen.add(value)
        # 简单方法：维护有序区间列表
        import bisect
        pos = bisect.bisect_left(self.intervals, [value])
        merged = [value, value]
        # 检查合并
        new_intervals = []
        inserted = False
        for interval in self.intervals:
            if interval[1] + 1 < merged[0]:
                new_intervals.append(interval)
            elif merged[1] + 1 < interval[0]:
                if not inserted: new_intervals.append(merged); inserted = True
                new_intervals.append(interval)
            else:
                merged[0] = min(merged[0], interval[0])
                merged[1] = max(merged[1], interval[1])
        if not inserted: new_intervals.append(merged)
        self.intervals = new_intervals
    def getIntervals(self):
        return self.intervals
```

```typescript
class SummaryRanges {
  private intervals: number[][] = []
  private seen = new Set<number>()
  addNum(value: number): void {
    if (this.seen.has(value)) return
    this.seen.add(value)
    const merged = [value, value]
    const newIntervals: number[][] = []
    let inserted = false
    for (const iv of this.intervals) {
      if (iv[1] + 1 < merged[0]) newIntervals.push(iv)
      else if (merged[1] + 1 < iv[0]) {
        if (!inserted) {
          newIntervals.push(merged)
          inserted = true
        }
        newIntervals.push(iv)
      } else {
        merged[0] = Math.min(merged[0], iv[0])
        merged[1] = Math.max(merged[1], iv[1])
      }
    }
    if (!inserted) newIntervals.push(merged)
    this.intervals = newIntervals
  }
  getIntervals(): number[][] {
    return this.intervals
  }
}
```

## 354. 俄罗斯套娃信封问题 (Russian Doll Envelopes)

**描述**：信封套娃，求最多能套多少层（宽高都严格小于）。

**思路**：按宽升序、宽相同按高降序排序，然后对高度求 LIS。

```python
import bisect
def maxEnvelopes(envelopes):
    envelopes.sort(key=lambda x: (x[0], -x[1]))
    tails = []
    for _, h in envelopes:
        pos = bisect.bisect_left(tails, h)
        if pos == len(tails): tails.append(h)
        else: tails[pos] = h
    return len(tails)
```

```typescript
function maxEnvelopes(envelopes: number[][]): number {
  envelopes.sort((a, b) => (a[0] === b[0] ? b[1] - a[1] : a[0] - b[0]))
  const tails: number[] = []
  for (const [, h] of envelopes) {
    let lo = 0,
      hi = tails.length
    while (lo < hi) {
      const mid = (lo + hi) >> 1
      if (tails[mid] < h) lo = mid + 1
      else hi = mid
    }
    tails[lo] = h
  }
  return tails.length
}
```

## 355. 设计推特 (Design Twitter)

**描述**：设计简化版 Twitter（发推、关注、取消关注、获取最近10条新闻流）。

**思路**：哈希表存关注关系和推文列表，合并 K 个有序链表取最近10条。

```python
import heapq
from collections import defaultdict
class Twitter:
    def __init__(self):
        self.time = 0; self.tweets = defaultdict(list)
        self.follows = defaultdict(set)
    def postTweet(self, userId, tweetId):
        self.tweets[userId].append((self.time, tweetId)); self.time += 1
    def getNewsFeed(self, userId):
        users = self.follows[userId] | {userId}
        heap = []
        for u in users:
            if self.tweets[u]:
                idx = len(self.tweets[u]) - 1
                t, tid = self.tweets[u][idx]
                heap.append((-t, tid, u, idx))
        heapq.heapify(heap)
        res = []
        while heap and len(res) < 10:
            _, tid, u, idx = heapq.heappop(heap)
            res.append(tid)
            if idx > 0:
                t2, tid2 = self.tweets[u][idx-1]
                heapq.heappush(heap, (-t2, tid2, u, idx-1))
        return res
    def follow(self, followerId, followeeId):
        self.follows[followerId].add(followeeId)
    def unfollow(self, followerId, followeeId):
        self.follows[followerId].discard(followeeId)
```

```typescript
class Twitter {
  private time = 0
  private tweets = new Map<number, [number, number][]>()
  private follows = new Map<number, Set<number>>()
  postTweet(userId: number, tweetId: number): void {
    if (!this.tweets.has(userId)) this.tweets.set(userId, [])
    this.tweets.get(userId)!.push([this.time++, tweetId])
  }
  getNewsFeed(userId: number): number[] {
    const users = new Set(this.follows.get(userId) || [])
    users.add(userId)
    const all: [number, number][] = []
    for (const u of users) all.push(...(this.tweets.get(u) || []))
    all.sort((a, b) => b[0] - a[0])
    return all.slice(0, 10).map(t => t[1])
  }
  follow(followerId: number, followeeId: number): void {
    if (!this.follows.has(followerId)) this.follows.set(followerId, new Set())
    this.follows.get(followerId)!.add(followeeId)
  }
  unfollow(followerId: number, followeeId: number): void {
    this.follows.get(followerId)?.delete(followeeId)
  }
}
```

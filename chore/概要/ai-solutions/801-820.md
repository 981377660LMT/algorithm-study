## 801. 使序列递增的最小交换次数 (Minimum Swaps To Make Sequences Increasing)

**描述**：两个等长数组，每步可交换同位置元素，求最少交换使两个数组都严格递增。

**思路**：DP 记录每个位置"换/不换"最小次数。

```python
def minSwap(nums1, nums2):
    n = len(nums1)
    keep, swap = 0, 1  # 不换/换的最小次数
    for i in range(1, n):
        nk, ns = float('inf'), float('inf')
        if nums1[i]>nums1[i-1] and nums2[i]>nums2[i-1]:
            nk = min(nk, keep); ns = min(ns, swap+1)
        if nums1[i]>nums2[i-1] and nums2[i]>nums1[i-1]:
            nk = min(nk, swap); ns = min(ns, keep+1)
        keep, swap = nk, ns
    return min(keep, swap)
```

```typescript
function minSwap(nums1: number[], nums2: number[]): number {
  let keep = 0,
    swap = 1
  for (let i = 1; i < nums1.length; i++) {
    let nk = Infinity,
      ns = Infinity
    if (nums1[i] > nums1[i - 1] && nums2[i] > nums2[i - 1]) {
      nk = Math.min(nk, keep)
      ns = Math.min(ns, swap + 1)
    }
    if (nums1[i] > nums2[i - 1] && nums2[i] > nums1[i - 1]) {
      nk = Math.min(nk, swap)
      ns = Math.min(ns, keep + 1)
    }
    keep = nk
    swap = ns
  }
  return Math.min(keep, swap)
}
```

## 802. 找到最终的安全状态 (Find Eventual Safe States)

**描述**：有向图中找不在任何环中的节点。

**思路**：反向拓扑排序，出度为 0 入队。

```python
def eventualSafeNodes(graph):
    n = len(graph)
    rg = [[] for _ in range(n)]
    outdeg = [0]*n
    for u in range(n):
        for v in graph[u]: rg[v].append(u)
        outdeg[u] = len(graph[u])
    from collections import deque
    q = deque(i for i in range(n) if outdeg[i]==0)
    safe = set()
    while q:
        u = q.popleft(); safe.add(u)
        for v in rg[u]:
            outdeg[v] -= 1
            if outdeg[v]==0: q.append(v)
    return sorted(safe)
```

```typescript
function eventualSafeNodes(graph: number[][]): number[] {
  const n = graph.length,
    rg: number[][] = Array.from({ length: n }, () => []),
    outdeg = new Array(n).fill(0)
  for (let u = 0; u < n; u++) {
    for (const v of graph[u]) rg[v].push(u)
    outdeg[u] = graph[u].length
  }
  const q: number[] = []
  for (let i = 0; i < n; i++) if (!outdeg[i]) q.push(i)
  const safe = new Set<number>()
  while (q.length) {
    const u = q.shift()!
    safe.add(u)
    for (const v of rg[u]) {
      outdeg[v]--
      if (!outdeg[v]) q.push(v)
    }
  }
  return [...safe].sort((a, b) => a - b)
}
```

## 806. 写字符串需要的行数 (Number of Lines To Write String)

**描述**：每行100宽，给定每个字母宽度，求行数和最后一行宽度。

**思路**：模拟。

```python
def numberOfLines(widths, s):
    lines, cur = 1, 0
    for c in s:
        w = widths[ord(c)-97]
        if cur + w > 100: lines += 1; cur = w
        else: cur += w
    return [lines, cur]
```

```typescript
function numberOfLines(widths: number[], s: string): number[] {
  let lines = 1,
    cur = 0
  for (const c of s) {
    const w = widths[c.charCodeAt(0) - 97]
    if (cur + w > 100) {
      lines++
      cur = w
    } else cur += w
  }
  return [lines, cur]
```

## 807. 保持城市天际线 (Max Increase to Keep City Skyline)

**描述**：在不改变任何一行/列最大值的前提下，最多能增加多少。

**思路**：grid[i][j] 最多增加到 min(行最大值, 列最大值)。

```python
def maxIncreaseKeepingSkyline(grid):
    n = len(grid)
    rowMax = [max(r) for r in grid]
    colMax = [max(grid[r][c] for r in range(n)) for c in range(n)]
    return sum(min(rowMax[i], colMax[j]) - grid[i][j] for i in range(n) for j in range(n))
```

```typescript
function maxIncreaseKeepingSkyline(grid: number[][]): number {
  const n = grid.length
  const rowMax = grid.map(r => Math.max(...r))
  const colMax = Array.from({ length: n }, (_, c) => Math.max(...grid.map(r => r[c])))
  let res = 0
  for (let i = 0; i < n; i++)
    for (let j = 0; j < n; j++) res += Math.min(rowMax[i], colMax[j]) - grid[i][j]
  return res
}
```

## 811. 子域名访问计数 (Subdomain Visit Count)

**描述**：统计域名及其所有父域名的访问次数。

**思路**：拆分域名逐层累计。

```python
def subdomainVisits(cpdomains):
    from collections import Counter
    cnt = Counter()
    for s in cpdomains:
        n, d = s.split(); n = int(n)
        parts = d.split('.')
        for i in range(len(parts)):
            cnt['.'.join(parts[i:])] += n
    return [f"{v} {k}" for k, v in cnt.items()]
```

```typescript
function subdomainVisits(cpdomains: string[]): string[] {
  const cnt = new Map<string, number>()
  for (const s of cpdomains) {
    const [ns, d] = s.split(' ')
    const n = +ns
    const parts = d.split('.')
    for (let i = 0; i < parts.length; i++) {
      const key = parts.slice(i).join('.')
      cnt.set(key, (cnt.get(key) ?? 0) + n)
    }
  }
  return [...cnt].map(([k, v]) => `${v} ${k}`)
}
```

## 814. 二叉树剪枝 (Binary Tree Pruning)

**描述**：删去所有不含 1 的子树。

**思路**：后序遍历，子树全 0 则裁。

```python
def pruneTree(root):
    if not root: return None
    root.left = pruneTree(root.left)
    root.right = pruneTree(root.right)
    if root.val == 0 and not root.left and not root.right: return None
    return root
```

```typescript
function pruneTree(root: TreeNode | null): TreeNode | null {
  if (!root) return null
  root.left = pruneTree(root.left)
  root.right = pruneTree(root.right)
  if (root.val === 0 && !root.left && !root.right) return null
  return root
}
```

## 815. 公交路线 (Bus Routes)

**描述**：给公交线路，求从 source 到 target 最少换乘次数。

**思路**：BFS，以线路为节点。

```python
def numBusesToDestination(routes, source, target):
    if source == target: return 0
    from collections import defaultdict, deque
    stop2bus = defaultdict(set)
    for i, r in enumerate(routes):
        for s in r: stop2bus[s].add(i)
    visited_bus, visited_stop = set(), {source}
    q = deque([source]); step = 0
    while q:
        step += 1
        for _ in range(len(q)):
            s = q.popleft()
            for bus in stop2bus[s]:
                if bus in visited_bus: continue
                visited_bus.add(bus)
                for ns in routes[bus]:
                    if ns == target: return step
                    if ns not in visited_stop: visited_stop.add(ns); q.append(ns)
    return -1
```

```typescript
function numBusesToDestination(routes: number[][], source: number, target: number): number {
  if (source === target) return 0
  const stop2bus = new Map<number, number[]>()
  for (let i = 0; i < routes.length; i++)
    for (const s of routes[i]) {
      if (!stop2bus.has(s)) stop2bus.set(s, [])
      stop2bus.get(s)!.push(i)
    }
  const vb = new Set<number>(),
    vs = new Set([source])
  let q = [source],
    step = 0
  while (q.length) {
    step++
    const nq: number[] = []
    for (const s of q)
      for (const bus of stop2bus.get(s) ?? []) {
        if (vb.has(bus)) continue
        vb.add(bus)
        for (const ns of routes[bus]) {
          if (ns === target) return step
          if (!vs.has(ns)) {
            vs.add(ns)
            nq.push(ns)
          }
        }
      }
    q = nq
  }
  return -1
}
```

## 819. 最常见的单词 (Most Common Word)

**描述**：不在 banned 列表中出现频率最高的单词。

**思路**：分词计数。

```python
def mostCommonWord(paragraph, banned):
    import re; from collections import Counter
    words = re.findall(r'\w+', paragraph.lower())
    ban = set(banned)
    cnt = Counter(w for w in words if w not in ban)
    return cnt.most_common(1)[0][0]
```

```typescript
function mostCommonWord(paragraph: string, banned: string[]): string {
  const ban = new Set(banned),
    cnt = new Map<string, number>()
  const words = paragraph.toLowerCase().match(/\w+/g) ?? []
  for (const w of words) if (!ban.has(w)) cnt.set(w, (cnt.get(w) ?? 0) + 1)
  let res = '',
    mx = 0
  for (const [k, v] of cnt)
    if (v > mx) {
      mx = v
      res = k
    }
  return res
}
```

## 820. 单词的压缩编码 (Short Encoding of Words)

**描述**：用后缀共享的方式编码单词列表，求最短参考字符串长度。

**思路**：去掉所有是其他单词后缀的单词。

```python
def minimumLengthEncoding(words):
    s = set(words)
    for w in words:
        for i in range(1, len(w)): s.discard(w[i:])
    return sum(len(w)+1 for w in s)
```

```typescript
function minimumLengthEncoding(words: string[]): number {
  const s = new Set(words)
  for (const w of words) for (let i = 1; i < w.length; i++) s.delete(w.slice(i))
  let res = 0
  for (const w of s) res += w.length + 1
  return res
}
```

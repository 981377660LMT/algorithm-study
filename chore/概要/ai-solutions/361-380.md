## 367. 有效的完全平方数 (Valid Perfect Square)

**描述**：不用 sqrt，判断正整数是否是完全平方数。

**思路**：二分查找。

```python
def isPerfectSquare(num):
    lo, hi = 1, num
    while lo <= hi:
        mid = (lo + hi) // 2
        sq = mid * mid
        if sq == num: return True
        elif sq < num: lo = mid + 1
        else: hi = mid - 1
    return False
```

```typescript
function isPerfectSquare(num: number): boolean {
  let lo = 1,
    hi = num
  while (lo <= hi) {
    const mid = lo + ((hi - lo) >> 1),
      sq = mid * mid
    if (sq === num) return true
    else if (sq < num) lo = mid + 1
    else hi = mid - 1
  }
  return false
}
```

## 368. 最大整除子集 (Largest Divisible Subset)

**描述**：找数组中最大的子集，其中任意一对都能整除。

**思路**：排序 + DP + 路径回溯，类似 LIS。

```python
def largestDivisibleSubset(nums):
    nums.sort()
    n = len(nums)
    dp = [1] * n; parent = [-1] * n
    max_idx = 0
    for i in range(1, n):
        for j in range(i):
            if nums[i] % nums[j] == 0 and dp[j] + 1 > dp[i]:
                dp[i] = dp[j] + 1; parent[i] = j
        if dp[i] > dp[max_idx]: max_idx = i
    res = []
    while max_idx != -1:
        res.append(nums[max_idx]); max_idx = parent[max_idx]
    return res[::-1]
```

```typescript
function largestDivisibleSubset(nums: number[]): number[] {
  nums.sort((a, b) => a - b)
  const n = nums.length,
    dp = new Array(n).fill(1),
    parent = new Array(n).fill(-1)
  let maxIdx = 0
  for (let i = 1; i < n; i++) {
    for (let j = 0; j < i; j++)
      if (nums[i] % nums[j] === 0 && dp[j] + 1 > dp[i]) {
        dp[i] = dp[j] + 1
        parent[i] = j
      }
    if (dp[i] > dp[maxIdx]) maxIdx = i
  }
  const res: number[] = []
  while (maxIdx !== -1) {
    res.push(nums[maxIdx])
    maxIdx = parent[maxIdx]
  }
  return res.reverse()
}
```

## 371. 两整数之和 (Sum of Two Integers)

**描述**：不使用 +/- 计算两整数之和。

**思路**：位运算，异或得无进位和，与后左移得进位，循环到无进位。

```python
def getSum(a, b):
    MASK = 0xFFFFFFFF; MAX = 0x7FFFFFFF
    while b & MASK:
        carry = (a & b) << 1
        a = a ^ b; b = carry
    return a & MASK if a > MAX else a
```

```typescript
function getSum(a: number, b: number): number {
  while (b) {
    const carry = (a & b) << 1
    a = a ^ b
    b = carry
  }
  return a
}
```

## 372. 超级次方 (Super Pow)

**描述**：计算 a^b mod 1337，b 用数组表示。

**思路**：快速幂，逐位扫描 `result = result^10 * a^digit`。

```python
def superPow(a, b):
    MOD = 1337
    def power(base, exp):
        result = 1; base %= MOD
        while exp:
            if exp & 1: result = result * base % MOD
            base = base * base % MOD; exp >>= 1
        return result
    result = 1
    for d in b:
        result = power(result, 10) * power(a, d) % MOD
    return result
```

```typescript
function superPow(a: number, b: number[]): number {
  const MOD = 1337
  function power(base: number, exp: number): number {
    let result = 1
    base %= MOD
    while (exp) {
      if (exp & 1) result = (result * base) % MOD
      base = (base * base) % MOD
      exp >>= 1
    }
    return result
  }
  let result = 1
  for (const d of b) result = (power(result, 10) * power(a, d)) % MOD
  return result
}
```

## 373. 查找和最小的 K 对数字 (Find K Pairs with Smallest Sums)

**描述**：两个升序数组，找和最小的 k 对。

**思路**：最小堆，初始放 (nums1[i]+nums2[0], i, 0)，每次弹出后扩展 j+1。

```python
import heapq
def kSmallestPairs(nums1, nums2, k):
    if not nums1 or not nums2: return []
    heap = [(nums1[i] + nums2[0], i, 0) for i in range(min(k, len(nums1)))]
    heapq.heapify(heap)
    res = []
    while heap and len(res) < k:
        _, i, j = heapq.heappop(heap)
        res.append([nums1[i], nums2[j]])
        if j + 1 < len(nums2):
            heapq.heappush(heap, (nums1[i] + nums2[j+1], i, j+1))
    return res
```

```typescript
function kSmallestPairs(nums1: number[], nums2: number[], k: number): number[][] {
  // 简化实现：使用排序
  const pairs: [number, number, number][] = []
  for (let i = 0; i < Math.min(nums1.length, k); i++)
    for (let j = 0; j < Math.min(nums2.length, k); j++) pairs.push([nums1[i] + nums2[j], i, j])
  pairs.sort((a, b) => a[0] - b[0])
  return pairs.slice(0, k).map(([, i, j]) => [nums1[i], nums2[j]])
}
```

## 374. 猜数字大小 (Guess Number Higher or Lower)

**描述**：二分猜数。

**思路**：经典二分查找。

```python
def guessNumber(n):
    lo, hi = 1, n
    while lo <= hi:
        mid = (lo + hi) // 2
        r = guess(mid)
        if r == 0: return mid
        elif r < 0: hi = mid - 1
        else: lo = mid + 1
```

```typescript
function guessNumber(n: number): number {
  let lo = 1,
    hi = n
  while (lo <= hi) {
    const mid = lo + ((hi - lo) >> 1),
      r = guess(mid)
    if (r === 0) return mid
    else if (r < 0) hi = mid - 1
    else lo = mid + 1
  }
  return lo
}
```

## 375. 猜数字大小 II (Guess Number Higher or Lower II)

**描述**：保证赢所需的最少金额（极小化极大）。

**思路**：区间 DP，`dp[i][j]` 表示猜 [i,j] 区间至少需要多少钱。

```python
def getMoneyAmount(n):
    dp = [[0]*(n+2) for _ in range(n+2)]
    for length in range(2, n+1):
        for i in range(1, n-length+2):
            j = i + length - 1
            dp[i][j] = float('inf')
            for k in range(i, j+1):
                cost = k + max(dp[i][k-1], dp[k+1][j])
                dp[i][j] = min(dp[i][j], cost)
    return dp[1][n]
```

```typescript
function getMoneyAmount(n: number): number {
  const dp = Array.from({ length: n + 2 }, () => new Array(n + 2).fill(0))
  for (let len = 2; len <= n; len++)
    for (let i = 1; i + len - 1 <= n; i++) {
      const j = i + len - 1
      dp[i][j] = Infinity
      for (let k = i; k <= j; k++)
        dp[i][j] = Math.min(dp[i][j], k + Math.max(dp[i][k - 1], dp[k + 1][j]))
    }
  return dp[1][n]
}
```

## 376. 摆动序列 (Wiggle Subsequence)

**描述**：找最长摆动子序列（交替上下波动）。

**思路**：贪心，跟踪 up/down 两个长度。

```python
def wiggleMaxLength(nums):
    up = down = 1
    for i in range(1, len(nums)):
        if nums[i] > nums[i-1]: up = down + 1
        elif nums[i] < nums[i-1]: down = up + 1
    return max(up, down)
```

```typescript
function wiggleMaxLength(nums: number[]): number {
  let up = 1,
    down = 1
  for (let i = 1; i < nums.length; i++) {
    if (nums[i] > nums[i - 1]) up = down + 1
    else if (nums[i] < nums[i - 1]) down = up + 1
  }
  return Math.max(up, down)
}
```

## 377. 组合总和 IV (Combination Sum IV)

**描述**：从数组中选数（可重复），组成目标值，求排列数。

**思路**：DP，`dp[i] = sum(dp[i - num])`，注意这是排列不是组合。

```python
def combinationSum4(nums, target):
    dp = [0] * (target + 1); dp[0] = 1
    for i in range(1, target + 1):
        for n in nums:
            if n <= i: dp[i] += dp[i - n]
    return dp[target]
```

```typescript
function combinationSum4(nums: number[], target: number): number {
  const dp = new Array(target + 1).fill(0)
  dp[0] = 1
  for (let i = 1; i <= target; i++) for (const n of nums) if (n <= i) dp[i] += dp[i - n]
  return dp[target]
}
```

## 378. 有序矩阵中第 K 小的元素 (Kth Smallest Element in a Sorted Matrix)

**描述**：行列均递增矩阵中找第 K 小元素。

**思路**：二分答案，统计矩阵中 ≤ mid 的个数（从左下角扫$O(n)$）。

```python
def kthSmallest(matrix, k):
    n = len(matrix)
    lo, hi = matrix[0][0], matrix[-1][-1]
    while lo < hi:
        mid = (lo + hi) // 2
        # 统计 <= mid 的元素数
        cnt = 0; j = n - 1
        for i in range(n):
            while j >= 0 and matrix[i][j] > mid: j -= 1
            cnt += j + 1
        if cnt < k: lo = mid + 1
        else: hi = mid
    return lo
```

```typescript
function kthSmallest(matrix: number[][], k: number): number {
  const n = matrix.length
  let lo = matrix[0][0],
    hi = matrix[n - 1][n - 1]
  while (lo < hi) {
    const mid = (lo + hi) >> 1
    let cnt = 0,
      j = n - 1
    for (let i = 0; i < n; i++) {
      while (j >= 0 && matrix[i][j] > mid) j--
      cnt += j + 1
    }
    if (cnt < k) lo = mid + 1
    else hi = mid
  }
  return lo
}
```

## 380. O(1) 时间插入、删除和获取随机元素 (Insert Delete GetRandom O(1))

**描述**：设计数据结构支持 O(1) 插入、删除、随机访问。

**思路**：数组 + 哈希表（值→下标），删除时与最后元素交换。

```python
import random
class RandomizedSet:
    def __init__(self):
        self.vals = []; self.idx = {}
    def insert(self, val):
        if val in self.idx: return False
        self.idx[val] = len(self.vals)
        self.vals.append(val)
        return True
    def remove(self, val):
        if val not in self.idx: return False
        i = self.idx[val]; last = self.vals[-1]
        self.vals[i] = last; self.idx[last] = i
        self.vals.pop(); del self.idx[val]
        return True
    def getRandom(self):
        return random.choice(self.vals)
```

```typescript
class RandomizedSet {
  private vals: number[] = []
  private idx = new Map<number, number>()
  insert(val: number): boolean {
    if (this.idx.has(val)) return false
    this.idx.set(val, this.vals.length)
    this.vals.push(val)
    return true
  }
  remove(val: number): boolean {
    if (!this.idx.has(val)) return false
    const i = this.idx.get(val)!,
      last = this.vals[this.vals.length - 1]
    this.vals[i] = last
    this.idx.set(last, i)
    this.vals.pop()
    this.idx.delete(val)
    return true
  }
  getRandom(): number {
    return this.vals[Math.floor(Math.random() * this.vals.length)]
  }
}
```

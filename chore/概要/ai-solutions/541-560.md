## 541. 反转字符串 II (Reverse String II)

**描述**：每 2k 个字符中前 k 个反转。

**思路**：步长 2k 遍历，切片反转。

```python
def reverseStr(s, k):
    s = list(s)
    for i in range(0, len(s), 2*k):
        s[i:i+k] = s[i:i+k][::-1]
    return ''.join(s)
```

```typescript
function reverseStr(s: string, k: number): string {
  const arr = [...s]
  for (let i = 0; i < arr.length; i += 2 * k) {
    let l = i,
      r = Math.min(i + k - 1, arr.length - 1)
    while (l < r) [arr[l++], arr[r--]] = [arr[r], arr[l]]
  }
  return arr.join('')
}
```

## 542. 01 矩阵 (01 Matrix)

**描述**：每个格到最近 0 的距离。

**思路**：多源 BFS，所有 0 同时入队。

```python
from collections import deque
def updateMatrix(mat):
    m, n = len(mat), len(mat[0])
    dist = [[float('inf')]*n for _ in range(m)]
    q = deque()
    for i in range(m):
        for j in range(n):
            if mat[i][j] == 0: dist[i][j] = 0; q.append((i,j))
    dirs = [(0,1),(0,-1),(1,0),(-1,0)]
    while q:
        x, y = q.popleft()
        for dx, dy in dirs:
            nx, ny = x+dx, y+dy
            if 0<=nx<m and 0<=ny<n and dist[nx][ny] > dist[x][y]+1:
                dist[nx][ny] = dist[x][y]+1; q.append((nx,ny))
    return dist
```

```typescript
function updateMatrix(mat: number[][]): number[][] {
  const m = mat.length,
    n = mat[0].length
  const dist = Array.from({ length: m }, () => new Array(n).fill(Infinity))
  const q: [number, number][] = []
  for (let i = 0; i < m; i++)
    for (let j = 0; j < n; j++)
      if (mat[i][j] === 0) {
        dist[i][j] = 0
        q.push([i, j])
      }
  const dirs = [
    [0, 1],
    [0, -1],
    [1, 0],
    [-1, 0]
  ]
  let head = 0
  while (head < q.length) {
    const [x, y] = q[head++]
    for (const [dx, dy] of dirs) {
      const nx = x + dx,
        ny = y + dy
      if (nx >= 0 && nx < m && ny >= 0 && ny < n && dist[nx][ny] > dist[x][y] + 1) {
        dist[nx][ny] = dist[x][y] + 1
        q.push([nx, ny])
      }
    }
  }
  return dist
}
```

## 543. 二叉树的直径 (Diameter of Binary Tree)

**描述**：任意两节点间路径的最大长度。

**思路**：DFS 返回深度，全局记录最大左+右。

```python
def diameterOfBinaryTree(root):
    res = [0]
    def depth(node):
        if not node: return 0
        l, r = depth(node.left), depth(node.right)
        res[0] = max(res[0], l + r)
        return max(l, r) + 1
    depth(root); return res[0]
```

```typescript
function diameterOfBinaryTree(root: TreeNode | null): number {
  let res = 0
  function depth(node: TreeNode | null): number {
    if (!node) return 0
    const l = depth(node.left),
      r = depth(node.right)
    res = Math.max(res, l + r)
    return Math.max(l, r) + 1
  }
  depth(root)
  return res
}
```

## 547. 省份数量 (Number of Provinces)

**描述**：n 个城市的连通分量数。

**思路**：并查集或 DFS。

```python
def findCircleNum(isConnected):
    n = len(isConnected); visited = [False]*n; cnt = 0
    def dfs(i):
        visited[i] = True
        for j in range(n):
            if isConnected[i][j] and not visited[j]: dfs(j)
    for i in range(n):
        if not visited[i]: dfs(i); cnt += 1
    return cnt
```

```typescript
function findCircleNum(isConnected: number[][]): number {
  const n = isConnected.length,
    visited = new Array(n).fill(false)
  let cnt = 0
  function dfs(i: number) {
    visited[i] = true
    for (let j = 0; j < n; j++) if (isConnected[i][j] && !visited[j]) dfs(j)
  }
  for (let i = 0; i < n; i++)
    if (!visited[i]) {
      dfs(i)
      cnt++
    }
  return cnt
}
```

## 552. 学生出勤记录 II (Student Attendance Record II)

**描述**：长度 n 的合法出勤串数量。

**思路**：DP，状态 (A 的次数, 末尾连续 L 数)。

```python
def checkRecord(n):
    MOD = 10**9 + 7
    # dp[a][l]: a=A次数(0/1), l=末尾连续L(0/1/2)
    dp = [[0]*3 for _ in range(2)]
    dp[0][0] = 1
    for _ in range(n):
        ndp = [[0]*3 for _ in range(2)]
        for a in range(2):
            for l in range(3):
                if dp[a][l] == 0: continue
                # 添加 P
                ndp[a][0] = (ndp[a][0] + dp[a][l]) % MOD
                # 添加 A
                if a == 0: ndp[1][0] = (ndp[1][0] + dp[a][l]) % MOD
                # 添加 L
                if l < 2: ndp[a][l+1] = (ndp[a][l+1] + dp[a][l]) % MOD
        dp = ndp
    return sum(dp[a][l] for a in range(2) for l in range(3)) % MOD
```

```typescript
function checkRecord(n: number): number {
  const MOD = 1e9 + 7
  let dp = Array.from({ length: 2 }, () => [0, 0, 0])
  dp[0][0] = 1
  for (let i = 0; i < n; i++) {
    const ndp = Array.from({ length: 2 }, () => [0, 0, 0])
    for (let a = 0; a < 2; a++)
      for (let l = 0; l < 3; l++) {
        if (!dp[a][l]) continue
        ndp[a][0] = (ndp[a][0] + dp[a][l]) % MOD
        if (a === 0) ndp[1][0] = (ndp[1][0] + dp[a][l]) % MOD
        if (l < 2) ndp[a][l + 1] = (ndp[a][l + 1] + dp[a][l]) % MOD
      }
    dp = ndp
  }
  let res = 0
  for (const row of dp) for (const v of row) res = (res + v) % MOD
  return res
}
```

## 553. 最优除法 (Optimal Division)

**描述**：加括号使除法结果最大。

**思路**：`a / (b/c/d/...) = a*c*d.../b`，即第一个除以（其余）。

```python
def optimalDivision(nums):
    if len(nums) == 1: return str(nums[0])
    if len(nums) == 2: return f"{nums[0]}/{nums[1]}"
    mid = '/'.join(map(str, nums[1:]))
    return f"{nums[0]}/({mid})"
```

```typescript
function optimalDivision(nums: number[]): string {
  if (nums.length === 1) return String(nums[0])
  if (nums.length === 2) return `${nums[0]}/${nums[1]}`
  return `${nums[0]}/(${nums.slice(1).join('/')})`
}
```

## 554. 砖墙 (Brick Wall)

**描述**：一面砖墙，画竖线穿过最少砖块。

**思路**：哈希统计每行缝隙位置出现次数，穿 = 行数 - 最多缝隙。

```python
from collections import Counter
def leastBricks(wall):
    cnt = Counter()
    for row in wall:
        s = 0
        for b in row[:-1]: s += b; cnt[s] += 1
    return len(wall) - (max(cnt.values()) if cnt else 0)
```

```typescript
function leastBricks(wall: number[][]): number {
  const cnt = new Map<number, number>()
  for (const row of wall) {
    let s = 0
    for (let i = 0; i < row.length - 1; i++) {
      s += row[i]
      cnt.set(s, (cnt.get(s) ?? 0) + 1)
    }
  }
  return wall.length - (cnt.size ? Math.max(...cnt.values()) : 0)
}
```

## 556. 下一个更大元素 III (Next Greater Element III)

**描述**：给定正整数 n，找最小的排列比 n 大的数。

**思路**：下一个排列算法。

```python
def nextGreaterElement(n):
    digits = list(str(n))
    i = len(digits)-2
    while i >= 0 and digits[i] >= digits[i+1]: i -= 1
    if i < 0: return -1
    j = len(digits)-1
    while digits[j] <= digits[i]: j -= 1
    digits[i], digits[j] = digits[j], digits[i]
    digits[i+1:] = digits[i+1:][::-1]
    res = int(''.join(digits))
    return res if res <= 2**31-1 else -1
```

```typescript
function nextGreaterElement(n: number): number {
  const digits = String(n).split('')
  let i = digits.length - 2
  while (i >= 0 && digits[i] >= digits[i + 1]) i--
  if (i < 0) return -1
  let j = digits.length - 1
  while (digits[j] <= digits[i]) j--
  ;[digits[i], digits[j]] = [digits[j], digits[i]]
  const res = Number(
    digits.slice(0, i + 1).join('') +
      digits
        .slice(i + 1)
        .reverse()
        .join('')
  )
  return res <= 2 ** 31 - 1 ? res : -1
}
```

## 559. N 叉树的最大深度 (Maximum Depth of N-ary Tree)

**描述**：N 叉树最大深度。

**思路**：递归。

```python
def maxDepth(root):
    if not root: return 0
    return 1 + max((maxDepth(c) for c in root.children), default=0)
```

```typescript
function maxDepth(root: Node | null): number {
  if (!root) return 0
  return 1 + Math.max(0, ...(root.children || []).map(maxDepth))
}
```

## 560. 和为 K 的子数组 (Subarray Sum Equals K)

**描述**：子数组和等于 k 的个数。

**思路**：前缀和 + 哈希。

```python
from collections import defaultdict
def subarraySum(nums, k):
    cnt = defaultdict(int); cnt[0] = 1; prefix = 0; res = 0
    for n in nums:
        prefix += n
        res += cnt[prefix - k]
        cnt[prefix] += 1
    return res
```

```typescript
function subarraySum(nums: number[], k: number): number {
  const cnt = new Map<number, number>([[0, 1]])
  let prefix = 0,
    res = 0
  for (const n of nums) {
    prefix += n
    res += cnt.get(prefix - k) ?? 0
    cnt.set(prefix, (cnt.get(prefix) ?? 0) + 1)
  }
  return res
}
```

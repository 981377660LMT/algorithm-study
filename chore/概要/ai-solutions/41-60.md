## 41. 缺失的第一个正数 (First Missing Positive)

**描述**：未排序整数数组，找缺失的最小正整数，要求 O(n) 时间 O(1) 空间。

**思路**：原地哈希——把每个数 x 放到下标 x-1 处，然后扫描第一个 nums[i] != i+1 的位置。

```python
def firstMissingPositive(nums):
    n = len(nums)
    for i in range(n):
        while 1 <= nums[i] <= n and nums[nums[i]-1] != nums[i]:
            nums[nums[i]-1], nums[i] = nums[i], nums[nums[i]-1]
    for i in range(n):
        if nums[i] != i + 1: return i + 1
    return n + 1
```

```typescript
function firstMissingPositive(nums: number[]): number {
  const n = nums.length
  for (let i = 0; i < n; i++)
    while (nums[i] >= 1 && nums[i] <= n && nums[nums[i] - 1] !== nums[i])
      [nums[nums[i] - 1], nums[i]] = [nums[i], nums[nums[i] - 1]]
  for (let i = 0; i < n; i++) if (nums[i] !== i + 1) return i + 1
  return n + 1
}
```

## 42. 接雨水 (Trapping Rain Water)

**描述**：给定柱状高度数组，计算能接多少雨水。

**思路**：双指针从两端收缩，维护左右最大高度，矮的一侧能确定水量。

```python
def trap(height):
    l, r = 0, len(height) - 1
    lmax = rmax = res = 0
    while l < r:
        if height[l] < height[r]:
            lmax = max(lmax, height[l])
            res += lmax - height[l]
            l += 1
        else:
            rmax = max(rmax, height[r])
            res += rmax - height[r]
            r -= 1
    return res
```

```typescript
function trap(height: number[]): number {
  let l = 0,
    r = height.length - 1,
    lmax = 0,
    rmax = 0,
    res = 0
  while (l < r) {
    if (height[l] < height[r]) {
      lmax = Math.max(lmax, height[l])
      res += lmax - height[l]
      l++
    } else {
      rmax = Math.max(rmax, height[r])
      res += rmax - height[r]
      r--
    }
  }
  return res
}
```

## 43. 字符串相乘 (Multiply Strings)

**描述**：以字符串形式给出两个非负整数，返回它们的乘积字符串。

**思路**：模拟竖式乘法，`num1[i] * num2[j]` 的结果加到 `pos[i+j]` 和 `pos[i+j+1]`。

```python
def multiply(num1, num2):
    m, n = len(num1), len(num2)
    pos = [0] * (m + n)
    for i in range(m - 1, -1, -1):
        for j in range(n - 1, -1, -1):
            mul = int(num1[i]) * int(num2[j])
            p1, p2 = i + j, i + j + 1
            total = mul + pos[p2]
            pos[p2] = total % 10
            pos[p1] += total // 10
    res = ''.join(map(str, pos)).lstrip('0')
    return res or '0'
```

```typescript
function multiply(num1: string, num2: string): string {
  const m = num1.length,
    n = num2.length,
    pos = new Array(m + n).fill(0)
  for (let i = m - 1; i >= 0; i--)
    for (let j = n - 1; j >= 0; j--) {
      const mul = +num1[i] * +num2[j],
        p1 = i + j,
        p2 = i + j + 1
      const total = mul + pos[p2]
      pos[p2] = total % 10
      pos[p1] += Math.floor(total / 10)
    }
  const res = pos.join('').replace(/^0+/, '')
  return res || '0'
}
```

## 44. 通配符匹配 (Wildcard Matching)

**描述**：实现支持 `?` 和 `*` 的模式匹配（`*` 匹配任意字符序列）。

**思路**：DP，`dp[i][j]` = s[:i] 是否匹配 p[:j]。`*` 可匹配空或任意前缀。

```python
def isMatch(s, p):
    m, n = len(s), len(p)
    dp = [[False] * (n+1) for _ in range(m+1)]
    dp[0][0] = True
    for j in range(1, n+1):
        dp[0][j] = dp[0][j-1] and p[j-1] == '*'
    for i in range(1, m+1):
        for j in range(1, n+1):
            if p[j-1] == '*':
                dp[i][j] = dp[i-1][j] or dp[i][j-1]
            else:
                dp[i][j] = dp[i-1][j-1] and (p[j-1] == '?' or p[j-1] == s[i-1])
    return dp[m][n]
```

```typescript
function isMatch(s: string, p: string): boolean {
  const m = s.length,
    n = p.length
  const dp = Array.from({ length: m + 1 }, () => Array(n + 1).fill(false))
  dp[0][0] = true
  for (let j = 1; j <= n; j++) dp[0][j] = dp[0][j - 1] && p[j - 1] === '*'
  for (let i = 1; i <= m; i++)
    for (let j = 1; j <= n; j++)
      if (p[j - 1] === '*') dp[i][j] = dp[i - 1][j] || dp[i][j - 1]
      else dp[i][j] = dp[i - 1][j - 1] && (p[j - 1] === '?' || p[j - 1] === s[i - 1])
  return dp[m][n]
}
```

## 45. 跳跃游戏 II (Jump Game II)

**描述**：数组每个元素表示最大跳跃步数，求到达末尾的最少跳跃次数。

**思路**：贪心BFS，维护当前层能到的最远位置，到达边界时跳跃次数+1。

```python
def jump(nums):
    jumps = cur_end = farthest = 0
    for i in range(len(nums) - 1):
        farthest = max(farthest, i + nums[i])
        if i == cur_end:
            jumps += 1
            cur_end = farthest
    return jumps
```

```typescript
function jump(nums: number[]): number {
  let jumps = 0,
    curEnd = 0,
    farthest = 0
  for (let i = 0; i < nums.length - 1; i++) {
    farthest = Math.max(farthest, i + nums[i])
    if (i === curEnd) {
      jumps++
      curEnd = farthest
    }
  }
  return jumps
}
```

## 46. 全排列 (Permutations)

**描述**：给定无重复数组，返回所有排列。

**思路**：回溯，已使用标记 / 交换法。

```python
def permute(nums):
    res = []
    def bt(path, used):
        if len(path) == len(nums): res.append(path[:]); return
        for i in range(len(nums)):
            if used[i]: continue
            used[i] = True; path.append(nums[i])
            bt(path, used)
            path.pop(); used[i] = False
    bt([], [False] * len(nums))
    return res
```

```typescript
function permute(nums: number[]): number[][] {
  const res: number[][] = [],
    used = new Array(nums.length).fill(false)
  const bt = (path: number[]) => {
    if (path.length === nums.length) {
      res.push([...path])
      return
    }
    for (let i = 0; i < nums.length; i++) {
      if (used[i]) continue
      used[i] = true
      path.push(nums[i])
      bt(path)
      path.pop()
      used[i] = false
    }
  }
  bt([])
  return res
}
```

## 47. 全排列 II (Permutations II)

**描述**：给定可重复数组，返回不重复的全排列。

**思路**：排序后回溯，同层跳过重复（`i > 0 && nums[i] == nums[i-1] && !used[i-1]`）。

```python
def permuteUnique(nums):
    nums.sort(); res = []
    used = [False] * len(nums)
    def bt(path):
        if len(path) == len(nums): res.append(path[:]); return
        for i in range(len(nums)):
            if used[i]: continue
            if i > 0 and nums[i] == nums[i-1] and not used[i-1]: continue
            used[i] = True; path.append(nums[i])
            bt(path)
            path.pop(); used[i] = False
    bt([])
    return res
```

```typescript
function permuteUnique(nums: number[]): number[][] {
  nums.sort((a, b) => a - b)
  const res: number[][] = [],
    used = new Array(nums.length).fill(false)

    if (path.length === nums.length) {
      res.push([...path])
      return
    }
    for (let i = 0; i < nums.length; i++) {
      if (used[i]) continue
      if (i > 0 && nums[i] === nums[i - 1] && !used[i - 1]) continue
      used[i] = true
      path.push(nums[i])
      bt(path)
      path.pop()
      used[i] = false
    }
  }
  bt([])
  return res
}
```

## 48. 旋转图像 (Rotate Image)

**描述**：将 n×n 矩阵顺时针旋转90度（原地）。

**思路**：先沿主对角线转置，再每行反转。

```python
def rotate(matrix):
    n = len(matrix)
    for i in range(n):
        for j in range(i+1, n):
            matrix[i][j], matrix[j][i] = matrix[j][i], matrix[i][j]
    for row in matrix:
        row.reverse()
```

```typescript
function rotate(matrix: number[][]): void {
  const n = matrix.length
  for (let i = 0; i < n; i++)
    for (let j = i + 1; j < n; j++) [matrix[i][j], matrix[j][i]] = [matrix[j][i], matrix[i][j]]
  for (const row of matrix) row.reverse()
}
```

## 49. 字母异位词分组 (Group Anagrams)

**描述**：把字母异位词分组。

**思路**：排序后作为键分组。

```python
from collections import defaultdict
def groupAnagrams(strs):
    groups = defaultdict(list)
    for s in strs:
        groups[tuple(sorted(s))].append(s)
    return list(groups.values())
```

```typescript
function groupAnagrams(strs: string[]): string[][] {
  const map = new Map<string, string[]>()
  for (const s of strs) {
    const key = [...s].sort().join('')
    if (!map.has(key)) map.set(key, [])
    map.get(key)!.push(s)
  }
  return [...map.values()]
}
```

## 50. Pow(x, n) (快速幂)

**描述**：实现 x 的 n 次幂。

**思路**：快速幂，n 为负取倒数，每次 n 右移一位，x 自乘。

```python
def myPow(x, n):
    if n < 0: x, n = 1/x, -n
    res = 1
    while n:
        if n & 1: res *= x
        x *= x; n >>= 1
    return res
```

```typescript
function myPow(x: number, n: number): number {
  if (n < 0) {
    x = 1 / x
    n = -n
  }
  let res = 1
  while (n) {
    if (n & 1) res *= x
    x *= x
    n = Math.floor(n / 2)
  }
  return res
}
```

## 51. N 皇后 (N-Queens)

**描述**：在 n×n 棋盘上放置 n 个皇后使其互不攻击，返回所有方案。

**思路**：回溯，按行放置，用集合记录列、主对角线、副对角线。

```python
def solveNQueens(n):
    res = []
    def bt(row, cols, diag1, diag2, board):
        if row == n: res.append([''.join(r) for r in board]); return
        for col in range(n):
            if col in cols or row-col in diag1 or row+col in diag2: continue
            board[row][col] = 'Q'
            bt(row+1, cols|{col}, diag1|{row-col}, diag2|{row+col}, board)
            board[row][col] = '.'
    bt(0, set(), set(), set(), [['.']*n for _ in range(n)])
    return res
```

```typescript
function solveNQueens(n: number): string[][] {
  const res: string[][] = []
  const cols = new Set<number>(),
    d1 = new Set<number>(),
    d2 = new Set<number>()
  const board = Array.from({ length: n }, () => '.'.repeat(n).split(''))
  const bt = (row: number) => {
    if (row === n) {
      res.push(board.map(r => r.join('')))
      return
    }
    for (let col = 0; col < n; col++) {
      if (cols.has(col) || d1.has(row - col) || d2.has(row + col)) continue
      board[row][col] = 'Q'
      cols.add(col)
      d1.add(row - col)
      d2.add(row + col)
      bt(row + 1)
      board[row][col] = '.'
      cols.delete(col)
      d1.delete(row - col)
      d2.delete(row + col)
    }
  }
  bt(0)
  return res
}
```

```python
def totalNQueens(n):

        nonlocal cnt
        if row == n: cnt += 1; return
        for col in range(n):
            if col in cols or row-col in d1 or row+col in d2: continue
            bt(row+1, cols|{col}, d1|{row-col}, d2|{row+col})
    bt(0, set(), set(), set())
    return cnt
```

## 54. 螺旋矩阵 (Spiral Matrix)

**描述**：按螺旋顺序返回矩阵中的所有元素。

**思路**：模拟，维护上下左右四个边界，依次遍历、收缩。

```python
def spiralOrder(matrix):
    res = []
    top, bottom, left, right = 0, len(matrix)-1, 0, len(matrix[0])-1
    while top <= bottom and left <= right:
        for j in range(left, right+1): res.append(matrix[top][j])
        top += 1
        for i in range(top, bottom+1): res.append(matrix[i][right])
        right -= 1
        if top <= bottom:
            for j in range(right, left-1, -1): res.append(matrix[bottom][j])
            bottom -= 1
        if left <= right:
            for i in range(bottom, top-1, -1): res.append(matrix[i][left])
            left += 1
    return res
```

```typescript
function spiralOrder(matrix: number[][]): number[] {
  const res: number[] = []
  let top = 0,
    bottom = matrix.length - 1,
    left = 0,
    right = matrix[0].length - 1
  while (top <= bottom && left <= right) {
    for (let j = left; j <= right; j++) res.push(matrix[top][j])
    top++
    for (let i = top; i <= bottom; i++) res.push(matrix[i][right])
    right--
    if (top <= bottom) {
      for (let j = right; j >= left; j--) res.push(matrix[bottom][j])
      bottom--
    }
    if (left <= right) {
      for (let i = bottom; i >= top; i--) res.push(matrix[i][left])
      left++
    }
  }
  return res
}
```

## 56. 合并区间 (Merge Intervals)

**描述**：合并所有重叠区间。

**思路**：按起点排序，遍历时若当前区间与上一个重叠则合并，否则新增。

```python
def merge(intervals):
    intervals.sort()
    res = [intervals[0]]
    for s, e in intervals[1:]:
        if s <= res[-1][1]:
            res[-1][1] = max(res[-1][1], e)
        else:
            res.append([s, e])
    return res
```

```typescript
function merge(intervals: number[][]): number[][] {
  intervals.sort((a, b) => a[0] - b[0])
  const res = [intervals[0]]
  for (let i = 1; i < intervals.length; i++) {
    if (intervals[i][0] <= res[res.length - 1][1])
      res[res.length - 1][1] = Math.max(res[res.length - 1][1], intervals[i][1])
    else res.push(intervals[i])
  }
  return res
}
```

## 57. 插入区间 (Insert Interval)

**描述**：在已排序不重叠区间中插入一个新区间并合并。

**思路**：三段处理——完全在左边的、有重叠需合并的、完全在右边的。

```python
def insert(intervals, newInterval):
    res = []
    i = 0; n = len(intervals)
    while i < n and intervals[i][1] < newInterval[0]:
        res.append(intervals[i]); i += 1
    while i < n and intervals[i][0] <= newInterval[1]:
        newInterval = [min(newInterval[0], intervals[i][0]), max(newInterval[1], intervals[i][1])]
        i += 1
    res.append(newInterval)
    while i < n:
        res.append(intervals[i]); i += 1
    return res
```

```typescript
function insert(intervals: number[][], newInterval: number[]): number[][] {
  const res: number[][] = []
  let i = 0
  while (i < intervals.length && intervals[i][1] < newInterval[0]) res.push(intervals[i++])
  while (i < intervals.length && intervals[i][0] <= newInterval[1]) {
    newInterval = [
      Math.min(newInterval[0], intervals[i][0]),
      Math.max(newInterval[1], intervals[i][1])
    ]
    i++
  }
  res.push(newInterval)
  while (i < intervals.length) res.push(intervals[i++])
  return res
}
```

```python
def generateMatrix(n):

        left += 1
    return matrix
```

## 60. 排列序列 (Permutation Sequence)

**描述**：返回 1~n 的第 k 个排列。

**思路**：数学，逐位确定：第 i 位有 `(n-i-1)!` 种后续排列，`k / factorial` 确定当前选哪个数。

```python
from math import factorial
def getPermutation(n, k):
    nums = list(range(1, n+1))
    k -= 1  # 转0-indexed
    res = ''
    for i in range(n, 0, -1):
        f = factorial(i - 1)
        idx = k // f
        res += str(nums[idx])
        nums.pop(idx)
        k %= f
    return res
```

```typescript
function getPermutation(n: number, k: number): string {
  const nums = Array.from({ length: n }, (_, i) => i + 1)
  const fact = [1]
  for (let i = 1; i <= n; i++) fact[i] = fact[i - 1] * i
  k--
  let res = ''
  for (let i = n; i > 0; i--) {
    const f = fact[i - 1]
    const idx = Math.floor(k / f)
    res += nums[idx]
    nums.splice(idx, 1)
    k %= f
  }
  return res
}
```

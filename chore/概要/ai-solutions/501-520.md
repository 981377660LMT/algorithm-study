## 501. 二叉搜索树中的众数 (Find Mode in BST)

**描述**：找出 BST 中出现频率最高的元素。

**思路**：中序遍历，利用 BST 有序性统计。

```python
def findMode(root):
    res, cur_val, cur_cnt, max_cnt = [], [None], [0], [0]
    def inorder(node):
        if not node: return
        inorder(node.left)
        if node.val == cur_val[0]: cur_cnt[0] += 1
        else: cur_val[0] = node.val; cur_cnt[0] = 1
        if cur_cnt[0] > max_cnt[0]: max_cnt[0] = cur_cnt[0]; res.clear(); res.append(node.val)
        elif cur_cnt[0] == max_cnt[0]: res.append(node.val)
        inorder(node.right)
    inorder(root); return res
```

```typescript
function findMode(root: TreeNode | null): number[] {
  let res: number[] = [],
    curVal = NaN,
    curCnt = 0,
    maxCnt = 0
  function inorder(node: TreeNode | null) {
    if (!node) return
    inorder(node.left)
    curCnt = node.val === curVal ? curCnt + 1 : 1
    curVal = node.val
    if (curCnt > maxCnt) {
      maxCnt = curCnt
      res = [curVal]
    } else if (curCnt === maxCnt) res.push(curVal)
    inorder(node.right)
  }
  inorder(root)
  return res
}
```

## 502. IPO

**描述**：最多 k 个项目，每个需要资本才能启动，求最大利润。

**思路**：贪心 + 大顶堆。按资本排序，每次选当前可启动项目中利润最大的。

```python
import heapq
def findMaximizedCapital(k, w, profits, capital):
    projects = sorted(zip(capital, profits))
    heap = []; i = 0
    for _ in range(k):
        while i < len(projects) and projects[i][0] <= w:
            heapq.heappush(heap, -projects[i][1]); i += 1
        if not heap: break
        w += -heapq.heappop(heap)
    return w
```

```typescript
function findMaximizedCapital(k: number, w: number, profits: number[], capital: number[]): number {
  const projects = capital.map((c, i) => [c, profits[i]]).sort((a, b) => a[0] - b[0])
  // 简化用排序代替堆
  const available: number[] = []
  let i = 0
  for (let j = 0; j < k; j++) {
    while (i < projects.length && projects[i][0] <= w) {
      available.push(projects[i][1])
      i++
    }
    available.sort((a, b) => b - a)
    if (!available.length) break
    w += available.shift()!
  }
  return w
}
```

## 503. 下一个更大元素 II (Next Greater Element II)

**描述**：循环数组中每个元素的下一个更大元素。

**思路**：单调栈 + 遍历两遍。

```python
def nextGreaterElements(nums):
    n = len(nums); res = [-1] * n; stk = []
    for i in range(2 * n):
        while stk and nums[stk[-1]] < nums[i % n]:
            res[stk.pop()] = nums[i % n]
        if i < n: stk.append(i)
    return res
```

```typescript
function nextGreaterElements(nums: number[]): number[] {
  const n = nums.length,
    res = new Array(n).fill(-1),
    stk: number[] = []
  for (let i = 0; i < 2 * n; i++) {
    while (stk.length && nums[stk[stk.length - 1]] < nums[i % n]) res[stk.pop()!] = nums[i % n]
    if (i < n) stk.push(i)
  }
  return res
}
```

```python
def convertToBase7(num):
    if num == 0: return "0"
    neg = num < 0; num = abs(num); res = []
    while num: res.append(str(num % 7)); num //= 7
    return ('-' if neg else '') + ''.join(res[::-1])
```

## 506. 相对名次 (Relative Ranks)

**描述**：按分数排名，前三名用 "Gold/Silver/Bronze Medal"。

**思路**：排序后建索引映射。

```python
def findRelativeRanks(score):
    rank = sorted(range(len(score)), key=lambda i: -score[i])
    medals = ["Gold Medal", "Silver Medal", "Bronze Medal"]
    res = [""] * len(score)
    for i, idx in enumerate(rank):
        res[idx] = medals[i] if i < 3 else str(i + 1)
    return res
```

```typescript
function findRelativeRanks(score: number[]): string[] {
  const rank = [...score.keys()].sort((a, b) => score[b] - score[a])
  const medals = ['Gold Medal', 'Silver Medal', 'Bronze Medal']
  const res = new Array(score.length)
  rank.forEach((idx, i) => (res[idx] = i < 3 ? medals[i] : String(i + 1)))
  return res
}
```

## 507. 完美数 (Perfect Number)

**描述**：判断正整数是否等于其所有正因子（不含自身）之和。

**思路**：枚举因子到 sqrt。

```python
def checkPerfectNumber(num):
    if num <= 1: return False
    s = 1
    for i in range(2, int(num**0.5)+1):
        if num % i == 0: s += i + num // i
    return s == num
```

```typescript
function checkPerfectNumber(num: number): boolean {
  if (num <= 1) return false
  let s = 1
  for (let i = 2; i * i <= num; i++) if (num % i === 0) s += i + num / i
  return s === num
}
```

## 508. 出现次数最多的子树元素和 (Most Frequent Subtree Sum)

**描述**：所有子树和中出现频率最高的。

**思路**：后序遍历统计。

```python
from collections import Counter
def findFrequentTreeSum(root):
    cnt = Counter()
    def dfs(node):
        if not node: return 0
        s = node.val + dfs(node.left) + dfs(node.right)
        cnt[s] += 1; return s
    dfs(root)
    mx = max(cnt.values())
    return [k for k, v in cnt.items() if v == mx]
```

```typescript
function findFrequentTreeSum(root: TreeNode | null): number[] {
  const cnt = new Map<number, number>()
  function dfs(node: TreeNode | null): number {
    if (!node) return 0
    const s = node.val + dfs(node.left) + dfs(node.right)
    cnt.set(s, (cnt.get(s) ?? 0) + 1)
    return s
  }
  dfs(root)
  const mx = Math.max(...cnt.values())
  return [...cnt.entries()].filter(([_, v]) => v === mx).map(([k]) => k)
}
```

## 509. 斐波那契数 (Fibonacci Number)

**思路**：迭代。

```python
def fib(n):
    a, b = 0, 1
    for _ in range(n): a, b = b, a + b
    return a
```

## 513. 找树左下角的值 (Find Bottom Left Tree Value)

**描述**：二叉树最底层最左边的值。

**思路**：BFS 从右到左遍历，最后访问的就是答案。

```python
from collections import deque
def findBottomLeftValue(root):
    q = deque([root])
    while q:
        node = q.popleft()
        if node.right: q.append(node.right)
        if node.left: q.append(node.left)
    return node.val
```

```typescript
function findBottomLeftValue(root: TreeNode): number {
  const q = [root]
  let node = root
  while (q.length) {
    node = q.shift()!
    if (node.right) q.push(node.right)
    if (node.left) q.push(node.left)
  }
  return node.val
}
```

## 514. 自由之路 (Freedom Trail)

**描述**：环形拨盘拼写字符串，最少步数。

**思路**：DP，`dp[i][j]` = 拼到第 i 个字符、盘子在位置 j 的最少步数。

```python
def findRotateSteps(ring, key):
    n, m = len(ring), len(key)
    dp = [[float('inf')]*n for _ in range(m+1)]
    dp[0][0] = 0
    for i in range(m):
        for j in range(n):
            if dp[i][j] == float('inf'): continue
            for k in range(n):
                if ring[k] == key[i]:
                    dist = min(abs(k-j), n-abs(k-j))
                    dp[i+1][k] = min(dp[i+1][k], dp[i][j]+dist+1)
    return min(dp[m])
```

```typescript
function findRotateSteps(ring: string, key: string): number {
  const n = ring.length,
    m = key.length
  const dp = Array.from({ length: m + 1 }, () => new Array(n).fill(Infinity))
  dp[0][0] = 0
  for (let i = 0; i < m; i++)
    for (let j = 0; j < n; j++) {
      if (dp[i][j] === Infinity) continue
      for (let k = 0; k < n; k++)
        if (ring[k] === key[i]) {
          const dist = Math.min(Math.abs(k - j), n - Math.abs(k - j))
          dp[i + 1][k] = Math.min(dp[i + 1][k], dp[i][j] + dist + 1)
        }
    }
  return Math.min(...dp[m])
}
```

## 515. 在每个树行中找最大值 (Find Largest Value in Each Tree Row)

**描述**：每一层的最大节点值。

**思路**：BFS 层序遍历。

```python
from collections import deque
def largestValues(root):
    if not root: return []
    res = []; q = deque([root])
    while q:
        mx = float('-inf')
        for _ in range(len(q)):
            node = q.popleft(); mx = max(mx, node.val)
            if node.left: q.append(node.left)
            if node.right: q.append(node.right)
        res.append(mx)
    return res
```

```typescript
function largestValues(root: TreeNode | null): number[] {
  if (!root) return []
  const res: number[] = [],
    q = [root]
  while (q.length) {
    let mx = -Infinity
    const size = q.length
    for (let i = 0; i < size; i++) {
      const node = q.shift()!
      mx = Math.max(mx, node.val)
      if (node.left) q.push(node.left)
      if (node.right) q.push(node.right)
    }
    res.push(mx)
  }
  return res
}
```

## 516. 最长回文子序列 (Longest Palindromic Subsequence)

**描述**：字符串最长回文子序列长度。

**思路**：区间 DP，`dp[i][j]` = s[i..j] 内最长回文子序列。

```python
def longestPalindromeSubseq(s):
    n = len(s)
    dp = [[0]*n for _ in range(n)]
    for i in range(n-1, -1, -1):
        dp[i][i] = 1
        for j in range(i+1, n):
            if s[i] == s[j]: dp[i][j] = dp[i+1][j-1] + 2
            else: dp[i][j] = max(dp[i+1][j], dp[i][j-1])
    return dp[0][n-1]
```

```typescript
function longestPalindromeSubseq(s: string): number {
  const n = s.length,
    dp = Array.from({ length: n }, () => new Array(n).fill(0))
  for (let i = n - 1; i >= 0; i--) {
    dp[i][i] = 1
    for (let j = i + 1; j < n; j++)
      dp[i][j] = s[i] === s[j] ? dp[i + 1][j - 1] + 2 : Math.max(dp[i + 1][j], dp[i][j - 1])
  }
  return dp[0][n - 1]
}
```

## 517. 超级洗衣机 (Super Washing Machines)

**描述**：n 台洗衣机各有若干衣服，每步每台可传一件给邻居，最少几步平衡。

**思路**：每台需要「流过」的衣服数的最大值。

```python
def findMinMoves(machines):
    total = sum(machines); n = len(machines)
    if total % n: return -1
    avg = total // n; res = flow = 0
    for m in machines:
        flow += m - avg  # 正=需右传，负=需左传
        res = max(res, abs(flow), m - avg)
    return res
```

```typescript
function findMinMoves(machines: number[]): number {
  const total = machines.reduce((a, b) => a + b, 0),
    n = machines.length
  if (total % n) return -1
  const avg = total / n
  let res = 0,
    flow = 0
  for (const m of machines) {
    flow += m - avg
    res = Math.max(res, Math.abs(flow), m - avg)
  }
  return res
}
```

## 518. 零钱兑换 II (Coin Change II)

**描述**：硬币组合数（完全背包）。

**思路**：完全背包计数。

```python
def change(amount, coins):
    dp = [0] * (amount + 1); dp[0] = 1
    for c in coins:
        for j in range(c, amount + 1):
            dp[j] += dp[j - c]
    return dp[amount]
```

```typescript
function change(amount: number, coins: number[]): number {
  const dp = new Array(amount + 1).fill(0)
  dp[0] = 1
  for (const c of coins) for (let j = c; j <= amount; j++) dp[j] += dp[j - c]
  return dp[amount]
}
```

## 520. 检测大写字母 (Detect Capital)

**思路**：三种模式判断。

```python
def detectCapitalUse(word):
    return word.isupper() or word.islower() or (word[0].isupper() and word[1:].islower())
```

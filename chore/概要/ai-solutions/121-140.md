## 121. 买卖股票的最佳时机 (Best Time to Buy and Sell Stock)

**描述**：一次买卖股票的最大利润。

**思路**：维护历史最低价，每天算当前价 - 最低价的最大利润。

```python
def maxProfit(prices):
    min_price, res = float('inf'), 0
    for p in prices:
        min_price = min(min_price, p)
        res = max(res, p - min_price)
    return res
```

```typescript
function maxProfit(prices: number[]): number {
  let minPrice = Infinity,
    res = 0
  for (const p of prices) {
    minPrice = Math.min(minPrice, p)
    res = Math.max(res, p - minPrice)
  }
  return res
}
```

## 124. 二叉树中的最大路径和 (Binary Tree Maximum Path Sum)

**描述**：求二叉树中任意路径的最大路径和（路径可以不过根）。

**思路**：后序遍历，每个节点返回"经过该节点往下的最大单侧和"，同时更新全局最大值。

```python
def maxPathSum(root):
    res = float('-inf')
    def dfs(node):
        nonlocal res
        if not node: return 0
        left = max(0, dfs(node.left))
        right = max(0, dfs(node.right))
        res = max(res, node.val + left + right)
        return node.val + max(left, right)
    dfs(root)
    return res
```

```typescript
function maxPathSum(root: TreeNode | null): number {
  let res = -Infinity
  const dfs = (node: TreeNode | null): number => {
    if (!node) return 0
    const left = Math.max(0, dfs(node.left))
    const right = Math.max(0, dfs(node.right))
    res = Math.max(res, node.val + left + right)
    return node.val + Math.max(left, right)
  }
  dfs(root)
  return res
}
```

## 125. 验证回文串 (Valid Palindrome)

**描述**：判断字符串是否为回文（只考虑字母和数字，忽略大小写）。

**思路**：双指针从两端向中间收缩，跳过非字母数字字符。

```python
def isPalindrome(s):
    l, r = 0, len(s) - 1
    while l < r:
        while l < r and not s[l].isalnum(): l += 1
        while l < r and not s[r].isalnum(): r -= 1
        if s[l].lower() != s[r].lower(): return False
        l += 1; r -= 1
    return True
```

```typescript
function isPalindrome(s: string): boolean {
  let l = 0,
    r = s.length - 1
  while (l < r) {
    while (l < r && !/[a-zA-Z0-9]/.test(s[l])) l++
    while (l < r && !/[a-zA-Z0-9]/.test(s[r])) r--
    if (s[l].toLowerCase() !== s[r].toLowerCase()) return false
    l++
    r--
  }
  return true
}
```

## 126. 单词接龙 II (Word Ladder II)

**描述**：返回从 beginWord 到 endWord 的所有最短转换序列。

**思路**：BFS建图 + DFS回溯找路径。先BFS记录每个单词的最短距离，再DFS按距离递减搜索路径。

```python
from collections import defaultdict, deque
def findLadders(beginWord, endWord, wordList):
    wordSet = set(wordList)
    if endWord not in wordSet: return []
    # BFS 建层
    layer = defaultdict(set)
    queue = deque([beginWord])
    visited = {beginWord}
    found = False
    while queue and not found:
        cur_visited = set()
        for _ in range(len(queue)):
            word = queue.popleft()
            for i in range(len(word)):
                for c in 'abcdefghijklmnopqrstuvwxyz':
                    nxt = word[:i] + c + word[i+1:]
                    if nxt in wordSet and nxt not in visited:
                        layer[nxt].add(word)
                        cur_visited.add(nxt)
                        if nxt == endWord: found = True
        visited |= cur_visited
        queue = deque(cur_visited)
    # DFS 回溯路径
    res = []
    def bt(word, path):
        if word == beginWord: res.append(path[:]); return
        for prev in layer[word]:
            path.append(prev); bt(prev, path); path.pop()
    if found:
        bt(endWord, [endWord])
        return [p[::-1] for p in res]
    return []
```

```typescript
function findLadders(beginWord: string, endWord: string, wordList: string[]): string[][] {
  const wordSet = new Set(wordList)
  if (!wordSet.has(endWord)) return []
  const layer = new Map<string, Set<string>>()
  let queue = [beginWord],
    visited = new Set([beginWord]),
    found = false
  while (queue.length && !found) {
    const curVisited = new Set<string>()
    for (const word of queue) {
      for (let i = 0; i < word.length; i++) {
        for (let c = 97; c <= 122; c++) {
          const nxt = word.slice(0, i) + String.fromCharCode(c) + word.slice(i + 1)
          if (wordSet.has(nxt) && !visited.has(nxt)) {
            if (!layer.has(nxt)) layer.set(nxt, new Set())
            layer.get(nxt)!.add(word)
            curVisited.add(nxt)
            if (nxt === endWord) found = true
          }
        }
      }
    }
    for (const w of curVisited) visited.add(w)
    queue = [...curVisited]
  }
  const res: string[][] = []
  const bt = (word: string, path: string[]) => {
    if (word === beginWord) {
      res.push([...path].reverse())
      return
    }
    for (const prev of layer.get(word) ?? []) {
      path.push(prev)
      bt(prev, path)
      path.pop()
    }
  }
  if (found) bt(endWord, [endWord])
  return res
}
```

## 127. 单词接龙 (Word Ladder)

**描述**：从 beginWord 到 endWord 的最短转换序列长度（每次改一个字母）。

**思路**：BFS，每次枚举单词每个位置的26种替换。

```python
from collections import deque
def ladderLength(beginWord, endWord, wordList):
    wordSet = set(wordList)
    if endWord not in wordSet: return 0
    queue = deque([(beginWord, 1)])
    visited = {beginWord}
    while queue:
        word, steps = queue.popleft()
        for i in range(len(word)):
            for c in 'abcdefghijklmnopqrstuvwxyz':
                nxt = word[:i] + c + word[i+1:]
                if nxt == endWord: return steps + 1
                if nxt in wordSet and nxt not in visited:
                    visited.add(nxt); queue.append((nxt, steps + 1))
    return 0
```

```typescript
function ladderLength(beginWord: string, endWord: string, wordList: string[]): number {
  const wordSet = new Set(wordList)
  if (!wordSet.has(endWord)) return 0
  const queue: [string, number][] = [[beginWord, 1]]
  const visited = new Set([beginWord])
  while (queue.length) {
    const [word, steps] = queue.shift()!
    for (let i = 0; i < word.length; i++)
      for (let c = 97; c <= 122; c++) {
        const nxt = word.slice(0, i) + String.fromCharCode(c) + word.slice(i + 1)
        if (nxt === endWord) return steps + 1
        if (wordSet.has(nxt) && !visited.has(nxt)) {
          visited.add(nxt)
          queue.push([nxt, steps + 1])
        }
      }
  }
  return 0
}
```

**思路**：哈希集合，对每个序列起点（`n-1` 不在集合中）向后扩展。

```python
def longestConsecutive(nums):
    s = set(nums); res = 0
    for n in s:
        if n - 1 not in s:
            cur = n
            while cur + 1 in s: cur += 1
            res = max(res, cur - n + 1)
    return res
```

```typescript
function longestConsecutive(nums: number[]): number {

    if (!s.has(n - 1)) {
      let cur = n
      while (s.has(cur + 1)) cur++

    }
  }
  return res
}
```

## 130. 被围绕的区域 (Surrounded Regions)

**描述**：将二维矩阵中被 X 完全围绕的 O 翻转为 X。

**思路**：从边界的 O 出发 DFS/BFS 标记不被翻转的，剩余 O 全翻转。

```python
def solve(board):
    m, n = len(board), len(board[0])
    def dfs(i, j):
        if i < 0 or i >= m or j < 0 or j >= n or board[i][j] != 'O': return
        board[i][j] = '#'
        for di, dj in [(0,1),(0,-1),(1,0),(-1,0)]: dfs(i+di, j+dj)
    for i in range(m):
        for j in range(n):
            if (i in (0, m-1) or j in (0, n-1)) and board[i][j] == 'O': dfs(i, j)
    for i in range(m):
        for j in range(n):
            if board[i][j] == 'O': board[i][j] = 'X'
            elif board[i][j] == '#': board[i][j] = 'O'
```

```typescript
function solve(board: string[][]): void {
  const m = board.length,
    n = board[0].length
  const dfs = (i: number, j: number) => {
    if (i < 0 || i >= m || j < 0 || j >= n || board[i][j] !== 'O') return
    board[i][j] = '#'
    ;[
      [0, 1],
      [0, -1],
      [1, 0],
      [-1, 0]
    ].forEach(([di, dj]) => dfs(i + di, j + dj))
  }
  for (let i = 0; i < m; i++)
    for (let j = 0; j < n; j++)
      if ((i === 0 || i === m - 1 || j === 0 || j === n - 1) && board[i][j] === 'O') dfs(i, j)
  for (let i = 0; i < m; i++)
    for (let j = 0; j < n; j++) {
      if (board[i][j] === 'O') board[i][j] = 'X'
      else if (board[i][j] === '#') board[i][j] = 'O'
    }
}
```

## 131. 分割回文串 (Palindrome Partitioning)

**描述**：将字符串分割为多个回文子串的所有方案。

**思路**：回溯，从当前位置枚举所有回文前缀。

```python
def partition(s):
    res = []
    def bt(start, path):
        if start == len(s): res.append(path[:]); return
        for end in range(start+1, len(s)+1):
            sub = s[start:end]
            if sub == sub[::-1]:
                path.append(sub); bt(end, path); path.pop()
    bt(0, []); return res
```

```typescript
function partition(s: string): string[][] {
  const res: string[][] = []
  const isPalin = (l: number, r: number) => {
    while (l < r) {
      if (s[l++] !== s[r--]) return false
    }
    return true
  }
  const bt = (start: number, path: string[]) => {
    if (start === s.length) {
      res.push([...path])
      return
    }
    for (let end = start; end < s.length; end++) {
      if (isPalin(start, end)) {
        path.push(s.slice(start, end + 1))
        bt(end + 1, path)
        path.pop()
      }
    }
  }
  bt(0, [])
  return res
}
```

## 132. 分割回文串 II (Palindrome Partitioning II)

**描述**：将字符串分割为回文子串的最少切割次数。

**思路**：DP，`dp[i]` = s[:i] 的最少切割数。中心扩展预处理回文。

```python
def minCut(s):
    n = len(s)
    dp = list(range(n))
    for center in range(n):
        for l, r in [(center, center), (center, center+1)]:
            while l >= 0 and r < n and s[l] == s[r]:
                dp[r] = min(dp[r], dp[l-1] + 1 if l > 0 else 0)
                l -= 1; r += 1
    return dp[-1]
```

```typescript
function minCut(s: string): number {
  const n = s.length,
    dp = Array.from({ length: n }, (_, i) => i)
  for (let c = 0; c < n; c++)
    for (let [l, r] of [
      [c, c],
      [c, c + 1]
    ])
      while (l >= 0 && r < n && s[l] === s[r]) {
        dp[r] = Math.min(dp[r], l > 0 ? dp[l - 1] + 1 : 0)
        l--
        r++
      }
  return dp[n - 1]
}
```

## 133. 克隆图 (Clone Graph)

**描述**：深拷贝无向图。

**思路**：哈希表 + DFS/BFS，记录已克隆的节点避免重复。

```python
def cloneGraph(node):
    if not node: return None
    cloned = {}
    def dfs(n):
        if n in cloned: return cloned[n]
        copy = Node(n.val)
        cloned[n] = copy
        copy.neighbors = [dfs(nb) for nb in n.neighbors]
        return copy
    return dfs(node)
```

```typescript
function cloneGraph(node: _Node | null): _Node | null {
  if (!node) return null
  const cloned = new Map<_Node, _Node>()
  const dfs = (n: _Node): _Node => {
    if (cloned.has(n)) return cloned.get(n)!
    const copy = new _Node(n.val)
    cloned.set(n, copy)
    copy.neighbors = n.neighbors.map(nb => dfs(nb!))
    return copy
  }
  return dfs(node)
}
```

## 134. 加油站 (Gas Station)

**描述**：环形路线加油站，判断能否走完一圈并返回起点。

**思路**：贪心，若总油量 >= 总消耗则一定有解；遍历中累积油量为负时重置起点。

```python
def canCompleteCircuit(gas, cost):
    if sum(gas) < sum(cost): return -1
    start = tank = 0
    for i in range(len(gas)):
        tank += gas[i] - cost[i]
        if tank < 0:
            start = i + 1; tank = 0
    return start
```

```typescript
function canCompleteCircuit(gas: number[], cost: number[]): number {
  if (gas.reduce((a, b) => a + b) < cost.reduce((a, b) => a + b)) return -1
  let start = 0,
    tank = 0
  for (let i = 0; i < gas.length; i++) {
    tank += gas[i] - cost[i]
    if (tank < 0) {
      start = i + 1
      tank = 0
    }
  }
  return start
}
```

## 135. 分发糖果 (Candy)

**描述**：每个孩子至少一个糖果，评分高于邻居的要比邻居多。求最少糖果数。

**思路**：两遍扫描，左→右保证右邻较高时递增，右→左保证左邻较高时递增，取 max。

```python
def candy(ratings):
    n = len(ratings)
    candies = [1] * n
    for i in range(1, n):
        if ratings[i] > ratings[i-1]: candies[i] = candies[i-1] + 1
    for i in range(n-2, -1, -1):
        if ratings[i] > ratings[i+1]: candies[i] = max(candies[i], candies[i+1] + 1)
    return sum(candies)
```

```typescript
function candy(ratings: number[]): number {
  const n = ratings.length,
    candies = new Array(n).fill(1)
  for (let i = 1; i < n; i++) if (ratings[i] > ratings[i - 1]) candies[i] = candies[i - 1] + 1
  for (let i = n - 2; i >= 0; i--)
    if (ratings[i] > ratings[i + 1]) candies[i] = Math.max(candies[i], candies[i + 1] + 1)
  return candies.reduce((a, b) => a + b)
}
```

## 139. 单词拆分 (Word Break)

**描述**：判断字符串是否能被字典中的单词拼接而成。

```python
def wordBreak(s, wordDict):
    words = set(wordDict)
    dp = [False] * (len(s) + 1)
    dp[0] = True
    for i in range(1, len(s) + 1):
        for j in range(i):
            if dp[j] and s[j:i] in words:
                dp[i] = True; break
    return dp[-1]
```

```typescript
function wordBreak(s: string, wordDict: string[]): boolean {
```

## 140. 单词拆分 II (Word Break II)

**描述**：返回所有可能的单词拆分结果。

**思路**：记忆化DFS/回溯 + 字典集合。

```python
def wordBreak(s, wordDict):
    words = set(wordDict)
    memo = {}
    def dfs(start):
        if start in memo: return memo[start]
        if start == len(s): return ['']
        res = []
        for end in range(start+1, len(s)+1):
            w = s[start:end]
            if w in words:
                for rest in dfs(end):
                    res.append(w + (' ' + rest if rest else ''))
        memo[start] = res
        return res
    return dfs(0)
```

```typescript
function wordBreak(s: string, wordDict: string[]): string[] {
  const words = new Set(wordDict),
    memo = new Map<number, string[]>()
  const dfs = (start: number): string[] => {
    if (memo.has(start)) return memo.get(start)!
    if (start === s.length) return ['']
    const res: string[] = []
    for (let end = start + 1; end <= s.length; end++) {
      const w = s.slice(start, end)
      if (words.has(w)) for (const rest of dfs(end)) res.push(rest ? w + ' ' + rest : w)
    }
    memo.set(start, res)
    return res
  }
  return dfs(0)
}
```

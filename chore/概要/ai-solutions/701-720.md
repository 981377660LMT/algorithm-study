## 701. 二叉搜索树中的插入操作 (Insert into a BST)

**描述**：BST 插入新值。

**思路**：递归找到空位插入。

```python
def insertIntoBST(root, val):
    if not root: return TreeNode(val)
    if val < root.val: root.left = insertIntoBST(root.left, val)
    else: root.right = insertIntoBST(root.right, val)
    return root
```

```typescript
function insertIntoBST(root: TreeNode | null, val: number): TreeNode {
  if (!root) return new TreeNode(val)
  if (val < root.val) root.left = insertIntoBST(root.left, val)
  else root.right = insertIntoBST(root.right, val)
  return root
}
```

## 703. 数据流中的第 K 大元素 (Kth Largest Element in a Stream)

**描述**：维护数据流中第 K 大的元素。

**思路**：小顶堆大小 k。

```python
import heapq
class KthLargest:
    def __init__(self, k, nums):
        self.k = k; self.heap = nums
        heapq.heapify(self.heap)
        while len(self.heap) > k: heapq.heappop(self.heap)
    def add(self, val):
        heapq.heappush(self.heap, val)
        if len(self.heap) > self.k: heapq.heappop(self.heap)
        return self.heap[0]
```

```typescript
class KthLargest {
  heap: number[]
  k: number
  constructor(k: number, nums: number[]) {
    this.k = k
    this.heap = []
    for (const n of nums) this.add(n)
  }
  add(val: number): number {
    this.heap.push(val)
    this.heap.sort((a, b) => a - b)
    while (this.heap.length > this.k) this.heap.shift()
    return this.heap[0]
  }
}
```

## 704. 二分查找 (Binary Search)

**描述**：有序数组二分查找。

**思路**：经典二分。

```python
def search(nums, target):
    lo, hi = 0, len(nums)-1
    while lo <= hi:
        mid = (lo+hi)//2
        if nums[mid] == target: return mid
        elif nums[mid] < target: lo = mid+1
        else: hi = mid-1
    return -1
```

```typescript
function search(nums: number[], target: number): number {
  let lo = 0,
    hi = nums.length - 1
  while (lo <= hi) {
    const mid = (lo + hi) >> 1
    if (nums[mid] === target) return mid
    if (nums[mid] < target) lo = mid + 1
    else hi = mid - 1
  }
  return -1
}
```

## 705. 设计哈希集合 (Design HashSet)

**描述**：不用内置哈希实现 HashSet。

**思路**：取模 + 拉链法。

```python
class MyHashSet:
    def __init__(self): self.size = 1001; self.buckets = [[] for _ in range(self.size)]
    def _hash(self, key): return key % self.size
    def add(self, key):
        b = self.buckets[self._hash(key)]
        if key not in b: b.append(key)
    def remove(self, key):
        b = self.buckets[self._hash(key)]
        if key in b: b.remove(key)
    def contains(self, key): return key in self.buckets[self._hash(key)]
```

```typescript
class MyHashSet {
  buckets: number[][] = Array.from({ length: 1001 }, () => [])
  add(key: number) {
    const b = this.buckets[key % 1001]
    if (!b.includes(key)) b.push(key)
  }
  remove(key: number) {
    const b = this.buckets[key % 1001]
    const i = b.indexOf(key)
    if (i >= 0) b.splice(i, 1)
  }
  contains(key: number): boolean {
    return this.buckets[key % 1001].includes(key)
  }
}
```

## 706. 设计哈希映射 (Design HashMap)

**描述**：不用内置哈希实现 HashMap。

**思路**：取模 + 拉链法。

```python
class MyHashMap:
    def __init__(self): self.size = 1001; self.buckets = [[] for _ in range(self.size)]
    def put(self, key, value):
        b = self.buckets[key % self.size]
        for i, (k, v) in enumerate(b):
            if k == key: b[i] = (key, value); return
        b.append((key, value))
    def get(self, key):
        for k, v in self.buckets[key % self.size]:
            if k == key: return v
        return -1
    def remove(self, key):
        b = self.buckets[key % self.size]
        for i, (k, v) in enumerate(b):
            if k == key: b.pop(i); return
```

```typescript

    else b.push([key, value])
```

## 707. 设计链表 (Design Linked List)

**描述**：实现单链表所有操作。

**思路**：哨兵头节点。

```python
class ListNode:
    def __init__(self, val=0): self.val = val; self.next = None

class MyLinkedList:
    def __init__(self): self.head = ListNode(); self.size = 0
    def get(self, index):
        if index < 0 or index >= self.size: return -1
        cur = self.head.next
        for _ in range(index): cur = cur.next
        return cur.val
    def addAtHead(self, val): self.addAtIndex(0, val)
    def addAtTail(self, val): self.addAtIndex(self.size, val)
    def addAtIndex(self, index, val):
        if index > self.size: return
        index = max(0, index); cur = self.head
        for _ in range(index): cur = cur.next
        node = ListNode(val); node.next = cur.next; cur.next = node; self.size += 1
    def deleteAtIndex(self, index):
        if index < 0 or index >= self.size: return
        cur = self.head
        for _ in range(index): cur = cur.next
        cur.next = cur.next.next; self.size -= 1
```

```typescript
class MyLinkedList {
  head = { val: 0, next: null as any }
  size = 0
  get(index: number): number {
    if (index < 0 || index >= this.size) return -1
    let cur = this.head.next
    for (let i = 0; i < index; i++) cur = cur.next
    return cur.val
  }
  addAtHead(val: number) {
    this.addAtIndex(0, val)
  }
  addAtTail(val: number) {
    this.addAtIndex(this.size, val)
  }
  addAtIndex(index: number, val: number) {
    if (index > this.size) return
    index = Math.max(0, index)
    let cur = this.head
    for (let i = 0; i < index; i++) cur = cur.next
    cur.next = { val, next: cur.next }
    this.size++
  }
  deleteAtIndex(index: number) {
    if (index < 0 || index >= this.size) return
    let cur = this.head
    for (let i = 0; i < index; i++) cur = cur.next
    cur.next = cur.next.next
    this.size--
  }
}
```

## 712. 两个字符串的最小 ASCII 删除和 (Minimum ASCII Delete Sum for Two Strings)

**描述**：删除使两个字符串相同的最小 ASCII 值之和。

**思路**：DP 类似编辑距离。

```python
def minimumDeleteSum(s1, s2):
    m, n = len(s1), len(s2)
    dp = [[0]*(n+1) for _ in range(m+1)]
    for i in range(1, m+1): dp[i][0] = dp[i-1][0]+ord(s1[i-1])
    for j in range(1, n+1): dp[0][j] = dp[0][j-1]+ord(s2[j-1])
    for i in range(1, m+1):
        for j in range(1, n+1):
            if s1[i-1] == s2[j-1]: dp[i][j] = dp[i-1][j-1]
            else: dp[i][j] = min(dp[i-1][j]+ord(s1[i-1]), dp[i][j-1]+ord(s2[j-1]))
    return dp[m][n]
```

```typescript
function minimumDeleteSum(s1: string, s2: string): number {
  const m = s1.length,
    n = s2.length
  const dp = Array.from({ length: m + 1 }, () => new Array(n + 1).fill(0))
  for (let i = 1; i <= m; i++) dp[i][0] = dp[i - 1][0] + s1.charCodeAt(i - 1)
  for (let j = 1; j <= n; j++) dp[0][j] = dp[0][j - 1] + s2.charCodeAt(j - 1)
  for (let i = 1; i <= m; i++)
    for (let j = 1; j <= n; j++)
      dp[i][j] =
        s1[i - 1] === s2[j - 1]
          ? dp[i - 1][j - 1]
          : Math.min(dp[i - 1][j] + s1.charCodeAt(i - 1), dp[i][j - 1] + s2.charCodeAt(j - 1))
  return dp[m][n]
}
```

## 713. 乘积小于 K 的子数组 (Subarray Product Less Than K)

**描述**：乘积严格小于 k 的连续子数组个数。

**思路**：滑动窗口。

```python
def numSubarrayProductLessThanK(nums, k):
    if k <= 1: return 0
    prod = 1; left = 0; res = 0
    for right, n in enumerate(nums):
        prod *= n
        while prod >= k: prod //= nums[left]; left += 1
        res += right - left + 1
    return res
```

```typescript
function numSubarrayProductLessThanK(nums: number[], k: number): number {
  if (k <= 1) return 0
  let prod = 1,
    left = 0,
    res = 0
  for (let right = 0; right < nums.length; right++) {
    prod *= nums[right]
    while (prod >= k) prod /= nums[left++]
    res += right - left + 1
  }
  return res
}
```

## 714. 买卖股票的最佳时机含手续费 (Best Time to Buy and Sell Stock with Transaction Fee)

**描述**：含手续费的股票交易最大利润。

**思路**：DP，cash 和 hold 两个状态。

```python
def maxProfit(prices, fee):
    cash, hold = 0, -prices[0]
    for p in prices[1:]:
        cash = max(cash, hold+p-fee)
        hold = max(hold, cash-p)
    return cash
```

```typescript
function maxProfit(prices: number[], fee: number): number {
  let cash = 0,
    hold = -prices[0]
  for (let i = 1; i < prices.length; i++) {
    cash = Math.max(cash, hold + prices[i] - fee)
    hold = Math.max(hold, cash - prices[i])
  }
  return cash
}
```

## 717. 1比特与2比特字符 (1-bit and 2-bit Characters)

**描述**：最后一个字符是否为 1 位字符(0)。

**思路**：从左到右跳，1开头跳2步，0开头跳1步。

```python
def isOneBitCharacter(bits):
    i = 0
    while i < len(bits) - 1:
        i += 2 if bits[i] else 1
    return i == len(bits) - 1
```

```typescript
function isOneBitCharacter(bits: number[]): boolean {
  let i = 0
  while (i < bits.length - 1) i += bits[i] ? 2 : 1
  return i === bits.length - 1
}
```

## 718. 最长重复子数组 (Maximum Length of Repeated Subarray)

**描述**：两数组最长公共子数组。

**思路**：DP。`dp[i][j]` = 以 A[i-1] 和 B[j-1] 结尾的最长公共子数组。

```python
def findLength(nums1, nums2):
    m, n = len(nums1), len(nums2)
    dp = [0]*(n+1); res = 0
    for i in range(1, m+1):
        for j in range(n, 0, -1):
            dp[j] = dp[j-1]+1 if nums1[i-1] == nums2[j-1] else 0
            res = max(res, dp[j])
    return res
```

```typescript
function findLength(nums1: number[], nums2: number[]): number {
  const m = nums1.length,
    n = nums2.length
  const dp = new Array(n + 1).fill(0)
  let res = 0
  for (let i = 1; i <= m; i++)
    for (let j = n; j >= 1; j--) {
      dp[j] = nums1[i - 1] === nums2[j - 1] ? dp[j - 1] + 1 : 0
      res = Math.max(res, dp[j])
    }
  return res
}
```

## 719. 找出第 K 小的数对距离 (Find K-th Smallest Pair Distance)

**描述**：所有数对距离中第 k 小的。

**思路**：排序 + 二分答案 + 双指针计数。

```python
def smallestDistancePair(nums, k):
    nums.sort()
    lo, hi = 0, nums[-1]-nums[0]
    while lo < hi:
        mid = (lo+hi)//2
        cnt = j = 0
        for i in range(len(nums)):
            while j < len(nums) and nums[j]-nums[i] <= mid: j += 1
            cnt += j - i - 1
        if cnt >= k: hi = mid
        else: lo = mid+1
    return lo
```

```typescript
function smallestDistancePair(nums: number[], k: number): number {
  nums.sort((a, b) => a - b)
  let lo = 0,
    hi = nums[nums.length - 1] - nums[0]
  while (lo < hi) {
    const mid = (lo + hi) >> 1
    let cnt = 0,
      j = 0
    for (let i = 0; i < nums.length; i++) {
      while (j < nums.length && nums[j] - nums[i] <= mid) j++
      cnt += j - i - 1
    }
    if (cnt >= k) hi = mid
    else lo = mid + 1
  }
  return lo
}
```

## 720. 词典中最长的单词 (Longest Word in Dictionary)

**描述**：能由字典中其他词逐字构建的最长词。

**思路**：排序后逐个检查前缀在 set 中。

```python
def longestWord(words):
    words.sort(); built = set(['']); res = ''
    for w in words:
        if w[:-1] in built:
            built.add(w)
            if len(w) > len(res): res = w
    return res
```

```typescript
function longestWord(words: string[]): string {
  words.sort()
  const built = new Set([''])
  let res = ''
  for (const w of words) {
    if (built.has(w.slice(0, -1))) {
      built.add(w)
      if (w.length > res.length) res = w
    }
  }
  return res
}
```

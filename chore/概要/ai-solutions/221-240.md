## 221. 最大正方形 (Maximal Square)

**描述**：在只含 '0' 和 '1' 的二维矩阵中，找全为 '1' 的最大正方形面积。

**思路**：DP，`dp[i][j] = min(dp[i-1][j], dp[i][j-1], dp[i-1][j-1]) + 1`（以(i,j)为右下角的最大正方形边长）。

```python
def maximalSquare(matrix):
    m, n = len(matrix), len(matrix[0])
    dp = [0] * (n + 1)
    res = 0
    for i in range(m):
        prev = 0
        for j in range(1, n + 1):
            tmp = dp[j]
            if matrix[i][j-1] == '1':
                dp[j] = min(dp[j], dp[j-1], prev) + 1
                res = max(res, dp[j])
            else: dp[j] = 0
            prev = tmp
    return res * res
```

```typescript
function maximalSquare(matrix: string[][]): number {
  const m = matrix.length,
    n = matrix[0].length
  const dp = new Array(n + 1).fill(0)
  let res = 0
  for (let i = 0; i < m; i++) {
    let prev = 0
    for (let j = 1; j <= n; j++) {
      const tmp = dp[j]
      if (matrix[i][j - 1] === '1') {
        dp[j] = Math.min(dp[j], dp[j - 1], prev) + 1
        res = Math.max(res, dp[j])
      } else dp[j] = 0
      prev = tmp
    }
  }
  return res * res
}
```

## 222. 完全二叉树的节点个数 (Count Complete Tree Nodes)

**描述**：O(log²n) 统计完全二叉树节点数。

**思路**：比较左右子树高度，相等则左子树满（节点 = 2^h-1 + 右子树递归），不等则右子树满。

```python
def countNodes(root):
    if not root: return 0
    lh = rh = 0
    l, r = root, root
    while l: lh += 1; l = l.left
    while r: rh += 1; r = r.right
    if lh == rh: return (1 << lh) - 1
    return 1 + countNodes(root.left) + countNodes(root.right)
```

```typescript
function countNodes(root: TreeNode | null): number {
  if (!root) return 0
  let lh = 0,
    rh = 0,
    l = root,
    r: TreeNode | null = root
  while (l) {
    lh++
    l = l.left!
  }
  while (r) {
    rh++
    r = r.right
  }
  if (lh === rh) return (1 << lh) - 1
  return 1 + countNodes(root.left) + countNodes(root.right)
}
```

## 223. 矩形面积 (Rectangle Area)

**描述**：计算两个矩形覆盖的总面积。

**思路**：两矩形面积之和 - 交集面积。

```python
def computeArea(ax1, ay1, ax2, ay2, bx1, by1, bx2, by2):
    area = (ax2-ax1)*(ay2-ay1) + (bx2-bx1)*(by2-by1)
    overlap_w = max(0, min(ax2, bx2) - max(ax1, bx1))
    overlap_h = max(0, min(ay2, by2) - max(ay1, by1))
    return area - overlap_w * overlap_h
```

```typescript
function computeArea(
  ax1: number,
  ay1: number,
  ax2: number,
  ay2: number,
  bx1: number,
  by1: number,
  bx2: number,
  by2: number
): number {
  const area = (ax2 - ax1) * (ay2 - ay1) + (bx2 - bx1) * (by2 - by1)
  const ow = Math.max(0, Math.min(ax2, bx2) - Math.max(ax1, bx1))
  const oh = Math.max(0, Math.min(ay2, by2) - Math.max(ay1, by1))
  return area - ow * oh
}
```

## 224. 基本计算器 (Basic Calculator)

**描述**：实现含 +、-、括号 的计算器。

**思路**：栈记录符号和累积结果，遇 `(` 压栈，遇 `)` 弹栈。

```python
def calculate(s):
    stack = []; res = 0; num = 0; sign = 1
    for c in s:
        if c.isdigit():
            num = num * 10 + int(c)
        elif c in '+-':
            res += sign * num; num = 0
            sign = 1 if c == '+' else -1
        elif c == '(':
            stack.append(res); stack.append(sign)
            res = 0; sign = 1
        elif c == ')':
            res += sign * num; num = 0
            res *= stack.pop()  # sign
            res += stack.pop()  # prev res
    return res + sign * num
```

```typescript
function calculate(s: string): number {
  const stack: number[] = []
  let res = 0,
    num = 0,
    sign = 1
  for (const c of s) {
    if (c >= '0' && c <= '9') num = num * 10 + +c
    else if (c === '+' || c === '-') {
      res += sign * num
      num = 0
      sign = c === '+' ? 1 : -1
    } else if (c === '(') {
      stack.push(res)
      stack.push(sign)
      res = 0
      sign = 1
    } else if (c === ')') {
      res += sign * num
      num = 0
      res *= stack.pop()!
      res += stack.pop()!
    }
  }
  return res + sign * num
}
```

## 225. 用队列实现栈 (Implement Stack using Queues)

**描述**：用队列实现栈。

**思路**：每次 push 后，将队前元素依次移到队尾，保持顺序。

```python
from collections import deque
class MyStack:
    def __init__(self): self.q = deque()
    def push(self, x):
        self.q.append(x)
        for _ in range(len(self.q) - 1): self.q.append(self.q.popleft())
    def pop(self): return self.q.popleft()
    def top(self): return self.q[0]
    def empty(self): return not self.q
```

```typescript
class MyStack {
  private q: number[] = []
  push(x: number): void {
    this.q.push(x)
    for (let i = 0; i < this.q.length - 1; i++) this.q.push(this.q.shift()!)
  }
  pop(): number {
    return this.q.shift()!
  }
  top(): number {
    return this.q[0]
  }
  empty(): boolean {
    return this.q.length === 0
  }
}
```

## 227. 基本计算器 II (Basic Calculator II)

**描述**：实现含 +、-、\*、/ 的计算器（无括号）。

**思路**：栈，遇到 \* / 立即计算与栈顶合并，最后栈求和。

```python
def calculate(s):
    stack = []; num = 0; op = '+'
    for i, c in enumerate(s):
        if c.isdigit(): num = num * 10 + int(c)
        if c in '+-*/' or i == len(s) - 1:
            if op == '+': stack.append(num)
            elif op == '-': stack.append(-num)
            elif op == '*': stack.append(stack.pop() * num)
            elif op == '/': stack.append(int(stack.pop() / num))
            op = c; num = 0
    return sum(stack)
```

```typescript
function calculate(s: string): number {
  const stack: number[] = []
  let num = 0,
    op = '+'
  for (let i = 0; i < s.length; i++) {
    const c = s[i]
    if (c >= '0' && c <= '9') num = num * 10 + +c
    if ('+-*/'.includes(c) || i === s.length - 1) {
      if (op === '+') stack.push(num)
      else if (op === '-') stack.push(-num)
      else if (op === '*') stack.push(stack.pop()! * num)
      else if (op === '/') stack.push(Math.trunc(stack.pop()! / num))
      op = c
      num = 0
    }
  }
  return stack.reduce((a, b) => a + b, 0)
}
```

## 228. 汇总区间 (Summary Ranges)

**描述**：将有序数组汇总为区间表示。

**思路**：双指针找连续段。

```python
def summaryRanges(nums):
    res = []; i = 0
    while i < len(nums):
        start = nums[i]
        while i + 1 < len(nums) and nums[i+1] == nums[i] + 1: i += 1
        if start == nums[i]: res.append(str(start))
        else: res.append(f"{start}->{nums[i]}")
        i += 1
    return res
```

```typescript
function summaryRanges(nums: number[]): string[] {
```

## 229. 多数元素 II (Majority Element II)

**描述**：找数组中出现次数超过 n/3 的所有元素。

**思路**：摩尔投票法扩展，最多两个候选。

```python
def majorityElement(nums):
    c1 = c2 = cnt1 = cnt2 = 0
    for n in nums:
        if n == c1: cnt1 += 1
        elif n == c2: cnt2 += 1
        elif cnt1 == 0: c1 = n; cnt1 = 1
        elif cnt2 == 0: c2 = n; cnt2 = 1
        else: cnt1 -= 1; cnt2 -= 1
    return [c for c in (c1, c2) if nums.count(c) > len(nums) // 3]
```

```typescript
function majorityElement(nums: number[]): number[] {
  let c1 = 0,
    c2 = 0,
    cnt1 = 0,
    cnt2 = 0
  for (const n of nums) {
    if (n === c1) cnt1++
    else if (n === c2) cnt2++
    else if (cnt1 === 0) {
      c1 = n
      cnt1 = 1
    } else if (cnt2 === 0) {
      c2 = n
      cnt2 = 1
    } else {
      cnt1--
      cnt2--
    }
  }
  const threshold = Math.floor(nums.length / 3)
  return [c1, c2].filter(c => nums.filter(x => x === c).length > threshold)
}
```

## 230. 二叉搜索树中第 K 小的元素 (Kth Smallest Element in a BST)

**描述**：返回 BST 中第 k 小的元素。

**思路**：中序遍历到第 k 个。

```python
def kthSmallest(root, k):
    stack = []
    while True:
        while root:
            stack.append(root); root = root.left
        root = stack.pop()
        k -= 1
        if k == 0: return root.val
        root = root.right
```

```typescript
function kthSmallest(root: TreeNode | null, k: number): number {
  const stack: TreeNode[] = []
  let cur = root
  while (true) {
    while (cur) {
      stack.push(cur)
      cur = cur.left
    }
    cur = stack.pop()!
    k--
    if (k === 0) return cur.val
    cur = cur.right
  }
}
```

## 232. 用栈实现队列 (Implement Queue using Stacks)

**描述**：用两个栈实现队列。

**思路**：in 栈接收 push，out 栈为空时把 in 全倒入 out，再从 out pop。

```python
class MyQueue:
    def __init__(self):
        self.sin = []; self.sout = []
    def push(self, x): self.sin.append(x)
    def pop(self):
        self.peek(); return self.sout.pop()
    def peek(self):
        if not self.sout:
            while self.sin: self.sout.append(self.sin.pop())
        return self.sout[-1]
    def empty(self): return not self.sin and not self.sout
```

```typescript
class MyQueue {
  private sin: number[] = []
  private sout: number[] = []
  push(x: number): void {
    this.sin.push(x)
  }
  pop(): number {
    this.peek()
    return this.sout.pop()!
  }
  peek(): number {
    if (!this.sout.length) while (this.sin.length) this.sout.push(this.sin.pop()!)
    return this.sout[this.sout.length - 1]
  }
  empty(): boolean {
    return !this.sin.length && !this.sout.length
  }
}
```

## 233. 数字 1 的个数 (Number of Digit Ones)

**描述**：1 ~ n 中数字 1 出现的总次数。

**思路**：逐位分析，每一位上 1 出现的次数由高位、当前位、低位共同决定。

```python
def countDigitOne(n):
    cnt = 0; factor = 1
    while factor <= n:
        low = n % factor
        cur = (n // factor) % 10
        high = n // (factor * 10)
        if cur == 0: cnt += high * factor
        elif cur == 1: cnt += high * factor + low + 1
        else: cnt += (high + 1) * factor
        factor *= 10
    return cnt
```

```typescript
function countDigitOne(n: number): number {
  let cnt = 0,
    factor = 1
  while (factor <= n) {
    const low = n % factor
    const cur = Math.floor(n / factor) % 10
    const high = Math.floor(n / (factor * 10))
    if (cur === 0) cnt += high * factor
    else if (cur === 1) cnt += high * factor + low + 1
    else cnt += (high + 1) * factor
    factor *= 10
  }
  return cnt
}
```

## 234. 回文链表 (Palindrome Linked List)

**描述**：O(n) 时间 O(1) 空间判断链表是否回文。

**思路**：快慢指针找中点→反转后半段→逐一比较。

```python
def isPalindrome(head):
    slow = fast = head
    while fast and fast.next:
        slow = slow.next; fast = fast.next.next
    prev = None
    while slow:
        slow.next, prev, slow = prev, slow, slow.next
    while prev:
        if head.val != prev.val: return False
        head = head.next; prev = prev.next
    return True
```

```typescript
function isPalindrome(head: ListNode | null): boolean {
  let slow = head,
    fast = head
  while (fast && fast.next) {
    slow = slow!.next
    fast = fast.next.next
  }
  let prev: ListNode | null = null
  while (slow) {
    const nxt = slow.next
    slow.next = prev
    prev = slow
    slow = nxt
  }
  while (prev) {
    if (head!.val !== prev.val) return false
    head = head!.next
    prev = prev.next
  }
  return true
}
```

## 235. 二叉搜索树的最近公共祖先 (Lowest Common Ancestor of a BST)

**思路**：利用BST性质，从根开始，都小往左，都大往右，分叉即为LCA。

```python
def lowestCommonAncestor(root, p, q):
    while root:
        if p.val < root.val and q.val < root.val: root = root.left
        elif p.val > root.val and q.val > root.val: root = root.right
        else: return root
```

```typescript

    else return root
  }
  return root
}
```

## 236. 二叉树的最近公共祖先 (Lowest Common Ancestor of a Binary Tree)

**描述**：普通二叉树找两节点的 LCA。

**思路**：递归，左右子树分别找 p/q，都找到则当前为 LCA，否则返回非空的那个。

```python
def lowestCommonAncestor(root, p, q):
    if not root or root == p or root == q: return root
    left = lowestCommonAncestor(root.left, p, q)
    right = lowestCommonAncestor(root.right, p, q)
    if left and right: return root
    return left or right
```

```typescript
function lowestCommonAncestor(root: TreeNode | null, p: TreeNode, q: TreeNode): TreeNode | null {
  if (!root || root === p || root === q) return root
  const left = lowestCommonAncestor(root.left, p, q)
  const right = lowestCommonAncestor(root.right, p, q)
  if (left && right) return root
  return left ?? right
}
```

## 238. 除自身以外数组的乘积 (Product of Array Except Self)

**描述**：返回数组中每个位置除自身外所有元素的乘积（不能用除法）。

**思路**：左右扫两遍，前缀积 × 后缀积。

```python
def productExceptSelf(nums):
    n = len(nums); res = [1] * n
    left = 1
    for i in range(n):
        res[i] = left; left *= nums[i]
    right = 1
    for i in range(n-1, -1, -1):
        res[i] *= right; right *= nums[i]
    return res
```

```typescript
function productExceptSelf(nums: number[]): number[] {
  const n = nums.length,
    res = new Array(n).fill(1)
  let left = 1
  for (let i = 0; i < n; i++) {
    res[i] = left
    left *= nums[i]
  }
  let right = 1
  for (let i = n - 1; i >= 0; i--) {
    res[i] *= right
    right *= nums[i]
  }
  return res
}
```

## 239. 滑动窗口最大值 (Sliding Window Maximum)

**描述**：大小为 k 的滑动窗口在数组上移动，返回每个位置的最大值。

**思路**：单调递减双端队列，队头始终是当前窗口最大值的下标。

```python
from collections import deque
def maxSlidingWindow(nums, k):
    dq = deque(); res = []
    for i, n in enumerate(nums):
        while dq and nums[dq[-1]] <= n: dq.pop()
        dq.append(i)
        if dq[0] <= i - k: dq.popleft()
        if i >= k - 1: res.append(nums[dq[0]])
    return res
```

```typescript
function maxSlidingWindow(nums: number[], k: number): number[] {
  const dq: number[] = [],
    res: number[] = []
  for (let i = 0; i < nums.length; i++) {
    while (dq.length && nums[dq[dq.length - 1]] <= nums[i]) dq.pop()
    dq.push(i)
    if (dq[0] <= i - k) dq.shift()
    if (i >= k - 1) res.push(nums[dq[0]])
  }
  return res
}
```

## 240. 搜索二维矩阵 II (Search a 2D Matrix II)

**描述**：每行每列递增的矩阵中搜索目标值。

**思路**：从右上角开始，大于目标往左，小于目标往下。

```python
def searchMatrix(matrix, target):
    m, n = len(matrix), len(matrix[0])
    i, j = 0, n - 1
    while i < m and j >= 0:
        if matrix[i][j] == target: return True
        elif matrix[i][j] > target: j -= 1
        else: i += 1
    return False
```


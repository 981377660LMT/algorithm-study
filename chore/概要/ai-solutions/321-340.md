## 321. 拼接最大数 (Create Maximum Number)

**描述**：从两个数组中分别取 i 和 k-i 个数(保持相对顺序)，拼成最大的长度 k 的数。

**思路**：枚举从 nums1 取 i 个从 nums2 取 k-i 个，分别求最大子序列，合并取最大。

```python
def maxNumber(nums1, nums2, k):
    def maxSub(nums, k):
        stack = []
        drop = len(nums) - k
        for n in nums:
            while drop and stack and stack[-1] < n: stack.pop(); drop -= 1
            stack.append(n)
        return stack[:k]
    def merge(a, b):
        res = []
        while a or b:
            if a >= b: res.append(a[0]); a = a[1:]
            else: res.append(b[0]); b = b[1:]
        return res
    res = []
    for i in range(k+1):
        if i <= len(nums1) and k-i <= len(nums2):
            candidate = merge(maxSub(nums1, i), maxSub(nums2, k-i))
            res = max(res, candidate)
    return res
```

```typescript
function maxNumber(nums1: number[], nums2: number[], k: number): number[] {
  function maxSub(nums: number[], k: number): number[] {
    const stack: number[] = []
    let drop = nums.length - k
    for (const n of nums) {
      while (drop && stack.length && stack[stack.length - 1] < n) {
        stack.pop()
        drop--
      }
      stack.push(n)
    }
    return stack.slice(0, k)
  }
  function merge(a: number[], b: number[]): number[] {
    const res: number[] = []
    let i = 0,
      j = 0
    while (i < a.length || j < b.length) {
      if (compare(a, i, b, j) >= 0) res.push(a[i++])
      else res.push(b[j++])
    }
    return res
  }
  function compare(a: number[], i: number, b: number[], j: number): number {
    while (i < a.length && j < b.length && a[i] === b[j]) {
      i++
      j++
    }
    if (i === a.length) return -1
    if (j === b.length) return 1
    return a[i] - b[j]
  }
  let res: number[] = []
  for (let i = 0; i <= k; i++) {
    if (i <= nums1.length && k - i <= nums2.length) {
      const candidate = merge(maxSub(nums1, i), maxSub(nums2, k - i))
      if (compare(candidate, 0, res, 0) > 0) res = candidate
    }
  }
  return res
}
```

## 322. 零钱兑换 (Coin Change)

**描述**：给定面额数组和金额，求凑成金额的最少硬币数。

**思路**：完全背包 DP，`dp[i] = min(dp[i - coin]) + 1`。

```python
def coinChange(coins, amount):
    dp = [float('inf')] * (amount + 1)
    dp[0] = 0
    for i in range(1, amount + 1):
        for c in coins:
            if c <= i: dp[i] = min(dp[i], dp[i - c] + 1)
    return dp[amount] if dp[amount] != float('inf') else -1
```

```typescript
function coinChange(coins: number[], amount: number): number {
  const dp = new Array(amount + 1).fill(Infinity)
  dp[0] = 0
  for (let i = 1; i <= amount; i++)
    for (const c of coins) if (c <= i) dp[i] = Math.min(dp[i], dp[i - c] + 1)
  return dp[amount] === Infinity ? -1 : dp[amount]
}
```

## 324. 摆动排序 II (Wiggle Sort II)

**描述**：将数组重新排列为 `nums[0] < nums[1] > nums[2] < nums[3] > ...`。

**思路**：找中位数，三向切分 + 虚拟索引映射。简化：排序后穿插放置。

```python
def wiggleSort(nums):
    nums.sort()
    half = (len(nums) + 1) // 2
    # 小半逆序放偶数位，大半逆序放奇数位
    small = nums[:half][::-1]
    large = nums[half:][::-1]
    for i in range(len(nums)):
        if i % 2 == 0: nums[i] = small[i // 2]
        else: nums[i] = large[i // 2]
```

```typescript
function wiggleSort(nums: number[]): void {
  nums.sort((a, b) => a - b)
  const half = Math.ceil(nums.length / 2)
  const small = nums.slice(0, half).reverse()
  const large = nums.slice(half).reverse()
  for (let i = 0; i < nums.length; i++) nums[i] = i % 2 === 0 ? small[i >> 1] : large[i >> 1]
}
```

## 327. 区间和的个数 (Count of Range Sum)

**描述**：统计前缀和差值在 [lower, upper] 范围内的对数。

**思路**：归并排序统计，在合并时对有序的两半用双指针统计满足条件的对。

```python
def countRangeSum(nums, lower, upper):
    pre = [0]
    for n in nums: pre.append(pre[-1] + n)
    def merge_count(arr):
        if len(arr) <= 1: return 0
        mid = len(arr) // 2
        left, right = arr[:mid], arr[mid:]
        cnt = merge_count(left) + merge_count(right)
        j = k = 0
        for l in left:
            while j < len(right) and right[j] - l < lower: j += 1
            while k < len(right) and right[k] - l <= upper: k += 1
            cnt += k - j
        arr[:] = sorted(arr)
        return cnt
    return merge_count(pre)
```

```typescript
function countRangeSum(nums: number[], lower: number, upper: number): number {
  const pre = [0]
  for (const n of nums) pre.push(pre[pre.length - 1] + n)
  function mergeCount(arr: number[]): number {
    if (arr.length <= 1) return 0
    const mid = arr.length >> 1
    const left = arr.slice(0, mid),
      right = arr.slice(mid)
    let cnt = mergeCount(left) + mergeCount(right)
    let j = 0,
      k = 0
    for (const l of left) {
      while (j < right.length && right[j] - l < lower) j++
      while (k < right.length && right[k] - l <= upper) k++
      cnt += k - j
    }
    const sorted = [...left, ...right].sort((a, b) => a - b)
    for (let i = 0; i < sorted.length; i++) arr[i] = sorted[i]
    return cnt
  }
  return mergeCount(pre)
}
```

## 328. 奇偶链表 (Odd Even Linked List)

**描述**：将链表奇数位和偶数位节点分组。

**思路**：双指针分别串奇、偶，最后奇尾接偶头。

```python
def oddEvenList(head):
    if not head: return head
    odd, even = head, head.next
    even_head = even
    while even and even.next:
        odd.next = even.next; odd = odd.next
        even.next = odd.next; even = even.next
    odd.next = even_head
    return head
```

```typescript
function oddEvenList(head: ListNode | null): ListNode | null {
  if (!head) return head
  let odd = head,
    even = head.next
  const evenHead = even
  while (even && even.next) {
    odd.next = even.next
    odd = odd.next
    even.next = odd.next
    even = even.next
  }
  odd.next = evenHead
  return head
}
```

## 329. 矩阵中的最长递增路径 (Longest Increasing Path in a Matrix)

**描述**：矩阵中寻找最长严格递增路径长度。

**思路**：DFS + 记忆化，每个格子缓存其最长路径。

```python
def longestIncreasingPath(matrix):
    m, n = len(matrix), len(matrix[0])
    memo = {}
    def dfs(i, j):
        if (i, j) in memo: return memo[(i, j)]
        res = 1
        for di, dj in ((-1,0),(1,0),(0,-1),(0,1)):
            ni, nj = i+di, j+dj
            if 0 <= ni < m and 0 <= nj < n and matrix[ni][nj] > matrix[i][j]:
                res = max(res, 1 + dfs(ni, nj))
        memo[(i, j)] = res
        return res
    return max(dfs(i, j) for i in range(m) for j in range(n))
```

```typescript
function longestIncreasingPath(matrix: number[][]): number {
  const m = matrix.length,
    n = matrix[0].length
  const memo = Array.from({ length: m }, () => new Array(n).fill(0))
  const dirs = [
    [-1, 0],
    [1, 0],
    [0, -1],
    [0, 1]
  ]
  function dfs(i: number, j: number): number {
    if (memo[i][j]) return memo[i][j]
    let res = 1
    for (const [di, dj] of dirs) {
      const ni = i + di,
        nj = j + dj
      if (ni >= 0 && ni < m && nj >= 0 && nj < n && matrix[ni][nj] > matrix[i][j])
        res = Math.max(res, 1 + dfs(ni, nj))
    }
    return (memo[i][j] = res)
  }
  let ans = 0
  for (let i = 0; i < m; i++) for (let j = 0; j < n; j++) ans = Math.max(ans, dfs(i, j))
  return ans
}
```

## 330. 按要求补齐数组 (Patching Array)

**描述**：给定排序数组和 n，求最少添加多少个数使得 [1,n] 都能被表示。

**思路**：贪心，维护当前能覆盖 [1, miss-1]，若 nums[i] <= miss 则扩展覆盖到 miss+nums[i]-1，否则补一个 miss。

```python
def minPatches(nums, n):
    miss = 1; i = 0; patches = 0
    while miss <= n:
        if i < len(nums) and nums[i] <= miss:
            miss += nums[i]; i += 1
        else:
            miss += miss; patches += 1
    return patches
```

```typescript
function minPatches(nums: number[], n: number): number {
  let miss = 1,
    i = 0,
    patches = 0
  while (miss <= n) {
    if (i < nums.length && nums[i] <= miss) {
      miss += nums[i]
      i++
    } else {
      miss += miss
      patches++
    }
  }
  return patches
}
```

## 331. 验证二叉树的前序序列化 (Verify Preorder Serialization of a Binary Tree)

**描述**：判断字符串是否是合法的前序序列化。

**思路**：出入度：根提供 2 个出度，每个非空节点消耗 1 入度 + 提供 2 出度，# 只消耗 1 入度。跟踪可用槽位。

```python
def isValidSerialization(preorder):
    slots = 1
    for node in preorder.split(','):
        slots -= 1
        if slots < 0: return False
        if node != '#': slots += 2
    return slots == 0
```

```typescript
function isValidSerialization(preorder: string): boolean {
  let slots = 1
  for (const node of preorder.split(',')) {
    slots--
    if (slots < 0) return false
    if (node !== '#') slots += 2
  }
  return slots === 0
}
```

## 332. 重新安排行程 (Reconstruct Itinerary)

**描述**：给定机票，找从 JFK 出发字典序最小的行程（欧拉路径）。

**思路**：Hierholzer 算法，DFS + 后序插入。

```python
def findItinerary(tickets):
    from collections import defaultdict
    graph = defaultdict(list)
    for a, b in sorted(tickets, reverse=True):
        graph[a].append(b)
    route = []
    def dfs(node):
        while graph[node]:
            dfs(graph[node].pop())
        route.append(node)
    dfs('JFK')
    return route[::-1]
```

```typescript
function findItinerary(tickets: string[][]): string[] {
  const graph = new Map<string, string[]>()
  tickets.sort().reverse()
  for (const [a, b] of tickets) {
    if (!graph.has(a)) graph.set(a, [])
    graph.get(a)!.push(b)
  }
  const route: string[] = []
  function dfs(node: string) {
    const dests = graph.get(node)
    while (dests && dests.length) dfs(dests.pop()!)
    route.push(node)
  }
  dfs('JFK')
  return route.reverse()
}
```

## 334. 递增的三元子序列 (Increasing Triplet Subsequence)

**描述**：判断数组中是否存在长度为 3 的递增子序列。

**思路**：维护两个变量 first、second（最小和次小），遍历中找到比 second 大的即返回 true。

```python
def increasingTriplet(nums):
    first = second = float('inf')
    for n in nums:
        if n <= first: first = n
        elif n <= second: second = n
        else: return True
    return False
```

```typescript
function increasingTriplet(nums: number[]): boolean {

    second = Infinity
  for (const n of nums) {
    if (n <= first) first = n
    else if (n <= second) second = n
    else return true
  }
  return false
}
```

## 335. 路径交叉 (Self Crossing)

**描述**：在螺旋路径中判断是否自相交。

**思路**：归纳分析，只有三种交叉模式需检查。

```python
def isSelfCrossing(distance):
    d = distance
    for i in range(3, len(d)):
        if d[i] >= d[i-2] and d[i-1] <= d[i-3]: return True
        if i >= 4 and d[i-1] == d[i-3] and d[i] + d[i-4] >= d[i-2]: return True
        if i >= 5 and d[i-2] >= d[i-4] and d[i-1] <= d[i-3] and d[i-1] >= d[i-3] - d[i-5] and d[i] >= d[i-2] - d[i-4]: return True
    return False
```

```typescript
function isSelfCrossing(distance: number[]): boolean {
  const d = distance
  for (let i = 3; i < d.length; i++) {
    if (d[i] >= d[i - 2] && d[i - 1] <= d[i - 3]) return true
    if (i >= 4 && d[i - 1] === d[i - 3] && d[i] + d[i - 4] >= d[i - 2]) return true
    if (
      i >= 5 &&
      d[i - 2] >= d[i - 4] &&
      d[i - 1] <= d[i - 3] &&
      d[i - 1] >= d[i - 3] - d[i - 5] &&
      d[i] >= d[i - 2] - d[i - 4]
    )
      return true
  }
  return false
}
```

## 336. 回文对 (Palindrome Pairs)

**描述**：给定单词数组，找所有 (i,j) 使得 words[i]+words[j] 是回文。

**思路**：哈希表存反转串,对每个词的每个前缀/后缀判断回文情况。

```python
def palindromePairs(words):
    # 用字典 {reversed word: index}
    rev_map = {w[::-1]: i for i, w in enumerate(words)}
    res = []
    for i, w in enumerate(words):
        for j in range(len(w) + 1):
            prefix, suffix = w[:j], w[j:]
            # prefix 是回文，reversed(suffix) + w 组成回文
            if prefix == prefix[::-1] and suffix in rev_map and rev_map[suffix] != i:
                res.append([rev_map[suffix], i])
            # suffix 是回文，w + reversed(prefix) 组成回文
            if j < len(w) and suffix == suffix[::-1] and prefix in rev_map and rev_map[prefix] != i:
                res.append([i, rev_map[prefix]])
    return res
```

```typescript
function palindromePairs(words: string[]): number[][] {
  const revMap = new Map<string, number>()
  for (let i = 0; i < words.length; i++) revMap.set([...words[i]].reverse().join(''), i)
  const res: number[][] = []
  const isPalin = (s: string) => {
    let l = 0,
      r = s.length - 1
    while (l < r) {
      if (s[l++] !== s[r--]) return false
    }
    return true
  }
  for (let i = 0; i < words.length; i++) {
    const w = words[i]
    for (let j = 0; j <= w.length; j++) {
      const prefix = w.slice(0, j),
        suffix = w.slice(j)
      if (isPalin(prefix) && revMap.has(suffix) && revMap.get(suffix) !== i)
        res.push([revMap.get(suffix)!, i])
      if (j < w.length && isPalin(suffix) && revMap.has(prefix) && revMap.get(prefix) !== i)
        res.push([i, revMap.get(prefix)!])
    }
  }
  return res
}
```

## 337. 打家劫舍 III (House Robber III)

**描述**：二叉树形结构，相邻节点不能同时偷，求最大金额。

**思路**：树形 DP，每个节点返回 (偷, 不偷) 两种情况。

```python
def rob(root):
    def dfs(node):
        if not node: return (0, 0)
        l = dfs(node.left); r = dfs(node.right)
        rob_it = node.val + l[1] + r[1]
        skip = max(l) + max(r)
        return (rob_it, skip)
    return max(dfs(root))
```

```typescript
function rob(root: TreeNode | null): number {
  function dfs(node: TreeNode | null): [number, number] {
    if (!node) return [0, 0]
    const l = dfs(node.left),
      r = dfs(node.right)
    return [node.val + l[1] + r[1], Math.max(...l) + Math.max(...r)]
  }
  return Math.max(...dfs(root))
}
```

## 338. 比特位计数 (Counting Bits)

**描述**：返回 0~n 每个数的二进制中 1 的个数。

**思路**：DP，`dp[i] = dp[i >> 1] + (i & 1)`。

```python
def countBits(n):
    dp = [0] * (n + 1)
    for i in range(1, n + 1):
        dp[i] = dp[i >> 1] + (i & 1)
    return dp
```

```typescript
function countBits(n: number): number[] {
  const dp = new Array(n + 1).fill(0)
  for (let i = 1; i <= n; i++) dp[i] = dp[i >> 1] + (i & 1)
  return dp
}
```

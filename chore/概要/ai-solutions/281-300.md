## 283. 移动零 (Move Zeroes)

**描述**：将数组中所有 0 移到末尾，保持非零元素相对顺序。

**思路**：双指针，非零元素往前填，剩余补零。

```python
def moveZeroes(nums):
    j = 0
    for i in range(len(nums)):
        if nums[i] != 0:
            nums[j], nums[i] = nums[i], nums[j]
            j += 1
```

```typescript
function moveZeroes(nums: number[]): void {
  let j = 0
  for (let i = 0; i < nums.length; i++) {
    if (nums[i] !== 0) {
      ;[nums[j], nums[i]] = [nums[i], nums[j]]
      j++
    }
  }
}
```

## 284. 窥视迭代器 (Peeking Iterator)

**描述**：实现支持 peek 的迭代器。

**思路**：缓存下一个值。

```python
class PeekingIterator:
    def __init__(self, iterator):
        self.it = iterator
        self._next = self.it.next() if self.it.hasNext() else None
    def peek(self): return self._next
    def next(self):
        val = self._next
        self._next = self.it.next() if self.it.hasNext() else None
        return val
    def hasNext(self): return self._next is not None
```

```typescript
class PeekingIterator {
  private _next: number | null
  constructor(private iterator: Iterator) {
    this._next = this.iterator.hasNext() ? this.iterator.next() : null
  }
  peek(): number {
    return this._next!
  }
  next(): number {
    const val = this._next!
    this._next = this.iterator.hasNext() ? this.iterator.next() : null
    return val
  }
  hasNext(): boolean {
    return this._next !== null
  }
}
```

## 287. 寻找重复数 (Find the Duplicate Number)

**描述**：n+1 个数在 [1,n] 范围内，只有一个重复（可能多次），O(1) 空间找出。

**思路**：快慢指针（Floyd 判圈），把值当作 next 指针，找环入口即为重复数。

```python
def findDuplicate(nums):
    slow = fast = nums[0]
    while True:
        slow = nums[slow]; fast = nums[nums[fast]]
        if slow == fast: break
    slow = nums[0]
    while slow != fast:
        slow = nums[slow]; fast = nums[fast]
    return slow
```

```typescript
function findDuplicate(nums: number[]): number {
  let slow = nums[0],
    fast = nums[0]
  do {
    slow = nums[slow]
    fast = nums[nums[fast]]
  } while (slow !== fast)
  slow = nums[0]
  while (slow !== fast) {
    slow = nums[slow]
    fast = nums[fast]
  }
  return slow
}
```

## 289. 生命游戏 (Game of Life)

**描述**：原地更新生命游戏。

**思路**：用 bit 编码（第2位存新状态），扫一遍更新，再右移一位。

```python
def gameOfLife(board):
    m, n = len(board), len(board[0])
    for i in range(m):
        for j in range(n):
            cnt = 0
            for di in (-1, 0, 1):
                for dj in (-1, 0, 1):
                    if di == 0 and dj == 0: continue
                    ni, nj = i+di, j+dj
                    if 0 <= ni < m and 0 <= nj < n:
                        cnt += board[ni][nj] & 1
            if board[i][j] & 1:
                if cnt in (2, 3): board[i][j] |= 2
            else:
                if cnt == 3: board[i][j] |= 2
    for i in range(m):
        for j in range(n):
            board[i][j] >>= 1
```

```typescript
function gameOfLife(board: number[][]): void {
  const m = board.length,
    n = board[0].length
  for (let i = 0; i < m; i++)
    for (let j = 0; j < n; j++) {
      let cnt = 0
      for (let di = -1; di <= 1; di++)
        for (let dj = -1; dj <= 1; dj++) {
          if (!di && !dj) continue
          const ni = i + di,
            nj = j + dj
          if (ni >= 0 && ni < m && nj >= 0 && nj < n) cnt += board[ni][nj] & 1
        }
      if (board[i][j] & 1) {
        if (cnt === 2 || cnt === 3) board[i][j] |= 2
      } else {
        if (cnt === 3) board[i][j] |= 2
      }
    }
  for (let i = 0; i < m; i++) for (let j = 0; j < n; j++) board[i][j] >>= 1
}
```

## 290. 单词规律 (Word Pattern)

**描述**：判断字符串是否遵循给定模式（双向映射）。

**思路**：两个哈希表互相映射。

```python
def wordPattern(pattern, s):
    words = s.split()
    if len(pattern) != len(words): return False
    return len(set(zip(pattern, words))) == len(set(pattern)) == len(set(words))
```

```typescript
function wordPattern(pattern: string, s: string): boolean {
  const words = s.split(' ')
  if (pattern.length !== words.length) return false
  const p2w = new Map<string, string>(),
    w2p = new Map<string, string>()
  for (let i = 0; i < pattern.length; i++) {
    const p = pattern[i],
      w = words[i]
    if ((p2w.has(p) && p2w.get(p) !== w) || (w2p.has(w) && w2p.get(w) !== p)) return false
    p2w.set(p, w)
    w2p.set(w, p)
  }
  return true
}
```

## 295. 数据流的中位数 (Find Median from Data Stream)

**描述**：支持动态添加数字并随时返回中位数。

**思路**：大顶堆存较小一半，小顶堆存较大一半，保持平衡。

```python
import heapq
class MedianFinder:
    def __init__(self):
        self.lo = []  # 大顶堆（取反）
        self.hi = []  # 小顶堆
    def addNum(self, num):
        heapq.heappush(self.lo, -num)
        heapq.heappush(self.hi, -heapq.heappop(self.lo))
        if len(self.hi) > len(self.lo):
            heapq.heappush(self.lo, -heapq.heappop(self.hi))
    def findMedian(self):
        if len(self.lo) > len(self.hi):
            return -self.lo[0]
        return (-self.lo[0] + self.hi[0]) / 2
```

```typescript
// 面试中可以用排序数组代替堆简化实现
class MedianFinder {
  private lo: number[] = [] // 大顶堆模拟：用sorted insert
  private hi: number[] = [] // sorted
  addNum(num: number): void {
    // 插入到lo（大顶堆用负数排序）
    this.lo.push(num)
    this.lo.sort((a, b) => b - a) // 降序
    this.hi.push(this.lo.shift()!)
    this.hi.sort((a, b) => a - b) // 升序
    if (this.hi.length > this.lo.length) {
      this.lo.push(this.hi.shift()!)
      this.lo.sort((a, b) => b - a)
    }
  }
  findMedian(): number {
    if (this.lo.length > this.hi.length) return this.lo[0]
    return (this.lo[0] + this.hi[0]) / 2
  }
}
```

## 297. 二叉树的序列化与反序列化 (Serialize and Deserialize Binary Tree)

**描述**：实现二叉树的序列化与反序列化。

**思路**：前序遍历，null 用特殊标记，反序列化递归构建。

```python
class Codec:
    def serialize(self, root):
        if not root: return "N"
        return str(root.val) + "," + self.serialize(root.left) + "," + self.serialize(root.right)
    def deserialize(self, data):
        vals = iter(data.split(","))
        def build():
            v = next(vals)
            if v == "N": return None
            node = TreeNode(int(v))
            node.left = build()
            node.right = build()
            return node
        return build()
```

```typescript
function serialize(root: TreeNode | null): string {
  if (!root) return 'N'
  return `${root.val},${serialize(root.left)},${serialize(root.right)}`
}
function deserialize(data: string): TreeNode | null {
  const vals = data.split(',')[Symbol.iterator]()
  function build(): TreeNode | null {
    const v = vals.next().value!
    if (v === 'N') return null
    const node = new TreeNode(Number(v))
    node.left = build()
    node.right = build()
    return node
  }
  return build()
}
```

## 299. 猜数字游戏 (Bulls and Cows)

**描述**：猜数字游戏，返回 "xAyB"（x 个位置和数字都对，y 个数字对但位置不对）。

**思路**：先扫一遍计 bulls，再用计数器统计 cows。

```python
def getHint(secret, guess):
    bulls = sum(s == g for s, g in zip(secret, guess))
    from collections import Counter
    cs, cg = Counter(secret), Counter(guess)
    total = sum((cs & cg).values())
    return f"{bulls}A{total - bulls}B"
```

```typescript
function getHint(secret: string, guess: string): string {
  let bulls = 0
  const cnt = new Array(10).fill(0)
  let cows = 0
  for (let i = 0; i < secret.length; i++) {
    if (secret[i] === guess[i]) bulls++
    else {
      if (cnt[+secret[i]] < 0) cows++
      if (cnt[+guess[i]] > 0) cows++
      cnt[+secret[i]]++
      cnt[+guess[i]]--
    }
  }
  return `${bulls}A${cows}B`
}
```

## 300. 最长递增子序列 (Longest Increasing Subsequence)

**描述**：找数组中最长严格递增子序列的长度。

**思路**：贪心+二分：维护 tails 数组，tails[i] 表示长度 i+1 的递增子序列的最小末尾，二分查找插入位置。O(n log n)。

```python
import bisect
def lengthOfLIS(nums):
    tails = []
    for n in nums:
        pos = bisect.bisect_left(tails, n)
        if pos == len(tails): tails.append(n)
        else: tails[pos] = n
    return len(tails)
```

```typescript
function lengthOfLIS(nums: number[]): number {
  const tails: number[] = []
  for (const n of nums) {
    let lo = 0,
      hi = tails.length
    while (lo < hi) {
      const mid = (lo + hi) >> 1
      if (tails[mid] < n) lo = mid + 1
      else hi = mid
    }
    tails[lo] = n
  }
  return tails.length
}
```

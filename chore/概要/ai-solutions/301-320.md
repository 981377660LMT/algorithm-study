## 301. 删除无效的括号 (Remove Invalid Parentheses)

**描述**：删除最少数量的无效括号，返回所有可能的有效字符串。

**思路**：BFS 逐层删除一个括号，第一层找到合法即停止；或先计算需删的左右括号数，DFS 回溯。

```python
def removeInvalidParentheses(s):
    def is_valid(s):
        cnt = 0
        for c in s:
            if c == '(': cnt += 1
            elif c == ')': cnt -= 1
            if cnt < 0: return False
        return cnt == 0
    # BFS
    level = {s}
    while True:
        valid = [x for x in level if is_valid(x)]
        if valid: return valid
        nxt = set()
        for x in level:
            for i in range(len(x)):
                if x[i] in '()': nxt.add(x[:i] + x[i+1:])
        level = nxt
```

```typescript
function removeInvalidParentheses(s: string): string[] {
  function isValid(s: string): boolean {
    let cnt = 0
    for (const c of s) {
      if (c === '(') cnt++
      else if (c === ')') cnt--
      if (cnt < 0) return false
    }
    return cnt === 0
  }
  let level = new Set([s])
  while (true) {
    const valid = [...level].filter(isValid)
    if (valid.length) return valid
    const nxt = new Set<string>()
    for (const x of level)
      for (let i = 0; i < x.length; i++)
        if (x[i] === '(' || x[i] === ')') nxt.add(x.slice(0, i) + x.slice(i + 1))
    level = nxt
  }
}
```

## 303. 区域和检索 - 数组不可变 (Range Sum Query - Immutable)

**描述**：多次查询数组区间和。

**思路**：前缀和预处理。

```python
class NumArray:
    def __init__(self, nums):
        self.pre = [0]
        for n in nums: self.pre.append(self.pre[-1] + n)
    def sumRange(self, left, right):
        return self.pre[right+1] - self.pre[left]
```

```typescript
class NumArray {
  private pre: number[]
  constructor(nums: number[]) {
    this.pre = [0]
    for (const n of nums) this.pre.push(this.pre[this.pre.length - 1] + n)
  }
  sumRange(left: number, right: number): number {
    return this.pre[right + 1] - this.pre[left]
  }
}
```

## 304. 二维区域和检索 - 矩阵不可变 (Range Sum Query 2D - Immutable)

**描述**：多次查询二维区间和。

**思路**：二维前缀和，容斥原理。

```python
class NumMatrix:
    def __init__(self, matrix):
        m, n = len(matrix), len(matrix[0])
        self.pre = [[0]*(n+1) for _ in range(m+1)]
        for i in range(m):
            for j in range(n):
                self.pre[i+1][j+1] = matrix[i][j] + self.pre[i][j+1] + self.pre[i+1][j] - self.pre[i][j]
    def sumRegion(self, r1, c1, r2, c2):
        return self.pre[r2+1][c2+1] - self.pre[r1][c2+1] - self.pre[r2+1][c1] + self.pre[r1][c1]
```

```typescript
class NumMatrix {
  private pre: number[][]
  constructor(matrix: number[][]) {
    const m = matrix.length,
      n = matrix[0].length
    this.pre = Array.from({ length: m + 1 }, () => new Array(n + 1).fill(0))
    for (let i = 0; i < m; i++)
      for (let j = 0; j < n; j++)
        this.pre[i + 1][j + 1] =
          matrix[i][j] + this.pre[i][j + 1] + this.pre[i + 1][j] - this.pre[i][j]
  }
  sumRegion(r1: number, c1: number, r2: number, c2: number): number {
    return this.pre[r2 + 1][c2 + 1] - this.pre[r1][c2 + 1] - this.pre[r2 + 1][c1] + this.pre[r1][c1]
  }
}
```

## 306. 累加数 (Additive Number)

**描述**：判断字符串是否为加法数（至少三个数，每个数等于前两个之和）。

**思路**：枚举前两个数的分割位置，然后验证后续是否符合。

```python
def isAdditiveNumber(num):
    n = len(num)
    for i in range(1, n):
        for j in range(i+1, n):
            a, b = num[:i], num[i:j]
            if (len(a) > 1 and a[0] == '0') or (len(b) > 1 and b[0] == '0'): continue
            while j < n:
                c = str(int(a) + int(b))
                if not num[j:].startswith(c): break
                j += len(c); a, b = b, c
            if j == n: return True
    return False
```

```typescript
function isAdditiveNumber(num: string): boolean {
  const n = num.length
  for (let i = 1; i < n; i++) {
    for (let j = i + 1; j < n; j++) {
      let a = num.slice(0, i),
        b = num.slice(i, j)
      if ((a.length > 1 && a[0] === '0') || (b.length > 1 && b[0] === '0')) continue
      let k = j
      while (k < n) {
        const c = String(BigInt(a) + BigInt(b))
        if (!num.startsWith(c, k)) break
        k += c.length
        a = b
        b = c
      }
      if (k === n) return true
    }
  }
  return false
}
```

## 307. 区域和检索 - 数组可修改 (Range Sum Query - Mutable)

**描述**：支持单点更新和区间求和。

**思路**：树状数组（BIT）。

```python
class NumArray:
    def __init__(self, nums):
        self.n = len(nums); self.nums = nums
        self.tree = [0] * (self.n + 1)
        for i, v in enumerate(nums): self._add(i + 1, v)
    def _add(self, i, delta):
        while i <= self.n: self.tree[i] += delta; i += i & (-i)
    def _query(self, i):
        s = 0
        while i > 0: s += self.tree[i]; i -= i & (-i)
        return s
    def update(self, index, val):
        self._add(index + 1, val - self.nums[index])
        self.nums[index] = val
    def sumRange(self, left, right):
        return self._query(right + 1) - self._query(left)
```

```typescript
class NumArray {
  private tree: number[]
  private nums: number[]
  private n: number
  constructor(nums: number[]) {
    this.n = nums.length
    this.nums = [...nums]
    this.tree = new Array(this.n + 1).fill(0)
    for (let i = 0; i < this.n; i++) this.add(i + 1, nums[i])
  }
  private add(i: number, delta: number) {
    for (; i <= this.n; i += i & -i) this.tree[i] += delta
  }
  private query(i: number) {
    let s = 0
    for (; i > 0; i -= i & -i) s += this.tree[i]
    return s
  }
  update(index: number, val: number) {
    this.add(index + 1, val - this.nums[index])
    this.nums[index] = val
  }
  sumRange(left: number, right: number) {
    return this.query(right + 1) - this.query(left)
  }
}
```

## 309. 买卖股票的最佳时机含冷冻期 (Best Time to Buy and Sell Stock with Cooldown)

**描述**：买卖股票，卖出后第二天不能买，求最大利润。

**思路**：状态机 DP，三个状态：持有 hold、不持有且冷冻 sold、不持有且不冷冻 rest。

```python
def maxProfit(prices):
    hold, sold, rest = float('-inf'), 0, 0
    for p in prices:
        hold, sold, rest = max(hold, rest - p), hold + p, max(rest, sold)
    return max(sold, rest)
```

```typescript
function maxProfit(prices: number[]): number {
  let hold = -Infinity,
    sold = 0,
    rest = 0
  for (const p of prices)
    [hold, sold, rest] = [Math.max(hold, rest - p), hold + p, Math.max(rest, sold)]
  return Math.max(sold, rest)
}
```

## 310. 最小高度树 (Minimum Height Trees)

**描述**：无向树中找使高度最小的根节点。

**思路**：拓扑排序剥叶子，最后剩 1~2 个节点即为答案。

```python
def findMinHeightTrees(n, edges):
    if n <= 2: return list(range(n))
    adj = [set() for _ in range(n)]
    for u, v in edges: adj[u].add(v); adj[v].add(u)
    leaves = [i for i in range(n) if len(adj[i]) == 1]
    remaining = n
    while remaining > 2:
        remaining -= len(leaves)
        new_leaves = []
        for leaf in leaves:
            nb = adj[leaf].pop()
            adj[nb].remove(leaf)
            if len(adj[nb]) == 1: new_leaves.append(nb)
        leaves = new_leaves
    return leaves
```

```typescript
function findMinHeightTrees(n: number, edges: number[][]): number[] {
  if (n <= 2) return Array.from({ length: n }, (_, i) => i)
  const adj: Set<number>[] = Array.from({ length: n }, () => new Set())
  for (const [u, v] of edges) {
    adj[u].add(v)
    adj[v].add(u)
  }
  let leaves = Array.from({ length: n }, (_, i) => i).filter(i => adj[i].size === 1)
  let remaining = n
  while (remaining > 2) {
    remaining -= leaves.length
    const newLeaves: number[] = []
    for (const leaf of leaves) {
      const nb = [...adj[leaf]][0]
      adj[nb].delete(leaf)
      if (adj[nb].size === 1) newLeaves.push(nb)
    }
    leaves = newLeaves
  }
  return leaves
}
```

## 312. 戳气球 (Burst Balloons)

**描述**：戳气球获得金币（左×当前×右），求最大金币。

**思路**：区间 DP，`dp[i][j]` 表示戳完 (i,j) 开区间内所有气球的最大金币，枚举最后被戳的气球 k。

```python
def maxCoins(nums):
    nums = [1] + nums + [1]
    n = len(nums)
    dp = [[0]*n for _ in range(n)]
    for length in range(2, n):
        for i in range(n - length):
            j = i + length
            for k in range(i+1, j):
                dp[i][j] = max(dp[i][j], dp[i][k] + dp[k][j] + nums[i]*nums[k]*nums[j])
    return dp[0][n-1]
```

```typescript
function maxCoins(nums: number[]): number {
  nums = [1, ...nums, 1]
  const n = nums.length
  const dp = Array.from({ length: n }, () => new Array(n).fill(0))
  for (let len = 2; len < n; len++)
    for (let i = 0; i + len < n; i++) {
      const j = i + len
      for (let k = i + 1; k < j; k++)
        dp[i][j] = Math.max(dp[i][j], dp[i][k] + dp[k][j] + nums[i] * nums[k] * nums[j])
    }
  return dp[0][n - 1]
}
```

## 313. 超级丑数 (Super Ugly Number)

**描述**：给定质因数集合 primes，找第 n 个超级丑数。

**思路**：类似 264 丑数 II，多指针 DP。

```python
def nthSuperUglyNumber(n, primes):
    dp = [1] * n
    ptrs = [0] * len(primes)
    for i in range(1, n):
        dp[i] = min(dp[ptrs[j]] * primes[j] for j in range(len(primes)))
        for j in range(len(primes)):
            if dp[ptrs[j]] * primes[j] == dp[i]: ptrs[j] += 1
    return dp[-1]
```

```typescript
function nthSuperUglyNumber(n: number, primes: number[]): number {
  const dp = new Array(n).fill(1)
  const ptrs = new Array(primes.length).fill(0)
  for (let i = 1; i < n; i++) {
    dp[i] = Math.min(...primes.map((p, j) => dp[ptrs[j]] * p))
    for (let j = 0; j < primes.length; j++) if (dp[ptrs[j]] * primes[j] === dp[i]) ptrs[j]++
  }
  return dp[n - 1]
}
```

## 315. 计算右侧小于当前元素的个数 (Count of Smaller Numbers After Self)

**描述**：返回数组中每个元素右侧比它小的元素个数。

**思路**：归并排序计逆序对，或 BIT + 离散化。

```python
def countSmaller(nums):
    # 归并排序
    res = [0] * len(nums)
    indexed = list(enumerate(nums))
    def merge_sort(arr):
        if len(arr) <= 1: return arr
        mid = len(arr) // 2
        left = merge_sort(arr[:mid])
        right = merge_sort(arr[mid:])
        merged = []; i = j = 0
        while i < len(left) and j < len(right):
            if left[i][1] <= right[j][1]:
                res[left[i][0]] += j
                merged.append(left[i]); i += 1
            else:
                merged.append(right[j]); j += 1
        while i < len(left):
            res[left[i][0]] += j
            merged.append(left[i]); i += 1
        while j < len(right):
            merged.append(right[j]); j += 1
        return merged
    merge_sort(indexed)
    return res
```

```typescript
function countSmaller(nums: number[]): number[] {
  const res = new Array(nums.length).fill(0)
  const indexed = nums.map((v, i) => [i, v] as [number, number])
  function mergeSort(arr: [number, number][]): [number, number][] {
    if (arr.length <= 1) return arr
    const mid = arr.length >> 1
    const left = mergeSort(arr.slice(0, mid))
    const right = mergeSort(arr.slice(mid))
    const merged: [number, number][] = []
    let i = 0,
      j = 0
    while (i < left.length && j < right.length) {
      if (left[i][1] <= right[j][1]) {
        res[left[i][0]] += j
        merged.push(left[i++])
      } else merged.push(right[j++])
    }
    while (i < left.length) {
      res[left[i][0]] += j
      merged.push(left[i++])
    }
    while (j < right.length) merged.push(right[j++])
    return merged
  }
  mergeSort(indexed)
  return res
}
```

## 316. 去除重复字母 (Remove Duplicate Letters)

**描述**：去除重复字母使结果字典序最小，每个字母只保留一个。

**思路**：单调栈 + 贪心，若栈顶大于当前且后面还有，弹出。

```python
def removeDuplicateLetters(s):
    last = {c: i for i, c in enumerate(s)}
    stack = []; seen = set()
    for i, c in enumerate(s):
        if c in seen: continue
        while stack and stack[-1] > c and last[stack[-1]] > i:
            seen.discard(stack.pop())
        stack.append(c); seen.add(c)
    return ''.join(stack)
```

```typescript
function removeDuplicateLetters(s: string): string {
  const last = new Map<string, number>()
  for (let i = 0; i < s.length; i++) last.set(s[i], i)
  const stack: string[] = [],
    seen = new Set<string>()
  for (let i = 0; i < s.length; i++) {
    if (seen.has(s[i])) continue
    while (
      stack.length &&
      stack[stack.length - 1] > s[i] &&
      last.get(stack[stack.length - 1])! > i
    ) {
      seen.delete(stack.pop()!)
    }
    stack.push(s[i])
    seen.add(s[i])
  }
  return stack.join('')
}
```

## 318. 最大单词长度乘积 (Maximum Product of Word Lengths)

**描述**：找两个不含公共字母的单词，长度乘积最大。

**思路**：位掩码表示每个单词包含的字母，两个掩码 AND 为 0 则无公共字母。

```python
def maxProduct(words):
    masks = [0] * len(words)
    for i, w in enumerate(words):
        for c in w: masks[i] |= 1 << (ord(c) - ord('a'))
    res = 0
    for i in range(len(words)):
        for j in range(i+1, len(words)):
            if masks[i] & masks[j] == 0:
                res = max(res, len(words[i]) * len(words[j]))
    return res
```

```typescript
function maxProduct(words: string[]): number {
  const masks = words.map(w => {
    let m = 0
    for (const c of w) m |= 1 << (c.charCodeAt(0) - 97)
    return m
  })
  let res = 0
  for (let i = 0; i < words.length; i++)
    for (let j = i + 1; j < words.length; j++)
      if (!(masks[i] & masks[j])) res = Math.max(res, words[i].length * words[j].length)
  return res
}
```

## 319. 灯泡开关 (Bulb Switcher)

**描述**：n 个灯泡，第 i 轮翻转 i 的倍数，最终亮着多少盏。

**思路**：只有完全平方数被翻转奇数次（因约数成对，平方数有一个不成对），答案 = `floor(sqrt(n))`。

```python
def bulbSwitch(n):
    return int(n ** 0.5)
```

```typescript
function bulbSwitch(n: number): number {
  return Math.floor(Math.sqrt(n))
}
```
